62 ethernet_init(void)
63 {
64  static const char signature[4] = {0x65, 0x74, 0x68, 0x01};
65 
66 #ifdef STATIC_DRIVER
67 
68  extern void STATIC_DRIVER;
69 
70  module = &STATIC_DRIVER;
71 
72  module->buffer = uip_buf;
73  module->buffer_size = UIP_BUFSIZE;
74  if(module->init(config.ethernet.addr)) {
75  #define _stringize(arg) #arg
76  #define stringize(arg) _stringize(arg)
77  log_message(stringize(STATIC_DRIVER), ": No hardware");
78  #undef _stringize
79  #undef stringize
80  error_exit();
81  }
82 
83 #else /* STATIC_DRIVER */
84 
85  struct mod_ctrl module_control = {cfs_read};
86  uint8_t byte;
87 
88  module_control.callerdata = cfs_open(config.ethernet.name, CFS_READ);
89  if(module_control.callerdata < 0) {
90  log_message(config.ethernet.name, ": File not found");
91  error_exit();
92  }
93 
94  byte = mod_load(&module_control);
95  if(byte != MLOAD_OK) {
96  log_message(config.ethernet.name, byte == MLOAD_ERR_MEM? ": Out of memory":
97  ": No module");
98  error_exit();
99  }
100 
101  cfs_close(module_control.callerdata);
102  module = module_control.module;
103 
104  for(byte = 0; byte < 4; ++byte) {
105  if(module->signature[byte] != signature[byte]) {
106  log_message(config.ethernet.name, ": No ETH driver");
107  error_exit();
108  }
109  }
110 
111  module->buffer = uip_buf;
112  module->buffer_size = UIP_BUFSIZE;
113  if(module->init(config.ethernet.addr)) {
114  log_message(config.ethernet.name, ": No hardware");
115  error_exit();
116  }
117 
118 #endif /* STATIC_DRIVER */
119 
120  uip_setethaddr(module->ethernet_address);
121 }
