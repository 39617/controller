248 PROCESS_THREAD(lwm2m_rd_client, ev, data)
249 {
250  static coap_packet_t request[1]; /* This way the packet can be treated as pointer as usual. */
251  static struct etimer et;
252 
253  PROCESS_BEGIN();
254 
255  printf("RD Client started with endpoint '%s'\n", endpoint);
256 
257  etimer_set(&et, 15 * CLOCK_SECOND);
258 
259  while(1) {
260  PROCESS_YIELD();
261 
262  if(etimer_expired(&et)) {
263  if(!has_network_access()) {
264  /* Wait until for a network to join */
265  } else if(use_bootstrap && bootstrapped == 0) {
266  if(update_bootstrap_server()) {
267  /* prepare request, TID is set by COAP_BLOCKING_REQUEST() */
268  coap_init_message(request, COAP_TYPE_CON, COAP_POST, 0);
269  coap_set_header_uri_path(request, "/bs");
270  coap_set_header_uri_query(request, endpoint);
271 
272  printf("Registering ID with bootstrap server [");
273  uip_debug_ipaddr_print(&bs_server_ipaddr);
274  printf("]:%u as '%s'\n", uip_ntohs(bs_server_port), endpoint);
275 
276  COAP_BLOCKING_REQUEST(&bs_server_ipaddr, bs_server_port, request,
277  client_chunk_handler);
278  bootstrapped++;
279  }
280  } else if(use_bootstrap && bootstrapped == 1) {
281  lwm2m_context_t context;
282  const lwm2m_instance_t *instance = NULL;
283  const lwm2m_resource_t *rsc;
284  const uint8_t *first;
285  int len;
286 
287  PRINTF("*** Bootstrap - checking for server info...\n");
288 
289  /* get the security object */
290  instance = get_first_instance_of_object(LWM2M_OBJECT_SECURITY_ID, &context);
291  if(instance != NULL) {
292  /* get the server URI */
293  context.resource_id = LWM2M_SECURITY_SERVER_URI;
294  rsc = get_resource(instance, &context);
295  first = lwm2m_object_get_resource_string(rsc, &context);
296  len = lwm2m_object_get_resource_strlen(rsc, &context);
297  if(first != NULL && len > 0) {
298  int start, end;
299  uip_ipaddr_t addr;
300  int32_t port;
301  uint8_t secure = 0;
302 
303  PRINTF("**** Found security instance using: %.*s\n", len, first);
304  /* TODO Should verify it is a URI */
305 
306  /* Check if secure */
307  secure = strncmp((const char *)first, "coaps:", 6) == 0;
308 
309  /* Only IPv6 supported */
310  start = index_of(first, 0, len, '[');
311  end = index_of(first, start, len, ']');
312  if(start > 0 && end > start &&
313  uiplib_ipaddrconv((const char *)&first[start], &addr)) {
314  if(first[end + 1] == ':' &&
315  lwm2m_plain_text_read_int(first + end + 2, len - end - 2, &port)) {
316  } else if(secure) {
321  port = COAP_DEFAULT_PORT;
322  } else {
323  port = COAP_DEFAULT_PORT;
324  }
325  PRINTF("Server address ");
326  PRINT6ADDR(&addr);
327  PRINTF(" port %" PRId32 "%s\n", port, secure ? " (secure)" : "");
328  if(secure) {
329  printf("Secure CoAP requested but not supported - can not bootstrap\n");
330  } else {
331  lwm2m_engine_register_with_server(&addr,
332  UIP_HTONS((uint16_t)port));
333  bootstrapped++;
334  }
335  } else {
336  printf("** failed to parse URI %.*s\n", len, first);
337  }
338  }
339  }
340 
341  if(bootstrapped == 1) {
342  /* Not ready. Lets retry with the bootstrap server again */
343  bootstrapped = 0;
344  }
345 
346  } else if(use_registration && !registered &&
347  update_registration_server()) {
348  int pos;
349  int len, i, j;
350  registered = 1;
351 
352  /* prepare request, TID is set by COAP_BLOCKING_REQUEST() */
353  coap_init_message(request, COAP_TYPE_CON, COAP_POST, 0);
354  coap_set_header_uri_path(request, "/rd");
355  coap_set_header_uri_query(request, endpoint);
356 
357  /* generate the rd data */
358  pos = 0;
359  for(i = 0; i < MAX_OBJECTS; i++) {
360  if(objects[i] != NULL) {
361  for(j = 0; j < objects[i]->count; j++) {
362  if(objects[i]->instances[j].flag & LWM2M_INSTANCE_FLAG_USED) {
363  len = snprintf(&rd_data[pos], sizeof(rd_data) - pos,
364  "%s<%d/%d>", pos > 0 ? "," : "",
365  objects[i]->id, objects[i]->instances[j].id);
366  if(len > 0 && len < sizeof(rd_data) - pos) {
367  pos += len;
368  }
369  }
370  }
371  }
372  }
373 
374  coap_set_payload(request, (uint8_t *)rd_data, pos);
375 
376  printf("Registering with [");
377  uip_debug_ipaddr_print(&server_ipaddr);
378  printf("]:%u lwm2m endpoint '%s': '%.*s'\n", uip_ntohs(server_port),
379  endpoint, pos, rd_data);
380  COAP_BLOCKING_REQUEST(&server_ipaddr, server_port, request,
381  client_chunk_handler);
382  }
383  /* for now only register once... registered = 0; */
384  etimer_set(&et, 15 * CLOCK_SECOND);
385  }
386  }
387  PROCESS_END();
388 }

390 void
391 lwm2m_engine_init(void)
392 {
393 #ifdef LWM2M_ENGINE_CLIENT_ENDPOINT_NAME
394 
395  snprintf(endpoint, sizeof(endpoint) - 1,
396  "?ep=" LWM2M_ENGINE_CLIENT_ENDPOINT_NAME);
397 
398 #else /* LWM2M_ENGINE_CLIENT_ENDPOINT_NAME */
399 
400  int len, i;
401  uint8_t state;
402  uip_ipaddr_t *ipaddr;
403  char client[sizeof(endpoint)];
404 
405  len = strlen(LWM2M_ENGINE_CLIENT_ENDPOINT_PREFIX);
406  /* ensure that this fits with the hex-nums */
407  if(len > sizeof(client) - 13) {
408  len = sizeof(client) - 13;
409  }
410  memcpy(client, LWM2M_ENGINE_CLIENT_ENDPOINT_PREFIX, len);
411 
412  /* pick an IP address that is PREFERRED or TENTATIVE */
413  ipaddr = NULL;
414  for(i = 0; i < UIP_DS6_ADDR_NB; i++) {
415  state = uip_ds6_if.addr_list[i].state;
416  if(uip_ds6_if.addr_list[i].isused &&
417  (state == ADDR_TENTATIVE || state == ADDR_PREFERRED)) {
418  ipaddr = &(uip_ds6_if.addr_list[i]).ipaddr;
419  break;
420  }
421  }
422 
423  if(ipaddr != NULL) {
424  for(i = 0; i < 6; i++) {
425  /* assume IPv6 for now */
426  uint8_t b = ipaddr->u8[10 + i];
427  client[len++] = (b >> 4) > 9 ? 'A' - 10 + (b >> 4) : '0' + (b >> 4);
428  client[len++] = (b & 0xf) > 9 ? 'A' - 10 + (b & 0xf) : '0' + (b & 0xf);
429  }
430  }
431 
432  /* a zero at end of string */
433  client[len] = 0;
434  /* create endpoint */
435  snprintf(endpoint, sizeof(endpoint) - 1, "?ep=%s", client);
436 
437 #endif /* LWM2M_ENGINE_CLIENT_ENDPOINT_NAME */
438 
439  rest_init_engine();
440  process_start(&lwm2m_rd_client, NULL);
441 }

451 static int
452 parse_next(const char **path, int *path_len, uint16_t *value)
453 {
454  char c;
455  *value = 0;
456  /* printf("parse_next: %p %d\n", *path, *path_len); */
457  if(*path_len == 0) {
458  return 0;
459  }
460  while(*path_len > 0) {
461  c = **path;
462  (*path)++;
463  *path_len = *path_len - 1;
464  if(c >= '0' && c <= '9') {
465  *value = *value * 10 + (c - '0');
466  } else if(c == '/') {
467  return 1;
468  } else {
469  /* error */
470  return -4;
471  }
472  }
473  return 1;
474 }

478  const char *path, int path_len,
479  lwm2m_context_t *context)
480 {
481  int ret;
482  if(context == NULL || object == NULL || path == NULL) {

486  /* get object id */
487  ret = 0;
488  ret += parse_next(&path, &path_len, &context->object_id);
489  ret += parse_next(&path, &path_len, &context->object_instance_id);
490  ret += parse_next(&path, &path_len, &context->resource_id);
491 
492  /* Set default reader/writer */
493  context->reader = &lwm2m_plain_text_reader;
494  context->writer = &oma_tlv_writer;
495 
496  return ret;
497 }

669  for(i = 0, len = 0; i < instance->count; i++) {
670  resource = &instance->resources[i];
671  len = 0;
672  if(lwm2m_object_is_resource_string(resource)) {
673  const uint8_t *value;
674  uint16_t slen;
675  value = lwm2m_object_get_resource_string(resource, context);
676  slen = lwm2m_object_get_resource_strlen(resource, context);
677  if(value != NULL) {
678  PRINTF("%s{\"n\":\"%u\",\"sv\":\"%.*s\"}", s,
679  resource->id, slen, value);
680  len = snprintf(&buffer[rdlen], size - rdlen,
681  "%s{\"n\":\"%u\",\"sv\":\"%.*s\"}", s,
682  resource->id, slen, value);
683  }
684  } else if(lwm2m_object_is_resource_int(resource)) {
685  int32_t value;

689  len = snprintf(&buffer[rdlen], size - rdlen,
690  "%s{\"n\":\"%u\",\"v\":%" PRId32 "}", s,
691  resource->id, value);
692  }
693  } else if(lwm2m_object_is_resource_floatfix(resource)) {

696  PRINTF("%s{\"n\":\"%u\",\"v\":%" PRId32 "}", s, resource->id,

698  len = snprintf(&buffer[rdlen], size - rdlen,
699  "%s{\"n\":\"%u\",\"v\":", s, resource->id);
700  rdlen += len;
701  if(len < 0 || rdlen >= size) {
702  return -1;
703  }
704 
705  len = lwm2m_plain_text_write_float32fix((uint8_t *)&buffer[rdlen],
706  size - rdlen,
707  value, LWM2M_FLOAT32_BITS);
708  if(len == 0) {
709  return -1;
710  }
711  rdlen += len;
712 
713  if(rdlen < size) {
714  buffer[rdlen] = '}';
715  }
716  len = 1;
717  }
718  } else if(lwm2m_object_is_resource_boolean(resource)) {
719  int value;
720  if(lwm2m_object_get_resource_boolean(resource, context, &value)) {
721  PRINTF("%s{\"n\":\"%u\",\"bv\":%s}", s, resource->id,
722  value ? "true" : "false");
723  len = snprintf(&buffer[rdlen], size - rdlen,
724  "%s{\"n\":\"%u\",\"bv\":%s}", s, resource->id,
725  value ? "true" : "false");
726  }
727  }
728  rdlen += len;
729  if(len < 0 || rdlen >= size) {
730  return -1;
731  }
732  if(len > 0) {
733  s = ",";
734  }
735  }
736  PRINTF("]}\n");
737  len = snprintf(&buffer[rdlen], size - rdlen, "]}");
738  rdlen += len;
739  if(len < 0 || rdlen >= size) {
740  return -1;
741  }
742 
743  return rdlen;

754 static unsigned int
755 lwm2m_engine_select_writer(lwm2m_context_t *context, unsigned int accept)
756 {
757  switch(accept) {
758  case LWM2M_TLV:
759  context->writer = &oma_tlv_writer;
760  break;
761  case LWM2M_TEXT_PLAIN:
762  case TEXT_PLAIN:
763  context->writer = &lwm2m_plain_text_writer;
764  break;
765  case LWM2M_JSON:
766  case APPLICATION_JSON:
767  context->writer = &lwm2m_json_writer;
768  break;
769  default:
770  PRINTF("Unknown Accept type %u, using LWM2M plain text\n", accept);
771  context->writer = &lwm2m_plain_text_writer;
772  /* Set the response type to plain text */
773  accept = LWM2M_TEXT_PLAIN;
774  break;
775  }
776  return accept;
777 }

786 lwm2m_engine_select_reader(lwm2m_context_t *context, unsigned int content_format)
787 {

790  context->reader = &oma_tlv_reader;
791  break;
792  case LWM2M_TEXT_PLAIN:
793  case TEXT_PLAIN:
794  context->reader = &lwm2m_plain_text_reader;
795  break;
796  default:
797  PRINTF("Unknown content type %u, using LWM2M plain text\n", accept);
798  context->reader = &lwm2m_plain_text_reader;
799  break;
800  }

802 /*---------------------------------------------------------------------------*/
803 void
804 lwm2m_engine_handler(const lwm2m_object_t *object,
805  void *request, void *response,
806  uint8_t *buffer, uint16_t preferred_size,
807  int32_t *offset)
808 {
809  int len;
810  const char *url;
811  unsigned int format;
812  unsigned int accept;
813  unsigned int content_type;
814  int depth;
815  lwm2m_context_t context;
816  rest_resource_flags_t method;
817  const lwm2m_instance_t *instance;
818 #if (DEBUG) & DEBUG_PRINT
819  const char *method_str;
820 #endif /* (DEBUG) & DEBUG_PRINT */
821 
822  method = REST.get_method_type(request);
823 
824  len = REST.get_url(request, &url);
825  if(!REST.get_header_content_type(request, &format)) {
826  PRINTF("No format given. Assume text plain...\n");
827  format = LWM2M_TEXT_PLAIN;
828  } else if(format == TEXT_PLAIN) {
829  /* CoAP content format text plain - assume LWM2M text plain */
830  format = LWM2M_TEXT_PLAIN;
831  }
832  if(!REST.get_header_accept(request, &accept)) {
833  PRINTF("No Accept header, using same as Content-format...\n");
834  accept = format;
835  }
836 
837  depth = lwm2m_engine_parse_context(object, url, len, &context);
838  PRINTF("Context: %u/%u/%u found: %d\n", context.object_id,
839  context.object_instance_id, context.resource_id, depth);
840 
841  /* Select reader and writer based on provided Content type and Accept headers */
842  lwm2m_engine_select_reader(&context, format);
843  content_type = lwm2m_engine_select_writer(&context, accept);
844 
845 #if (DEBUG) & DEBUG_PRINT
846  /* for debugging */
847  if(method == METHOD_GET) {
848  method_str = "GET";
849  } else if(method == METHOD_POST) {
850  method_str = "POST";
851  } else if(method == METHOD_PUT) {
852  method_str = "PUT";
853  } else if(method == METHOD_DELETE) {
854  method_str = "DELETE";
855  } else {
856  method_str = "UNKNOWN";
857  }
858  PRINTF("%s Called Path:%.*s Format:%d ID:%d bsize:%u\n", method_str, len,
859  url, format, object->id, preferred_size);
860  if(format == LWM2M_TEXT_PLAIN) {
861  /* a string */
862  const uint8_t *data;
863  int plen = REST.get_request_payload(request, &data);
864  if(plen > 0) {
865  PRINTF("Data: '%.*s'\n", plen, (char *)data);
866  }
867  }
868 #endif /* (DEBUG) & DEBUG_PRINT */
869 
870  instance = get_instance(object, &context, depth);
871 
872  /* from POST */
873  if(depth > 1 && instance == NULL) {
874  if(method != METHOD_PUT && method != METHOD_POST) {
875  PRINTF("Error - do not have instance %d\n", context.object_instance_id);
876  REST.set_response_status(response, NOT_FOUND_4_04);
877  return;
878  } else {
879  const uint8_t *data;
880  int i, len, plen, pos;
881  oma_tlv_t tlv;
882  PRINTF(">>> CREATE ? %d/%d\n", context.object_id,
883  context.object_instance_id);
884 
885  for(i = 0; i < object->count; i++) {
886  if((object->instances[i].flag & LWM2M_INSTANCE_FLAG_USED) == 0) {
887  /* allocate this instance */
888  object->instances[i].flag |= LWM2M_INSTANCE_FLAG_USED;
889  object->instances[i].id = context.object_instance_id;
890  context.object_instance_index = i;
891  PRINTF("Created instance: %d\n", context.object_instance_id);
892  REST.set_response_status(response, CREATED_2_01);
893  instance = &object->instances[i];
894  break;
895  }
896  }
897 
898  if(instance == NULL) {
899  /* could for some reason not create the instance */
900  REST.set_response_status(response, NOT_ACCEPTABLE_4_06);
901  return;
902  }
903 
904  plen = REST.get_request_payload(request, &data);
905  if(plen == 0) {
906  /* do nothing more */
907  return;
908  }
909  PRINTF("Payload: ");
910  for(i = 0; i < plen; i++) {
911  PRINTF("%02x", data[i]);
912  }
913  PRINTF("\n");
914 
915  pos = 0;
916  do {
917  len = oma_tlv_read(&tlv, (uint8_t *)&data[pos], plen - pos);
918  PRINTF("Found TLV type=%u id=%u len=%lu\n",
919  tlv.type, tlv.id, (unsigned long)tlv.length);
920  /* here we need to do callbacks or write value */
921  if(tlv.type == OMA_TLV_TYPE_RESOURCE) {
922  context.resource_id = tlv.id;
923  const lwm2m_resource_t *rsc = get_resource(instance, &context);
924  if(rsc != NULL) {
925  /* write the value to the resource */
926  if(lwm2m_object_is_resource_string(rsc)) {
927  PRINTF(" new string value for /%d/%d/%d = %.*s\n",
928  context.object_id, context.object_instance_id,
929  context.resource_id, (int)tlv.length, tlv.value);
930  lwm2m_object_set_resource_string(rsc, &context,
931  tlv.length, tlv.value);
932  } else if(lwm2m_object_is_resource_int(rsc)) {
933  PRINTF(" new int value for /%d/%d/%d = %" PRId32 "\n",
934  context.object_id, context.object_instance_id,
935  context.resource_id, oma_tlv_get_int32(&tlv));
936  lwm2m_object_set_resource_int(rsc, &context,
937  oma_tlv_get_int32(&tlv));
938  } else if(lwm2m_object_is_resource_floatfix(rsc)) {
939  int32_t value;
940  if(oma_tlv_float32_to_fix(&tlv, &value, LWM2M_FLOAT32_BITS)) {
941  PRINTF(" new float value for /%d/%d/%d = %" PRId32 "\n",
942  context.object_id, context.object_instance_id,
943  context.resource_id, value >> LWM2M_FLOAT32_BITS);
944  lwm2m_object_set_resource_floatfix(rsc, &context, value);
945  } else {
946  PRINTF(" new float value for /%d/%d/%d: FAILED\n",
947  context.object_id, context.object_instance_id,
948  context.resource_id);
949  }
950  } else if(lwm2m_object_is_resource_boolean(rsc)) {
951  PRINTF(" new boolean value for /%d/%d/%d = %" PRId32 "\n",
952  context.object_id, context.object_instance_id,
953  context.resource_id, oma_tlv_get_int32(&tlv));
954  lwm2m_object_set_resource_boolean(rsc, &context,
955  oma_tlv_get_int32(&tlv) != 0);
956  }
957  }
958  }
959  pos = pos + len;
960  } while(len > 0 && pos < plen);
961  }
962  return;
963  }
964 
965  if(depth == 3) {
966  const lwm2m_resource_t *resource = get_resource(instance, &context);
967  size_t content_len = 0;
968  if(resource == NULL) {
969  PRINTF("Error - do not have resource %d\n", context.resource_id);
970  REST.set_response_status(response, NOT_FOUND_4_04);
971  return;
972  }
973  /* HANDLE PUT */
974  if(method == METHOD_PUT) {
975  if(lwm2m_object_is_resource_callback(resource)) {
976  if(resource->value.callback.write != NULL) {
977  /* pick a reader ??? */
978  if(format == LWM2M_TEXT_PLAIN) {
979  /* a string */
980  const uint8_t *data;
981  int plen = REST.get_request_payload(request, &data);
982  context.reader = &lwm2m_plain_text_reader;
983  PRINTF("PUT Callback with data: '%.*s'\n", plen, data);
984  /* no specific reader for plain text */
985  content_len = resource->value.callback.write(&context, data, plen,
986  buffer, preferred_size);
987  PRINTF("content_len:%u\n", (unsigned int)content_len);
988  REST.set_response_status(response, CHANGED_2_04);
989  } else {
990  PRINTF("PUT callback with format %d\n", format);
991  REST.set_response_status(response, NOT_ACCEPTABLE_4_06);
992  }
993  } else {
994  PRINTF("PUT - no write callback\n");
995  REST.set_response_status(response, METHOD_NOT_ALLOWED_4_05);
996  }
997  } else {
998  PRINTF("PUT on non-callback resource!\n");
999  REST.set_response_status(response, METHOD_NOT_ALLOWED_4_05);
1000  }
1001  /* HANDLE GET */
1002  } else if(method == METHOD_GET) {
1003  if(lwm2m_object_is_resource_string(resource)) {
1004  const uint8_t *value;
1005  value = lwm2m_object_get_resource_string(resource, &context);
1006  if(value != NULL) {
1007  uint16_t len = lwm2m_object_get_resource_strlen(resource, &context);
1008  PRINTF("Get string value: %.*s\n", (int)len, (char *)value);
1009  content_len = context.writer->write_string(&context, buffer,
1010  preferred_size, (const char *)value, len);
1011  }
1012  } else if(lwm2m_object_is_resource_int(resource)) {
1013  int32_t value;
1014  if(lwm2m_object_get_resource_int(resource, &context, &value)) {
1015  content_len = context.writer->write_int(&context, buffer, preferred_size, value);
1016  }
1017  } else if(lwm2m_object_is_resource_floatfix(resource)) {
1018  int32_t value;
1019  if(lwm2m_object_get_resource_floatfix(resource, &context, &value)) {
1020  /* export FLOATFIX */
1021  PRINTF("Exporting %d-bit fix as float: %" PRId32 "\n",
1022  LWM2M_FLOAT32_BITS, value);
1023  content_len = context.writer->write_float32fix(&context, buffer,
1024  preferred_size, value, LWM2M_FLOAT32_BITS);
1025  }
1026  } else if(lwm2m_object_is_resource_callback(resource)) {
1027  if(resource->value.callback.read != NULL) {
1028  content_len = resource->value.callback.read(&context,
1029  buffer, preferred_size);
1030  } else {
1031  REST.set_response_status(response, METHOD_NOT_ALLOWED_4_05);
1032  return;
1033  }
1034  }
1035  if(content_len > 0) {
1036  REST.set_response_payload(response, buffer, content_len);
1037  REST.set_header_content_type(response, content_type);
1038  } else {
1039  /* failed to produce output - it is an internal error */
1040  REST.set_response_status(response, INTERNAL_SERVER_ERROR_5_00);
1041  }
1042  /* Handle POST */
1043  } else if(method == METHOD_POST) {
1044  if(lwm2m_object_is_resource_callback(resource)) {
1045  if(resource->value.callback.exec != NULL) {
1046  const uint8_t *data;
1047  int plen = REST.get_request_payload(request, &data);
1048  PRINTF("Execute Callback with data: '%.*s'\n", plen, data);
1049  content_len = resource->value.callback.exec(&context,
1050  data, plen,
1051  buffer, preferred_size);
1052  REST.set_response_status(response, CHANGED_2_04);
1053  } else {
1054  PRINTF("Execute callback - no exec callback\n");
1055  REST.set_response_status(response, METHOD_NOT_ALLOWED_4_05);
1056  }
1057  } else {
1058  PRINTF("Resource post but no callback resource\n");
1059  REST.set_response_status(response, METHOD_NOT_ALLOWED_4_05);
1060  }
1061  }
1062  } else if(depth == 2) {
1063  /* produce an instance response */
1064  if(method != METHOD_GET) {
1065  REST.set_response_status(response, METHOD_NOT_ALLOWED_4_05);
1066  } else if(instance == NULL) {
1067  REST.set_response_status(response, NOT_FOUND_4_04);
1068  } else {
1069  int rdlen;
1070  if(accept == APPLICATION_LINK_FORMAT) {
1071  rdlen = write_rd_link_data(object, instance,
1072  (char *)buffer, preferred_size);
1073  } else {
1074  rdlen = write_rd_json_data(&context, object, instance,
1075  (char *)buffer, preferred_size);
1076  }
1077  if(rdlen < 0) {
1078  PRINTF("Failed to generate instance response\n");
1079  REST.set_response_status(response, SERVICE_UNAVAILABLE_5_03);
1080  return;
1081  }
1082  REST.set_response_payload(response, buffer, rdlen);
1083  if(accept == APPLICATION_LINK_FORMAT) {
1084  REST.set_header_content_type(response, REST.type.APPLICATION_LINK_FORMAT);
1085  } else {
1086  REST.set_header_content_type(response, LWM2M_JSON);
1087  }
1088  }
1089  } else if(depth == 1) {
1090  /* produce a list of instances */
1091  if(method != METHOD_GET) {
1092  REST.set_response_status(response, METHOD_NOT_ALLOWED_4_05);
1093  } else {
1094  int rdlen;
1095  PRINTF("Sending instance list for object %u\n", object->id);
1096  /* TODO: if(accept == APPLICATION_LINK_FORMAT) { */
1097  rdlen = write_object_instances_link(object, (char *)buffer, preferred_size);
1098  if(rdlen < 0) {
1099  PRINTF("Failed to generate object response\n");
1100  REST.set_response_status(response, SERVICE_UNAVAILABLE_5_03);
1101  return;
1102  }
1103  REST.set_header_content_type(response, REST.type.APPLICATION_LINK_FORMAT);
1104  REST.set_response_payload(response, buffer, rdlen);
1105  }
1106  }
1107 }
