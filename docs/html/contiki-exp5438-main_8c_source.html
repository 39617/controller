73 static void
74 set_rime_addr(void)
75 {
76  linkaddr_t addr;
77  int i;
78 
79  memset(&addr, 0, sizeof(linkaddr_t));
80 #if NETSTACK_CONF_WITH_IPV6
81  memcpy(addr.u8, node_mac, sizeof(addr.u8));
82 #else
83  if(node_id == 0) {
84  for(i = 0; i < sizeof(linkaddr_t); ++i) {
85  addr.u8[i] = node_mac[7 - i];
86  }
87  } else {
88  addr.u8[0] = node_id & 0xff;
89  addr.u8[1] = node_id >> 8;
90  }
91 #endif
92  linkaddr_set_node_addr(&addr);
93  printf("Rime addr ");
94  for(i = 0; i < sizeof(addr.u8) - 1; i++) {
95  printf("%u.", addr.u8[i]);
96  }
97  printf("%u\n", addr.u8[i]);
98 }

322

325  dint();
326  irq_energest = energest_type_time(ENERGEST_TYPE_IRQ);
327  eint();
328  watchdog_start();
329  ENERGEST_SWITCH(ENERGEST_TYPE_LPM, ENERGEST_TYPE_CPU);
330  }
