639  dataBuffer++;

642  } else { /* Acknowledge was not detected. */
643  status = 0xFF;
644  }

646  SPT0 = 1; /* Generate a stop condition. */
647  while(IICBSY0) ; /* Wait until the I2C bus status flag is cleared. */
648  }
