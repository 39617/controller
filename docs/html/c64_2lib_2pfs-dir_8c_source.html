61 int __fastcall__
62 pfs_opendir(struct cfs_dir *dirp, register const char *name)
63 {
64  static int fd;
65  static char buf[2 + 1] = "$";
66 
67  /* Set up the actual file name that is sent to the DOS.
68  ** We accept "0:", "1:", "/", and "." as directory names.
69  */
70  if(name == NULL || name[0] == '\0' || (name[0] == '.' || name[0] == '/') && name[1] == '\0') {
71  buf[1] = '\0';
72  } else if((name[0] == '0' || name[0] == '1') && name[1] == ':' && name[2] == '\0') {
73  buf[1] = name[0];
74  buf[2] = '\0';
75  } else {
76  return -1;
77  }
78 
79  /* Open the directory on a disk, for reading. */
80  fd = pfs_open(buf, CFS_READ);
81  if(fd >= 0) {
82  ((DIR *)dirp)->fd = fd;
83 
84  /* Skip the load address. */
85  if(_pfs_dirread(dirp, buf + 1, 2)) {
86  return 0;
87  } else {
88  pfs_close(fd);
89  }
90  }
91 
92  return -1;
93 }

96 pfs_readdir(struct cfs_dir *dirp, register struct cfs_dirent *dirent)
97 {
98  register unsigned char *b;
99  register unsigned char i;
100  register unsigned char count;
101  unsigned char buffer[0x40];
102  static unsigned char s;
103  static unsigned char j;
104 
105  /* Skip the BASIC line-link. */
106  if(!_pfs_dirread(dirp, buffer, 2)) {
107  return -1;
108  }
109 
110  /* Read the number of blocks. It's a two-byte number; but, the size is
111  ** a four-byte number. Zero the size; then, put the block number in
112  ** the first two bytes. It works because the 6502 CPU's numbers are
113  ** little-endian.
114  */
115  dirent->size = 0;
116  if(!_pfs_dirread(dirp, &dirent->size, 2)) {
117  return -1;
118  }
119 
120  /* Read the next file entry into a buffer. */
121  for(count = 0, b = buffer; count < sizeof(buffer); ++b) {
122  if(!_pfs_dirread1(dirp, b)) {
123  return -1;
124  }
125  ++count;
126  if(*b == '\0') {
127  break;
128  }
129  }
130 
131  /* The end of the directory was reached if the buffer contains "blocks free."
132  ** It is sufficient here to check for the leading 'b'. The buffer will have
133  ** at least one byte if we come here.
134  */
135  if(buffer[0] == 'b') {
136  return -1;
137  }
138 
139  /* Parse the buffer for the file-name. */
140  b = buffer;
141  j = 0;
142  s = 0;
143  i = 0;
144  while(i < count) {
145  switch(s) {
146  case 0:
147  /* Search for the start of the file-name. */
148  if(*b == '"') {
149  s = 1;
150  }
151  break;
152  case 1:
153  /* Within the file-name. */
154  if(*b == '"') {
155  /* The end of the file-name was found. */
156  dirent->name[j] = '\0';
157  return 0;
158  } else if(j < sizeof(dirent->name) - 1) {
159  dirent->name[j] = *b;
160  ++j;
161  }
162  break;
163  }
164  ++b;
165  ++i;
166  }
167 
168  /* The file-name is too long for the buffer. Return what could be read. */
169  dirent->name[j] = '\0';
170  return 0;
171 }
