65 PROCESS_THREAD(sensor_demo_process, ev, data)
66 {
67  static struct etimer etimer;
68  static unsigned long _button_pressed;
69  static int sensor_value = 0;
70 
71  PROCESS_BEGIN();
72  PROCESS_PAUSE();
73 
74  SENSORS_ACTIVATE(button_sensor);
75 
76  SENSORS_ACTIVATE(radio_sensor);
77 
78 #ifdef X_NUCLEO_IKS01A1
79  SENSORS_ACTIVATE(temperature_sensor);
80  SENSORS_ACTIVATE(humidity_sensor);
81  SENSORS_ACTIVATE(pressure_sensor);
82  SENSORS_ACTIVATE(magneto_sensor);
83  SENSORS_ACTIVATE(acceleration_sensor);
84  SENSORS_ACTIVATE(gyroscope_sensor);
85 #endif /*X_NUCLEO_IKS01A1*/
86 
87  while(1) {
88  etimer_set(&etimer, PRINT_INTERVAL);
89 
90  PROCESS_WAIT_EVENT();
91  if(ev == sensors_event && data == &button_sensor) {
92  printf("Sensor event detected: Button Pressed.\n\n");
93  printf("Toggling Leds\n");
94  _button_pressed++;
95  leds_toggle(LEDS_ALL);
96  }
97 
98  printf("Button state:\t%s (pressed %lu times)\n", button_sensor.value(0) ? "Released" : "Pressed",
99  _button_pressed);
100 
101 #ifdef X_NUCLEO_IKS01A1
102  printf("LEDs status:\tRED:n/a GREEN:%s\n", leds_get() & LEDS_GREEN ? "on" : "off");
103 #else
104  printf("LEDs status:\tRED:%s GREEN:%s\n", leds_get() & LEDS_RED ? "on" : "off",
105  leds_get() & LEDS_GREEN ? "on" : "off");
106 #endif /*X_NUCLEO_IKS01A1*/
107  sensor_value = radio_sensor.value(RADIO_SENSOR_LAST_PACKET);
108  printf("Radio (RSSI):\t%d.%d dBm\n", sensor_value / 10, ABS_VALUE(sensor_value) % 10);
109  printf("Radio (LQI):\t%d\n", radio_sensor.value(RADIO_SENSOR_LAST_VALUE));
110 
111 #ifdef X_NUCLEO_IKS01A1
112  sensor_value = temperature_sensor.value(0);
113  printf("Temperature:\t%d.%d C\n", sensor_value / 10, ABS_VALUE(sensor_value) % 10);
114 
115  sensor_value = humidity_sensor.value(0);
116  printf("Humidity:\t%d.%d rH\n", sensor_value / 10, ABS_VALUE(sensor_value) % 10);
117 
118  sensor_value = pressure_sensor.value(0);
119  printf("Pressure:\t%d.%d mbar\n", sensor_value / 10, ABS_VALUE(sensor_value) % 10);
120 
121  /* NOTE: this demo uses the mapping of ST Nucleo sensors on Contiki sensor API.
122  * For a real use case of sensors like acceleration, magneto and gyroscope,
123  * it is better to directly call the ST lib to get the three value (X/Y/Z)
124  * at once.
125  */
126  printf("Magneto:\t%d/%d/%d (X/Y/Z) mgauss\n", magneto_sensor.value(X_AXIS),
127  magneto_sensor.value(Y_AXIS),
128  magneto_sensor.value(Z_AXIS));
129 
130  printf("Acceleration:\t%d/%d/%d (X/Y/Z) mg\n", acceleration_sensor.value(X_AXIS),
131  acceleration_sensor.value(Y_AXIS),
132  acceleration_sensor.value(Z_AXIS));
133 
134  printf("Gyroscope:\t%d/%d/%d (X/Y/Z) mdps\n", gyroscope_sensor.value(X_AXIS),
135  gyroscope_sensor.value(Y_AXIS),
136  gyroscope_sensor.value(Z_AXIS));
137 #endif /*X_NUCLEO_IKS01A1*/
138 
139  printf("\n");
140  }
141 
142  PROCESS_END();
143 }
