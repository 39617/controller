58 /* Called when the URL present in the global "url" variable should be
59  * opened. It will call the hostname resolver as well as the HTTP
60  * client requester.
61  */
62 static void
63 start_get(void)
64 {
65  uip_ipaddr_t addr;
66  unsigned char i;
67  static char host[32];
68  char *file;
69  register char *urlptr;
70 
71  /* Trim off any spaces in the end of the url. */
72  urlptr = url + strlen(url) - 1;
73  while(*urlptr == ' ' && urlptr > url) {
74  *urlptr = 0;
75  --urlptr;
76  }
77 
78  /* Don't even try to go further if the URL is empty. */
79  if(urlptr == url) {
80  return;
81  }
82 
83  /* See if the URL starts with http://, otherwise prepend it. */
84  if(strncmp(url, http_http, 7) != 0) {
85  while(urlptr >= url) {
86  *(urlptr + 7) = *urlptr;
87  --urlptr;
88  }
89  strncpy(url, http_http, 7);
90  }
91 
92  /* Find host part of the URL. */
93  urlptr = &url[7];
94  for(i = 0; i < sizeof(host); ++i) {
95  if(*urlptr == 0 ||
96  *urlptr == '/' ||
97  *urlptr == ' ' ||
98  *urlptr == ':') {
99  host[i] = 0;
100  break;
101  }
102  host[i] = *urlptr;
103  ++urlptr;
104  }
105 
106  /* XXX: Here we should find the port part of the URL, but this isn't
107  currently done because of laziness from the programmer's side
108  :-) */
109 
110  /* Find file part of the URL. */
111  while(*urlptr != '/' && *urlptr != 0) {
112  ++urlptr;
113  }
114  if(*urlptr == '/') {
115  file = urlptr;
116  } else {
117  file = "/";
118  }
119 
120 #if UIP_UDP
121  /* First check if the host is an IP address. */
122  if(uiplib_ipaddrconv(host, &addr) == 0) {
123  uip_ipaddr_t *addrptr;
124  /* Try to lookup the hostname. If it fails, we initiate a hostname
125  lookup and print out an informative message on the
126  statusbar. */
127  if(resolv_lookup(host, &addrptr) != RESOLV_STATUS_CACHED) {
128  resolv_query(host);
129  puts("Resolving host...");
130  return;
131  }
132  uip_ipaddr_copy(&addr, addrptr);
133  }
134 #else /* UIP_UDP */
135  uiplib_ipaddrconv(host, &addr);
136 #endif /* UIP_UDP */
137 
138  /* The hostname we present in the hostname table, so we send out the
139  initial GET request. */
140  if(webclient_get(host, 80, file) == 0) {
141  puts("Out of memory error");
142  } else {
143  puts("Connecting...");
144  }
145 }
