98 static int8_t
99 get_status(uint8_t mask, uint8_t *val)
100 {
101  uint8_t status, print_msg;
102 
103  /* Retrieve the status of the power management block */
104  if(pm_get_state(&status) != PM_SUCCESS) {
105  printf("Failed to retrieve the power management status\n");
106  return PM_ERROR;
107  }
108 
109  if(!mask) {
110  printf("STATUS %u\n", status);
111  *val = PM_IDLE;
112  return PM_SUCCESS;
113  }
114 
115  /* Read back ony the requested status bit */
116  switch(mask) {
117  case PM_SYSOFF_ON_MASK:
118  print_msg = (status & mask) ? PM_SYSOFF_ON : PM_SYSOFF_OFF;
119  break;
120  case PM_TIMER_ENABLED_MASK:
121  print_msg = (status & mask) ? PM_TIMER_ENABLED : PM_TIMER_DISABLED;
122  break;
123  case PM_AWAITING_RTC_EVENT_MASK:
124  print_msg = (status & mask) ? PM_AWAITING_RTC_EVENT : PM_AWAITING_RTC_DIS;
125  break;
126  default:
127  return PM_ERROR;
128  }
129 
130  printf("Status -> %s\n", print_pm(print_msg));
131  *val = print_msg;
132  return PM_SUCCESS;
133 }

135 PROCESS_THREAD(test_remote_pm, ev, data)
136 {
137  static uint8_t aux;
138 
139  PROCESS_BEGIN();
140 
141  /* Configures the pins and initializes the driver */
142  if(pm_init() != PM_SUCCESS) {
143  printf("Failed to initialize\n");
144  PROCESS_EXIT();
145  }
146 
147  /* Get the current firmware version to track the driver implementation on the
148  * low-power MCU
149  */
150  if(pm_get_firmware_version(&aux) != PM_SUCCESS) {
151  printf("Failed to retrieve PIC's fw version\n");
152  PROCESS_EXIT();
153  }
154  printf("Firmware PIC-Power Manager Version: V0.%u\n", aux);
155 
156  if(aux != PM_EXPECTED_VERSION) {
157  printf("Unexpected firmware version\n");
158  PROCESS_EXIT();
159  }
160 
161  /* Enables the nano timer, the power management block will be driven by the
162  * nano timer now, putting the node in shutdown mode every minute (as
163  * default). For this test you need to disconnect the USB cable off to power
164  * the RE-Mote ONLY from the external battery, so after veryfing the above
165  * works, disconnect the USB cable and hit the reset button
166  * When the nano timer is enabled, the external battery is shutdown for a
167  * couple of usecs, so the CC2538 is restarted. After this as the nano timer
168  * is enabled, then it will run as intended
169  */
170 
171  if(get_status(PM_TIMER_ENABLED_MASK, &aux) != PM_SUCCESS) {
172  PROCESS_EXIT();
173  }
174 
175  if(aux == PM_TIMER_DISABLED) {
176  printf("Enabling the nano Timer...\n");
177  if(pm_enable_timer() != PM_SUCCESS) {
178  printf("Failed to set the nano Timer\n");
179  PROCESS_EXIT();
180  }
181  }
182 
183  /* At this point as the RE-Mote is powered over USB you should see the prints,
184  * disconnect the USB cable and power only with the external battery. If
185  * something fails, then you should not see the red LED blinking
186  */
187  aux = ENTER_SHUTDOWN_COUNT;
188 
189  /* Open the broadcast channel */
190  broadcast_open(&bc, BROADCAST_CHANNEL, &bc_rx);
191 
192  /* Send a message */
193  packetbuf_copyfrom(&aux, sizeof(aux));
194  broadcast_send(&bc);
195 
196  /* And wait a few seconds before going to sleep */
197  while(1) {
198  etimer_set(&et, CLOCK_SECOND);
199  PROCESS_WAIT_EVENT();
200 
201  /* Enter shutdown mode before the shutdown period (1 min default) expires */
202  if(!aux) {
203  /* Say goodnight */
204  PM_SHUTDOWN_NOW;
205  printf("Goodnight!\n");
206  PROCESS_EXIT();
207  }
208 
209  aux--;
210  leds_toggle(LEDS_RED);
211  }
212 
213  PROCESS_END();
214 }
