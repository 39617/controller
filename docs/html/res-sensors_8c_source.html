49 /*
50  * Generic resource handler for any sensor in this example. Ultimately gets
51  * called by all handlers and populates the CoAP response
52  */
53 static void
54 res_get_handler_all(int sens_type, void *request, void *response,
55  uint8_t *buffer, uint16_t preferred_size, int32_t *offset)
56 {
57  unsigned int accept = -1;
58  const cc26xx_web_demo_sensor_reading_t *reading;
59 
60  reading = cc26xx_web_demo_sensor_lookup(sens_type);
61 
62  if(reading == NULL) {
63  REST.set_response_status(response, REST.status.NOT_FOUND);
64  REST.set_response_payload(response, coap_server_not_found_msg,
65  strlen(coap_server_not_found_msg));
66  return;
67  }
68 
69  REST.get_header_accept(request, &accept);
70 
71  if(accept == -1 || accept == REST.type.TEXT_PLAIN) {
72  REST.set_header_content_type(response, REST.type.TEXT_PLAIN);
73  snprintf((char *)buffer, REST_MAX_CHUNK_SIZE, "%s", reading->converted);
74 
75  REST.set_response_payload(response, (uint8_t *)buffer,
76  strlen((char *)buffer));
77  } else if(accept == REST.type.APPLICATION_JSON) {
78  REST.set_header_content_type(response, REST.type.APPLICATION_JSON);
79  snprintf((char *)buffer, REST_MAX_CHUNK_SIZE, "{\"%s\":%s}",
80  reading->descr, reading->converted);
81 
82  REST.set_response_payload(response, buffer, strlen((char *)buffer));
83  } else if(accept == REST.type.APPLICATION_XML) {
84  REST.set_header_content_type(response, REST.type.APPLICATION_XML);
85  snprintf((char *)buffer, REST_MAX_CHUNK_SIZE,
86  "<%s val=\"%s\" unit=\"%s\"/>", reading->xml_element,
87  reading->converted, reading->units);
88 
89  REST.set_response_payload(response, buffer, strlen((char *)buffer));
90  } else {
91  REST.set_response_status(response, REST.status.NOT_ACCEPTABLE);
92  REST.set_response_payload(response, coap_server_supported_msg,
93  strlen(coap_server_supported_msg));
94  }
95 }
