69 void
70 res_get_handler(void *request, void *response, uint8_t *buffer, uint16_t preferred_size, int32_t *offset)
71 {
72  coap_packet_t *const coap_req = (coap_packet_t *)request;
73 
74  PRINTF("/separate ");
75  if(separate_active) {
76  PRINTF("REJECTED ");
77  coap_separate_reject();
78  } else {
79  PRINTF("STORED ");
80  separate_active = 1;
81 
82  /* Take over and skip response by engine. */
83  coap_separate_accept(request, &separate_store->request_metadata);
84  /* Be aware to respect the Block2 option, which is also stored in the coap_separate_t. */
85 
86  snprintf(separate_store->buffer, MAX_PLUGFEST_PAYLOAD,
87  "Type: %u\nCode: %u\nMID: %u", coap_req->type, coap_req->code,
88  coap_req->mid);
89  }
90 
91  PRINTF("(%s %u)\n", coap_req->type == COAP_TYPE_CON ? "CON" : "NON", coap_req->mid);
92 }
93 static void
94 res_resume_handler()
95 {
96  if(separate_active) {
97  PRINTF("/separate ");
98  coap_transaction_t *transaction = NULL;
99  if((transaction = coap_new_transaction(separate_store->request_metadata.mid,
100  &separate_store->request_metadata.addr,
101  separate_store->request_metadata.port))) {
102  PRINTF(
103  "RESPONSE (%s %u)\n", separate_store->request_metadata.type == COAP_TYPE_CON ? "CON" : "NON", separate_store->request_metadata.mid);
104 
105  coap_packet_t response[1]; /* This way the packet can be treated as pointer as usual. */
106 
107  /* Restore the request information for the response. */
108  coap_separate_resume(response, &separate_store->request_metadata, CONTENT_2_05);
109 
110  REST.set_header_content_type(response, REST.type.TEXT_PLAIN);
111  coap_set_payload(response, separate_store->buffer,
112  strlen(separate_store->buffer));
113 
114  /*
115  * Be aware to respect the Block2 option, which is also stored in the coap_separate_t.
116  * As it is a critical option, this example resource pretends to handle it for compliance.
117  */
118  coap_set_header_block2(response,
119  separate_store->request_metadata.block2_num, 0,
120  separate_store->request_metadata.block2_size);
121 
122  /* Warning: No check for serialization error. */
123  transaction->packet_len = coap_serialize_message(response,
124  transaction->packet);
125  coap_send_transaction(transaction);
126  /* The engine will clear the transaction (right after send for NON, after acked for CON). */
127 
128  separate_active = 0;
129  } else {
130  PRINTF("ERROR (transaction)\n");
131  }
132  } /* if (separate_active) */
133 }
