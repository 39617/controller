591 HAL_StatusTypeDef SD_IO_WriteCmd(uint8_t Cmd, uint32_t Arg, uint8_t Crc, uint8_t Response)
592 {
593  uint32_t counter = 0x00;
594  uint8_t frame[6] = {0};
595 
596  /* Prepare Frame to send */
597  frame[0] = (Cmd | 0x40); /* Construct byte 1 */
598  frame[1] = (uint8_t)(Arg >> 24); /* Construct byte 2 */
599  frame[2] = (uint8_t)(Arg >> 16); /* Construct byte 3 */
600  frame[3] = (uint8_t)(Arg >> 8); /* Construct byte 4 */
601  frame[4] = (uint8_t)(Arg); /* Construct byte 5 */
602  frame[5] = (Crc); /* Construct byte 6 */
603 
604  /* SD chip select low */
605  SD_CS_LOW();
606 
607  /* Send Frame */
608  for (counter = 0; counter < 6; counter++)
609  {
610  SD_IO_WriteByte(frame[counter]); /* Send the Cmd bytes */
611  }
612 
613  if(Response != SD_NO_RESPONSE_EXPECTED)
614  {
615  return SD_IO_WaitResponse(Response);
616  }
617 
618  return HAL_OK;
619 }
