55 input(struct tcp_socket *s, void *ptr,
56  const uint8_t *inputptr, int inputdatalen)
57 {
58  printf("input %d bytes '%s'\n", inputdatalen, inputptr);
59  if(!get_received) {
60  /* See if we have a full GET request in the buffer. */
61  if(strncmp((char *)inputptr, "GET /", 5) == 0 &&
62  atoi((char *)&inputptr[5]) != 0) {
63  bytes_to_send = atoi((char *)&inputptr[5]);
64  printf("bytes_to_send %d\n", bytes_to_send);
65  return 0;
66  }
67  printf("inputptr '%.*s'\n", inputdatalen, inputptr);
68  /* Return the number of data bytes we received, to keep them all
69  in the buffer. */
70  return inputdatalen;
71  } else {
72  /* Discard everything */
73  return 0; /* all data consumed */
74  }
75 }
