65 static void
66 tcpip_handler(void)
67 {
68  memset(buf, 0, MAX_PAYLOAD_LEN);
69  if(uip_newdata()) {
70  leds_on(LEDS_RED);
71  len = uip_datalen();
72  memcpy(buf, uip_appdata, len);
73  PRINTF("%u bytes from [", len);
74  PRINT6ADDR(&UIP_IP_BUF->srcipaddr);
75  PRINTF("]:%u\n", UIP_HTONS(UIP_UDP_BUF->srcport));
76 #if SERVER_REPLY
77  uip_ipaddr_copy(&server_conn->ripaddr, &UIP_IP_BUF->srcipaddr);
78  server_conn->rport = UIP_UDP_BUF->srcport;
79 
80  uip_udp_packet_send(server_conn, buf, len);
81  /* Restore server connection to allow data from any node */
82  uip_create_unspecified(&server_conn->ripaddr);
83  server_conn->rport = 0;
84 #endif
85  }
86  leds_off(LEDS_RED);
87  return;
88 }
