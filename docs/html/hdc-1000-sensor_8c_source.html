187 {

211 static int
212 value(int type)
213 {
214  int rv;
215  float temp;
216  float hum;
217 
218  if(enabled != HDC_1000_SENSOR_STATUS_READINGS_READY) {

221  }
222 
223  if((type != HDC_1000_SENSOR_TYPE_TEMP) &&
224  type != HDC_1000_SENSOR_TYPE_HUMIDITY) {
225  PRINTF("Invalid type\n");
226  return CC26XX_SENSOR_READING_ERROR;
227  } else {
228  convert(&temp, &hum);
229  PRINTF("HDC: %04X %04X t=%d h=%d\n", raw_temp, raw_hum,
230  (int)(temp * 100), (int)(hum * 100));
231 
232  if(type == HDC_1000_SENSOR_TYPE_TEMP) {
233  rv = (int)(temp * 100);
234  } else if(type == HDC_1000_SENSOR_TYPE_HUMIDITY) {
235  rv = (int)(hum * 100);
236  }
237  }

251 static int
252 configure(int type, int enable)
253 {
254  switch(type) {
255  case SENSORS_HW_INIT:
256  raw_temp = 0;
257  raw_hum = 0;
258  memset(&data, 0, sizeof(data));
259 
260  sensor_init();
261  enabled = HDC_1000_SENSOR_STATUS_INITIALISED;
262  break;
263  case SENSORS_ACTIVE:
264  /* Must be initialised first */
265  if(enabled == HDC_1000_SENSOR_STATUS_DISABLED) {
266  return HDC_1000_SENSOR_STATUS_DISABLED;
267  }
268  if(enable) {
269  start();
270  ctimer_set(&startup_timer, SENSOR_STARTUP_DELAY, notify_ready, NULL);
271  enabled = HDC_1000_SENSOR_STATUS_TAKING_READINGS;
272  } else {
273  ctimer_stop(&startup_timer);
274  enabled = HDC_1000_SENSOR_STATUS_INITIALISED;
275  }
276  break;
277  default:
278  break;
279  }
280  return enabled;
281 }
