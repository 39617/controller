51 void fill_packet(volatile packet_t *p) {
52  static volatile uint8_t count=0;
53  volatile uint8_t i;
54  p->length = PAYLOAD_LEN;
55  p->offset = 0;
56  for(i=0; i<PAYLOAD_LEN; i++) {
57  p->data[i] = count++;
58  }
59 
60  /* acks get treated differently, even in promiscuous mode */
61  /* setting the third bit makes sure that we never send an ack */
62  /* or any valid 802.15.4-2006 packet */
63  p->data[0] |= (1 << 3);
64 }
