150 
151  PRINTF("Processes running\n");
152 
153  leds_off(LEDS_ALL);
154 
155  watchdog_start();
156 
157  /*
158  * This is the scheduler loop.
159  */
160  while(1) {
161 
162  do {
163  /* Reset watchdog. */
164  watchdog_periodic();
165  r = process_run();

167 
168  ENERGEST_SWITCH(ENERGEST_TYPE_CPU, ENERGEST_TYPE_LPM);
169 
170  watchdog_stop();

172  watchdog_start();
173 
174  ENERGEST_SWITCH(ENERGEST_TYPE_LPM, ENERGEST_TYPE_CPU);
175 
176  }
177 
178  return 0;
