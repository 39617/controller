161 static void
162 get_bmp_reading()
163 {
164  int value;
165  clock_time_t next = SENSOR_READING_PERIOD +
166  (random_rand() % SENSOR_READING_RANDOM);
167 
168  value = bmp_280_sensor.value(BMP_280_SENSOR_TYPE_PRESS);
169  if(value != CC26XX_SENSOR_READING_ERROR) {
170  printf("BAR: Pressure=%d.%02d hPa\n", value / 100, value % 100);
171  } else {
172  printf("BAR: Pressure Read Error\n");
173  }
174 
175  value = bmp_280_sensor.value(BMP_280_SENSOR_TYPE_TEMP);
176  if(value != CC26XX_SENSOR_READING_ERROR) {
177  printf("BAR: Temp=%d.%02d C\n", value / 100, value % 100);
178  } else {
179  printf("BAR: Temperature Read Error\n");
180  }
181 
182  SENSORS_DEACTIVATE(bmp_280_sensor);
183 
184  ctimer_set(&bmp_timer, next, init_bmp_reading, NULL);
185 }

375 PROCESS_THREAD(cc26xx_demo_process, ev, data)
376 {
377 
378  PROCESS_BEGIN();
379 
380  printf("CC26XX demo\n");
381 
382  init_sensors();
383 
384  /* Init the BLE advertisement daemon */
385  rf_ble_beacond_config(0, BOARD_STRING);
386  rf_ble_beacond_start();
387 
388  etimer_set(&et, CC26XX_DEMO_LOOP_INTERVAL);
389  get_sync_sensor_readings();
390  init_sensor_readings();
391 
392  while(1) {
393 
394  PROCESS_YIELD();
395 
396  if(ev == PROCESS_EVENT_TIMER) {
397  if(data == &et) {
398  leds_toggle(CC26XX_DEMO_LEDS_PERIODIC);
399 
400  get_sync_sensor_readings();
401 
402  etimer_set(&et, CC26XX_DEMO_LOOP_INTERVAL);
403  }
404  } else if(ev == sensors_event) {
405  if(data == CC26XX_DEMO_SENSOR_1) {
406  printf("Left: Pin %d, press duration %d clock ticks\n",
407  (CC26XX_DEMO_SENSOR_1)->value(BUTTON_SENSOR_VALUE_STATE),
408  (CC26XX_DEMO_SENSOR_1)->value(BUTTON_SENSOR_VALUE_DURATION));
409 
410  if((CC26XX_DEMO_SENSOR_1)->value(BUTTON_SENSOR_VALUE_DURATION) >
411  CLOCK_SECOND) {
412  printf("Long button press!\n");
413  }
414 
415  leds_toggle(CC26XX_DEMO_LEDS_BUTTON);
416  } else if(data == CC26XX_DEMO_SENSOR_2) {
417  leds_on(CC26XX_DEMO_LEDS_REBOOT);
418  watchdog_reboot();
419  } else if(data == CC26XX_DEMO_SENSOR_3) {
420  printf("Up\n");
421  } else if(data == CC26XX_DEMO_SENSOR_4) {
422  printf("Down\n");
423  } else if(data == CC26XX_DEMO_SENSOR_5) {
424 #if BOARD_SENSORTAG
425  if(buzzer_state()) {
426  buzzer_stop();
427  } else {
428  buzzer_start(1000);
429  }
430  } else if(ev == sensors_event && data == &bmp_280_sensor) {
431  get_bmp_reading();
432  } else if(ev == sensors_event && data == &opt_3001_sensor) {
433  get_light_reading();
434  } else if(ev == sensors_event && data == &hdc_1000_sensor) {
435  get_hdc_reading();
436  } else if(ev == sensors_event && data == &tmp_007_sensor) {
437  get_tmp_reading();
438  } else if(ev == sensors_event && data == &mpu_9250_sensor) {
439  get_mpu_reading();
440 #elif BOARD_SMARTRF06EB
441  printf("Sel: Pin %d, press duration %d clock ticks\n",
442  button_select_sensor.value(BUTTON_SENSOR_VALUE_STATE),
443  button_select_sensor.value(BUTTON_SENSOR_VALUE_DURATION));
444 #endif
445  }
446  }
447  }
448 
449  PROCESS_END();
450 }
