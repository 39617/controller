49 ISR(CC2420_IRQ, cc24240_fifop_interrupt)
50 {
51  ENERGEST_ON(ENERGEST_TYPE_IRQ);
52 
53  /* check TA0IV - if 4 (CCR2 => FIFOP interrupt ) */
54  if(P1IV == (2 * (1 + CC2420_FIFOP_PIN))) {
55  if(cc2420_interrupt()) {
56  LPM4_EXIT;
57  }
58  }
59  ENERGEST_OFF(ENERGEST_TYPE_IRQ);
60 }
