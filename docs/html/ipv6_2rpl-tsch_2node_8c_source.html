66 static void
67 print_network_status(void)
68 {
69  int i;
70  uint8_t state;
71  uip_ds6_defrt_t *default_route;
72 #if RPL_WITH_STORING
73  uip_ds6_route_t *route;
74 #endif /* RPL_WITH_STORING */
75 #if RPL_WITH_NON_STORING
76  rpl_ns_node_t *link;
77 #endif /* RPL_WITH_NON_STORING */
78 
79  PRINTF("--- Network status ---\n");
80 
81  /* Our IPv6 addresses */
82  PRINTF("- Server IPv6 addresses:\n");
83  for(i = 0; i < UIP_DS6_ADDR_NB; i++) {
84  state = uip_ds6_if.addr_list[i].state;
85  if(uip_ds6_if.addr_list[i].isused &&
86  (state == ADDR_TENTATIVE || state == ADDR_PREFERRED)) {
87  PRINTF("-- ");
88  PRINT6ADDR(&uip_ds6_if.addr_list[i].ipaddr);
89  PRINTF("\n");
90  }
91  }
92 
93  /* Our default route */
94  PRINTF("- Default route:\n");
95  default_route = uip_ds6_defrt_lookup(uip_ds6_defrt_choose());
96  if(default_route != NULL) {
97  PRINTF("-- ");
98  PRINT6ADDR(&default_route->ipaddr);
99  PRINTF(" (lifetime: %lu seconds)\n", (unsigned long)default_route->lifetime.interval);
100  } else {
101  PRINTF("-- None\n");
102  }
103 
104 #if RPL_WITH_STORING
105  /* Our routing entries */
106  PRINTF("- Routing entries (%u in total):\n", uip_ds6_route_num_routes());
107  route = uip_ds6_route_head();
108  while(route != NULL) {
109  PRINTF("-- ");
110  PRINT6ADDR(&route->ipaddr);
111  PRINTF(" via ");
112  PRINT6ADDR(uip_ds6_route_nexthop(route));
113  PRINTF(" (lifetime: %lu seconds)\n", (unsigned long)route->state.lifetime);
114  route = uip_ds6_route_next(route);
115  }
116 #endif
117 
118 #if RPL_WITH_NON_STORING
119  /* Our routing links */
120  PRINTF("- Routing links (%u in total):\n", rpl_ns_num_nodes());
121  link = rpl_ns_node_head();
122  while(link != NULL) {
123  uip_ipaddr_t child_ipaddr;
124  uip_ipaddr_t parent_ipaddr;
125  rpl_ns_get_node_global_addr(&child_ipaddr, link);
126  rpl_ns_get_node_global_addr(&parent_ipaddr, link->parent);
127  PRINTF("-- ");
128  PRINT6ADDR(&child_ipaddr);
129  if(link->parent == NULL) {
130  memset(&parent_ipaddr, 0, sizeof(parent_ipaddr));
131  PRINTF(" --- DODAG root ");
132  } else {
133  PRINTF(" to ");
134  PRINT6ADDR(&parent_ipaddr);
135  }
136  PRINTF(" (lifetime: %lu seconds)\n", (unsigned long)link->lifetime);
137  link = rpl_ns_node_next(link);
138  }
139 #endif
140 
141  PRINTF("----------------------\n");
142 }

144 static void
145 net_init(uip_ipaddr_t *br_prefix)
146 {
147  uip_ipaddr_t global_ipaddr;
148 
149  if(br_prefix) { /* We are RPL root. Will be set automatically
150  as TSCH pan coordinator via the tsch-rpl module */
151  memcpy(&global_ipaddr, br_prefix, 16);
152  uip_ds6_set_addr_iid(&global_ipaddr, &uip_lladdr);
153  uip_ds6_addr_add(&global_ipaddr, 0, ADDR_AUTOCONF);
154  rpl_set_root(RPL_DEFAULT_INSTANCE, &global_ipaddr);
155  rpl_set_prefix(rpl_get_any_dag(), br_prefix, 64);
156  rpl_repair_root(RPL_DEFAULT_INSTANCE);
157  }
158 
159  NETSTACK_MAC.on();
160 }

162 PROCESS_THREAD(node_process, ev, data)
163 {
164  static struct etimer et;
165  PROCESS_BEGIN();
166 
167  /* 3 possible roles:
168  * - role_6ln: simple node, will join any network, secured or not
169  * - role_6dr: DAG root, will advertise (unsecured) beacons
170  * - role_6dr_sec: DAG root, will advertise secured beacons
171  * */
172  static int is_coordinator = 0;
173  static enum { role_6ln, role_6dr, role_6dr_sec } node_role;
174  node_role = role_6ln;
175 
176  int coordinator_candidate = 0;
177 
178 #ifdef CONTIKI_TARGET_Z1
179  /* Set node with MAC address c1:0c:00:00:00:00:01 as coordinator,
180  * convenient in cooja for regression tests using z1 nodes
181  * */
182  extern unsigned char node_mac[8];
183  unsigned char coordinator_mac[8] = { 0xc1, 0x0c, 0x00, 0x00, 0x00, 0x00, 0x00, 0x01 };
184 
185  coordinator_candidate = (memcmp(node_mac, coordinator_mac, 8) == 0);
186 #elif CONTIKI_TARGET_COOJA
187  coordinator_candidate = (node_id == 1);
188 #endif
189 
190  if(coordinator_candidate) {
191  if(LLSEC802154_ENABLED) {
192  node_role = role_6dr_sec;
193  } else {
194  node_role = role_6dr;
195  }
196  } else {
197  node_role = role_6ln;
198  }
199 
200 #if CONFIG_VIA_BUTTON
201  {
202 #define CONFIG_WAIT_TIME 5
203 
204  SENSORS_ACTIVATE(button_sensor);
205  etimer_set(&et, CLOCK_SECOND * CONFIG_WAIT_TIME);
206 
207  while(!etimer_expired(&et)) {
208  printf("Init: current role: %s. Will start in %u seconds. Press user button to toggle mode.\n",
209  node_role == role_6ln ? "6ln" : (node_role == role_6dr) ? "6dr" : "6dr-sec",
210  CONFIG_WAIT_TIME);
211  PROCESS_WAIT_EVENT_UNTIL(((ev == sensors_event) &&
212  (data == &button_sensor) && button_sensor.value(0) > 0)
213  || etimer_expired(&et));
214  if(ev == sensors_event && data == &button_sensor && button_sensor.value(0) > 0) {
215  node_role = (node_role + 1) % 3;
216  if(LLSEC802154_ENABLED == 0 && node_role == role_6dr_sec) {
217  node_role = (node_role + 1) % 3;
218  }
219  etimer_restart(&et);
220  }
221  }
222  }
223 
224 #endif /* CONFIG_VIA_BUTTON */
225 
226  printf("Init: node starting with role %s\n",
227  node_role == role_6ln ? "6ln" : (node_role == role_6dr) ? "6dr" : "6dr-sec");
228 
229  tsch_set_pan_secured(LLSEC802154_ENABLED && (node_role == role_6dr_sec));
230  is_coordinator = node_role > role_6ln;
231 
232  if(is_coordinator) {
233  uip_ipaddr_t prefix;
234  uip_ip6addr(&prefix, UIP_DS6_DEFAULT_PREFIX, 0, 0, 0, 0, 0, 0, 0);
235  net_init(&prefix);
236  } else {
237  net_init(NULL);
238  }
239 
240 #if WITH_ORCHESTRA
241  orchestra_init();
242 #endif /* WITH_ORCHESTRA */
243 
244  /* Print out routing tables every minute */
245  etimer_set(&et, CLOCK_SECOND * 60);
246  while(1) {
247  print_network_status();
248  PROCESS_YIELD_UNTIL(etimer_expired(&et));
249  etimer_reset(&et);
250  }
251 
252  PROCESS_END();
253 }
