539 void SpiritTimerComputeWakeUpValues(float fDesiredMsec , uint8_t* pcCounter , uint8_t* pcPrescaler)
540 {
541  float rco_freq,err;
542  uint32_t n;
543 
544  rco_freq=((float)SpiritTimerGetRcoFrequency())/1000;
545 
546  /* N cycles in the time base of the timer:
547  - clock of the timer is RCO frequency
548  - divide times 1000 more because we have an input in ms (variable rco_freq is already this frequency divided by 1000)
549  */
550  n=(uint32_t)(fDesiredMsec*rco_freq);
551 
552  /* check if it is possible to reach that target with prescaler and counter of spirit1 */
553  if(n/0xFF>0xFD)
554  {
555  /* if not return the maximum possible value */
556  (*pcCounter) = 0xFF;
557  (*pcPrescaler) = 0xFF;
558  return;
559  }
560 
561  /* prescaler is really 2 as min value */
562  (*pcPrescaler)=(n/0xFF)+2;
563  (*pcCounter) = n / (*pcPrescaler);
564 
565  /* check if the error is minimum */
566  err=S_ABS((float)(*pcCounter)*(*pcPrescaler)/rco_freq-fDesiredMsec);
567 
568  if((*pcCounter)<=254)
569  {
570  if(S_ABS((float)((*pcCounter)+1)*(*pcPrescaler)/rco_freq-fDesiredMsec)<err)
571  (*pcCounter)=(*pcCounter)+1;
572  }
573 
574  /* decrement prescaler and counter according to the logic of this timer in spirit1 */
575  (*pcPrescaler)--;
576  if((*pcCounter)>1)
577  (*pcCounter)--;
578  else
579  (*pcCounter)=1;
580 }
