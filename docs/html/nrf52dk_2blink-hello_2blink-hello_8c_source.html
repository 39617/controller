58 static void handle_event(process_event_t ev, process_data_t data, struct blink_process_ctx *ctx)
59 {
60  if (ev == PROCESS_EVENT_TIMER && etimer_expired(&ctx->et_blink)) {
61  leds_toggle(ctx->led);

63  printf("Blink %d\n", ctx->led);
64  } else if (ev == sensors_event && data == ctx->button) {
