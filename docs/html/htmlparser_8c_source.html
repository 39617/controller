35 /* htmlparser.c:
36  *
37  * Implements a very simplistic HTML parser. It recognizes HTML links
38  * (<a href>-tags), HTML img alt tags, a few text flow break tags
39 G * (<br>, <p>, <h>), the <li> tag (but does not even try to
40  * distinguish between <ol> or <ul>) as well as HTML comment tags
41  * (<!-- -->).
42  *
43  * To save memory, the HTML parser is state machine driver, which
44  * means that it will shave off one character from the HTML page,
45  * process that character, and return to the next. Another way of
46  * doing it would be to buffer a number of characters and process them
47  * together.
48  *
49  * The main function in this file is the htmlparser_parse() function
50  * which takes a htmlparser_state structur and a part of an HTML file
51  * as an argument. The htmlparser_parse() function will call the
52  * helper functions parse_char() and parse_tag(). Those functions will
53  * in turn call the two callback functions htmlparser_char() and
54  * htmlparser_tag(). Those functions must be implemented by the using
55  * module (e.g., a web browser program).
56  *
57  * htmlparser_char() will be called for every non-tag character.
58  *
59  * htmlparser_tag() will be called whenever a full tag has been found.
60  *
61  */

258 static char
259 lowercase(char c)
260 {
261  /* XXX: This is a *brute force* approach to lower-case
262  converting and should *not* be used anywhere else! It
263  works for our purposes, however (i.e., HTML tags). */
264  if(c > 0x40) {
265  return (c & 0x1f) | 0x60;
266  } else {
267  return c;
268  }
269 }

289 static void
290 add_char(unsigned char c)
291 {
292  if(s.wordlen < WWW_CONF_WEBPAGE_WIDTH - 1 && c < 0x80) {
293  s.word[s.wordlen] = c;
294  ++s.wordlen;
295  }

297 /*-----------------------------------------------------------------------------------*/
298 static void
299 do_word(void)
300 {
301  if(s.wordlen > 0) {
302  if(s.majorstate == MAJORSTATE_LINK) {
303  if(s.word[s.wordlen - 1] != ISO_space) {
304  add_char(ISO_space);
305  }
306  } else if(s.majorstate >= MAJORSTATE_DISCARD) {
307  s.wordlen = 0;
308  } else {
309  s.word[s.wordlen] = '\0';
310  htmlparser_word(s.word, s.wordlen);
311  s.wordlen = 0;
312  }
313  }
314 }

323 static unsigned char
324 find_tag(char *tag)
325 {
326  static unsigned char first, last, i, tabi;
327  static char tagc;
328 
329  first = TAG_FIRST;
330  last = TAG_LAST;
331  i = 0;
332 
333  do {
334  tagc = tag[i];
335 
336  if((tagc == 0 || tagc == ISO_slash) && tags[first][i] == 0) {
337  return first;
338  }
339 
340  tabi = first;
341 
342  /* First, find first matching tag from table. */
343  while(tagc > (tags[tabi])[i] && tabi < last) {
344  ++tabi;
345  }
346  first = tabi;
347 
348  /* Second, find last matching tag from table. */
349  while(tagc == (tags[tabi])[i] && tabi < last) {
350  ++tabi;
351  }
352  last = tabi;
353 
354  /* If first and last matching tags are equal, we have a non-match
355  and return. Else we continue with the next character. */
356  ++i;
357 
358  } while(last != first);
359  return TAG_LAST;
360 }
361 /*-----------------------------------------------------------------------------------*/
362 static void
363 parse_tag(void)
364 {
365  static char *tagattrparam;
366  static unsigned char tag;
367  static unsigned char size;
368 
369  tag = find_tag(s.tag);
370  /* If we are inside a <script> we mustn't interpret any tags
371  (inside JavaScript strings) but wait for the </script>. */
372  if(s.majorstate == MAJORSTATE_SCRIPT && tag != TAG_SLASHSCRIPT) {
373  return;
374  }
375 
376  PRINTF(("Parsing tag '%s' '%s' '%s'\n", s.tag, s.tagattr, s.tagattrparam));
377 
378  switch(tag) {
379  case TAG_P:
380  case TAG_H1:
381  case TAG_H2:
382  case TAG_H3:
383  case TAG_H4:
384  newline();
385  /* FALLTHROUGH */
386  case TAG_BR:
387  case TAG_TR:
388  case TAG_SLASHDIV:
389  case TAG_SLASHH:
390  newline();
391  break;
392  case TAG_LI:
393  if(s.tagattr[0] == 0) {
394  newline();
395  add_char(ISO_asterisk);
396  add_char(ISO_space);
397  }
398  break;
399  case TAG_SCRIPT:
400  switch_majorstate(MAJORSTATE_SCRIPT);
401  break;
402  case TAG_STYLE:
403  case TAG_SELECT:
404  switch_majorstate(MAJORSTATE_DISCARD);
405  break;
406  case TAG_SLASHSCRIPT:
407  case TAG_SLASHSTYLE:
408  case TAG_SLASHSELECT:
409  do_word();
410  switch_majorstate(s.lastmajorstate);
411  break;
412  case TAG_BODY:
413  do_word();
414  s.majorstate = s.lastmajorstate = MAJORSTATE_BODY;
415  break;
416  case TAG_IMG:
417  if(strncmp(s.tagattr, html_alt, sizeof(html_alt)) == 0 && s.tagattrparam[0] != 0) {
418  add_char(ISO_lt);
419  tagattrparam = &s.tagattrparam[0];
420  while(*tagattrparam) {
421  add_char(*tagattrparam);
422  ++tagattrparam;
423  }
424  add_char(ISO_gt);
425  do_word();
426  }
427  break;
428  case TAG_A:
429  PRINTF(("A %s %s\n", s.tagattr, s.tagattrparam));
430  if(strncmp(s.tagattr, html_href, sizeof(html_href)) == 0 && s.tagattrparam[0] != 0) {
431  strcpy(s.linkurl, s.tagattrparam);
432  do_word();
433  switch_majorstate(MAJORSTATE_LINK);
434  }
435  break;
436  case TAG_SLASHA:
437  if(s.majorstate == MAJORSTATE_LINK) {
438  switch_majorstate(s.lastmajorstate);
439  s.word[s.wordlen] = 0;
440  htmlparser_link(s.word, s.wordlen, s.linkurl);
441  s.wordlen = 0;
442  }
443  break;
444 #if WWW_CONF_FORMS
445  case TAG_FORM:
446  /* First check if we are called at the end of a form tag. If
447  so, we should propagate the form action. */
448  if(s.tagattr[0] == 0 && s.formaction[0] != 0) {
449  htmlparser_form(s.formaction);
450  init_input();
451  } else {
452  PRINTF(("Form tag\n"));
453  switch_majorstate(MAJORSTATE_FORM);
454  if(strncmp(s.tagattr, html_action, sizeof(html_action)) == 0) {
455  PRINTF(("Form action '%s'\n", s.tagattrparam));
456  strncpy(s.formaction, s.tagattrparam, WWW_CONF_MAX_FORMACTIONLEN - 1);
457  }
458  }
459  break;
460  case TAG_SLASHFORM:
461  switch_majorstate(MAJORSTATE_BODY);
462  s.formaction[0] = 0;
463  break;
464  case TAG_INPUT:
465  if(s.majorstate == MAJORSTATE_FORM) {
466  /* First check if we are called at the end of an input tag. If
467  so, we should render the input widget. */
468  if(s.tagattr[0] == 0 && s.inputname[0] != 0) {
469  PRINTF(("Render input type %d\n", s.inputtype));
470  switch(s.inputtype) {
471  case HTMLPARSER_INPUTTYPE_NONE:
472  case HTMLPARSER_INPUTTYPE_TEXT:
473  case HTMLPARSER_INPUTTYPE_HIDDEN:
474  htmlparser_inputfield(s.inputtype, s.inputvaluesize, s.inputvalue, s.inputname);
475  break;
476  case HTMLPARSER_INPUTTYPE_SUBMIT:
477  case HTMLPARSER_INPUTTYPE_IMAGE:
478  htmlparser_submitbutton(s.inputvalue, s.inputname);
479  break;
480  }
481  init_input();
482  } else {
483  PRINTF(("Input '%s' '%s'\n", s.tagattr, s.tagattrparam));
484  if(strncmp(s.tagattr, html_type, sizeof(html_type)) == 0) {
485  if(strncmp(s.tagattrparam, html_submit, sizeof(html_submit)) == 0) {
486  s.inputtype = HTMLPARSER_INPUTTYPE_SUBMIT;
487  } else if(strncmp(s.tagattrparam, html_image, sizeof(html_image)) == 0) {
488  s.inputtype = HTMLPARSER_INPUTTYPE_IMAGE;
489  } else if(strncmp(s.tagattrparam, html_text, sizeof(html_text)) == 0) {
490  s.inputtype = HTMLPARSER_INPUTTYPE_TEXT;
491  } else if(strncmp(s.tagattrparam, html_hidden, sizeof(html_hidden)) == 0) {
492  s.inputtype = HTMLPARSER_INPUTTYPE_HIDDEN;
493  } else {
494  s.inputtype = HTMLPARSER_INPUTTYPE_OTHER;
495  }
496  } else if(strncmp(s.tagattr, html_name, sizeof(html_name)) == 0) {
497  strncpy(s.inputname, s.tagattrparam, WWW_CONF_MAX_INPUTNAMELEN);
498  } else if(strncmp(s.tagattr, html_alt, sizeof(html_alt)) == 0 &&
499  s.inputtype == HTMLPARSER_INPUTTYPE_IMAGE) {
500  strncpy(s.inputvalue, s.tagattrparam, WWW_CONF_MAX_INPUTVALUELEN);
501  } else if(strncmp(s.tagattr, html_value, sizeof(html_value)) == 0) {
502  strncpy(s.inputvalue, s.tagattrparam, WWW_CONF_MAX_INPUTVALUELEN);
503  } else if(strncmp(s.tagattr, html_size, sizeof(html_size)) == 0) {
504  size = 0;
505  if(s.tagattrparam[0] >= '0' &&
506  s.tagattrparam[0] <= '9') {
507  size = s.tagattrparam[0] - '0';
508  if(s.tagattrparam[1] >= '0' &&
509  s.tagattrparam[1] <= '9') {
510  size = size * 10 + (s.tagattrparam[1] - '0');
511  }
512  }
513  if(size >= WWW_CONF_MAX_INPUTVALUELEN) {
514  size = WWW_CONF_MAX_INPUTVALUELEN - 1;
515  }
516  s.inputvaluesize = size;
517  }
518  }
519  }
520  break;
521 #endif /* WWW_CONF_FORMS */
522  }
523 }

525 static uint16_t
526 parse_word(char *data, uint8_t dlen)
527 {
528  static uint8_t i;
529  static uint8_t len;
530  unsigned char c;
531 
532  len = dlen;
533 
534  switch(s.minorstate) {
535  case MINORSTATE_TEXT:
536  for(i = 0; i < len; ++i) {
537  c = data[i];
538  if(iswhitespace(c)) {
539  do_word();
540  } else if(c == ISO_lt) {
541  s.minorstate = MINORSTATE_TAG;
542  s.tagptr = 0;
543  break;
544  } else if(c == ISO_ampersand) {
545  s.minorstate = MINORSTATE_EXTCHAR;
546  break;
547  } else {
548  add_char(c);
549  }
550  }
551  break;
552  case MINORSTATE_EXTCHAR:
553  for(i = 0; i < len; ++i) {
554  c = data[i];
555  if(c == ISO_semicolon) {
556  s.minorstate = MINORSTATE_TEXT;
557  add_char(' ');
558  break;
559  } else if(iswhitespace(c)) {
560  s.minorstate = MINORSTATE_TEXT;
561  add_char('&');
562  add_char(' ');
563  break;
564  }
565  }
566  break;
567  case MINORSTATE_TAG:
568  /* If we are inside a <srcipt> we mustn't mistake a JavaScript
569  equation with a '<' as a tag. So we check for the very next
570  character to be a '/' as we're only interested in parsing
571  the </script>. */
572  if(s.majorstate == MAJORSTATE_SCRIPT && data[0] != ISO_slash) {
573  s.minorstate = MINORSTATE_TEXT;
574  break;
575  }
576 
577  /* We are currently parsing within the name of a tag. We check
578  for the end of a tag (the '>' character) or whitespace (which
579  indicates that we should parse a tag attr argument
580  instead). */
581  for(i = 0; i < len; ++i) {
582  c = data[i];
583  if(c == ISO_gt) {
584  /* Full tag found. We continue parsing regular text. */
585  s.minorstate = MINORSTATE_TEXT;
586  s.tagattrptr = s.tagattrparamptr = 0;
587  endtagfound();
588  parse_tag();
589  break;
590  } else if(iswhitespace(c)) {
591  /* The name of the tag found. We continue parsing the tag
592  attr.*/
593  s.minorstate = MINORSTATE_TAGATTR;
594  s.tagattrptr = 0;
595  endtagfound();
596  break;
597  } else {
598  /* Keep track of the name of the tag, but convert it to
599  lower case. */
600  s.tag[s.tagptr] = lowercase(c);
601  ++s.tagptr;
602  /* Check if the ->tag field is full. If so, we just eat up
603  any data left in the tag. */
604  if(s.tagptr == sizeof(s.tag)) {
605  s.minorstate = MINORSTATE_TAGEND;
606  break;
607  }
608  }
609 
610  /* Check for HTML comment, indicated by <!-- */
611  if(s.tagptr == 3 &&
612  s.tag[0] == ISO_bang &&
613  s.tag[1] == ISO_dash &&
614  s.tag[2] == ISO_dash) {
615  PRINTF(("Starting comment...\n"));
616  s.minorstate = MINORSTATE_HTMLCOMMENT;
617  s.tagptr = 0;
618  endtagfound();
619  break;
620  }
621  }
622  break;
623  case MINORSTATE_TAGATTR:
624  /* We parse the "tag attr", i.e., the "href" in <a
625  href="...">. */
626  for(i = 0; i < len; ++i) {
627  c = data[i];
628  if(c == ISO_gt) {
629  /* Full tag found. */
630  s.minorstate = MINORSTATE_TEXT;
631  s.tagattrparamptr = 0;
632  s.tagattrptr = 0;
633  endtagfound();
634  parse_tag();
635  s.tagptr = 0;
636  endtagfound();
637  break;
638  } else if(iswhitespace(c)) {
639  if(s.tagattrptr == 0) {
640  /* Discard leading spaces. */
641  } else {
642  /* A non-leading space is the end of the attribute. */
643  s.tagattrparamptr = 0;
644  endtagfound();
645  parse_tag();
646  s.minorstate = MINORSTATE_TAGATTRSPACE;
647  break;
648  }
649  } else if(c == ISO_eq) {
650  s.minorstate = MINORSTATE_TAGATTRPARAMNQ;
651  s.tagattrparamptr = 0;
652  endtagfound();
653  break;
654  } else {
655  s.tagattr[s.tagattrptr] = lowercase(c);
656  ++s.tagattrptr;
657  /* Check if the "tagattr" field is full. If so, we just eat
658  up any data left in the tag. */
659  if(s.tagattrptr == sizeof(s.tagattr)) {
660  s.minorstate = MINORSTATE_TAGEND;
661  break;
662  }
663  }
664  }
665  break;
666  case MINORSTATE_TAGATTRSPACE:
667  for(i = 0; i < len; ++i) {
668  c = data[i];
669  if(iswhitespace(c)) {

756  s.minorstate = MINORSTATE_TAGEND;
757  break;
758  }
759  }
760  }

762  case MINORSTATE_HTMLCOMMENT:
763  for(i = 0; i < len; ++i) {
764  c = data[i];
765  if(c == ISO_dash) {
766  ++s.tagptr;
767  } else if(c == ISO_gt && s.tagptr > 0) {

773  }
774  }

778  for(i = 0; i < len; ++i) {

783  parse_tag();
784  break;
785  }
786  }
787  break;
788  default:
789  i = 0;
790  break;
791  }
792  if(i >= len) {
793  return len;
794  }
795  return i + 1;
796 }
