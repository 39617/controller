46 enum PHIDGET_RELAY_STATUSTYPES {

53 static enum PHIDGET_RELAY_STATUSTYPES _RELAY_STATUS = 0x00;

57 void
58 relay_enable(uint8_t pin)
59 {
60 
61  if(!(_RELAY_STATUS & INITED)) {
62 
63  _RELAY_STATUS |= INITED;
64 
65  /* Selects the pin to be configure as the control pin of the relay module */
66  controlPin = (1 << pin);
67 
68  /* Configures the control pin */
69  P6SEL &= ~controlPin;
70  P6DIR |= controlPin;
71  }
72 }

74 
75 void
76 relay_on()
77 {
78  if((_RELAY_STATUS & INITED)) {
79  P6OUT |= controlPin;
80  }
81 }
82 /*---------------------------------------------------------------------------*/
83 void
84 relay_off()
85 {
86  if((_RELAY_STATUS & INITED)) {

89 }

92 int8_t
93 relay_toggle()
94 {
95  if((_RELAY_STATUS & INITED)) {
96  P6OUT ^= controlPin;
97  if((P6OUT & controlPin)) {
98  return 1;
99  }
100  return 0;
101  }
102  return -1;
