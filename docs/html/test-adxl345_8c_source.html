59 /* As several interrupts can be mapped to one interrupt pin, when interrupt
60  * strikes, the adxl345 interrupt source register is read. This function prints
61  * out which interrupts occurred. Note that this will include all interrupts,
62  * even those mapped to 'the other' pin, and those that will always signal even
63  * if not enabled (such as watermark).
64  */
65 void
66 print_int(uint16_t reg)
67 {
68  if(reg & ADXL345_INT_FREEFALL) {
69  printf("Freefall ");
70  }
71  if(reg & ADXL345_INT_INACTIVITY) {
72  printf("InActivity ");
73  }
74  if(reg & ADXL345_INT_ACTIVITY) {
75  printf("Activity ");
76  }
77  if(reg & ADXL345_INT_DOUBLETAP) {
78  printf("DoubleTap ");
79  }
80  if(reg & ADXL345_INT_TAP) {
81  printf("Tap ");
82  }
83  printf("\n");
84 }

100 void

110  printf("~~[%u] Tap detected! (0x%02X) -- ",
111  ((uint16_t)clock_time()) / 128, reg);
112  }
113  print_int(reg);
114 }
