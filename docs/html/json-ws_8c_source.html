111 static int
112 cfg_get(struct jsontree_context *js_ctx)
113 {
114  const char *path = jsontree_path_name(js_ctx, js_ctx->depth - 1);
115 
116  if(strncmp(path, "host", 4) == 0) {
117  jsontree_write_string(js_ctx, callback_host);
118  } else if(strncmp(path, "port", 4) == 0) {
119  jsontree_write_int(js_ctx, callback_port);
120  } else if(strncmp(path, "interval", 8) == 0) {
121  jsontree_write_int(js_ctx, callback_interval);
122  } else if(strncmp(path, "path", 4) == 0) {
123  jsontree_write_string(js_ctx, callback_path);
124  } else if(strncmp(path, "appdata", 7) == 0) {
125  jsontree_write_string(js_ctx, callback_appdata[0] == '\0' ? "" : "***");
126  } else if(strncmp(path, "proto", 5) == 0) {
127  jsontree_write_string(js_ctx, callback_proto);
128  }
129  return 0;
130 }
131 static int
132 cfg_set(struct jsontree_context *js_ctx, struct jsonparse_state *parser)
133 {
134  int type;
135  int update = 0;
136 
137  while((type = jsonparse_next(parser)) != 0) {
138  if(type == JSON_TYPE_PAIR_NAME) {
139  if(jsonparse_strcmp_value(parser, "host") == 0) {
140  json_copy_string(parser, callback_host, sizeof(callback_host));
141  update++;
142  } else if(jsonparse_strcmp_value(parser, "path") == 0) {
143  json_copy_string(parser, callback_path, sizeof(callback_path));
144  update++;
145  } else if(jsonparse_strcmp_value(parser, "appdata") == 0) {
146  json_copy_string(parser, callback_appdata, sizeof(callback_appdata));
147  update++;
148  } else if(jsonparse_strcmp_value(parser, "proto") == 0) {
149  json_copy_string(parser, callback_proto, sizeof(callback_proto));
150  update++;
151  } else if(jsonparse_strcmp_value(parser, "port") == 0) {
152  jsonparse_next(parser);
153  jsonparse_next(parser);
154  callback_port = jsonparse_get_value_as_int(parser);
155  if(callback_port == 0) {
156  callback_port = CALLBACK_PORT;
157  }
158  update++;
159  } else if(jsonparse_strcmp_value(parser, "interval") == 0) {
160  jsonparse_next(parser);
161  jsonparse_next(parser);
162  callback_interval = jsonparse_get_value_as_int(parser);
163  if(callback_interval == 0) {
164  callback_interval = SEND_INTERVAL;
165  }
166  update++;
167  }
168  }
169  }
170  if(update && callback_json_path != NULL) {
171 #if WITH_UDP
172  if(strncmp(callback_proto, "udp", 3) == 0) {
173  json_ws_udp_setup(callback_host, callback_port);
174  }
175 #endif
176  ctimer_set(&periodic_timer, CLOCK_SECOND * callback_interval,
177  periodic, NULL);
178  }
179  return 0;
180 }

284 static
285 PT_THREAD(send_values(struct httpd_ws_state *s))
286 {
287  json_putchar_context = s;
288 
289  PSOCK_BEGIN(&s->sout);
290 
291  s->json.putchar = json_putchar;
292  s->outbuf_pos = 0;
293 
294  if(s->json.values[0] == NULL) {
295  /* Nothing to do */
296 
297  } else if(s->request_type == HTTPD_WS_POST &&
298  s->state == HTTPD_WS_STATE_OUTPUT) {
299  /* Set value */
300  struct jsontree_value *v;
301  struct jsontree_callback *c;
302 
303  while((v = jsontree_find_next(&s->json, JSON_TYPE_CALLBACK)) != NULL) {
304  c = (struct jsontree_callback *)v;
305  if(c->set != NULL) {
306  struct jsonparse_state js;
307 
308  jsonparse_setup(&js, s->inputbuf, s->content_len);
309  c->set(&s->json, &js);
310  }
311  }
312  memcpy(s->outbuf, "{\"Status\":\"OK\"}", 15);
313  s->outbuf_pos = 15;
314 
315  } else {
316  /* Get value */
317  while(jsontree_print_next(&s->json) && s->json.path <= s->json.depth) {
318  if(s->outbuf_pos >= UIP_TCP_MSS) {
319  SEND_STRING(&s->sout, s->outbuf, UIP_TCP_MSS);
320  s->outbuf_pos -= UIP_TCP_MSS;
321  if(s->outbuf_pos > 0) {
322  memcpy(s->outbuf, &s->outbuf[UIP_TCP_MSS], s->outbuf_pos);
323  }
324  }
325  }
326  }
327 
328  if(s->outbuf_pos > 0) {
329  SEND_STRING(&s->sout, s->outbuf, s->outbuf_pos);
330  s->outbuf_pos = 0;
331  }
332  PSOCK_END(&s->sout);
333 }

335 struct jsontree_value *
336 find_json_path(struct jsontree_context *json, const char *path)
337 {
338  struct jsontree_value *v;
339  const char *start;
340  const char *end;
341  int len;
342 
343  v = json->values[0];
344  start = path;
345  do {
346  end = strchr(start, '/');
347  if(end == start) {
348  break;
349  }
350  if(end != NULL) {
351  len = end - start;
352  end++;
353  } else {
354  len = strlen(start);
355  }
356  if(v->type != JSON_TYPE_OBJECT) {
357  v = NULL;
358  } else {
359  struct jsontree_object *o;
360  int i;
361 
362  o = (struct jsontree_object *)v;
363  v = NULL;
364  for(i = 0; i < o->count; i++) {
365  if(strncmp(start, o->pairs[i].name, len) == 0) {
366  v = o->pairs[i].value;
367  json->index[json->depth] = i;
368  json->depth++;
369  json->values[json->depth] = v;
370  json->index[json->depth] = 0;
371  break;
372  }
373  }
374  }
375  start = end;
376  } while(end != NULL && *end != '\0' && v != NULL);
377  json->callback_state = 0;
378  return v;
379 }

381 static int
382 calculate_json_size(const char *path, struct jsontree_value *v)
383 {
384  /* check size of JSON expression */
385  struct jsontree_context json;
386 
387  json.values[0] = (v == NULL) ? (struct jsontree_value *)tree : v;
388  jsontree_reset(&json);
389 
390  if(path != NULL) {
391  find_json_path(&json, path);
392  }
393 
394  json.path = json.depth;
395  json.putchar = json_putchar_count;
396  putchar_size = 0;
397  while(jsontree_print_next(&json) && json.path <= json.depth);
398 
399  return putchar_size;
400 }

444 static void
445 periodic(void *ptr)
446 {
447  struct httpd_ws_state *s;
448  int callback_size;
449 
450  if(callback_json_path != NULL && strlen(callback_host) > 2) {
451  ctimer_restart(&periodic_timer);
452 
453  if(strncmp(callback_proto, "http", 4) == 0) {
454  callback_size = calculate_json_size(callback_json_path, NULL);
455  s = httpd_ws_request(HTTPD_WS_POST, callback_host, NULL, callback_port,
456  callback_path, http_content_type_json,
457  callback_size, send_values);
458  if(s != NULL) {
459  PRINTF("PERIODIC POST %s\n", callback_json_path);
460 #if JSON_POST_EXTRA_HEADER
461  s->output_extra_headers = output_headers;
462 #endif
463  s->json.values[0] = (struct jsontree_value *)tree;
464  jsontree_reset(&s->json);
465  find_json_path(&s->json, callback_json_path);
466  s->json.path = s->json.depth;
467  } else {
468  PRINTF("PERIODIC CALLBACK FAILED\n");
469  }
470 #if WITH_COSM
471  } else if(strncmp(callback_proto, "cosm", 4) == 0) {
472  callback_size = calculate_json_size(NULL, (struct jsontree_value *)
473  &cosm_tree);
474  /* printf("JSON Size:%d\n", callback_size); */
475  s = httpd_ws_request(HTTPD_WS_PUT, callback_host, "api.pachube.com",
476  callback_port, callback_path,
477  http_content_type_json, callback_size, send_values);
478  /* host = cosm host */
479  /* path => path to datastream / data point */
480  s->output_extra_headers = output_headers;
481  s->json.values[0] = (struct jsontree_value *)&cosm_tree;
482  jsontree_reset(&s->json);
483  s->json.path = 0;
484 
485  PRINTF("PERIODIC cosm callback: %d\n", callback_size);
486 #endif /* WITH_COSM */
487  }
488 #if WITH_UDP
489  else {
490  callback_size = calculate_json_size(callback_json_path, NULL);
491  PRINTF("PERIODIC UDP size: %d\n", callback_size);
492  json_ws_udp_send(tree, callback_json_path);
493  }
494 #endif /* WITH_UDP */
495  } else {
496  printf("PERIODIC CALLBACK - nothing todo\n");
497  }
498 }
