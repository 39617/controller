86 void ping(void);
87 SYSCALLS_DEFINE_SINGLETON(ping, ping_dcd)
88 {
89  uint64_t start, mid, end;
90  uint64_t diff1 = 0, diff2 = 0;
91  double diff1_d, diff2_d;
92  int i = 0;
93  int cnt;
94 
95  while(((diff1 + diff2) >> DURATION_SECONDS_SHAMT) < CPU_FREQ) {
96  start = _rdtsc();
97  pong(&mid, &cnt);
98  end = _rdtsc();
99 
100 #ifdef SAMPLE_METADATA
101  assert(cnt == i);
102 #endif
103 
104  /* exclude the warm-up round */
105  if(i != 0) {
106  diff1 += mid - start;
107  diff2 += end - mid;
108  }
109 
110  i++;
111  }
112 
113  diff1_d = diff1;
114  diff2_d = diff2;
115 
116  diff1_d /= i - 1;
117  diff2_d /= i - 1;
118 
119  puts( "Sample protection domain ping-pong switching latency measurements:");
120  printf(" %u iterations\n", i - 1);
121  printf(" Avg. # cycles ping -> pong: %.2f\n", diff1_d);
122  printf(" + Avg. # cycles pong -> ping: %.2f\n", diff2_d);
123  puts( " ----------------------------------------");
124  printf(" Avg. # cycles round-trip: %.2f\n", diff1_d + diff2_d);
125 }
