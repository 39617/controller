55 // AONBatMonTemperatureGetDegC()
56 // Returns sign extended temperature in Deg C (-256 .. +255)
57 //
58 //*****************************************************************************
59 int32_t
60 AONBatMonTemperatureGetDegC( void )
61 {
62  int32_t signedTemp ; // Signed extended temperature with 8 fractional bits
63  int32_t tempCorrection ; // Voltage dependent temp correction with 8 fractional bits
64  int8_t voltageSlope ; // Signed byte value representing the TEMP slope with battery voltage, in degrees C/V, with 4 fractional bits.
65 
66  //
67  // Shift left then right to sign extend the BATMON_TEMP field
68  //
69  signedTemp = ((((int32_t)HWREG( AON_BATMON_BASE + AON_BATMON_O_TEMP ))
70  << ( 32 - AON_BATMON_TEMP_INT_W - AON_BATMON_TEMP_INT_S ))
71  >> ( 32 - AON_BATMON_TEMP_INT_W - AON_BATMON_TEMP_INT_S ));
72 
73  //
74  // Typecasting voltageSlope to int8_t prior to assignment in order to make sure sign extension works properly
75  // Using byte read (HWREGB) in order to make more efficient code since voltageSlope is assigned to bits[7:0] of FCFG1_O_MISC_TRIM
76  //
77  voltageSlope = ((int8_t)HWREGB( FCFG1_BASE + FCFG1_O_MISC_TRIM ));
78  tempCorrection = (( voltageSlope * (((int32_t)HWREG( AON_BATMON_BASE + AON_BATMON_O_BAT )) - 0x300 )) >> 4 );
79 
80  return ((( signedTemp - tempCorrection ) + 0x80 ) >> 8 );
81 }
