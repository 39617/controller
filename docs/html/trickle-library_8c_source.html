50 #define IMIN 16 /* ticks */
51 #define IMAX 10 /* doublings */
52 #define REDUNDANCY_CONST 2
53 
54 /* Networking */
55 #define TRICKLE_PROTO_PORT 30001
56 static struct uip_udp_conn *trickle_conn;
57 static uip_ipaddr_t ipaddr; /* destination: link-local all-nodes multicast */
58 
59 /*
60  * For this 'protocol', nodes exchange a token (1 byte) at a frequency
61  * governed by trickle. A node detects an inconsistency when it receives a
62  * token different than the one it knows.
63  * In this case, either:
64  * - 'they' have a 'newer' token and we also update our own value, or
65  * - 'we' have a 'newer' token, in which case we trigger an inconsistency
66  * without updating our value.
67  * In this context, 'newer' is defined in serial number arithmetic terms.
68  *
69  * Every NEW_TOKEN_INTERVAL clock ticks each node will generate a new token
70  * with probability 1/NEW_TOKEN_PROB. This is controlled by etimer et.
71  */
72 #define NEW_TOKEN_INTERVAL 10 * CLOCK_SECOND
73 #define NEW_TOKEN_PROB 2
74 static uint8_t token;
75 static struct etimer et; /* Used to periodically generate inconsistencies */
76 /*---------------------------------------------------------------------------*/
77 PROCESS(trickle_protocol_process, "Trickle Protocol process");
78 AUTOSTART_PROCESSES(&trickle_protocol_process);
79 /*---------------------------------------------------------------------------*/
80 static void
81 tcpip_handler(void)
82 {
83  leds_on(LEDS_GREEN);
84  if(uip_newdata()) {
85  PRINTF("At %lu (I=%lu, c=%u): ",
86  (unsigned long)clock_time(), (unsigned long)tt.i_cur, tt.c);
87  PRINTF("Our token=0x%02x, theirs=0x%02x\n", token,
88  ((uint8_t *)uip_appdata)[0]);
89  if(token == ((uint8_t *)uip_appdata)[0]) {
90  PRINTF("Consistent RX\n");
91  trickle_timer_consistency(&tt);
92  } else {
93  if((signed char)(token - ((uint8_t *)uip_appdata)[0]) < 0) {
94  PRINTF("Theirs is newer. Update\n");
95  token = ((uint8_t *)uip_appdata)[0];
96  } else {
97  PRINTF("They are behind\n");
98  }
99  trickle_timer_inconsistency(&tt);
100 
101  /*
102  * Here tt.ct.etimer.timer.{start + interval} points to time t in the
103  * current interval. However, between t and I it points to the interval's
104  * end so if you're going to use this, do so with caution.
105  */
106  PRINTF("At %lu: Trickle inconsistency. Scheduled TX for %lu\n",
107  (unsigned long)clock_time(),
108  (unsigned long)(tt.ct.etimer.timer.start +
109  tt.ct.etimer.timer.interval));
110  }
111  }
112  leds_off(LEDS_GREEN);
113  return;
114 }
115 /*---------------------------------------------------------------------------*/
116 static void
117 trickle_tx(void *ptr, uint8_t suppress)
118 {
119  /* *ptr is a pointer to the trickle_timer that triggered this callback. In
120  * his example we know that ptr points to tt. However, we pretend that we did
121  * not know (which would be the case if we e.g. had multiple trickle timers)
122  * and cast it to a local struct trickle_timer* */
123  struct trickle_timer *loc_tt = (struct trickle_timer *)ptr;
124 
125  if(suppress == TRICKLE_TIMER_TX_SUPPRESS) {
126  return;
127  }
128 
129  leds_on(LEDS_RED);
130 
131  PRINTF("At %lu (I=%lu, c=%u): ",
132  (unsigned long)clock_time(), (unsigned long)loc_tt->i_cur,
133  loc_tt->c);
134  PRINTF("Trickle TX token 0x%02x\n", token);
135 
136  /* Instead of changing ->ripaddr around by ourselves, we could have used
137  * uip_udp_packet_sendto which would have done it for us. However it puts an
138  * extra ~20 bytes on stack and the cc2x3x micros hate it, so we stick with
139  * send() */
140 
141  /* Destination IP: link-local all-nodes multicast */
142  uip_ipaddr_copy(&trickle_conn->ripaddr, &ipaddr);
143  uip_udp_packet_send(trickle_conn, &token, sizeof(token));
144 
145  /* Restore to 'accept incoming from any IP' */
146  uip_create_unspecified(&trickle_conn->ripaddr);
147 
148  leds_off(LEDS_RED);
149 }
150 /*---------------------------------------------------------------------------*/
151 PROCESS_THREAD(trickle_protocol_process, ev, data)
152 {
153  PROCESS_BEGIN();
154 
155  PRINTF("Trickle protocol started\n");
156 
157  uip_create_linklocal_allnodes_mcast(&ipaddr); /* Store for later */
158 
159  trickle_conn = udp_new(NULL, UIP_HTONS(TRICKLE_PROTO_PORT), NULL);
160  udp_bind(trickle_conn, UIP_HTONS(TRICKLE_PROTO_PORT));
161 
162  PRINTF("Connection: local/remote port %u/%u\n",
163  UIP_HTONS(trickle_conn->lport), UIP_HTONS(trickle_conn->rport));
164 
165  token = 0;
166 
167  trickle_timer_config(&tt, IMIN, IMAX, REDUNDANCY_CONST);
168  trickle_timer_set(&tt, trickle_tx, &tt);
169  /*
170  * At this point trickle is started and is running the first interval. All
171  * nodes 'agree' that token == 0. This will change when one of them randomly
172  * decides to generate a new one
173  */
174  etimer_set(&et, NEW_TOKEN_INTERVAL);
175 
176  while(1) {
177  PROCESS_YIELD();
178  if(ev == tcpip_event) {
179  tcpip_handler();
180  } else if(etimer_expired(&et)) {
181  /* Periodically (and randomly) generate a new token. This will trigger
182  * a trickle inconsistency */
183  if((random_rand() % NEW_TOKEN_PROB) == 0) {
184  token++;
185  PRINTF("At %lu: Generating a new token 0x%02x\n",
186  (unsigned long)clock_time(), token);
187  trickle_timer_reset_event(&tt);
188  }
189  etimer_set(&et, NEW_TOKEN_INTERVAL);
190  }
191  }
192  PROCESS_END();
193 }
