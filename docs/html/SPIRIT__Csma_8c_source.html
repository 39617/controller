120 {
121  uint8_t tempRegValue[5];
122 
123  /* Check the parameters */
124  s_assert_param(IS_SPIRIT_FUNCTIONAL_STATE(pxCsmaInit->xCsmaPersistentMode));
125  s_assert_param(IS_CCA_PERIOD(pxCsmaInit->xMultiplierTbit));
126  s_assert_param(IS_CSMA_LENGTH(pxCsmaInit->xCcaLength));
127  s_assert_param(IS_BU_COUNTER_SEED(pxCsmaInit->nBuCounterSeed));
128  s_assert_param(IS_BU_PRESCALER(pxCsmaInit->cBuPrescaler));
129  s_assert_param(IS_CMAX_NB(pxCsmaInit->cMaxNb));
130 
131  /* CSMA BU counter seed (MSB) config */
132  tempRegValue[0] = (uint8_t)(pxCsmaInit->nBuCounterSeed >> 8);
133 
134  /* CSMA BU counter seed (LSB) config */
135  tempRegValue[1] = (uint8_t) pxCsmaInit->nBuCounterSeed;
136 
137  /* CSMA BU prescaler config and CCA period config */
138  tempRegValue[2] = (pxCsmaInit->cBuPrescaler << 2) | pxCsmaInit->xMultiplierTbit;
139 
140  /* CSMA CCA length config and max number of back-off */
141  tempRegValue[3] = (pxCsmaInit->xCcaLength | pxCsmaInit->cMaxNb);
142 
143  /* Reads the PROTOCOL1_BASE register value, to write the SEED_RELOAD and CSMA_PERS_ON fields */
144  g_xStatus = SpiritSpiReadRegisters(PROTOCOL1_BASE, 1, &tempRegValue[4]);
145 
146  /* Writes the new value for persistent mode */
147  if(pxCsmaInit->xCsmaPersistentMode==S_ENABLE)
148  {
149  tempRegValue[4] |= PROTOCOL1_CSMA_PERS_ON_MASK;
150  }
151  else
152  {
153  tempRegValue[4] &= ~PROTOCOL1_CSMA_PERS_ON_MASK;
154  }
155 
156  /* Writes PROTOCOL1_BASE register */
157  g_xStatus = SpiritSpiWriteRegisters(PROTOCOL1_BASE, 1, &tempRegValue[4]);
158 
159  /* Writes CSMA_CONFIGx_BASE registers */
160  g_xStatus = SpiritSpiWriteRegisters(CSMA_CONFIG3_BASE, 4, tempRegValue);
161 
162 }
