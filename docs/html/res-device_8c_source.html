50 static uint16_t
51 detect_chip(void)
52 {
53  if(ti_lib_chipinfo_chip_family_is_cc26xx()) {
54  if(ti_lib_chipinfo_supports_ieee_802_15_4() == true) {
55  if(ti_lib_chipinfo_supports_ble() == true) {
56  return 2650;
57  } else {
58  return 2630;
59  }
60  } else {
61  return 2640;
62  }
63  } else if(ti_lib_chipinfo_chip_family_is_cc13xx()) {
64  if(ti_lib_chipinfo_supports_ble() == false &&
65  ti_lib_chipinfo_supports_ieee_802_15_4() == false) {
66  return 1310;
67  } else if(ti_lib_chipinfo_supports_ble() == true &&
68  ti_lib_chipinfo_supports_ieee_802_15_4() == true) {
69  return 1350;
70  }
71  }
72 
73  return 0;
74 }

78  uint16_t preferred_size, int32_t *offset)
