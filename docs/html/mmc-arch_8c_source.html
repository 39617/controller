52 mmc_arch_init(void)
53 {
54  static uint8_t init_done;
55 
56  if(init_done) {
57  return;
58  }
59 
60  GPIO_SET_INPUT(USD_SEL_PORT_BASE, USD_SEL_PIN_MASK);
61  GPIO_SOFTWARE_CONTROL(USD_SEL_PORT_BASE, USD_SEL_PIN_MASK);
62  ioc_set_over(USD_SEL_PORT, USD_SEL_PIN, IOC_OVERRIDE_DIS);
63 
64  spix_cs_init(USD_CSN_PORT, USD_CSN_PIN);
65  spix_init(USD_SPI_INSTANCE);
66  spix_set_mode(USD_SPI_INSTANCE, SSI_CR0_FRF_MOTOROLA, 0, 0, 8);
67 
68  init_done = 1;
69 }

71 bool
72 mmc_arch_get_cd(uint8_t dev)
73 {
74  mmc_arch_init();
75 
76  if(GPIO_IS_OUTPUT(USD_SEL_PORT_BASE, USD_SEL_PIN_MASK)) {
77  /* Card previously detected and powered */
78  return true;
79  } else if(GPIO_READ_PIN(USD_SEL_PORT_BASE, USD_SEL_PIN_MASK)) {
80  /* Card inserted -> power it */
81  GPIO_SET_OUTPUT(USD_SEL_PORT_BASE, USD_SEL_PIN_MASK);
82  GPIO_CLR_PIN(USD_SEL_PORT_BASE, USD_SEL_PIN_MASK);
83  return true;
84  } else {
85  /* No card detected */
86  return false;
87  }
88 }

112 void
113 mmc_arch_spi_xfer(uint8_t dev, const void *tx_buf, size_t tx_cnt,
114  void *rx_buf, size_t rx_cnt)
115 {
116  const uint8_t *tx_buf_u8 = tx_buf;
117  uint8_t *rx_buf_u8 = rx_buf;
118 
119  while(tx_cnt || rx_cnt) {
120  SPIX_WAITFORTxREADY(USD_SPI_INSTANCE);
121  if(tx_cnt) {
122  SPIX_BUF(USD_SPI_INSTANCE) = *tx_buf_u8++;
123  tx_cnt--;
124  } else {
125  SPIX_BUF(USD_SPI_INSTANCE) = 0;
126  }
127  SPIX_WAITFOREOTx(USD_SPI_INSTANCE);
128  SPIX_WAITFOREORx(USD_SPI_INSTANCE);
129  if(rx_cnt) {
130  *rx_buf_u8++ = SPIX_BUF(USD_SPI_INSTANCE);
131  rx_cnt--;
132  } else {
133  SPIX_BUF(USD_SPI_INSTANCE);
134  }
135  }
136 }
