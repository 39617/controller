133 /* option format serialization */
134 #define COAP_SERIALIZE_INT_OPTION(number, field, text) \
135  if(IS_OPTION(coap_pkt, number)) { \
136  PRINTF(text " [%u]\n", (unsigned int)coap_pkt->field); \
137  option += coap_serialize_int_option(number, current_number, option, coap_pkt->field); \
138  current_number = number; \
139  }
140 #define COAP_SERIALIZE_BYTE_OPTION(number, field, text) \
141  if(IS_OPTION(coap_pkt, number)) { \
142  PRINTF(text " %u [0x%02X%02X%02X%02X%02X%02X%02X%02X]\n", (unsigned int)coap_pkt->field##_len, \
143  coap_pkt->field[0], \
144  coap_pkt->field[1], \
145  coap_pkt->field[2], \
146  coap_pkt->field[3], \
147  coap_pkt->field[4], \
148  coap_pkt->field[5], \
149  coap_pkt->field[6], \
150  coap_pkt->field[7] \
151  ); /* FIXME always prints 8 bytes */ \
152  option += coap_serialize_array_option(number, current_number, option, coap_pkt->field, coap_pkt->field##_len, '\0'); \
153  current_number = number; \
154  }
155 #define COAP_SERIALIZE_STRING_OPTION(number, field, splitter, text) \
156  if(IS_OPTION(coap_pkt, number)) { \
157  PRINTF(text " [%.*s]\n", (int)coap_pkt->field##_len, coap_pkt->field); \
158  option += coap_serialize_array_option(number, current_number, option, (uint8_t *)coap_pkt->field, coap_pkt->field##_len, splitter); \
159  current_number = number; \
160  }
161 #define COAP_SERIALIZE_BLOCK_OPTION(number, field, text) \
162  if(IS_OPTION(coap_pkt, number)) \
163  { \
164  PRINTF(text " [%lu%s (%u B/blk)]\n", (unsigned long)coap_pkt->field##_num, coap_pkt->field##_more ? "+" : "", coap_pkt->field##_size); \
165  uint32_t block = coap_pkt->field##_num << 4; \
166  if(coap_pkt->field##_more) { block |= 0x8; } \
167  block |= 0xF & coap_log_2(coap_pkt->field##_size / 16); \
168  PRINTF(text " encoded: 0x%lX\n", (unsigned long)block); \
169  option += coap_serialize_int_option(number, current_number, option, block); \
170  current_number = number; \
171  }
