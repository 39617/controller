83  if(st_lib_hal_gpio_read_pin(st_lib_a_led_gpio_port[RADIO_SHIELD_LED],
84  st_lib_a_led_gpio_pin[RADIO_SHIELD_LED])) {
85  ret |= LEDS_RED;

87 #endif /*X_NUCLEO_IKS01A1*/
