110 PROCESS_THREAD(sensors_test_process, ev, data)
111 {
112  static struct etimer et;
113 
114  /* Sensor Values */
115  static int rv;
116  static struct sensors_sensor *sensor;
117  static float sane = 0;
118  static int dec;
119  static float frac;
120 
121  PROCESS_BEGIN();
122 
123  PRINTF("========================\n");
124  PRINTF("Starting Sensor Example.\n");
125  PRINTF("========================\n");
126 
127  /* Set an etimer. We take sensor readings when it expires and reset it. */
128  etimer_set(&et, CLOCK_SECOND * 2);
129 
130  while(1) {
131 
132  PROCESS_WAIT_EVENT_UNTIL(etimer_expired(&et));
133 
134  /*
135  * Request some ADC conversions
136  * Return value -1 means sensor not available or turned off in conf
137  */
138  sensor = sensors_find(ADC_SENSOR);
139  if(sensor) {
140  PRINTF("------------------\n");
141  leds_on(LEDS_RED);
142  /*
143  * Temperature:
144  * Using 1.25V ref. voltage (1250mV).
145  * Typical AD Output at 25°C: 1480
146  * Typical Co-efficient : 4.5 mV/°C
147  *
148  * Thus, at 12bit decimation (and ignoring the VDD co-efficient as well
149  * as offsets due to lack of calibration):
150  *
151  * AD - 1480
152  * T = 25 + ---------
153  * 4.5
154  */
155  rv = sensor->value(ADC_SENSOR_TYPE_TEMP);
156  if(rv != -1) {
157  sane = 25 + ((rv - 1480) / 4.5);
158  dec = sane;
159  frac = sane - dec;
160  PRINTF(" Temp=%d.%02u C (%d)\n", dec, (unsigned int)(frac*100), rv);
161  }
162  /*
163  * Power Supply Voltage.
164  * Using 1.25V ref. voltage.
165  * AD Conversion on VDD/3
166  *
167  * Thus, at 12bit resolution:
168  *
169  * ADC x 1.25 x 3
170  * Supply = -------------- V
171  * 2047
172  */
173  rv = sensor->value(ADC_SENSOR_TYPE_VDD);
174  if(rv != -1) {
175  sane = rv * 3.75 / 2047;
176  dec = sane;
177  frac = sane - dec;
178  PRINTF("Supply=%d.%02u V (%d)\n", dec, (unsigned int)(frac*100), rv);
179  /* Store rv temporarily in dec so we can use it for the battery */
180  dec = rv;
181  }
182  /*
183  * Battery Voltage - ToDo
184  * rv = sensor->value(ADC_SENSOR_TYPE_BATTERY);
185  */
186 
187  leds_off(LEDS_RED);
188  }
189  etimer_reset(&et);
190  }
191  PROCESS_END();
192 }
