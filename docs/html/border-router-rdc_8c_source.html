58 /* a structure for calling back when packet data is coming back
59  from radio... */
60 struct tx_callback {
61  mac_callback_t cback;

63  struct packetbuf_attr attrs[PACKETBUF_NUM_ATTRS];
64  struct packetbuf_addr addrs[PACKETBUF_NUM_ADDRS];

67 static struct tx_callback callbacks[MAX_CALLBACKS];
68 /*---------------------------------------------------------------------------*/
69 void packet_sent(uint8_t sessionid, uint8_t status, uint8_t tx)
70 {
71  if(sessionid < MAX_CALLBACKS) {
72  struct tx_callback *callback;
73  callback = &callbacks[sessionid];
74  packetbuf_clear();
75  packetbuf_attr_copyfrom(callback->attrs, callback->addrs);
76  mac_call_sent_callback(callback->cback, callback->ptr, status, tx);
77  } else {
78  PRINTF("*** ERROR: too high session id %d\n", sessionid);
79  }
80 }

82 static int
83 setup_callback(mac_callback_t sent, void *ptr)
84 {
85  struct tx_callback *callback;
86  int tmp = callback_pos;
87  callback = &callbacks[callback_pos];
88  callback->cback = sent;
89  callback->ptr = ptr;
90  packetbuf_attr_copyto(callback->attrs, callback->addrs);
91 
92  callback_pos++;
93  if(callback_pos >= MAX_CALLBACKS) {
94  callback_pos = 0;
95  }
96 
97  return tmp;
98 }

100 static void
101 send_packet(mac_callback_t sent, void *ptr)
102 {
103  int size;
104  /* 3 bytes per packet attribute is required for serialization */
105  uint8_t buf[PACKETBUF_NUM_ATTRS * 3 + PACKETBUF_SIZE + 3];
106  uint8_t sid;
107 
108  packetbuf_set_addr(PACKETBUF_ADDR_SENDER, &linkaddr_node_addr);
109 
110  /* ack or not ? */
111  packetbuf_set_attr(PACKETBUF_ATTR_MAC_ACK, 1);
112 
113  if(NETSTACK_FRAMER.create() < 0) {
114  /* Failed to allocate space for headers */
115  PRINTF("br-rdc: send failed, too large header\n");
116  mac_call_sent_callback(sent, ptr, MAC_TX_ERR_FATAL, 1);
117 
118  } else {
119  /* here we send the data over SLIP to the radio-chip */
120  size = 0;
121 #if SERIALIZE_ATTRIBUTES
122  size = packetutils_serialize_atts(&buf[3], sizeof(buf) - 3);
123 #endif
124  if(size < 0 || size + packetbuf_totlen() + 3 > sizeof(buf)) {
125  PRINTF("br-rdc: send failed, too large header\n");
126  mac_call_sent_callback(sent, ptr, MAC_TX_ERR_FATAL, 1);
127  } else {
128  sid = setup_callback(sent, ptr);
129 
130  buf[0] = '!';
131  buf[1] = 'S';
132  buf[2] = sid; /* sequence or session number for this packet */
133 
134  /* Copy packet data */
135  memcpy(&buf[3 + size], packetbuf_hdrptr(), packetbuf_totlen());
136 
137  write_to_slip(buf, packetbuf_totlen() + size + 3);
138  }
139  }
140 }
