65  /* 3 possible roles:

67  * - role_6dg: DAG root, will advertise (unsecured) beacons
68  * */

71  node_role = role_6ln;
72 
73 #if CONFIG_VIA_BUTTON
74  {
75 #define CONFIG_WAIT_TIME 5
76  SENSORS_ACTIVATE(button_sensor);
77  etimer_set(&et, CLOCK_SECOND * CONFIG_WAIT_TIME);
78 
79  while(!etimer_expired(&et)) {
80  printf("Init: current role: %s. Will start in %u seconds.\n",
81  node_role == role_6ln ? "6ln" : "6dr",
82  CONFIG_WAIT_TIME);
83  PROCESS_WAIT_EVENT_UNTIL(((ev == sensors_event) &&
84  (data == &button_sensor) && button_sensor.value(0) > 0)
85  || etimer_expired(&et));
86  if(ev == sensors_event && data == &button_sensor && button_sensor.value(0) > 0) {
87  node_role = (node_role + 1) % 2;
88  etimer_restart(&et);
89  }
90  }
91  }
92 
93 #endif /* CONFIG_VIA_BUTTON */
94
