67  if(uip_len > 0) {

70  uip_neighbor_add(&IPBUF->srcipaddr, (struct uip_neighbor_addr *)&BUF->src);
71  tcpip_input();
72  } else
73 #endif /* NETSTACK_CONF_WITH_IPV6 */
74  if(BUF->type == uip_htons(UIP_ETHTYPE_IP)) {
75  uip_len -= sizeof(struct uip_eth_hdr);
76  tcpip_input();
77  } else if(BUF->type == uip_htons(UIP_ETHTYPE_ARP)) {
78  uip_arp_arpin();

81  uip_len is set to a value > 0. */

83  ethernet_send();
