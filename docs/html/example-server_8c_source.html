88 static struct node *
89 add_node(const uip_ipaddr_t *addr)
90 {
91  int i;
92  for(i = 0; i < node_count; i++) {
93  if(uip_ipaddr_cmp(&nodes[i].ipaddr, addr)) {
94  /* Node already added */
95  return &nodes[i];
96  }
97  }
98  if(node_count < MAX_NODES) {
99  uip_ipaddr_copy(&nodes[node_count].ipaddr, addr);
100  return &nodes[node_count++];
101  }
102  return NULL;
103 }

168 static void
169 client_chunk_handler(void *response)
170 {
171  const uint8_t *chunk;
172  unsigned int format;
173  int len = coap_get_payload(response, &chunk);
174  coap_get_header_content_format(response, &format);
175 
176  /* if(len > 0) { */
177  /* printf("|%.*s (%d,%d)", len, (char *)chunk, len, format); */
178  /* } */
179  if(current_target != NULL && fetching_type) {
180  if(len > sizeof(current_target->type) - 1) {
181  len = sizeof(current_target->type) - 1;
182  }
183  memcpy(current_target->type, chunk, len);
184  current_target->type[len] = 0;
185  current_target->flags |= NODE_HAS_TYPE;
186 
187  PRINTF("\nNODE ");
188  PRINT6ADDR(&current_target->ipaddr);
189  PRINTF(" HAS TYPE %s\n", current_target->type);
190  } else {
191  /* otherwise update the current value */
192  if(format == LWM2M_TLV) {
193  oma_tlv_t tlv;
194  /* we can only read int32 for now ? */
195  if(oma_tlv_read(&tlv, chunk, len) > 0) {
196  /* printf("TLV.type=%d len=%d id=%d value[0]=%d\n", */
197  /* tlv.type, tlv.length, tlv.id, tlv.value[0]); */
198 
199  int value = oma_tlv_get_int32(&tlv);
200  snprintf(current_value, sizeof(current_value), "%d", value);
201  }
202  } else {
203  if(len > sizeof(current_value) - 1) {
204  len = sizeof(current_value) - 1;
205  }
206  memcpy(current_value, chunk, len);
207  current_value[len] = 0;
208  }
209  }
210 }

226 #if UIP_CONF_ROUTER
227 
236 #if 0
237 /* Mode 1 - 64 bits inline */
238  uip_ip6addr(&ipaddr, UIP_DS6_DEFAULT_PREFIX, 0, 0, 0, 0, 0, 0, 1);
239 #elif 1
240 /* Mode 2 - 16 bits inline */
241  uip_ip6addr(&ipaddr, UIP_DS6_DEFAULT_PREFIX, 0, 0, 0, 0, 0x00ff, 0xfe00, 1);
242 #else
243 /* Mode 3 - derived from link local (MAC) address */
244  uip_ip6addr(&ipaddr, UIP_DS6_DEFAULT_PREFIX, 0, 0, 0, 0, 0, 0, 0);
245  uip_ds6_set_addr_iid(&ipaddr, &uip_lladdr);
246 #endif
247 
248  uip_ds6_addr_add(&ipaddr, 0, ADDR_MANUAL);
249  root_if = uip_ds6_addr_lookup(&ipaddr);
250  if(root_if != NULL) {
251  dag = rpl_set_root(RPL_DEFAULT_INSTANCE, &ipaddr);
252  uip_ip6addr(&ipaddr, UIP_DS6_DEFAULT_PREFIX, 0, 0, 0, 0, 0, 0, 0);
253  rpl_set_prefix(dag, &ipaddr, 64);
254  PRINTF("created a new RPL dag\n");
255  } else {
256  PRINTF("failed to create a new RPL DAG\n");
257  }
258 #endif /* UIP_CONF_ROUTER */
259 
260  PRINTF("IPv6 addresses: ");
261  for(i = 0; i < UIP_DS6_ADDR_NB; i++) {
262  state = uip_ds6_if.addr_list[i].state;
263  if(state == ADDR_TENTATIVE || state == ADDR_PREFERRED) {
264  PRINT6ADDR(&uip_ds6_if.addr_list[i].ipaddr);
265  PRINTF("\n");
266  /* hack to make address "final" */
267  if (state == ADDR_TENTATIVE) {
268  uip_ds6_if.addr_list[i].state = ADDR_PREFERRED;
269  }
270  }
271  }

274 PROCESS_THREAD(router_process, ev, data)
275 {
276  /* This way the packet can be treated as pointer as usual. */
277  static coap_packet_t request[1];
278  static struct etimer timer;
279  uip_ds6_route_t *r;
280  uip_ipaddr_t *nexthop;
281  int n;
282 
283  PROCESS_BEGIN();
284 
285  PROCESS_PAUSE();
286 
287  /* receives all CoAP messages */
288  coap_init_engine();
289 
290  setup_network();
291 
292  /* The data sink runs with a 100% duty cycle in order to ensure high
293  packet reception rates. */
294  NETSTACK_MAC.off(1);
295 
296  while(1) {
297  etimer_set(&timer, CLOCK_SECOND * 5);
298  PROCESS_YIELD();
299 
300  /* Handle serial line input */
301  if(ev == serial_line_event_message) {
302  serial_protocol_input((char *) data);
303  }
304 
305  if(etimer_expired(&timer)) {
306  current_target = NULL;
307  n = 0;
308  for(r = uip_ds6_route_head(); r != NULL; r = uip_ds6_route_next(r)) {
309  current_target = add_node(&r->ipaddr);
310  if(current_target == NULL ||
311  (current_target->flags & NODE_HAS_TYPE) != 0 ||
312  current_target->retries > 5) {
313  continue;
314  }
315  PRINTF(" ");
316  PRINT6ADDR(&r->ipaddr);
317  PRINTF(" -> ");
318  nexthop = uip_ds6_route_nexthop(r);
319  if(nexthop != NULL) {
320  PRINT6ADDR(nexthop);
321  PRINTF("\n");
322  } else {
323  PRINTF("-");
324  }
325  PRINTF("\n");
326  n++;
327  break;
328  }
329  }
330 
331  /* This is a node type discovery */
332  if(current_target != NULL &&
333  (current_target->flags & NODE_HAS_TYPE) == 0 &&
334  current_target->retries < 6) {
335 
336  /* prepare request, TID is set by COAP_BLOCKING_REQUEST() */
337  coap_init_message(request, COAP_TYPE_CON, COAP_GET, 0);
338  coap_set_header_uri_path(request, URL_DEVICE_MODEL);
339 
340  current_target->retries++;
341 
342  PRINTF("CoAP request to [");
343  PRINT6ADDR(&current_target->ipaddr);
344  PRINTF("]:%u (%u tx)\n", UIP_HTONS(REMOTE_PORT),
345  current_target->retries);
346 
347  fetching_type = 1;
348  COAP_BLOCKING_REQUEST(&current_target->ipaddr, REMOTE_PORT, request,
349  client_chunk_handler);
350  fetching_type = 0;
351  strncpy(current_uri, URL_LIGHT_CONTROL, sizeof(current_uri));
352  printf("\n--Done--\n");
353  }
354 
355  /* If having a type this is another type of request */
356  if(current_target != NULL &&
357  (current_target->flags & NODE_HAS_TYPE) && strlen(current_uri) > 0) {
358  /* prepare request, TID is set by COAP_BLOCKING_REQUEST() */
359  coap_init_message(request, COAP_TYPE_CON, current_request, 0);
360  coap_set_header_uri_path(request, current_uri);
361 
362  if(strlen(current_value) > 0) {
363  coap_set_payload(request, (uint8_t *)current_value,
364  strlen(current_value));
365  }
366 
367  PRINTF("CoAP request to [");
368  PRINT6ADDR(&current_target->ipaddr);
369  PRINTF("]:%u %s\n", UIP_HTONS(REMOTE_PORT), current_uri);
370 
371  COAP_BLOCKING_REQUEST(&current_target->ipaddr, REMOTE_PORT, request,
372  client_chunk_handler);
373 
374  /* print out result of command */
375  if(current_request == COAP_PUT) {
376  printf("s ");
377  } else {
378  printf("g ");
379  }
380  uip_debug_ipaddr_print(&current_target->ipaddr);
381  printf(" %s %s\n", current_uri, current_value);
382 
383  current_target = NULL;
384  current_uri[0] = 0;
385  current_value[0] = 0;
386 
387  }
388  }
389 
390  PROCESS_END();
391 }
