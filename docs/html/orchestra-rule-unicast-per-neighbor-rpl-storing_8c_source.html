68 static uint16_t
69 get_node_timeslot(const linkaddr_t *addr)
70 {
71  if(addr != NULL && ORCHESTRA_UNICAST_PERIOD > 0) {
72  return ORCHESTRA_LINKADDR_HASH(addr) % ORCHESTRA_UNICAST_PERIOD;
73  } else {
74  return 0xffff;
75  }
76 }

78 static int
79 neighbor_has_uc_link(const linkaddr_t *linkaddr)
80 {
81  if(linkaddr != NULL && !linkaddr_cmp(linkaddr, &linkaddr_null)) {
82  if((orchestra_parent_knows_us || !ORCHESTRA_UNICAST_SENDER_BASED)
83  && linkaddr_cmp(&orchestra_parent_linkaddr, linkaddr)) {
84  return 1;
85  }
86  if(nbr_table_get_from_lladdr(nbr_routes, (linkaddr_t *)linkaddr) != NULL) {
87  return 1;
88  }
89  }
90  return 0;
91 }

93 static void
94 add_uc_link(const linkaddr_t *linkaddr)
95 {
96  if(linkaddr != NULL) {
97  uint16_t timeslot = get_node_timeslot(linkaddr);
98  uint8_t link_options = ORCHESTRA_UNICAST_SENDER_BASED ? LINK_OPTION_RX : LINK_OPTION_TX | UNICAST_SLOT_SHARED_FLAG;
99 
100  if(timeslot == get_node_timeslot(&linkaddr_node_addr)) {
101  /* This is also our timeslot, add necessary flags */
102  link_options |= ORCHESTRA_UNICAST_SENDER_BASED ? LINK_OPTION_TX | UNICAST_SLOT_SHARED_FLAG: LINK_OPTION_RX;
103  }
104 
105  /* Add/update link */
106  tsch_schedule_add_link(sf_unicast, link_options, LINK_TYPE_NORMAL, &tsch_broadcast_address,
107  timeslot, channel_offset);
108  }
109 }
110 /*---------------------------------------------------------------------------*/
111 static void
112 remove_uc_link(const linkaddr_t *linkaddr)
113 {
114  uint16_t timeslot;
115  struct tsch_link *l;
116 
117  if(linkaddr == NULL) {
118  return;
119  }
120 
121  timeslot = get_node_timeslot(linkaddr);
122  l = tsch_schedule_get_link_by_timeslot(sf_unicast, timeslot);
123  if(l == NULL) {
124  return;
125  }
126  /* Does our current parent need this timeslot? */
127  if(timeslot == get_node_timeslot(&orchestra_parent_linkaddr)) {
128  /* Yes, this timeslot is being used, return */
129  return;
130  }
131  /* Does any other child need this timeslot?
132  * (lookup all route next hops) */
133  nbr_table_item_t *item = nbr_table_head(nbr_routes);
134  while(item != NULL) {
135  linkaddr_t *addr = nbr_table_get_lladdr(nbr_routes, item);
136  if(timeslot == get_node_timeslot(addr)) {
137  /* Yes, this timeslot is being used, return */
138  return;
139  }
140  item = nbr_table_next(nbr_routes, item);
141  }
142 
143  /* Do we need this timeslot? */
144  if(timeslot == get_node_timeslot(&linkaddr_node_addr)) {
145  /* This is our link, keep it but update the link options */
146  uint8_t link_options = ORCHESTRA_UNICAST_SENDER_BASED ? LINK_OPTION_TX | UNICAST_SLOT_SHARED_FLAG: LINK_OPTION_RX;
147  tsch_schedule_add_link(sf_unicast, link_options, LINK_TYPE_NORMAL, &tsch_broadcast_address,
148  timeslot, channel_offset);
149  } else {
150  /* Remove link */
151  tsch_schedule_remove_link(sf_unicast, l);
152  }
153 }

167 static int
168 select_packet(uint16_t *slotframe, uint16_t *timeslot)
169 {
170  /* Select data packets we have a unicast link to */
171  const linkaddr_t *dest = packetbuf_addr(PACKETBUF_ADDR_RECEIVER);
172  if(packetbuf_attr(PACKETBUF_ATTR_FRAME_TYPE) == FRAME802154_DATAFRAME
173  && neighbor_has_uc_link(dest)) {
174  if(slotframe != NULL) {
175  *slotframe = slotframe_handle;
176  }
177  if(timeslot != NULL) {
178  *timeslot = ORCHESTRA_UNICAST_SENDER_BASED ? get_node_timeslot(&linkaddr_node_addr) : get_node_timeslot(dest);
179  }
180  return 1;
181  }
182  return 0;
183 }

185 static void
186 new_time_source(const struct tsch_neighbor *old, const struct tsch_neighbor *new)
187 {
188  if(new != old) {
189  const linkaddr_t *old_addr = old != NULL ? &old->addr : NULL;
190  const linkaddr_t *new_addr = new != NULL ? &new->addr : NULL;
191  if(new_addr != NULL) {
192  linkaddr_copy(&orchestra_parent_linkaddr, new_addr);
193  } else {
194  linkaddr_copy(&orchestra_parent_linkaddr, &linkaddr_null);
195  }
196  remove_uc_link(old_addr);
197  add_uc_link(new_addr);
198  }
199 }

211  LINK_TYPE_NORMAL, &tsch_broadcast_address,
212  timeslot, channel_offset);
213 }
