52 void
53 leds_arch_init(void)
54 {
55  /* Initialize LED2 (Green) and LED3 (Blue) */
56  GPIO_SOFTWARE_CONTROL(GPIO_B_BASE, LEDS_PORTB_PIN_MASK);
57  GPIO_SET_OUTPUT(GPIO_B_BASE, LEDS_PORTB_PIN_MASK);
58  GPIO_CLR_PIN(GPIO_B_BASE, LEDS_PORTB_PIN_MASK);
59 
60  /* Initialize LED1 (Red) */
61  GPIO_SOFTWARE_CONTROL(LEDS_RED_PORT_BASE, LEDS_RED_PIN_MASK);
62  GPIO_SET_OUTPUT(LEDS_RED_PORT_BASE, LEDS_RED_PIN_MASK);
63  GPIO_CLR_PIN(LEDS_RED_PORT_BASE, LEDS_RED_PIN_MASK);
64 }
65 /*---------------------------------------------------------------------------*/
66 unsigned char
67 leds_arch_get(void)
68 {
69  uint8_t mask_leds;
70 
71  mask_leds = GPIO_READ_PIN(LEDS_GREEN_PORT_BASE, LEDS_GREEN_PIN_MASK) == 0 ? 0: LEDS_GREEN;
72  mask_leds |= GPIO_READ_PIN(LEDS_BLUE_PORT_BASE, LEDS_BLUE_PIN_MASK) == 0 ? 0 : LEDS_BLUE;
73  mask_leds |= GPIO_READ_PIN(LEDS_RED_PORT_BASE, LEDS_RED_PIN_MASK) == 0 ? 0 : LEDS_RED;
74 
75  return mask_leds;
76 }
