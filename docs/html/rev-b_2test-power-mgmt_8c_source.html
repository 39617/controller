74 PROCESS_THREAD(test_remote_pm, ev, data)
75 {
76  static uint8_t aux;
77  static uint16_t voltage;
78  static uint32_t cycles;
79 
80  PROCESS_BEGIN();
81 
82  aux = 0;
83  cycles = 0;
84 
85  /* Initialize the power management block and signal the low-power PIC */
86  if(pm_enable() != PM_SUCCESS) {
87  printf("PM: Failed to initialize\n");
88  TEST_LEDS_FAIL;
89  }
90 
91  printf("PM: enabled!\n");
92 
93  /* Retrieve the firmware version and check expected */
94  if((pm_get_fw_ver(&aux) == PM_ERROR) ||
95  (aux != PM_EXPECTED_VERSION)) {
96  printf("PM: unexpected version 0x%02X\n", aux);
97  TEST_LEDS_FAIL;
98  }
99 
100  printf("PM: firmware version 0x%02X OK\n", aux);
101 
102  /* Read the battery voltage level */
103  if(pm_get_voltage(&voltage) != PM_SUCCESS) {
104  printf("PM: error retrieving voltage\n");
105  TEST_LEDS_FAIL;
106  }
107 
108  printf("PM: Voltage (raw) = %u\n", voltage);
109 
110  /* Note: if running the following test while the RE-Mote is powered over USB
111  * will show the command execution, but it will not put the board in shutdown
112  * mode. If powering the RE-Mote with an external battery the shutdown mode
113  * will operate as intended, and the RE-Mote will restart and run the tests
114  * from the start after waking-up off the shutdown mode.
115  *
116  * The first test shows how to use the "soft" shutdown mode, being the low
117  * power PIC the one counting cycles and restarting the system off shutdown
118  * mode.
119  *
120  * Each restart cycle is tracked by the low-power PIC, we can use this value
121  * to determine how many times we have entered shutdown mode, thus choosing
122  * a specific configuration or behaviour. For the next examples we are going
123  * to trigger a "soft" mode each even number, and "hard" if odd.
124  */
125 
126  cycles = pm_get_num_cycles();
127  printf("PM: cycle number %lu\n", cycles);
128 
129  if((cycles % 2) == 0) {
130  /* Set the timeout */
131  if(pm_set_timeout(PM_SOFT_SHTDN_5_7_SEC) != PM_SUCCESS) {
132  printf("PM: error setting timeout for soft shutdown mode\n");
133  TEST_LEDS_FAIL;
134  }
135 
136  printf("PM: Soft shutdown, timeout set to %lu\n", pm_get_timeout());
137 
138  leds_off(LEDS_ALL);
139  leds_on(LEDS_PURPLE);
140 
141  /* Wait just enough to be able to check the LED result */
142  etimer_set(&et, CLOCK_SECOND * 3);
143  PROCESS_WAIT_EVENT_UNTIL(etimer_expired(&et));
144 
145  /* Enter soft shut*/
146  if(pm_shutdown_now(PM_SOFT_SLEEP_CONFIG) == PM_SUCCESS) {
147  printf("PM: Good night!\n");
148  } else {
149  printf("PM: error shutting down the system!\n");
150  TEST_LEDS_FAIL;
151  }
152 
153  /* Exit and wait the next cycle */
154  PROCESS_EXIT();
155  }
156 
157  /* Configure the RTCC to schedule a "hard" restart of the shutdown mode,
158  * waking up from a RTCC interrupt to the low-power PIC
159  */
160  printf("PM\n");
161 
162  if(rtcc_get_time_date(simple_td) == AB08_ERROR) {
163  printf("PM: Couldn't read time and date\n");
164  TEST_LEDS_FAIL;
165  }
166 
167  printf("PM: Configured time: ");
168  rtcc_print(RTCC_PRINT_DATE_DEC);
169 
170  /* Configure the RTCC to trigger an alarm tick */
171  printf("\nPM: Setting an alarm to tick in %u seconds\n", TEST_ALARM_SECOND);
172  if(rtcc_date_increment_seconds(simple_td, TEST_ALARM_SECOND) == AB08_ERROR) {
173  printf("PM: could not increment the next alarm date\n");
174  TEST_LEDS_FAIL;
175  }
176 
177  /* Set the timeout to zero to avoid the PIC being awake while waiting for the
178  * RTCC system to kick-in
179  */
180  if(pm_set_timeout(0x00) != PM_SUCCESS) {
181  printf("PM: couldn't clear the shutdown period\n");
182  TEST_LEDS_FAIL;
183  }
184 
185  /* We use the RTCC_REPEAT_DAY as we want the RTCC to match the given date */
186  if(rtcc_set_alarm_time_date(simple_td, RTCC_ALARM_ON, RTCC_REPEAT_DAY,
187  RTCC_TRIGGER_INT2) == AB08_ERROR) {
188  printf("PM: couldn't set the alarm\n");
189  TEST_LEDS_FAIL;
190  }
191 
192  printf("PM: Alarm set to match: ");
193  rtcc_print(RTCC_PRINT_ALARM_DEC);
194 
195  leds_off(LEDS_ALL);
196  leds_on(LEDS_BLUE);
197 
198  etimer_set(&et, CLOCK_SECOND * 3);
199  PROCESS_WAIT_EVENT_UNTIL(etimer_expired(&et));
200 
201  if(pm_shutdown_now(PM_HARD_SLEEP_CONFIG) == PM_SUCCESS) {
202  printf("PM: good night!\n");
203  } else {
204  printf("PM: error shutting down the system!\n");
205  TEST_LEDS_FAIL;
206  }
207 
208  PROCESS_END();
209 }
