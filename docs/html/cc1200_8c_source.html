555 PROCESS(cc1200_process, "CC1200 driver");
556 /*---------------------------------------------------------------------------*/
557 PROCESS_THREAD(cc1200_process, ev, data)
558 {
559 
560  PROCESS_POLLHANDLER(pollhandler());
561 
562  PROCESS_BEGIN();
563 
564 #if CC1200_USE_RX_WATCHDOG
565  while(1) {
566 
567  if((rf_flags & (RF_ON | RF_TX_ACTIVE)) == RF_ON) {
568 
569  PROCESS_WAIT_EVENT_UNTIL(etimer_expired(&et));
570  etimer_reset(&et);
571 
572  /*
573  * We are on and not in TX. As every function of this driver
574  * assures that we are in RX mode
575  * (using BUSYWAIT_UNTIL_STATE(STATE_RX, ...) construct) in
576  * either rx_rx(), idle_calibrate_rx() or transmit(),
577  * something probably went wrong in the rx interrupt handler
578  * if we are not in RX at this point.
579  */
580 
581  if(cc1200_arch_gpio0_read_pin() == 0) {
582 
583  /*
584  * GPIO de-asserts as soon as we leave RX for what reason ever. No
585  * reason to check RX as long as it is asserted (we are receiving a
586  * packet). We should never interfere with the rx interrupt if we
587  * check GPIO0 in advance...
588  */
589 
590  LOCK_SPI();
591  if(state() != STATE_RX) {
592  WARNING("RF: RX watchdog triggered!\n");
593  rx_rx();
594  }
595  RELEASE_SPI();
596 
597  }
598 
599  } else {
600  PROCESS_YIELD();
601  }
602 
603  }
604 #endif /* #if CC1200_USE_RX_WATCHDOG */
605 
606  PROCESS_YIELD_UNTIL(ev == PROCESS_EVENT_EXIT);
607 
608  PROCESS_END();
609 
610 }

612 /* Handle tasks left over from rx interrupt or because SPI was locked */
613 static void
614 pollhandler(void)
615 {
616 
617  if((rf_flags & (RF_ON + RF_POLL_RX_INTERRUPT)) ==
618  (RF_ON + RF_POLL_RX_INTERRUPT)) {
619  cc1200_rx_interrupt();
620  }
621 
622  if(rf_flags & RF_UPDATE_CHANNEL) {
623  /* We couldn't set the channel because we were busy. Try again now. */
624  set_channel(new_rf_channel);
625  }
626 
627  if(rx_pkt_len > 0) {
628 
629  int len;
630

730 /*---------------------------------------------------------------------------*/
731 /* Send the packet that has previously been prepared. */
732 static int
733 transmit(unsigned short transmit_len)
734 {
735 
736  uint8_t was_off = 0;
737  int ret = RADIO_TX_OK;
738 
739  INFO("RF: Transmit (%d)\n", transmit_len);
740 
741  if(transmit_len != tx_pkt_len) {
742  ERROR("RF: TX length mismatch!\n");
743  return RADIO_TX_ERR;
744  }
745 
746  /* TX ongoing. Inhibit channel update & ACK as soon as possible */
747  rf_flags |= RF_TX_ACTIVE;
748 
749  if(!(rf_flags & RF_ON)) {
750  /* Radio is off - turn it on */
751  was_off = 1;
752  on();
753  /* Radio is in RX now (and calibrated...) */
754  }
755 
756  if(tx_mode_value & RADIO_TX_MODE_SEND_ON_CCA) {
757  /* Perform clear channel assessment */
758  if(!channel_clear()) {
759  /* Channel occupied */
760  RIMESTATS_ADD(contentiondrop);
761  if(was_off) {
762  off();
763  }
764  rf_flags &= ~RF_TX_ACTIVE;
765  return RADIO_TX_COLLISION;
766  }
767  }
768 
769  /*
770  * Lock SPI here because "on()" and "channel_clear()"
771  * won't work if SPI is locked!
772  */
773  LOCK_SPI();
774 
775  /*
776  * Make sure we start from a sane state. idle() also disables
777  * the GPIO interrupt(s).
778  */
779  idle();
780 
781  /* Update output power */
782  if(new_txpower != txpower) {
783  update_txpower(new_txpower);
784  }
785 
786 #if !CC1200_AUTOCAL
787  /* Perform manual calibration unless just turned on */
788  if(!was_off) {
789  calibrate();
790  }
791 #endif
792 
793  RIMESTATS_ADD(lltx);
794 
795  /* Send data using TX FIFO */
796  if(idle_tx_rx((const uint8_t *)tx_pkt, tx_pkt_len) == RADIO_TX_OK) {
797 
798  /*
799  * TXOFF_MODE is set to RX,
800  * let's wait until we are in RX and turn on the GPIO IRQs
801  * again as they were turned off in idle()
802  */
803 
804  BUSYWAIT_UNTIL_STATE(STATE_RX,
805  CC1200_RF_CFG.tx_rx_turnaround);
806 
807  ENABLE_GPIO_INTERRUPTS();
808 
809  } else {
810 
811  /*
812  * Something went wrong during TX, idle_tx_rx() returns in IDLE
813  * state in this case.
814  * Turn on RX again unless we turn off anyway
815  */
816 
817  ret = RADIO_TX_ERR;
818  if(!was_off) {
819 #ifdef RF_FORCE_CALIBRATION
820  rf_flags |= RF_FORCE_CALIBRATION;
821 #endif
822  idle_calibrate_rx();
823  }
824  }
825 
826  /* Release SPI here because "off()" won't work if SPI is locked! */
827  RELEASE_SPI();
828 
829  if(was_off) {
830  off();
831  }
832 
833  /* TX completed */
834  rf_flags &= ~RF_TX_ACTIVE;
835 
836  return ret;
837 
838 }

858 /* Read a received packet into a buffer. */
859 static int
860 read(void *buf, unsigned short buf_len)
861 {
862 
863  int len = 0;
864 
865  if(rx_pkt_len > 0) {
866 
867  int8_t rssi = rx_pkt[rx_pkt_len - 2];
868  /* CRC is already checked */
869  uint8_t crc_lqi = rx_pkt[rx_pkt_len - 1];
870 
871  len = rx_pkt_len - APPENDIX_LEN;
872 
873  if(len > buf_len) {
874 
875  ERROR("RF: Failed to read packet (too big)!\n");
876 
877  } else {
878 
879  INFO("RF: Read (%d bytes, %d dBm)\n", len, rssi);
880 
881  memcpy((void *)buf, (const void *)rx_pkt, len);
882 
883  /* Release rx_pkt */
884  rx_pkt_len = 0;

891  RIMESTATS_ADD(llrx);
892  }
893 
894  }

899 /*---------------------------------------------------------------------------*/
900 /*
901  * Perform a Clear-Channel Assessment (CCA) to find out if there is a
902  * packet in the air or not.
903  */
904 static int
905 channel_clear(void)
906 {
907 
908  uint8_t cca, was_off = 0;
909 
910  if(SPI_IS_LOCKED()) {
911  /* Probably locked in rx interrupt. Return "channel occupied" */
912  return 0;
913  }
914 
915  if(!(rf_flags & RF_ON)) {
916  /* We are off */
917  was_off = 1;
918  on();
919  }
920 
921  LOCK_SPI();
922 
923  RF_ASSERT(state() == STATE_RX);
924 
925  /*
926  * At this point we should be in RX. If GPIO0 is set, we are currently
927  * receiving a packet, no need to check the RSSI. Or is there any situation
928  * when we want access the channel even if we are currently receiving a
929  * packet???
930  */
931 
932  if(cc1200_arch_gpio0_read_pin() == 1) {
933  /* Channel occupied */
934  INFO("RF: CCA (0)\n");
935  cca = 0;
936  } else {
937 
938  uint8_t rssi0;
939 
940  /* Update CCA threshold */
941  if(new_cca_threshold != cca_threshold) {
942  update_cca_threshold(new_cca_threshold);
943  }
944 
945  /* Wait for CARRIER_SENSE_VALID signal */
946  BUSYWAIT_UNTIL(((rssi0 = single_read(CC1200_RSSI0))
947  & CC1200_CARRIER_SENSE_VALID),
948  RTIMER_SECOND / 100);
949  RF_ASSERT(rssi0 & CC1200_CARRIER_SENSE_VALID);
950 
951  if(rssi0 & CC1200_CARRIER_SENSE) {
952  /* Channel occupied */
953  INFO("RF: CCA (0)\n");
954  cca = 0;
955  } else {
956  /* Channel clear */
957  INFO("RF: CCA (1)\n");
958  cca = 1;
959  }
960 
961  }
962 
963  RELEASE_SPI();
964 
965  if(was_off) {
966  off();
967  }
968 
969  return cca;
970 
971 }

1771 /*---------------------------------------------------------------------------*/
1772 /* Fill TX FIFO, start TX and wait for TX to complete (blocking!). */
1773 static int
1774 idle_tx_rx(const uint8_t *payload, uint16_t payload_len)
1775 {
1776 
1777 #if (CC1200_MAX_PAYLOAD_LEN > (CC1200_FIFO_SIZE - PHR_LEN))
1778  uint16_t bytes_left_to_write;
1779  uint8_t to_write;
1780  const uint8_t *p;
1781 #endif
1782 
1783 #if CC1200_802154G
1784  /* Prepare PHR for 802.15.4g frames */
1785  struct {
1786  uint8_t phra;
1787  uint8_t phrb;
1788  } phr;
1789 #if CC1200_802154G_CRC16
1790  payload_len += 2;
1791 #else
1792  payload_len += 4;
1793 #endif
1794  /* Frame length */
1795  phr.phrb = (uint8_t)(payload_len & 0x00FF);
1796  phr.phra = (uint8_t)((payload_len >> 8) & 0x0007);
1797 #if CC1200_802154G_WHITENING
1798  /* Enable Whitening */
1799  phr.phra |= (1 << 3);
1800 #endif /* #if CC1200_802154G_WHITENING */
1801 #if CC1200_802154G_CRC16
1802  /* FCS type 1, 2 Byte CRC */
1803  phr.phra |= (1 << 4);
1804 #endif /* #if CC1200_802154G_CRC16 */
1805 #endif /* #if CC1200_802154G */
1806 
1807  /* Prepare for RX */
1808  rf_flags &= ~RF_RX_PROCESSING_PKT;
1809  strobe(CC1200_SFRX);
1810 
1811  /* Flush TX FIFO */
1812  strobe(CC1200_SFTX);
1813 
1814 #if USE_SFSTXON
1815  /*
1816  * Enable synthesizer. Saves us a few s especially if it takes
1817  * long enough to fill the FIFO. This strobe must not be
1818  * send before SFTX!
1819  */
1820  strobe(CC1200_SFSTXON);
1821 #endif
1822 
1823  /* Configure GPIO0 to detect TX state */
1824  single_write(CC1200_IOCFG0, CC1200_IOCFG_MARC_2PIN_STATUS_0);
1825 
1826 #if (CC1200_MAX_PAYLOAD_LEN > (CC1200_FIFO_SIZE - PHR_LEN))
1827  /*
1828  * We already checked that GPIO2 is used if
1829  * CC1200_MAX_PAYLOAD_LEN > 127 / 126 in the header of this file
1830  */
1831  single_write(CC1200_IOCFG2, CC1200_IOCFG_TXFIFO_THR);
1832 #endif
1833 
1834 #if CC1200_802154G
1835  /* Write PHR */
1836  burst_write(CC1200_TXFIFO, (uint8_t *)&phr, PHR_LEN);
1837 #else
1838  /* Write length byte */
1839  burst_write(CC1200_TXFIFO, (uint8_t *)&payload_len, PHR_LEN);
1840 #endif /* #if CC1200_802154G */
1841 
1842  /*
1843  * Fill FIFO with data. If SPI is slow it might make sense
1844  * to divide this process into several chunks.
1845  * The best solution would be to perform TX FIFO refill
1846  * using an interrupt, but we are blocking here (= in TX) anyway...
1847  */
1848 
1849 #if (CC1200_MAX_PAYLOAD_LEN > (CC1200_FIFO_SIZE - PHR_LEN))
1850  to_write = MIN(payload_len, (CC1200_FIFO_SIZE - PHR_LEN));
1851  burst_write(CC1200_TXFIFO, payload, to_write);
1852  bytes_left_to_write = payload_len - to_write;
1853  p = payload + to_write;
1854 #else
1855  burst_write(CC1200_TXFIFO, payload, payload_len);
1856 #endif
1857 
1858 #if USE_SFSTXON
1859  /* Wait for synthesizer to be ready */
1860  BUSYWAIT_UNTIL_STATE(STATE_FSTXON, RTIMER_SECOND / 100);
1861 #endif
1862 
1863  /* Start TX */
1864  strobe(CC1200_STX);
1865 
1866  /* Wait for TX to start. */
1867  BUSYWAIT_UNTIL((cc1200_arch_gpio0_read_pin() == 1), RTIMER_SECOND / 100);
1868 
1869  /* Turned off at the latest in idle() */
1870  TX_LEDS_ON();
1871 
1872  /* Turned off at the latest in idle() */
1873  ENERGEST_ON(ENERGEST_TYPE_TRANSMIT);
1874 
1875  if((cc1200_arch_gpio0_read_pin() == 0) &&
1876  (single_read(CC1200_NUM_TXBYTES) != 0)) {
1877 
1878  /*
1879  * TX didn't start in time. We also check NUM_TXBYES
1880  * in case we missed the rising edge of the GPIO signal
1881  */
1882 
1883  ERROR("RF: TX doesn't start!\n");
1884 #if (CC1200_MAX_PAYLOAD_LEN > (CC1200_FIFO_SIZE - PHR_LEN))
1885  single_write(CC1200_IOCFG2, GPIO2_IOCFG);
1886 #endif
1887  idle();
1888 
1889  /* Re-configure GPIO0 */
1890  single_write(CC1200_IOCFG0, GPIO0_IOCFG);
1891 
1892  return RADIO_TX_ERR;
1893 
1894  }
1895 
1896 #if (CC1200_MAX_PAYLOAD_LEN > (CC1200_FIFO_SIZE - PHR_LEN))
1897  if(bytes_left_to_write != 0) {
1898  rtimer_clock_t t0;
1899  uint8_t s;
1900  t0 = RTIMER_NOW();
1901  do {
1902  if((bytes_left_to_write != 0) &&
1903  (cc1200_arch_gpio2_read_pin() == 0)) {
1904  /* TX TIFO is drained below FIFO_THRESHOLD. Re-fill... */
1905  to_write = MIN(bytes_left_to_write, FIFO_THRESHOLD);
1906  burst_write(CC1200_TXFIFO, p, to_write);
1907  bytes_left_to_write -= to_write;
1908  p += to_write;
1909  t0 += CC1200_RF_CFG.tx_pkt_lifetime;
1910  }
1911  } while((cc1200_arch_gpio0_read_pin() == 1) &&
1912  RTIMER_CLOCK_LT(RTIMER_NOW(), t0 + CC1200_RF_CFG.tx_pkt_lifetime));
1913 
1914  /*
1915  * At this point we either left TX or a timeout occurred. If all went
1916  * well, we are in RX (or at least settling) now.
1917  * If we didn't manage to refill the TX FIFO, an underflow might
1918  * have occur-ed - the radio might be still in TX here!
1919  */
1920 
1921  s = state();
1922  if((s != STATE_RX) && (s != STATE_SETTLING)) {
1923 
1924  /*
1925  * Something bad happened. Wait for radio to enter a
1926  * stable state (in case of an error we are in TX here)
1927  */
1928 
1929  INFO("RF: TX failure!\n");
1930  BUSYWAIT_UNTIL((state() != STATE_TX), RTIMER_SECOND / 100);
1931  /* Re-configure GPIO2 */
1932  single_write(CC1200_IOCFG2, GPIO2_IOCFG);
1933  idle();
1934 
1935  /* Re-configure GPIO0 */
1936  single_write(CC1200_IOCFG0, GPIO0_IOCFG);
1937 
1938  return RADIO_TX_ERR;
1939 
1940  }
1941 
1942  } else {
1943  /* Wait for TX to complete */
1944  BUSYWAIT_UNTIL((cc1200_arch_gpio0_read_pin() == 0),
1945  CC1200_RF_CFG.tx_pkt_lifetime);
1946  }
1947 #else
1948  /* Wait for TX to complete */
1949  BUSYWAIT_UNTIL((cc1200_arch_gpio0_read_pin() == 0),
1950  CC1200_RF_CFG.tx_pkt_lifetime);
1951 #endif
1952 
1953  if(cc1200_arch_gpio0_read_pin() == 1) {
1954  /* TX takes to long - abort */
1955  ERROR("RF: TX takes to long!\n");
1956 #if (CC1200_MAX_PAYLOAD_LEN > (CC1200_FIFO_SIZE - PHR_LEN))
1957  /* Re-configure GPIO2 */
1958  single_write(CC1200_IOCFG2, GPIO2_IOCFG);
1959 #endif
1960  idle();
1961 
1962  /* Re-configure GPIO0 */
1963  single_write(CC1200_IOCFG0, GPIO0_IOCFG);
1964 
1965  return RADIO_TX_ERR;
1966 
1967  }
1968 
1969 #if (CC1200_MAX_PAYLOAD_LEN > (CC1200_FIFO_SIZE - PHR_LEN))
1970  /* Re-configure GPIO2 */
1971  single_write(CC1200_IOCFG2, GPIO2_IOCFG);
1972 #endif
1973 
1974  /* Re-configure GPIO0 */
1975  single_write(CC1200_IOCFG0, GPIO0_IOCFG);
1976 
1977  TX_LEDS_OFF();
1978 
1979  ENERGEST_OFF(ENERGEST_TYPE_TRANSMIT);
1980  ENERGEST_ON(ENERGEST_TYPE_LISTEN);
1981 
1982  return RADIO_TX_OK;
1983 
1984 }
