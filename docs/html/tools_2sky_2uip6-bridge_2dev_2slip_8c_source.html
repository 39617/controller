72 /*
73  * Variables begin and end manage the buffer space in a cyclic
74  * fashion. The first used byte is at begin and end is one byte past
75  * the last. I.e. [begin, end) is the actively used space.
76  *
77  * If begin != pkt_end we have a packet at [begin, pkt_end),
78  * furthermore, if state == STATE_TWOPACKETS we have one more packet at
79  * [pkt_end, end). If more bytes arrive in state STATE_TWOPACKETS
80  * they are discarded.
81  */
82

163 /* Upper half does the polling. */
164 static uint16_t
165 slip_poll_handler(uint8_t *outbuf, uint16_t blen)
166 {
167  /* This is a hack and won't work across buffer edge! */
168  if(rxbuf[begin] == 'C') {
169  int i;
170  if(begin < end && (end - begin) >= 6
171  && memcmp(&rxbuf[begin], "CLIENT", 6) == 0) {
172  state = STATE_TWOPACKETS; /* Interrupts do nothing. */
173  memset(&rxbuf[begin], 0x0, 6);
174 
175  rxbuf_init();
176 
177  for(i = 0; i < 13; i++) {
178  slip_arch_writeb("CLIENTSERVER\300"[i]);
179  }
180  return 0;
181  }
182  } else if(rxbuf[begin] == '?') {
183  int i, j;
184  char* hexchar = "0123456789abcdef";
185  if(begin < end && (end - begin) >= 2
186  && rxbuf[begin + 1] == 'M') {
187  state = STATE_TWOPACKETS; /* Interrupts do nothing. */
188  rxbuf[begin] = 0;
189  rxbuf[begin + 1] = 0;
190 
191  rxbuf_init();
192 
193  /* this is just a test so far... just to see if it works */
194  slip_arch_writeb('!');
195  slip_arch_writeb('M');
196  for(j = 0; j < 8; j++) {
197  slip_arch_writeb(hexchar[ds2411_id[j] >> 4]);
198  slip_arch_writeb(hexchar[ds2411_id[j] & 15]);
199  }
200  slip_arch_writeb(SLIP_END);
201  return 0;
202  }
203  }
204 
205  /*
206  * Interrupt can not change begin but may change pkt_end.
207  * If pkt_end != begin it will not change again.
208  */
209  if(begin != pkt_end) {
210  uint16_t len;
211 
212  if(begin < pkt_end) {
213  len = pkt_end - begin;
214  if(len > blen) {
215  len = 0;
216  } else {
217  memcpy(outbuf, &rxbuf[begin], len);
218  }
219  } else {
220  len = (RX_BUFSIZE - begin) + (pkt_end - 0);
221  if(len > blen) {
222  len = 0;
223  } else {
224  unsigned i;
225  for(i = begin; i < RX_BUFSIZE; i++) {
226  *outbuf++ = rxbuf[i];
227  }
228  for(i = 0; i < pkt_end; i++) {
229  *outbuf++ = rxbuf[i];
230  }
231  }
232  }
233 
234  /* Remove data from buffer together with the copied packet. */
235  begin = pkt_end;
236  if(state == STATE_TWOPACKETS) {
237  pkt_end = end;
238  state = STATE_OK; /* Assume no bytes where lost! */
239 
240  /* One more packet is buffered, need to be polled again! */
241  process_poll(&slip_process);
242  }
243  return len;
244  }
245 
246  return 0;
247 }

249 PROCESS_THREAD(slip_process, ev, data)
250 {
251  PROCESS_BEGIN();
252 
253  rxbuf_init();
254 
255  while(1) {
256  PROCESS_YIELD_UNTIL(ev == PROCESS_EVENT_POLL);
257 
258  slip_active = 1;
259 
260  /* Move packet from rxbuf to buffer provided by uIP. */
261  uip_len = slip_poll_handler(&uip_buf[UIP_LLH_LEN],
262  UIP_BUFSIZE - UIP_LLH_LEN);
263 #if !NETSTACK_CONF_WITH_IPV6
264  if(uip_len == 4 && strncmp((char*)&uip_buf[UIP_LLH_LEN], "?IPA", 4) == 0) {
265  char buf[8];
266  memcpy(&buf[0], "=IPA", 4);
267  memcpy(&buf[4], &uip_hostaddr, 4);
268  if(input_callback) {
269  input_callback();
270  }
271  slip_write(buf, 8);
272  } else if(uip_len > 0
273  && uip_len == (((uint16_t)(BUF->len[0]) << 8) + BUF->len[1])
274  && uip_ipchksum() == 0xffff) {
275 #define IP_DF 0x40
276  if(BUF->ipid[0] == 0 && BUF->ipid[1] == 0 && BUF->ipoffset[0] & IP_DF) {
277  static uint16_t ip_id;
278  uint16_t nid = ip_id++;
279  BUF->ipid[0] = nid >> 8;
280  BUF->ipid[1] = nid;
281  nid = uip_htons(nid);
282  nid = ~nid; /* negate */
283  BUF->ipchksum += nid; /* add */
284  if(BUF->ipchksum < nid) { /* 1-complement overflow? */
285  BUF->ipchksum++;
286  }
287  }
288  if(tcpip_input_callback) {
289  tcpip_input_callback();
290  } else {
291  tcpip_input();
292  }
293  } else {
294  uip_clear_buf();
295  SLIP_STATISTICS(slip_ip_drop++);
296  }
297 #else /* NETSTACK_CONF_WITH_IPV6 */
298  if(uip_len > 0) {
299  if(tcpip_input_callback) {
300  tcpip_input_callback();
301  } else {
302  tcpip_input();
303  }
304  }
305 #endif /* NETSTACK_CONF_WITH_IPV6 */
306  }
307 
308  PROCESS_END();
309 }

311 int
312 slip_input_byte(unsigned char c)
313 {
314  switch(state) {
315  case STATE_RUBBISH:
316  if(c == SLIP_END) {
317  state = STATE_OK;
318  }
319  return 0;
320 
321  case STATE_TWOPACKETS: /* Two packets are already buffered! */
322  return 0;
323 
324  case STATE_ESC:
325  if(c == SLIP_ESC_END) {
326  c = SLIP_END;
327  } else if(c == SLIP_ESC_ESC) {
328  c = SLIP_ESC;
329  } else {
330  state = STATE_RUBBISH;
331  SLIP_STATISTICS(slip_rubbish++);
332  end = pkt_end; /* remove rubbish */
333  return 0;
334  }
335  state = STATE_OK;
336  break;
337 
338  case STATE_OK:
339  if(c == SLIP_ESC) {
340  state = STATE_ESC;
341  return 0;
342  } else if(c == SLIP_END) {
343  /*
344  * We have a new packet, possibly of zero length.
345  *
346  * There may already be one packet buffered.
347  */
348  if(end != pkt_end) { /* Non zero length. */
349  if(begin == pkt_end) { /* None buffered. */
350  pkt_end = end;
351  } else {
352  state = STATE_TWOPACKETS;
353  SLIP_STATISTICS(slip_twopackets++);
354  }
355  process_poll(&slip_process);
356  return 1;
357  }
358  return 0;
359  }
360  break;
361  }
362 
363  /* add_char: */
364  {
365  unsigned next;
366  next = end + 1;
367  if(next == RX_BUFSIZE) {
368  next = 0;
369  }
370  if(next == begin) { /* rxbuf is full */
371  state = STATE_RUBBISH;
372  SLIP_STATISTICS(slip_overflow++);
373  end = pkt_end; /* remove rubbish */
374  return 0;
375  }
376  rxbuf[end] = c;
377  end = next;
378  }
379 
380  /* There could be a separate poll routine for this. */
381  if(c == 'T' && rxbuf[begin] == 'C') {
382  process_poll(&slip_process);
383  return 1;
384  }
385 
386  return 0;
387 }
