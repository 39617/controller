298 static void
299 at_cmd_adc_callback(struct at_cmd *cmd, uint8_t len, char *data)
300 {
301  /* Format: AT&ADC=N where N is 4-7, it can be "*" to read all */
302  uint8_t i, pin;
303  uint16_t res[4];
304  char read_result[24];
305 
306  if(strncmp(&data[7], "*", 1) == 0) {
307  pin = 8;
308  } else {
309  pin = atoi(&data[7]);
310  }
311 
312  if((pin < 4) || (pin > 8)) {
313  AT_RESPONSE(AT_DEFAULT_RESPONSE_ERROR);
314  return;
315  }
316 
317  if(pin < 8) {
318  config_gpio(GPIO_A_NUM, pin, HWTEST_GPIO_INPUT);
319  ioc_set_over(GPIO_A_NUM, pin, IOC_OVERRIDE_ANA);
320  res[pin - 4] = adc_get((SOC_ADC_ADCCON_CH_AIN0 + pin),
321  SOC_ADC_ADCCON_REF_AVDD5,
322  SOC_ADC_ADCCON_DIV_512);
323  res[pin - 4] = res[pin - 4] / 10;
324  PRINTF("ADC%u: %04d\n", pin, res[pin - 4]);
325  snprintf(read_result, 5, "%04d", res[pin - 4]);
326  } else {
327  for(i = 4; i < 8; i++) {
328  config_gpio(GPIO_A_NUM, i, HWTEST_GPIO_INPUT);
329  ioc_set_over(GPIO_A_NUM, i, IOC_OVERRIDE_ANA);
330  res[i - 4] = adc_get((SOC_ADC_ADCCON_CH_AIN0 + i),
331  SOC_ADC_ADCCON_REF_AVDD5,
332  SOC_ADC_ADCCON_DIV_512);
333  res[i - 4] = res[i - 4] / 10;
334  }
335  snprintf(read_result, 24, "%04d %04d %04d %04d", res[0], res[1],
336  res[2], res[3]);
337  }
338 
339  AT_RESPONSE(read_result);
340  AT_RESPONSE(AT_DEFAULT_RESPONSE_OK);
341 }

343 static void
344 at_cmd_read_callback(struct at_cmd *cmd, uint8_t len, char *data)
345 {
346  /* Format: AT&READ=PN where P(ort)N(number), N can be "*" to read all */
347  uint8_t port, pin;
348  char read_result[5];
349 
350  if(strncmp(&data[9], "*", 1) == 0) {
351  pin = 0xFF;
352  } else {
353  pin = atoi(&data[9]);
354  }
355 
356  if((pin < 0) || (pin > 7)) {
357  if(pin != 0xFF) {
358  AT_RESPONSE(AT_DEFAULT_RESPONSE_ERROR);
359  return;
360  }
361  }
362 
363  if(pin < 8) {
364  pin = GPIO_PIN_MASK(pin);
365  }
366 
367  /* Exclude PA0-PA3 */
368  if(strncmp(&data[8], "A", 1) == 0) {
369  port = GPIO_A_NUM;
370  if(pin < 0x1F) {
371  AT_RESPONSE(AT_DEFAULT_RESPONSE_ERROR);
372  return;
373  } else {
374  if(pin == 0xFF) {
375  pin = 0xF0;
376  }
377  }
378  } else if(strncmp(&data[8], "B", 1) == 0) {
379  port = GPIO_B_NUM;
380  } else if(strncmp(&data[8], "C", 1) == 0) {
381  port = GPIO_C_NUM;
382  } else if(strncmp(&data[8], "D", 1) == 0) {
383  port = GPIO_D_NUM;
384  } else {
385  AT_RESPONSE(AT_DEFAULT_RESPONSE_ERROR);
386  return;
387  }
388 
389  config_gpio(port, pin, HWTEST_GPIO_INPUT);
390  snprintf(read_result, 5, "0x%02X",
391  (uint16_t)GPIO_READ_PIN(GPIO_PORT_TO_BASE(port), pin));
392  AT_RESPONSE(read_result);
393  AT_RESPONSE(AT_DEFAULT_RESPONSE_OK);
394 }

396 static void
397 at_cmd_sha256_callback(struct at_cmd *cmd, uint8_t len, char *data)
398 {
399  /* Format: AT&SHA256=s, where s is a string up to 64 bytes */
400  uint8_t i;
401  char tmp[4], sha256[32], sha256_res[64];
402  static sha256_state_t state;
403 
404  crypto_init();
405  if(sha256_init(&state) != CRYPTO_SUCCESS) {
406  AT_RESPONSE(AT_DEFAULT_RESPONSE_ERROR);
407  return;
408  }
409 
410  if(sha256_process(&state, &data[10],
411  len - (cmd->cmd_hdr_len)) != CRYPTO_SUCCESS) {
412  AT_RESPONSE(AT_DEFAULT_RESPONSE_ERROR);
413  return;
414  }
415 
416  if(sha256_done(&state, sha256) != CRYPTO_SUCCESS) {
417  AT_RESPONSE(AT_DEFAULT_RESPONSE_ERROR);
418  return;
419  }
420 
421  crypto_disable();
422 
423  PRINTF("Input: %s:\n", &data[10]);
424  snprintf(tmp, 3, "%02X", sha256[0]);
425  strncpy(sha256_res, tmp, 3);
426  for(i = 1; i < 32; i++) {
427  PRINTF("0x%02X ", sha256[i]);
428  snprintf(tmp, 3, "%02X", sha256[i]);
429  strcat(sha256_res, tmp);
430  }
431  PRINTF("\nSHA256: %s\n", sha256_res);
432  AT_RESPONSE(sha256_res);
433  AT_RESPONSE(AT_DEFAULT_RESPONSE_OK);
434 }
