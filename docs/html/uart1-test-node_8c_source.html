165 /*---------------------------------------------------------------------------*/
166 PROCESS_THREAD(rx_data_process, ev, data)
167 {
168  PROCESS_BEGIN();
169 
170  /* Process is polled whenever data is available from uart isr */
171  uint8_t c;
172 
173  while(1) {
174  PROCESS_YIELD_UNTIL(ev == PROCESS_EVENT_POLL);
175  /* Read RX ringbuffer. ASCII chars Output when LF is seen.
176  If overflowed, strings are skipped */
177  do {
178  if (get_ringbuf(&c) == -1) {
179  break; /* No more rx char's in ringbuffer */
180  } else {
181  if (rx_buf_index == RX_BUFFER_SIZE) { /* Skip current content if buffer full */
182  rx_buf_index = 0;
183  }
184  rx_buf[rx_buf_index++] = c;
185  if ((c == '\n')||(c == '\r')) {
186  rx_buf[rx_buf_index] = '\0';
187  printf("RX on UART1: %s", rx_buf);
188  /* Signal event to coap clients.
189  Demo assumes data is consumed before new data comes in */
190  event_coap_rx_uart1_handler();
191  rx_buf_index = 0;
192  }
193  }
194  } while (1);
195  }
196  PROCESS_END();
197 }

225 /* Simple ringbuffer
226  if tail==head, no data has been written yet on that position. So, empty buffer
227  is also initial state */
228 static int
229 get_ringbuf(uint8_t *c)
230 {
231  int return_val = 0;
232 
233  uart1_disable_interrupts();
234  if (head_index != ((tail_index + 1)&RINGBUF_MAX_INDEX)) {
235  tail_index = ((tail_index + 1)& RINGBUF_MAX_INDEX);
236  *c = ringbuf[tail_index];
237  } else {
238  return_val = -1;
239  }
240  uart1_enable_interrupts();
241  return return_val;
242 }
243 
244 static int
245 put_ringbuf(uint8_t c)
246 {
247  int return_val = 0;
248 
249  uart1_disable_interrupts();
250  if (head_index != tail_index) {
251  ringbuf[head_index] = c;
252  head_index = ((head_index+1)&RINGBUF_MAX_INDEX);
253  } else {
254  return_val = -1;
255  }
256  uart1_enable_interrupts();
257  return return_val;
258 }
