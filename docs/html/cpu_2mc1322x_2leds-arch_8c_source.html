45 void leds_arch_init(void)
46 {
47  /* set gpio func_sel to gpio (3) */
48  GPIO->FUNC_SEL.LED_ARCH_RED = 3;
49  GPIO->FUNC_SEL.LED_ARCH_GREEN = 3;
50  GPIO->FUNC_SEL.LED_ARCH_BLUE = 3;
51 
52  /* set led gpios to output */
53  GPIO->PAD_DIR.LED_ARCH_RED = 1;
54  GPIO->PAD_DIR.LED_ARCH_GREEN = 1;
55  GPIO->PAD_DIR.LED_ARCH_BLUE = 1;
56 }

67 void leds_arch_set(unsigned char leds)
68 {
69  if(leds & LEDS_RED) { GPIO->DATA.LED_ARCH_RED = 1; } else { gpio_reset(LED_ARCH_RED); }
70  if(leds & LEDS_GREEN) { gpio_set(LED_ARCH_GREEN); } else { gpio_reset(LED_ARCH_GREEN); }
71  if(leds & LEDS_BLUE) { gpio_set(LED_ARCH_BLUE); } else { gpio_reset(LED_ARCH_BLUE); }
72 }
