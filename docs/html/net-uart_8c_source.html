100 /*
101  * \brief Attempts to convert a string representation of an IPv6 address to a
102  * numeric one.
103  * \param buf The buffer with the string to be converted.
104  * \return ADDRESS_CONVERSION_OK or ADDRESS_CONVERSION_ERROR
105  *
106  * ToDo: Add support for NAT64 conversion in case the incoming address is a v4
107  * This is now supported in the current master, so when we pull it in this will
108  * be very straightforward.
109  */
110 static int
111 set_new_ip_address(char *buf)
112 {
113  /*
114  * uiplib_ip6addrconv will immediately start writing into the supplied buffer
115  * even if it subsequently fails. Thus, pass an intermediate buffer
116  */
117  uip_ip6addr_t tmp_addr;
118 
119  int rv = uiplib_ip6addrconv(buf, &tmp_addr);
120 
121  if(rv == ADDRESS_CONVERSION_OK) {
122  /* Conversion OK, copy to our main buffer */
123  memcpy(&remote_addr, &tmp_addr, sizeof(remote_addr));
124 
125  PRINTF("Updated remote address ");
126  PRINT6ADDR(&remote_addr);
127  PRINTF("\n");
128  }
129 
130  return rv;
131 }

252 PROCESS_THREAD(net_uart_process, ev, data)
253 {
254  PROCESS_BEGIN();
255 
256  printf("CC26XX Net UART Process\n");
257 
258  set_config_defaults();
259 
260  udp_conn = udp_new(NULL, UIP_HTONS(0), NULL);
261  udp_bind(udp_conn, UIP_HTONS(REMOTE_PORT));
262 
263  if(udp_conn == NULL) {
264  printf("No UDP connection available, exiting the process!\n");
265  PROCESS_EXIT();
266  }
267 
268  httpd_simple_register_post_handler(&remote_port_handler);
269  httpd_simple_register_post_handler(&remote_ipv6_handler);
270  httpd_simple_register_post_handler(&on_off_handler);
271 
272  while(1) {
273 
274  PROCESS_YIELD();
275 
276  if(ev == serial_line_event_message) {
277  /*
278  * If the message contains a new IP address, save it and go back to
279  * waiting.
280  */
281  if(set_new_ip_address((char *)data) == ADDRESS_CONVERSION_ERROR) {
282  /* Not an IP address in the message. Send to current destination */
283  memset(buffer, 0, MAX_MSG_SIZE);
284 
285  /* We need to add a line feed, thus never fill the entire buffer */
286  msg_len = MIN(strlen(data), MAX_MSG_SIZE - 1);
287  memcpy(buffer, data, msg_len);
288 
289  /* Add a line feed */
290  buffer[msg_len] = 0x0A;
291  msg_len++;
292 
293  uip_udp_packet_sendto(
294  udp_conn, buffer, msg_len, &remote_addr,
295  UIP_HTONS(cc26xx_web_demo_config.net_uart.remote_port));
296  }
297  } else if(ev == tcpip_event) {
298  net_input();
299  } else if(ev == cc26xx_web_demo_config_loaded_event) {
300  /*
301  * New config. Check if it's possible to update the remote address.
302  * The port will have been updated already
303  */
304  set_new_ip_address(cc26xx_web_demo_config.net_uart.remote_address);
305 
306  if(cc26xx_web_demo_config.net_uart.enable == 1) {
307  keep_uart_on();
308  }
309  } else if(ev == cc26xx_web_demo_load_config_defaults) {
310  set_config_defaults();
311  }
312  }
313 
314  PROCESS_END();
315 }
