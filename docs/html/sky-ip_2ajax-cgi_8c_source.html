66 httpd_cgifunction
67 httpd_cgi(char *name)
68 {
69  struct httpd_cgi_call *f;
70 
71  /* Find the matching name in the table, return the function. */
72  for(f = calls; f != NULL; f = f->next) {
73  if(strncmp(f->name, name, strlen(f->name)) == 0) {
74  return f->function;
75  }
76  }
77  return nullfunction;
78 }
