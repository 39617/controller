255 int8_t mac_translateIcmpLinkLayer(lltype_t target)
256 {
257  uint16_t icmp_opt_offset = 0;
258  int16_t len = UIP_IP_BUF->len[1] | (UIP_IP_BUF->len[0] << 8);
259 
260  uint16_t iplen;
261 
262  uint8_t i;
263 
264  int16_t sizechange;
265 
266  uint8_t llbuf[16];
267 
268  //Figure out offset to start of options
269  switch(UIP_ICMP_BUF->type) {
270  case ICMP6_NS:
271  case ICMP6_NA:
272  icmp_opt_offset = 24;
273  break;
274 
275  case ICMP6_RS:
276  icmp_opt_offset = 8;
277  break;
278 
279  case ICMP6_RA:
280  icmp_opt_offset = 16;
281  break;
282 
283  case ICMP6_REDIRECT:
284  icmp_opt_offset = 40;
285  break;
286 
288  case ICMP6_DST_UNREACH:
289  case ICMP6_PACKET_TOO_BIG:
290  case ICMP6_TIME_EXCEEDED:
291  case ICMP6_PARAM_PROB:
292  case ICMP6_ECHO_REQUEST:
293  case ICMP6_ECHO_REPLY:
294  return 0;
295  break;
296 
297  default:
298  return -1;
299  }
300 
301  //Figure out length of options
302  len -= icmp_opt_offset;
303 
304  //Sanity check
305  if (len < 8) return -2;
306 
307 #ifdef DEBUG
308  uint16_t offset_bk = icmp_opt_offset;
309  print_tgaddr(offset_bk);
310 #endif /* DEBUG */
311 
312  //While we have options to do...
313  while (len >= 8){
314  //If we have one of these, we have something useful!
315  if (((UIP_ICMP_OPTS(icmp_opt_offset)->type) == UIP_ND6_OPT_SLLAO) ||
316  ((UIP_ICMP_OPTS(icmp_opt_offset)->type) == UIP_ND6_OPT_TLLAO) ) {
317 
318  /* Shrinking the buffer may thrash things, so we store the old
319  link-layer address */
320  for(i = 0; i < (UIP_ICMP_OPTS(icmp_opt_offset)->length*8 - 2); i++) {
321  llbuf[i] = UIP_ICMP_OPTS(icmp_opt_offset)->data[i];
322  }
323 
324  //Shrink/grow buffer as needed
325  if (target == ll_802154_type) {
326  //Current is 802.3, Hence current link-layer option is 6 extra bytes
327  sizechange = 8;
328  slide(UIP_ICMP_OPTS(icmp_opt_offset)->data + 6, len - 6, sizechange);
329  } else if (target == ll_8023_type) {
330  /* Current is 802.15.4, Hence current link-layer option is 14 extra
331  * bytes.
332  * (Actual LL is 8 bytes, but total Option field length is in multiples of
333  * 8 Bytes, hence 8 + 2 = 10 (2 is one from type + one from length).
334  * So, closest is 16 bytes, then 16 bytes for
335  * total Optional field length -> 2 bytes for type + length leaves 14 )
336  */
337  sizechange = -8;
338  slide(UIP_ICMP_OPTS(icmp_opt_offset)->data + 14, len - 14, sizechange);
339  } else {
340  return -3; //Uh-oh!
341  }
342 
343  //Translate addresses
344  if (target == ll_802154_type) {
345  mac_createSicslowpanLongAddr(llbuf, (uip_lladdr_t *)UIP_ICMP_OPTS(icmp_opt_offset)->data);
346  } else {
347  mac_createEthernetAddr(UIP_ICMP_OPTS(icmp_opt_offset)->data, (uip_lladdr_t *)llbuf);
348  }
349 
350  //Adjust the length
351  if (target == ll_802154_type) {
352  UIP_ICMP_OPTS(icmp_opt_offset)->length = 2;
353  } else {
354  UIP_ICMP_OPTS(icmp_opt_offset)->length = 1;
355  }
356 
357  //Adjust the IP header length, as well as uIP length
358  iplen = UIP_IP_BUF->len[1] | (UIP_IP_BUF->len[0]<<8);
359  iplen += sizechange;
360  len += sizechange;
361 
362  UIP_IP_BUF->len[1] = (uint8_t)iplen;
363  UIP_IP_BUF->len[0] = (uint8_t)(iplen >> 8);
364 
365  uip_len += sizechange;
366 
367  //We broke ICMP checksum, be sure to fix that
368  UIP_ICMP_BUF->icmpchksum = 0;
369  UIP_ICMP_BUF->icmpchksum = ~uip_icmp6chksum();
370 
371  //Finally set up next run in while loop
372  len -= 8 * UIP_ICMP_OPTS(icmp_opt_offset)->length;
373  icmp_opt_offset += 8 * UIP_ICMP_OPTS(icmp_opt_offset)->length;
374  } else {
375 
376  //Not an option we care about, ignore it
377  len -= 8 * UIP_ICMP_OPTS(icmp_opt_offset)->length;
378 
379  //This shouldn't happen!
380  if (UIP_ICMP_OPTS(icmp_opt_offset)->length == 0) {
381  PRINTF("Option in ND packet has length zero, error?\n");
382  len = 0;
383  }
384 
385  icmp_opt_offset += 8 * UIP_ICMP_OPTS(icmp_opt_offset)->length;
386 
387  } //If ICMP_OPT is one we care about
388 
389  } //while(len >= 8)
390 
391 #ifdef DEBUG
392  print_tgaddr(offset_bk);
393 #endif /* DEBUG */
394 
395  return 0;
396 
397 }

450  uint8_t index = 0;
451  uint8_t i;

453

455 
456  //Special case - if the address is our address, we just copy over what we know to be

461  return 1;

463 
464 
465  // TODO: Averificação dos BIT Translate e Multicas tem de se apagar
466  //Check if we need to do anything:
467  if ((lowpan->addr[3] == 0xff) && (lowpan->addr[4] == 0xfe) &&
468  ((lowpan->addr[0] & TRANSLATE_BIT_MASK) == 0) &&
469  ((lowpan->addr[0] & MULTICAST_BIT_MASK) == 0) &&
470  (lowpan->addr[0] & LOCAL_BIT_MASK)) {
471 
473  ethernet[0] = lowpan->addr[0];
474  ethernet[1] = lowpan->addr[1];
475  ethernet[2] = lowpan->addr[2];
476  ethernet[3] = lowpan->addr[5];
477  ethernet[4] = lowpan->addr[6];
478  ethernet[5] = lowpan->addr[7];
479 
480 
481  } else {
482 
484  for (i = 0; i < prefixCounter; i++) {
485  //Check the current prefix - if it fails, check next one
486 
487 
488  if ((lowpan->addr[0] == prefixBuffer[i][0]) &&
489  (lowpan->addr[3] == prefixBuffer[i][1]) &&
490  (lowpan->addr[4] == prefixBuffer[i][2])) {
491  break;
492  }
493 
494  }
495 
496  index = i;
497 
498  //Deal with overflow, iron-fist style
499  if (index >= PREFIX_BUFFER_SIZE) {
500  index = 0;
501  prefixCounter = PREFIX_BUFFER_SIZE;
502  } else {
503  //Are we making a new one?
504  if (index == prefixCounter) {
505  prefixCounter++;
506  }
507  }
508 
509  //Copy the prefix over, no matter if we have a new or old one
510  prefixBuffer[index][0] = lowpan->addr[0];
511  prefixBuffer[index][1] = lowpan->addr[3];
512  prefixBuffer[index][2] = lowpan->addr[4];
513 
514  //Create ethernet MAC address now
515  ethernet[1] = lowpan->addr[1];
516  ethernet[2] = lowpan->addr[2];
517  ethernet[3] = lowpan->addr[5];
518  ethernet[4] = lowpan->addr[6];
519  ethernet[5] = lowpan->addr[7];
520 
521 
522  ethernet[0] = TRANSLATE_BIT_MASK | LOCAL_BIT_MASK | (index << 3);
523  }
524 
525 #else

545 void slide(uint8_t * data, uint8_t length, int16_t slide)
546 {
547  //Sanity checks
548  if (!length) return;
549  if (!slide) return;
550 
551  uint8_t i = 0;
552 
553  while(length) {
554  length--;
555 
556  //If we are sliding up, we do from the top of the buffer down
557  if (slide > 0) {
558  *(data + length + slide) = *(data + length);
559 
560  //If we are sliding down, we do from the bottom of the buffer up
561  } else {
562  *(data + slide + i) = *(data + i);
563  }
564 
565  i++;
566  }
567 }
