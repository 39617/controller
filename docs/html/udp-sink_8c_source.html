95 static void
96 tcpip_handler(void)
97 {
98  uint8_t *appdata;
99  linkaddr_t sender;
100  uint8_t seqno;
101  uint8_t hops;
102 
103  if(uip_newdata()) {
104  appdata = (uint8_t *)uip_appdata;
105  sender.u8[0] = UIP_IP_BUF->srcipaddr.u8[15];
106  sender.u8[1] = UIP_IP_BUF->srcipaddr.u8[14];
107  seqno = *appdata;
108  hops = uip_ds6_if.cur_hop_limit - UIP_IP_BUF->ttl + 1;
109  collect_common_recv(&sender, seqno, hops,
110  appdata + 2, uip_datalen() - 2);
111  }
112 }

134 PROCESS_THREAD(udp_server_process, ev, data)
135 {
136  uip_ipaddr_t ipaddr;
137  struct uip_ds6_addr *root_if;
138 
139  PROCESS_BEGIN();
140 
141  PROCESS_PAUSE();
142 
143  SENSORS_ACTIVATE(button_sensor);
144 
145  PRINTF("UDP server started\n");
146 
147 #if UIP_CONF_ROUTER
148  uip_ip6addr(&ipaddr, UIP_DS6_DEFAULT_PREFIX, 0, 0, 0, 0, 0, 0, 1);
149  /* uip_ds6_set_addr_iid(&ipaddr, &uip_lladdr); */
150  uip_ds6_addr_add(&ipaddr, 0, ADDR_MANUAL);
151  root_if = uip_ds6_addr_lookup(&ipaddr);
152  if(root_if != NULL) {
153  rpl_dag_t *dag;
154  dag = rpl_set_root(RPL_DEFAULT_INSTANCE,(uip_ip6addr_t *)&ipaddr);
155  uip_ip6addr(&ipaddr, UIP_DS6_DEFAULT_PREFIX, 0, 0, 0, 0, 0, 0, 0);
156  rpl_set_prefix(dag, &ipaddr, 64);
157  PRINTF("created a new RPL dag\n");
158  } else {
159  PRINTF("failed to create a new RPL DAG\n");
160  }
161 #endif /* UIP_CONF_ROUTER */
162 
163  print_local_addresses();
164 
165  /* The data sink runs with a 100% duty cycle in order to ensure high
166  packet reception rates. */
167  NETSTACK_RDC.off(1);
168 
169  server_conn = udp_new(NULL, UIP_HTONS(UDP_CLIENT_PORT), NULL);
170  udp_bind(server_conn, UIP_HTONS(UDP_SERVER_PORT));
171 
172  PRINTF("Created a server connection with remote address ");
173  PRINT6ADDR(&server_conn->ripaddr);
174  PRINTF(" local/remote port %u/%u\n", UIP_HTONS(server_conn->lport),
175  UIP_HTONS(server_conn->rport));
176 
177  while(1) {
178  PROCESS_YIELD();
179  if(ev == tcpip_event) {
180  tcpip_handler();
181  } else if (ev == sensors_event && data == &button_sensor) {
182  PRINTF("Initiating global repair\n");
183  rpl_repair_root(RPL_DEFAULT_INSTANCE);
184  }
185  }
186 
187  PROCESS_END();
188 }
