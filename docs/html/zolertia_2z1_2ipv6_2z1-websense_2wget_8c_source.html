81 PROCESS_THREAD(wget_process, ev, data)
82 {
83  PROCESS_BEGIN();
84 
85  PRINTF("wget: fetching %s\n", file);
86 #if STATS
87  fetch_counter = 0;
88  fetch_started = clock_time();
89 #endif /* STATS */
90  LEDS_ON(LEDS_YELLOW);
91  if(webclient_get(server, port, file) == 0) {
92  PRINTF("wget: failed to connect\n");
93  LEDS_OFF(LEDS_YELLOW);
94  fetch_running = 0;
95  call_done(WGET_CONNECT_FAILED);
96  } else {
97  while(fetch_running) {
98  PROCESS_WAIT_EVENT();
99  if(ev == tcpip_event) {
100  webclient_appcall(data);
101  }
102  }
103  }
104 
105  PROCESS_END();
106 }

108 void
109 webclient_datahandler(char *data, uint16_t len)
110 {
111  if(len == 0) {
112 #if STATS
113  clock_time_t elapsed;
114  elapsed = clock_time() - fetch_started;
115  PRINTF("wget: recv %lu bytes during %lu sec (",
116  fetch_counter, (elapsed / CLOCK_SECOND));
117 #if CLOCK_SECOND == 128
118  PRINTF("%lu.%02lus, ", (unsigned long)elapsed >> 7,
119  (unsigned long)((elapsed & 127) * 100) / 128);
120  PRINTF("%lu byte/sec ", (fetch_counter * 128L) / elapsed);
121 #endif
122  PRINTF("%lu tick): ", (unsigned long)elapsed);
123  if(elapsed > CLOCK_SECOND) {
124  PRINTF("%lu", fetch_counter / (elapsed / CLOCK_SECOND));
125  } else {
126  PRINTF("-");
127  }
128  PRINTF(" byte/sec\n");
129 #else /* STATS */
130  PRINTF("wget: complete\n");
131 #endif /* STATS */
132 
133  fetch_running = 0;
134  call_done(WGET_OK);
135  LEDS_OFF(LEDS_RED | LEDS_YELLOW);
136  } else {
137 #if STATS
138  fetch_counter += len;
139 #endif /* STATS */
140  if(callbacks != NULL && callbacks->data != NULL) {
141  /* Assume the buffer is large enough for ending zero */
142  data[len] = '\0';
143  callbacks->data(data, len);
144  }
145  }
146 }
