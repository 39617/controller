135 /*
136  * Baud rate defines used in uart_init() to set the values of UART_IBRD and
137  * UART_FBRD in order to achieve the configured baud rates.
138  */
139 #define UART_CLOCK_RATE SYS_CTRL_SYS_CLOCK
140 #define UART_CTL_HSE_VALUE 0
141 #define UART_CTL_VALUE (UART_CTL_RXE | UART_CTL_TXE | (UART_CTL_HSE_VALUE << 5))
142 
143 /* DIV_ROUND() divides integers while avoiding a rounding error: */
144 #define DIV_ROUND(num, denom) (((num) + (denom) / 2) / (denom))
145 
146 #define BAUD2BRD(baud) DIV_ROUND(UART_CLOCK_RATE << (UART_CTL_HSE_VALUE + 2), (baud))
147 #define BAUD2IBRD(baud) (BAUD2BRD(baud) >> 6)
148 #define BAUD2FBRD(baud) (BAUD2BRD(baud) & 0x3f)

203 static void
204 reset(uint32_t uart_base)
205 {
206  uint32_t lchr;
207 
208  /* Make sure the UART is disabled before trying to configure it */
209  REG(uart_base + UART_CTL) = UART_CTL_VALUE;
210 
211  /* Clear error status */
212  REG(uart_base + UART_ECR) = 0xFF;
213 
214  /* Store LCHR configuration */
215  lchr = REG(uart_base + UART_LCRH);
216 
217  /* Flush FIFOs by clearing LCHR.FEN */
218  REG(uart_base + UART_LCRH) = 0;
219 
220  /* Restore LCHR configuration */
221  REG(uart_base + UART_LCRH) = lchr;
222 
223  /* UART Enable */
224  REG(uart_base + UART_CTL) |= UART_CTL_UARTEN;
225 }

227 static bool
228 permit_pm1(void)
229 {
230  const uart_regs_t *regs;
231 
232  for(regs = &uart_regs[0]; regs < &uart_regs[UART_INSTANCE_COUNT]; regs++) {
233  if((REG(regs->base + UART_FR) & UART_FR_BUSY) != 0) {
234  return false;
235  }
236  }
237 
238  return true;
239 }
240 /*---------------------------------------------------------------------------*/
241 void
242 uart_init(uint8_t uart)
243 {
244  const uart_regs_t *regs;
245 
246  if(uart >= UART_INSTANCE_COUNT) {
247  return;
248  }
249  regs = &uart_regs[uart];
250  if(regs->rx.port < 0 || regs->tx.port < 0) {
251  return;
252  }
253 
254  lpm_register_peripheral(permit_pm1);
255 
256  /* Enable clock for the UART while Running, in Sleep and Deep Sleep */
257  REG(SYS_CTRL_RCGCUART) |= regs->sys_ctrl_rcgcuart_uart;
258  REG(SYS_CTRL_SCGCUART) |= regs->sys_ctrl_scgcuart_uart;
259  REG(SYS_CTRL_DCGCUART) |= regs->sys_ctrl_dcgcuart_uart;
260 
261  /* Run on SYS_DIV */
262  REG(regs->base + UART_CC) = 0;
263 
264  /*
265  * Select the UARTx RX pin by writing to the IOC_UARTRXD_UARTn register
266  *
267  * The value to be written will be on of the IOC_INPUT_SEL_Pxn defines from
268  * ioc.h. The value can also be calculated as:
269  *
270  * (port << 3) + pin
271  */
272  REG(regs->ioc_uartrxd_uart) = (regs->rx.port << 3) + regs->rx.pin;
273 
274  /*
275  * Pad Control for the TX pin:
276  * - Set function to UARTn TX
277  * - Output Enable
278  */
279  ioc_set_sel(regs->tx.port, regs->tx.pin, regs->ioc_pxx_sel_uart_txd);
280  ioc_set_over(regs->tx.port, regs->tx.pin, IOC_OVERRIDE_OE);
281 
282  /* Set RX and TX pins to peripheral mode */
283  GPIO_PERIPHERAL_CONTROL(GPIO_PORT_TO_BASE(regs->tx.port),
284  GPIO_PIN_MASK(regs->tx.pin));
285  GPIO_PERIPHERAL_CONTROL(GPIO_PORT_TO_BASE(regs->rx.port),
286  GPIO_PIN_MASK(regs->rx.pin));
287 
288  /*
289  * UART Interrupt Masks:
290  * Acknowledge RX and RX Timeout
291  * Acknowledge Framing, Overrun and Break Errors
292  */
293  REG(regs->base + UART_IM) = UART_IM_RXIM | UART_IM_RTIM;
294  REG(regs->base + UART_IM) |= UART_IM_OEIM | UART_IM_BEIM | UART_IM_FEIM;
295 
296  REG(regs->base + UART_IFLS) =
297  UART_IFLS_RXIFLSEL_1_8 | UART_IFLS_TXIFLSEL_1_2;
298 
299  /* Make sure the UART is disabled before trying to configure it */
300  REG(regs->base + UART_CTL) = UART_CTL_VALUE;
301 
302  /* Baud Rate Generation */
303  REG(regs->base + UART_IBRD) = regs->ibrd;
304  REG(regs->base + UART_FBRD) = regs->fbrd;
305 
306  /* UART Control: 8N1 with FIFOs */
307  REG(regs->base + UART_LCRH) = UART_LCRH_WLEN_8 | UART_LCRH_FEN;
308 
309  /*
310  * Enable hardware flow control (RTS/CTS) if requested.
311  * Note that hardware flow control is available only on UART1.
312  */
313  if(regs->cts.port >= 0) {
314  REG(IOC_UARTCTS_UART1) = ioc_input_sel(regs->cts.port, regs->cts.pin);
315  GPIO_PERIPHERAL_CONTROL(GPIO_PORT_TO_BASE(regs->cts.port), GPIO_PIN_MASK(regs->cts.pin));
316  ioc_set_over(regs->cts.port, regs->cts.pin, IOC_OVERRIDE_DIS);
317  REG(UART_1_BASE + UART_CTL) |= UART_CTL_CTSEN;
318  }
319 
320  if(regs->rts.port >= 0) {
321  ioc_set_sel(regs->rts.port, regs->rts.pin, IOC_PXX_SEL_UART1_RTS);
322  GPIO_PERIPHERAL_CONTROL(GPIO_PORT_TO_BASE(regs->rts.port), GPIO_PIN_MASK(regs->rts.pin));
323  ioc_set_over(regs->rts.port, regs->rts.pin, IOC_OVERRIDE_OE);
324  REG(UART_1_BASE + UART_CTL) |= UART_CTL_RTSEN;
325  }
326 
327  /* UART Enable */
328  REG(regs->base + UART_CTL) |= UART_CTL_UARTEN;
329 
330  /* Enable UART0 Interrupts */
331  NVIC_EnableIRQ(regs->nvic_int);
332 }

344 void
345 uart_write_byte(uint8_t uart, uint8_t b)
346 {
347  uint32_t uart_base;
348 
349  if(uart >= UART_INSTANCE_COUNT) {
350  return;
351  }
352  uart_base = uart_regs[uart].base;
353 
354  /* Block if the TX FIFO is full */
355  while(REG(uart_base + UART_FR) & UART_FR_TXFF);
356 
357  REG(uart_base + UART_DR) = b;
358 }
359 /*---------------------------------------------------------------------------*/
360 static void
361 uart_isr(uint8_t uart)
362 {
363  uint32_t uart_base;
364  uint16_t mis;
365 
366  ENERGEST_ON(ENERGEST_TYPE_IRQ);
367 
368  uart_base = uart_regs[uart].base;
369 
370  /* Store the current MIS and clear all flags early, except the RTM flag.
371  * This will clear itself when we read out the entire FIFO contents */
372  mis = REG(uart_base + UART_MIS) & 0x0000FFFF;
373 
374  REG(uart_base + UART_ICR) = 0x0000FFBF;
375 
376  if(mis & (UART_MIS_RXMIS | UART_MIS_RTMIS)) {
377  while(!(REG(uart_base + UART_FR) & UART_FR_RXFE)) {
378  if(input_handler[uart] != NULL) {
379  input_handler[uart]((unsigned char)(REG(uart_base + UART_DR) & 0xFF));
380  } else {
381  /* To prevent an Overrun Error, we need to flush the FIFO even if we
382  * don't have an input_handler. Use mis as a data trash can */
383  mis = REG(uart_base + UART_DR);
384  }
385  }
386  } else if(mis & (UART_MIS_OEMIS | UART_MIS_BEMIS | UART_MIS_FEMIS)) {
387  /* ISR triggered due to some error condition */
388  reset(uart_base);
389  }
390 
391  ENERGEST_OFF(ENERGEST_TYPE_IRQ);
392 }
