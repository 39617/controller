86 def udpReceive():
87  """RUNS ON SEPARATE THREAD """
88  while True:
89  data, addr = s_rx.recvfrom(128)
90  data = data.replace("\"","").strip("\0") # strip termination byte and possible
91  node_data_key = node_data[addr[0]]
92  # Indicate node is connected
93  node_data_key[NODE_CONNECTED] = CONNECT_WATCHDOG
94  print addr[0] + ' (' + node_data_key[NODE_ALIAS] + ') : ' + data
95  # Write new data at index in data buffer. Data buffer is view of plot
96  data_lock.acquire()
