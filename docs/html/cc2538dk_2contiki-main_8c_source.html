88 static void
89 fade(unsigned char l)
90 {
91  volatile int i;
92  int k, j;
93  for(k = 0; k < 800; ++k) {
94  j = k > 400 ? 800 - k : k;
95 
96  leds_on(l);
97  for(i = 0; i < j; ++i) {
98  asm("nop");
99  }
100  leds_off(l);
101  for(i = 0; i < 400 - j; ++i) {
102  asm("nop");
103  }
104  }
105 }

107 static void
108 set_rf_params(void)
109 {
110  uint16_t short_addr;
111  uint8_t ext_addr[8];
112 
113  ieee_addr_cpy_to(ext_addr, 8);
114 
115  short_addr = ext_addr[7];
116  short_addr |= ext_addr[6] << 8;
117 
118  /* Populate linkaddr_node_addr. Maintain endianness */
119  memcpy(&linkaddr_node_addr, &ext_addr[8 - LINKADDR_SIZE], LINKADDR_SIZE);
120 
121 #if STARTUP_CONF_VERBOSE
122  {
123  int i;
124  printf("Rime configured with address ");
125  for(i = 0; i < LINKADDR_SIZE - 1; i++) {
126  printf("%02x:", linkaddr_node_addr.u8[i]);
127  }
128  printf("%02x\n", linkaddr_node_addr.u8[i]);
129  }
130 #endif
131 
132  NETSTACK_RADIO.set_value(RADIO_PARAM_PAN_ID, IEEE802154_PANID);
133  NETSTACK_RADIO.set_value(RADIO_PARAM_16BIT_ADDR, short_addr);
134  NETSTACK_RADIO.set_value(RADIO_PARAM_CHANNEL, CC2538_RF_CHANNEL);
135  NETSTACK_RADIO.set_object(RADIO_PARAM_64BIT_ADDR, ext_addr, 8);
136 }
