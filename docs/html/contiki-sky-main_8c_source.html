440 #if !PROCESS_CONF_NO_PROCESS_NAMES

443  autostart_start(autostart_processes);

445  /*
446  * This is the scheduler loop.
447  */
448 #if DCOSYNCH_CONF_ENABLED
449  timer_set(&mgt_timer, DCOSYNCH_PERIOD * CLOCK_SECOND);
450 #endif
451 
452  /* watchdog_stop();*/
453  while(1) {
454  int r;
455  do {
456  /* Reset watchdog. */
457  watchdog_periodic();
458  r = process_run();
459  } while(r > 0);
460 
461  /*
462  * Idle processing.
463  */
464  int s = splhigh(); /* Disable interrupts. */
465  /* uart1_active is for avoiding LPM3 when still sending or receiving */
466  if(process_nevents() != 0 || uart1_active()) {
467  splx(s); /* Re-enable interrupts. */
468  } else {
469  static unsigned long irq_energest = 0;
470 
471 #if DCOSYNCH_CONF_ENABLED
472  /* before going down to sleep possibly do some management */
473  if(timer_expired(&mgt_timer)) {
474  watchdog_periodic();
475  timer_reset(&mgt_timer);
476  msp430_sync_dco();
477 #if CC2420_CONF_SFD_TIMESTAMPS
478  cc2420_arch_sfd_init();
479 #endif /* CC2420_CONF_SFD_TIMESTAMPS */
480  }
481 #endif
482 
483  /* Re-enable interrupts and go to sleep atomically. */
484  ENERGEST_SWITCH(ENERGEST_TYPE_CPU, ENERGEST_TYPE_LPM);
485  /* We only want to measure the processing done in IRQs when we
486  are asleep, so we discard the processing time done when we
487  were awake. */
488  energest_type_set(ENERGEST_TYPE_IRQ, irq_energest);
489  watchdog_stop();
490  /* check if the DCO needs to be on - if so - only LPM 1 */
491  if (msp430_dco_required) {
492  _BIS_SR(GIE | CPUOFF); /* LPM1 sleep for DMA to work!. */
493  } else {
494  _BIS_SR(GIE | SCG0 | SCG1 | CPUOFF); /* LPM3 sleep. This
495  statement will block
496  until the CPU is
497  woken up by an
498  interrupt that sets
499  the wake up flag. */
500  }
501  /* We get the current processing time for interrupts that was
502  done during the LPM and store it for next time around. */
503  dint();
504  irq_energest = energest_type_time(ENERGEST_TYPE_IRQ);
505  eint();
506  watchdog_start();
507  ENERGEST_SWITCH(ENERGEST_TYPE_LPM, ENERGEST_TYPE_CPU);
508  }
509  }
510 
511  return 0;
512 }
