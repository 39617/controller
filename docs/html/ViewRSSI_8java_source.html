79  // Draw the light grey channels from 11 to 26

130  public void handleInput() throws IOException {
131  BufferedReader reader = new BufferedReader(new InputStreamReader(System.in));
132  while(true) {
133  String line = reader.readLine();
134  if (line.startsWith("RSSI:")) {
135  try {
136  String[] parts = line.substring(5).split(" ");
137  for (int i = 0, n = parts.length; i < n; i++) {
138  rssi[i] = Integer.parseInt(parts[i]); // An RSSI=0 corresponds to -100 dBm, an RSSI=100 corresponds to 0 dBm
139  if (rssi[i] > rssiMax[i]) rssiMax[i] = rssi[i];
140  else if (rssiMax[i] > 0) rssiMax[i] = rssiMax[i] - DECREASE_RSSI;
141  }
142  } catch (Exception e) {
143  e.printStackTrace(); /* Report error, but do not fail... */
144  }
145  repaint();
146  }
147  }
148  }
