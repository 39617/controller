123 httpd_cgi(char *name)
124 {
125  struct httpd_cgi_call *f;
126 
127  /* Find the matching name in the table, return the function. */
128  for(f = calls; f != NULL; f = f->next) {
129  if(strncmp(f->name, name, strlen(f->name)) == 0) {
130  return f->function;
131  }
132  }
133  return nullfunction;
134 }

252 static unsigned short
253 make_addresses(void *p)
254 {
255  uint8_t i, j = 0;
256  uint16_t numprinted;
257 
258  numprinted = httpd_snprintf((char *)uip_appdata, uip_mss(),httpd_cgi_addrh);
259  for (i=0; i<UIP_DS6_ADDR_NB;i++) {
260  if (uip_ds6_if.addr_list[i].isused) {
261  j++;
262  numprinted += httpd_cgi_sprint_ip6(uip_ds6_if.addr_list[i].ipaddr, uip_appdata + numprinted);
263  numprinted += httpd_snprintf((char *)uip_appdata+numprinted, uip_mss()-numprinted, httpd_cgi_addrb);
264  }
265  }
266 //if (j==0) numprinted += httpd_snprintf((char *)uip_appdata+numprinted, uip_mss()-numprinted, httpd_cgi_addrn);
267  numprinted += httpd_snprintf((char *)uip_appdata+numprinted, uip_mss()-numprinted, httpd_cgi_addrf, UIP_DS6_ADDR_NB-j);
268  return numprinted;
269 }

310 static unsigned short
311 make_routes(void *p)
312 {
313  static const char httpd_cgi_rtes1[] HTTPD_STRING_ATTR = "(%u (via ";
314  static const char httpd_cgi_rtes2[] HTTPD_STRING_ATTR = ") %lus<br>";
315  static const char httpd_cgi_rtes3[] HTTPD_STRING_ATTR = ")<br>";
316  uint8_t j=0;
317  uint16_t numprinted;
318  uip_ds6_route_t *r;
319 
320  numprinted = httpd_snprintf((char *)uip_appdata, uip_mss(),httpd_cgi_addrh);
321  for(r = uip_ds6_route_head();
322  r != NULL;
323  r = uip_ds6_route_next(r)) {
324  j++;
325  numprinted += httpd_cgi_sprint_ip6(r->ipaddr, uip_appdata + numprinted);
326  numprinted += httpd_snprintf((char *)uip_appdata+numprinted, uip_mss()-numprinted, httpd_cgi_rtes1, r->length);
327  numprinted += httpd_cgi_sprint_ip6(*(uip_ds6_route_nexthop(r)), uip_appdata + numprinted);
328  if(r->state.lifetime < 3600) {
329  numprinted += httpd_snprintf((char *)uip_appdata+numprinted, uip_mss()-numprinted, httpd_cgi_rtes2, r->state.lifetime);
330  } else {
331  numprinted += httpd_snprintf((char *)uip_appdata+numprinted, uip_mss()-numprinted, httpd_cgi_rtes3);
332  }
333  }
334  if (j==0) numprinted += httpd_snprintf((char *)uip_appdata+numprinted, uip_mss()-numprinted, httpd_cgi_addrn);
335  numprinted += httpd_snprintf((char *)uip_appdata+numprinted, uip_mss()-numprinted, httpd_cgi_addrf,UIP_DS6_ROUTE_NB-j);
336  return numprinted;
337 }
