175 static void
176 conf_post_handler(void *request, void *response, uint8_t *buffer,
177  uint16_t preferred_size, int32_t *offset)
178 {
179  const char *ptr = NULL;
180  char tmp_buf[16];
181  unsigned long interval = 0;
182  unsigned long duration = 0;
183  uint8_t mode = VERY_SLEEPY_MODE_OFF;
184  uint8_t post_status = POST_STATUS_NONE;
185  int rv;
186 
187  rv = REST.get_post_variable(request, "mode", &ptr);
188  if(rv && rv < 16) {
189  memset(tmp_buf, 0, sizeof(tmp_buf));
190  memcpy(tmp_buf, ptr, rv);
191  rv = atoi(tmp_buf);
192 
193  if(rv == 1) {
194  mode = VERY_SLEEPY_MODE_ON;
195  post_status |= POST_STATUS_HAS_MODE;
196  } else if(rv == 0) {
197  mode = VERY_SLEEPY_MODE_OFF;
198  post_status |= POST_STATUS_HAS_MODE;
199  } else {
200  post_status = POST_STATUS_BAD;
201  }
202  }
203 
204  rv = REST.get_post_variable(request, "duration", &ptr);
205  if(rv && rv < 16) {
206  memset(tmp_buf, 0, sizeof(tmp_buf));
207  memcpy(tmp_buf, ptr, rv);
208  rv = atoi(tmp_buf);
209 
210  duration = (unsigned long)rv;
211  if(duration < NORMAL_OP_DURATION_MIN || duration > NORMAL_OP_DURATION_MAX) {
212  post_status = POST_STATUS_BAD;
213  } else {
214  post_status |= POST_STATUS_HAS_DURATION;
215  }
216  }
217 
218  rv = REST.get_post_variable(request, "interval", &ptr);
219  if(rv && rv < 16) {
220  memset(tmp_buf, 0, sizeof(tmp_buf));
221  memcpy(tmp_buf, ptr, rv);
222  rv = atoi(tmp_buf);
223  interval = (unsigned long)rv;
224  if(interval < PERIODIC_INTERVAL_MIN || interval > PERIODIC_INTERVAL_MAX) {
225  post_status = POST_STATUS_BAD;
226  } else {
227  post_status |= POST_STATUS_HAS_INTERVAL;
228  }
229  }
230 
231  if((post_status & POST_STATUS_BAD) == POST_STATUS_BAD ||
232  post_status == POST_STATUS_NONE) {
233  REST.set_response_status(response, REST.status.BAD_REQUEST);
234  snprintf((char *)buffer, REST_MAX_CHUNK_SIZE,
235  "mode=0|1&duration=[%u,%u]&interval=[%u,%u]",
236  NORMAL_OP_DURATION_MIN, NORMAL_OP_DURATION_MAX,
237  PERIODIC_INTERVAL_MIN, PERIODIC_INTERVAL_MAX);
238 
239  REST.set_response_payload(response, buffer, strlen((char *)buffer));
240  return;
241  }
242 
243  /* Values are sane. Update the config and notify the process */
244  if(post_status & POST_STATUS_HAS_MODE) {
245  config.mode = mode;
246  }
247 
248  if(post_status & POST_STATUS_HAS_INTERVAL) {
249  config.interval = interval;
250  }
251 
252  if(post_status & POST_STATUS_HAS_DURATION) {
253  config.duration = duration;
254  }
255 
256  process_post(&very_sleepy_demo_process, event_new_config, NULL);
257 }

264 /*
265  * If our preferred parent is not NBR_REACHABLE in the ND cache, NUD will send
266  * a unicast NS and wait for NA. If NA fails then the neighbour will be removed
267  * from the ND cache and the default route will be deleted. To prevent this,
268  * keep the MAC on until the parent becomes NBR_REACHABLE. We also keep the MAC
269  * on if we are about to do RPL probing.
270  *
271  * In all cases, the radio will be locked on for KEEP_MAC_ON_MIN_PERIOD secs
272  */
273 static uint8_t
274 keep_mac_on(void)
275 {
276  uip_ds6_nbr_t *nbr;
277  uint8_t rv = MAC_CAN_BE_TURNED_OFF;
278 
279  if(!stimer_expired(&st_min_mac_on_duration)) {
280  return MAC_MUST_STAY_ON;
281  }
282 
283 #if RPL_WITH_PROBING
284  /* Determine if we are about to send a RPL probe */
285  if(CLOCK_LT(etimer_expiration_time(
286  &rpl_get_default_instance()->probing_timer.etimer),
287  (clock_time() + PERIODIC_INTERVAL))) {
288  rv = MAC_MUST_STAY_ON;
289  }
290 #endif
291 
292  /* It's OK to pass a NULL pointer, the callee checks and returns NULL */
293  nbr = uip_ds6_nbr_lookup(uip_ds6_defrt_choose());
294 
295  if(nbr == NULL) {
296  /* We don't have a default route, or it's not reachable (NUD likely). */
297  rv = MAC_MUST_STAY_ON;
298  } else {
299  if(nbr->state != NBR_REACHABLE) {
300  rv = MAC_MUST_STAY_ON;
301  }
302  }
303 
304  if(rv == MAC_MUST_STAY_ON && stimer_expired(&st_min_mac_on_duration)) {
305  stimer_set(&st_min_mac_on_duration, KEEP_MAC_ON_MIN_PERIOD);
306  }
307 
308  return rv;
309 }

313 {

330 PROCESS_THREAD(very_sleepy_demo_process, ev, data)
331 {
332  uint8_t mac_keep_on;
333 
334  PROCESS_BEGIN();
335 
336  SENSORS_ACTIVATE(batmon_sensor);
337 
338  config.mode = VERY_SLEEPY_MODE_OFF;
339  config.interval = PERIODIC_INTERVAL_DEFAULT;
340  config.duration = NORMAL_OP_DURATION_DEFAULT;
341 
342  state = STATE_NORMAL;
343 
344  event_new_config = process_alloc_event();
345 
346  rest_init_engine();
347 
348  readings_resource.flags += IS_OBSERVABLE;
349  rest_activate_resource(&readings_resource, "sen/readings");
350  rest_activate_resource(&very_sleepy_conf, "very_sleepy_config");
351 
352  printf("Very Sleepy Demo Process\n");
353 
354  switch_to_normal();
355 
356  etimer_set(&et_periodic, PERIODIC_INTERVAL);
357 
358  while(1) {
359 
360  PROCESS_YIELD();
361 
362  if(ev == sensors_event && data == &button_left_sensor) {
363  switch_to_normal();
364  }
365 
366  if(ev == event_new_config) {
367  stimer_set(&st_interval, config.interval);
368  stimer_set(&st_duration, config.duration);
369  }
370 
371  if((ev == PROCESS_EVENT_TIMER && data == &et_periodic) ||
372  (ev == sensors_event && data == &button_left_sensor) ||
373  (ev == event_new_config)) {
374 
375  /*
376  * Determine if the stack is about to do essential network maintenance
377  * and, if so, keep the MAC layer on
378  */
379  mac_keep_on = keep_mac_on();
380 
381  if(mac_keep_on == MAC_MUST_STAY_ON || state != STATE_VERY_SLEEPY) {
382  leds_on(LEDS_GREEN);
383  NETSTACK_MAC.on();
384  }
385 
386  /*
387  * Next, switch between normal and very sleepy mode depending on config,
388  * send notifications to observers as required.
389  */
390  if(state == STATE_NOTIFY_OBSERVERS) {
391  REST.notify_subscribers(&readings_resource);
392  state = STATE_NORMAL;
393  }
394 
395  if(state == STATE_NORMAL) {
396  if(stimer_expired(&st_duration)) {
397  stimer_set(&st_duration, config.duration);
398  if(config.mode == VERY_SLEEPY_MODE_ON) {
399  switch_to_very_sleepy();
400  }
401  }
402  } else if(state == STATE_VERY_SLEEPY) {
403  if(stimer_expired(&st_interval)) {
404  switch_to_normal();
405  }
406  }
407 
408  if(mac_keep_on == MAC_CAN_BE_TURNED_OFF && state == STATE_VERY_SLEEPY) {
409  leds_off(LEDS_GREEN);
410  NETSTACK_MAC.off(0);
411  } else {
412  leds_on(LEDS_GREEN);
413  NETSTACK_MAC.on();
414  }
415 
416  /* Schedule next pass */
417  etimer_set(&et_periodic, PERIODIC_INTERVAL);
418  }
419  }
420 
421  PROCESS_END();
422 }
