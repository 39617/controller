54  PRINTA("--- Network status ---\n");
55  /* Our IPv6 addresses */
56  PRINTA("- Server IPv6 addresses:\n");
57  for(i = 0; i < UIP_DS6_ADDR_NB; i++) {
58  state = uip_ds6_if.addr_list[i].state;
59  if(uip_ds6_if.addr_list[i].isused &&
60  (state == ADDR_TENTATIVE || state == ADDR_PREFERRED)) {
61  PRINTA("-- ");
62  uip_debug_ipaddr_print(&uip_ds6_if.addr_list[i].ipaddr);
63  PRINTA("\n");
64  }
65  }
66  /* Our default route */
67  PRINTA("- Default route:\n");
68  default_route = uip_ds6_defrt_lookup(uip_ds6_defrt_choose());
69  if(default_route != NULL) {
70  PRINTA("-- ");
71  uip_debug_ipaddr_print(&default_route->ipaddr);;
72  PRINTA(" (lifetime: %lu seconds)\n", (unsigned long)default_route->lifetime.interval);
73  } else {
74  PRINTA("-- None\n");
75  }
76  /* Our routing entries */
77  PRINTA("- Routing entries (%u in total):\n", uip_ds6_route_num_routes());
78  route = uip_ds6_route_head();
79  while(route != NULL) {
80  PRINTA("-- ");
81  uip_debug_ipaddr_print(&route->ipaddr);

83  uip_debug_ipaddr_print(uip_ds6_route_nexthop(route));
84  PRINTA(" (lifetime: %lu seconds)\n", (unsigned long)route->state.lifetime);
85  route = uip_ds6_route_next(route);
86  }
87  PRINTA("----------------------\n");
