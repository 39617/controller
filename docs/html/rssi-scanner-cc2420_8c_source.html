52 /* This assumes that the CC2420 is always on and "stable" */
53 static void
54 set_frq(int c)
55 {
56  int f;
57  /* We can read even other channels with CC2420! */
58  /* Fc = 2048 + FSCTRL ; Fc = 2405 + 5(k-11) MHz, k=11,12, ... , 26 */
59  f = c + 352; /* Start from 2400 MHz to 2485 MHz, */
60 
61  /* Write the new channel value */
62  CC2420_SPI_ENABLE();
63  SPI_WRITE_FAST(CC2420_FSCTRL);
64  SPI_WRITE_FAST((uint8_t)(f >> 8));
65  SPI_WRITE_FAST((uint8_t)(f & 0xff));
66  SPI_WAITFORTx_ENDED();
67  SPI_WRITE(0);
68 
69  /* Send the strobe */
70  SPI_WRITE(CC2420_SRXON);
71  CC2420_SPI_DISABLE();
72 }
73 static void
74 do_rssi(void)
75 {
76  int channel;
77  printf("RSSI:");
78  for(channel = 0; channel <= 85; ++channel) {
79  set_frq(channel);
80  printf("%d ", cc2420_rssi() + 100);
81  }
82  printf("\n");
83 }
