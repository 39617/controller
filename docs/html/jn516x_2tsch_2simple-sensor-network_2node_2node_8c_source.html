111 static void
112 my_sprintf(char * udp_buf, int8_t value)
113 {
114  /* Fill the buffer with 4 ASCII chars */
115  if (value < 0) {
116  *udp_buf++ = '-';
117  } else {
118  *udp_buf++ = '+';
119  }
120  value = abs(value);
121  *udp_buf++ = value/100 + '0';
122  value %= 100;
123  *udp_buf++ = value/10 + '0';
124  value %= 10;
125  *udp_buf++ = value + '0';
126  *udp_buf = 0;
127 }

135 PROCESS_THREAD(node_process, ev, data)
136 {
137  PROCESS_BEGIN();
138 
139  static int sample_count = 0;
140  static struct etimer et;
141  extern unsigned char node_mac[8];
142 
143  leds_init();
144 
145  /* 3 possible roles:
146  * - role_6ln: simple node, will join any network, secured or not
147  * - role_6dr: DAG root, will advertise (unsecured) beacons
148  * - role_6dr_sec: DAG root, will advertise secured beacons
149  * */
150  static int is_coordinator = 0;
151  static enum { role_6ln, role_6dr, role_6dr_sec } node_role;
152 
153  /* Set node with ID == 1 as coordinator, handy in Cooja. */
154  if(node_id == 1) {
155  if(LLSEC802154_ENABLED) {
156  node_role = role_6dr_sec;
157  } else {
158  node_role = role_6dr;
159  }
160  } else {
161  node_role = role_6ln;
162  }
163 
164  printf("Init: node starting with role %s\n",
165  node_role == role_6ln ? "6ln" : (node_role == role_6dr) ? "6dr" : "6dr-sec");
166 
167 #if WITH_TSCH
168  tsch_set_pan_secured(LLSEC802154_ENABLED && (node_role == role_6dr_sec));
169 #endif /* WITH_TSCH */
170  is_coordinator = node_role > role_6ln;
171 
172  if(is_coordinator) {
173  uip_ipaddr_t prefix;
174  uip_ip6addr(&prefix, 0xbbbb, 0, 0, 0, 0, 0, 0, 0);
175  rpl_tools_init(&prefix);
176  } else {
177  rpl_tools_init(NULL);
178  }
179 
180  /* Selected waveform depends on LS byte of MAC */
181  selected_waveform = node_mac[7] % NUMBER_OF_WAVEFORMS;
182  printf("LS-Byte=0x%x; waveform=%d\n", node_mac[7], selected_waveform);
183 
184  process_start(&led_process, NULL);
185 
186  /* Listen to any host on 8185 */
187  udp_conn_rx = udp_new(NULL, 0, NULL);
188  udp_bind(udp_conn_rx, UIP_HTONS(8185));
189 
190  /* Wait for timer event
191  On timer event, handle next sample */
192  etimer_set(&et, INTERVAL*CLOCK_SECOND);
193  while(1) {
194  PROCESS_WAIT_EVENT_UNTIL(etimer_expired(&et) || (ev == tcpip_event));
195  if (ev == tcpip_event) {
196  udp_rx_handler();
197  }
198  if (etimer_expired(&et) ) {
199  /* Restart timer */
200  total_time += INTERVAL;
201  if (host_found) {
202  /* Make sample count dependent on asn. After a disconnect, waveforms remain
203  synchronous. Use node_mac to create phase offset between waveforms in different nodes */
204  sample_count = ((tsch_current_asn.ls4b/((1000/(TSCH_CONF_DEFAULT_TIMESLOT_LENGTH/1000)))/INTERVAL)+node_mac[7]) % (SIZE_OF_WAVEFORM-1);
205  printf("%d sec. waveform=%s. cnt=%d. value=%d\n", total_time, waveform_table[selected_waveform].str, sample_count, waveform_table[selected_waveform].table[sample_count]);
206  my_sprintf(udp_buf, waveform_table[selected_waveform].table[sample_count]);
207  uip_udp_packet_send(udp_conn_tx, udp_buf, strlen(udp_buf));
208  /* Switch LED on and start blink timer (callback timer) */
209  process_post(&led_process, PROCESS_EVENT_CONTINUE, post_mssg);
210  } else {
211  printf("No host\n");
212  }
213  etimer_restart(&et);
214  if (total_time%60 == 0) {
215  /* Print network status once per minute */
216  print_network_status();
217  }
218  }
219  }
220  PROCESS_END();
221 }
