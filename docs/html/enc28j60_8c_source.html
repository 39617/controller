66 static void
67 setregbank(uint8_t address)
68 {
69  /* set the bank (if needed) */
70  if((address & BANK_MASK) != Enc28j60Bank) {
71  write_op(ENC28J60_BIT_FIELD_CLR, ECON1, (ECON1_BSEL1 | ECON1_BSEL0));
72  write_op(ENC28J60_BIT_FIELD_SET, ECON1, (address & BANK_MASK) >> 5);
73  Enc28j60Bank = (address & BANK_MASK);
74  }

155 {
156  /* set the PHY register address */

159  writereg(MIWRL, data);

165  clock_delay(15);

178 static void
179 reset(void)
180 {
181  PRINTF("resetting chip\n");
182 
183  enc28j60_arch_spi_init();
184 
185  /* Wait for OST */
186  while((readreg(ESTAT) & ESTAT_CLKRDY) == 0);
187 
188  softreset();
189 
190  /* do bank 0 stuff */
191  /* initialize receive buffer */
192  /* 16-bit transfers, must write low byte first */
193  /* set receive buffer start address */
194  NextPacketPtr = RXSTART_INIT;
195  /* Rx start */
196  writereg(ERXSTL, RXSTART_INIT & 0xFF);
197  writereg(ERXSTH, RXSTART_INIT >> 8);
198  /* set receive pointer address */
199  writereg(ERXRDPTL, RXSTART_INIT & 0xFF);
200  writereg(ERXRDPTH, RXSTART_INIT >> 8);
201  /* RX end */
202  writereg(ERXNDL, RXSTOP_INIT & 0xFF);
203  writereg(ERXNDH, RXSTOP_INIT >> 8);
204  /* TX start */
205  writereg(ETXSTL, TXSTART_INIT & 0xFF);
206  writereg(ETXSTH, TXSTART_INIT >> 8);
207  /* TX end */
208  writereg(ETXNDL, TXSTOP_INIT & 0xFF);
209  writereg(ETXNDH, TXSTOP_INIT >> 8);
210 
211  /* do bank 1 stuff, packet filter: */
212 #if CETIC_6LBR_TRANSPARENTBRIDGE
213  /*Enter Promiscuous mode (capture all packets) */
214  writereg(ERXFCON, 0);
215 #else
216  writereg(ERXFCON, ERXFCON_UCEN | ERXFCON_MCEN | ERXFCON_BCEN);
217 #endif
218 
219  /* do bank 2 stuff */
220  /* enable MAC receive */
221  writereg(MACON1, MACON1_MARXEN | MACON1_TXPAUS | MACON1_RXPAUS);
222  /* bring MAC out of reset */
223  writereg(MACON2, 0x00);
224  /* enable automatic padding to 60bytes and CRC operations */
225  writereg(MACON3, readreg(MACON3) | (MACON3_PADCFG0 + MACON3_TXCRCEN +
226  MACON3_FULDPX + MACON3_FRMLNEN));
227  /* set inter-frame gap (non-back-to-back) */
228  writereg(MAIPGL, 0x12);
229  writereg(MAIPGH, 0x0C);
230  /* set inter-frame gap (back-to-back) */
231  writereg(MABBIPG, 0x12);
232  /* Set the maximum packet size which the controller will accept */
233  /* Do not send packets longer than MAX_FRAMELEN: */
234  writereg(MAMXFLL, MAX_FRAMELEN & 0xFF);
235  writereg(MAMXFLH, MAX_FRAMELEN >> 8);
236 
237  /* do bank 3 stuff */
238  /* Set MAC address */
239  /* NOTE: MAC address in ENC28J60 is byte-backward */
240  writereg(MAADR5, enc_mac_addr[0]);
241  writereg(MAADR4, enc_mac_addr[1]);
242  writereg(MAADR3, enc_mac_addr[2]);
243  writereg(MAADR2, enc_mac_addr[3]);
244  writereg(MAADR1, enc_mac_addr[4]);
245  writereg(MAADR0, enc_mac_addr[5]);
246 
247  /* no loopback of transmitted frames */
248  writephy(PHCON2, PHCON2_HDLDIS);
249  /* Turn on reception */
250  writereg(ECON1, ECON1_RXEN);
251  clock_delay(10);
252  writephy(PHLCON, 0x476);
253  clock_delay(20);
254 
255  PRINTF("REV %u\n", readreg(EREVID));
256 }

276 int
277 enc28j60_send(const uint8_t *data, uint16_t datalen)
278 {
279  int padding = 0;
280 
281  if(!initialized) {
282  return -1;
283  }
284 
285  /*
286  1. Appropriately program the ETXST pointer to point to an unused
287  location in memory. It will point to the per packet control
288  byte. In the example, it would be programmed to 0120h. It is
289  recommended that an even address be used for ETXST.
290  2. Use the WBM SPI command to write the per packet control byte,
291  the destination address, the source MAC address, the
292  type/length and the data payload.
293  3. Appropriately program the ETXND pointer. It should point to the
294  last byte in the data payload. In the example, it would be
295  programmed to 0156h.
296  4. Clear EIR.TXIF, set EIE.TXIE and set EIE.INTIE to enable an
297  interrupt when done (if desired).
298  5. Start the transmission process by setting
299  ECON1.TXRTS.
300  */
301 
302  /* Set up the transmit buffer pointer */
303  writereg(ETXSTL, TXSTART_INIT & 0xff);
304  writereg(ETXSTH, TXSTART_INIT >> 8);
305  writereg(EWRPTL, TXSTART_INIT & 0xff);
306  writereg(EWRPTH, TXSTART_INIT >> 8);
307 
308  /* Write the transmission control register as the first byte of the
309  output packet. We write 0x00 to indicate that the default
310  configuration (the values in MACON3) will be used. */
311 #define WITH_MANUAL_PADDING 1
312 #if WITH_MANUAL_PADDING
313 #if UIP_LLH_LEN == 16
314 #define PADDING_MIN_SIZE (60 + 2)
315 #else
316 #define PADDING_MIN_SIZE 60
317 #endif
318  writedatabyte(0x0B); /* POVERRIDE, PCRCEN, PHUGEEN. Not PPADEN */
319  if(datalen < PADDING_MIN_SIZE) {
320  padding = PADDING_MIN_SIZE - datalen;
321  } else {
322  padding = 0;
323  }
324 #else /* WITH_MANUAL_PADDING */
325  writedatabyte(0x00); /* MACON3 */
326  padding = 0;
327 #endif /* WITH_MANUAL_PADDING */
328 
329  /* Write a pointer to the last data byte. */
330 #if UIP_LLH_LEN == 16
331  writereg(ETXNDL, (TXSTART_INIT + datalen - 2 + 0 + padding) & 0xff);
332  writereg(ETXNDH, (TXSTART_INIT + datalen - 2 + 0 + padding) >> 8);
333 #else
334  writereg(ETXNDL, (TXSTART_INIT + datalen + 0 + padding) & 0xff);
335  writereg(ETXNDH, (TXSTART_INIT + datalen + 0 + padding) >> 8);
336 #endif
337 
338 #if UIP_LLH_LEN == 16
339  /* Write Ethernet header */
340  writedata(data, 14);
341  /* Write Ehternet payload */
342  writedata(data + 16, datalen - 16);
343 #else
344  writedata(data, datalen);
345 #endif
346  if(padding > 0) {
347  uint8_t padding_buf[60];
348  memset(padding_buf, 0, padding);
349  writedata(padding_buf, padding);
350  }
351 
352  /* Clear EIR.TXIF */
353  writereg(EIR, readreg(EIR) & (~EIR_TXIF));
354 
355  /* Don't care about interrupts for now */
356 
357  /* Send the packet */
358  writereg(ECON1, readreg(ECON1) | ECON1_TXRTS);
359  while((readreg(ECON1) & ECON1_TXRTS) > 0);
360 
361  if((readreg(ESTAT) & ESTAT_TXABRT) != 0) {
362  PRINTF("tx err: %d: %02x:%02x:%02x:%02x:%02x:%02x\n", datalen,
363  0xff&data[0], 0xff&data[1], 0xff&data[2],
364  0xff&data[3], 0xff&data[4], 0xff&data[5]);
365  } else {
366  PRINTF("tx: %d: %02x:%02x:%02x:%02x:%02x:%02x\n", datalen,
367  0xff&data[0], 0xff&data[1], 0xff&data[2],
368  0xff&data[3], 0xff&data[4], 0xff&data[5]);
369  }
370  sent_packets++;
371  PRINTF("sent_packets %d\n", sent_packets);
372  return datalen;
373 }
374 /*---------------------------------------------------------------------------*/
375 int
376 enc28j60_read(uint8_t *buffer, uint16_t bufsize)
377 {
378  uint16_t rxstat;
379  uint16_t len;
380  int n;
381 
382  /* check if a packet has been received and buffered */
383  if((n = readreg(EPKTCNT)) == 0){
384  return (0);
385  }
386  PRINTF("EPKTCNT 0x%02x\n", n);
387 
388  /* Set the read pointer to the start of the received packet */
389  writereg(ERDPTL, (NextPacketPtr));
390  writereg(ERDPTH, (NextPacketPtr) >> 8);
391  /* Read the next packet pointer */
392  NextPacketPtr = readdatabyte();
393  NextPacketPtr |= readdatabyte() << 8;
394  /* read the packet length */
395  len = readdatabyte();
396  len |= readdatabyte() << 8;
397  /*remove the CRC count */
398  len -=4;
399  rxstat = readdatabyte();
400  rxstat |= readdatabyte() << 8;
401  /* limit retrieve length */
402  if(len > bufsize - 1) {
403  len = bufsize - 1;
404  }
405  /* check CRC and symbol errors (see datasheet page 44, table 7-3): */
406  /* The ERXFCON.CRCEN is set by default. Normally we should not */
407  /* need to check this. */
408  if((rxstat & 0x80) == 0) {
409  /* invalid */
410  len = 0;
411  } else {
412  /* copy the packet from the receive buffer */
413  len += 2;
414 #if UIP_LLH_LEN == 16
415  /* Read Ethernet header */
416  readdata(buffer, 14);
417  /* Read Ethernet payload */
418  readdata(buffer + 16, len - 14);
419 #else
420  readdata(buffer, len);
421 #endif
422  }
423 
424  /* Move the RX read pointer to the start of the next received packet */
425  /* This frees the memory we just read out */
426  writereg(ERXRDPTL, (NextPacketPtr));
427  writereg(ERXRDPTH, (NextPacketPtr) >> 8);
428  /* decrement the packet counter indicate we are done with this packet */
429  writereg(ECON2, readreg(ECON2) | ECON2_PKTDEC);
430 
431  PRINTF("rx: %d: %02x:%02x:%02x:%02x:%02x:%02x\n", len,
432  0xff&buffer[0], 0xff&buffer[1], 0xff&buffer[2],
433  0xff&buffer[3], 0xff&buffer[4], 0xff&buffer[5]);
434 
435  received_packets++;
436  PRINTF("received_packets %d\n", received_packets);
437  return (len);
438 }
