100 /*
101  * This function is called by the ctimer present in each neighbor
102  * table entry. The function removes the neighbor from the table
103  * because it has become too old.
104  */
105 static void
106 remove_neighbor(void *n)
107 {
108  struct example_neighbor *e = n;
109 
110  list_remove(neighbor_table, e);
111  memb_free(&neighbor_mem, e);
112 }
113 /*---------------------------------------------------------------------------*/
114 /*
115  * This function is called when an incoming announcement arrives. The
116  * function checks the neighbor table to see if the neighbor is
117  * already present in the list. If the neighbor is not present in the
118  * list, a new neighbor table entry is allocated and is added to the
119  * neighbor table.
120  */
121 static void
122 received_announcement(struct announcement *a,
123  const linkaddr_t *from,
124  uint16_t id, uint16_t value)
125 {
126  struct example_neighbor *e;
127 
128  /* printf("Got announcement from %d.%d, id %d, value %d\n",
129  from->u8[0], from->u8[1], id, value);*/
130 
131  /* We received an announcement from a neighbor so we need to update
132  the neighbor list, or add a new entry to the table. */
133  for(e = list_head(neighbor_table); e != NULL; e = e->next) {
134  if(linkaddr_cmp(from, &e->addr)) {
135  /* Our neighbor was found, so we update the timeout. */
136  ctimer_set(&e->ctimer, NEIGHBOR_TIMEOUT, remove_neighbor, e);
137  return;
138  }
139  }
140 
141  /* The neighbor was not found in the list, so we add a new entry by
142  allocating memory from the neighbor_mem pool, fill in the
143  necessary fields, and add it to the list. */
144  e = memb_alloc(&neighbor_mem);
145  if(e != NULL) {
146  linkaddr_copy(&e->addr, from);
147  list_add(neighbor_table, e);
148  ctimer_set(&e->ctimer, NEIGHBOR_TIMEOUT, remove_neighbor, e);
149  }
150 }

163 /*
164  * This function is called to forward a packet. The function picks a
165  * random neighbor from the neighbor list and returns its address. The
166  * multihop layer sends the packet to this address. If no neighbor is
167  * found, the function returns NULL to signal to the multihop layer
168  * that the packet should be dropped.
169  */
170 static linkaddr_t *
171 forward(struct multihop_conn *c,
172  const linkaddr_t *originator, const linkaddr_t *dest,
173  const linkaddr_t *prevhop, uint8_t hops)
174 {
175  /* Find a random neighbor to send to. */
176  int num, i;
177  struct example_neighbor *n;
178 
179  if(list_length(neighbor_table) > 0) {
180  num = random_rand() % list_length(neighbor_table);
181  i = 0;
182  for(n = list_head(neighbor_table); n != NULL && i != num; n = n->next) {
183  ++i;
184  }
185  if(n != NULL) {
186  printf("%d.%d: Forwarding packet to %d.%d (%d in list), hops %d\n",
187  linkaddr_node_addr.u8[0], linkaddr_node_addr.u8[1],
188  n->addr.u8[0], n->addr.u8[1], num,
189  packetbuf_attr(PACKETBUF_ATTR_HOPS));
190  return &n->addr;
191  }
192  }
193  printf("%d.%d: did not find a neighbor to foward to\n",
194  linkaddr_node_addr.u8[0], linkaddr_node_addr.u8[1]);
195  return NULL;
196 }
