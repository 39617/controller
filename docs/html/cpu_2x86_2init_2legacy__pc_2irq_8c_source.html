37 static void
38 spurious_irq7_handler(void)
39 {
40  /*
41  * NOTE: Originally IRQ7 was used for the parallel port interrupts. Nowadays,
42  * though, it is only used if some other IRQ (i.e.: a PCIx interrupt) is
43  * mapped to it. In this case we will have to check the PIC ISR register in
44  * order to confirm this was a real interrupt.
45  *
46  * In case of a spurious interrupt, we should NEVER send an EOI here so the PIC
47  * doesn't trigger the next queued interrupt.
48  */
49 }
