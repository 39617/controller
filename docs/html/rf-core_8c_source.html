126 uint_fast8_t
127 rf_core_send_cmd(uint32_t cmd, uint32_t *status)
128 {
129  uint32_t timeout_count = 0;
130  bool interrupts_disabled;
131  bool is_radio_op = false;
132 
133  /*
134  * If cmd is 4-byte aligned, then it's either a radio OP or an immediate
135  * command. Clear the status field if it's a radio OP
136  */
137  if((cmd & 0x03) == 0) {
138  uint32_t cmd_type;
139  cmd_type = ((rfc_command_t *)cmd)->commandNo & RF_CORE_COMMAND_TYPE_MASK;
140  if(cmd_type == RF_CORE_COMMAND_TYPE_IEEE_FG_RADIO_OP ||
141  cmd_type == RF_CORE_COMMAND_TYPE_RADIO_OP) {
142  is_radio_op = true;
143  ((rfc_radioOp_t *)cmd)->status = RF_CORE_RADIO_OP_STATUS_IDLE;
144  }
145  }
146 
147  /*
148  * Make sure ContikiMAC doesn't turn us off from within an interrupt while
149  * we are accessing RF Core registers
150  */
151  interrupts_disabled = ti_lib_int_master_disable();
152 
153  if(!rf_core_is_accessible()) {
154  PRINTF("rf_core_send_cmd: RF was off\n");
155  if(!interrupts_disabled) {
156  ti_lib_int_master_enable();
157  }
158  return RF_CORE_CMD_ERROR;
159  }
160 
161  if(is_radio_op) {
162  uint16_t command_no = ((rfc_radioOp_t *)cmd)->commandNo;
163  if((command_no & RF_CORE_COMMAND_PROTOCOL_MASK) != RF_CORE_COMMAND_PROTOCOL_COMMON &&
164  (command_no & RF_CORE_COMMAND_TYPE_MASK) == RF_CORE_COMMAND_TYPE_RADIO_OP) {
165  last_radio_op = (rfc_radioOp_t *)cmd;
166  }
167  }
168 
169  HWREG(RFC_DBELL_BASE + RFC_DBELL_O_CMDR) = cmd;
170  do {
171  *status = HWREG(RFC_DBELL_BASE + RFC_DBELL_O_CMDSTA);
172  if(++timeout_count > 50000) {
173  PRINTF("rf_core_send_cmd: 0x%08lx Timeout\n", cmd);
174  if(!interrupts_disabled) {
175  ti_lib_int_master_enable();
176  }
177  return RF_CORE_CMD_ERROR;
178  }
179  } while((*status & RF_CORE_CMDSTA_RESULT_MASK) == RF_CORE_CMDSTA_PENDING);
180 
181  if(!interrupts_disabled) {
182  ti_lib_int_master_enable();
183  }
184 
185  /*
186  * If we reach here the command is no longer pending. It is either completed
187  * successfully or with error
188  */
189  return (*status & RF_CORE_CMDSTA_RESULT_MASK) == RF_CORE_CMDSTA_DONE;
190 }
191 /*---------------------------------------------------------------------------*/
192 uint_fast8_t
193 rf_core_wait_cmd_done(void *cmd)
194 {
195  volatile rfc_radioOp_t *command = (rfc_radioOp_t *)cmd;
196  uint32_t timeout_cnt = 0;
197 
198  /*
199  * 0xn4nn=DONE, 0x0400=DONE_OK while all other "DONE" values means done
200  * but with some kind of error (ref. "Common radio operation status codes")
201  */
202  do {
203  if(++timeout_cnt > 500000) {
204  return RF_CORE_CMD_ERROR;
205  }
206  } while((command->status & RF_CORE_RADIO_OP_MASKED_STATUS)
207  != RF_CORE_RADIO_OP_MASKED_STATUS_DONE);
208 
209  return (command->status & RF_CORE_RADIO_OP_MASKED_STATUS)
210  == RF_CORE_RADIO_OP_STATUS_DONE_OK;
211 }

213 static int
214 fs_powerdown(void)
215 {
216  rfc_CMD_FS_POWERDOWN_t cmd;
217  uint32_t cmd_status;
218 
219  rf_core_init_radio_op((rfc_radioOp_t *)&cmd, sizeof(cmd), CMD_FS_POWERDOWN);
220 
221  if(rf_core_send_cmd((uint32_t)&cmd, &cmd_status) != RF_CORE_CMD_OK) {
222  PRINTF("fs_powerdown: CMDSTA=0x%08lx\n", cmd_status);
223  return RF_CORE_CMD_ERROR;
224  }
225 
226  if(rf_core_wait_cmd_done(&cmd) != RF_CORE_CMD_OK) {
227  PRINTF("fs_powerdown: CMDSTA=0x%08lx, status=0x%04x\n",
228  cmd_status, cmd.status);
229  return RF_CORE_CMD_ERROR;
230  }
231 
232  return RF_CORE_CMD_OK;
233 }

235 int
236 rf_core_power_up()
237 {
238  uint32_t cmd_status;
239  bool interrupts_disabled = ti_lib_int_master_disable();
240 
241  ti_lib_int_pend_clear(INT_RFC_CPE_0);
242  ti_lib_int_pend_clear(INT_RFC_CPE_1);
243  ti_lib_int_disable(INT_RFC_CPE_0);
244  ti_lib_int_disable(INT_RFC_CPE_1);
245 
246  /* Enable RF Core power domain */
247  ti_lib_prcm_power_domain_on(PRCM_DOMAIN_RFCORE);
248  while(ti_lib_prcm_power_domain_status(PRCM_DOMAIN_RFCORE)
249  != PRCM_DOMAIN_POWER_ON);
250 
251  ti_lib_prcm_domain_enable(PRCM_DOMAIN_RFCORE);
252  ti_lib_prcm_load_set();
253  while(!ti_lib_prcm_load_get());
254 
255  HWREG(RFC_DBELL_NONBUF_BASE + RFC_DBELL_O_RFCPEIFG) = 0x0;
256  HWREG(RFC_DBELL_NONBUF_BASE + RFC_DBELL_O_RFCPEIEN) = 0x0;
257  ti_lib_int_enable(INT_RFC_CPE_0);
258  ti_lib_int_enable(INT_RFC_CPE_1);
259 
260  if(!interrupts_disabled) {
261  ti_lib_int_master_enable();
262  }
263 
264  /* Let CPE boot */
265  HWREG(RFC_PWR_NONBUF_BASE + RFC_PWR_O_PWMCLKEN) = RF_CORE_CLOCKS_MASK;
266 
267  /* Send ping (to verify RFCore is ready and alive) */
268  if(rf_core_send_cmd(CMDR_DIR_CMD(CMD_PING), &cmd_status) != RF_CORE_CMD_OK) {
269  PRINTF("rf_core_power_up: CMD_PING fail, CMDSTA=0x%08lx\n", cmd_status);
270  return RF_CORE_CMD_ERROR;
271  }
272 
273  return RF_CORE_CMD_OK;
274 }

276 uint8_t
277 rf_core_start_rat(void)
278 {
279  uint32_t cmd_status;
280  rfc_CMD_SYNC_START_RAT_t cmd_start;
281 
282  /* Start radio timer (RAT) */
283  rf_core_init_radio_op((rfc_radioOp_t *)&cmd_start, sizeof(cmd_start), CMD_SYNC_START_RAT);
284 
285  /* copy the value and send back */
286  cmd_start.rat0 = rat_offset;
287 
288  if(rf_core_send_cmd((uint32_t)&cmd_start, &cmd_status) != RF_CORE_CMD_OK) {
289  PRINTF("rf_core_get_rat_rtc_offset: SYNC_START_RAT fail, CMDSTA=0x%08lx\n",
290  cmd_status);
291  return RF_CORE_CMD_ERROR;
292  }
293 
294  /* Wait until done (?) */
295  if(rf_core_wait_cmd_done(&cmd_start) != RF_CORE_CMD_OK) {
296  PRINTF("rf_core_cmd_ok: SYNC_START_RAT wait, CMDSTA=0x%08lx, status=0x%04x\n",
297  cmd_status, cmd_start.status);
298  return RF_CORE_CMD_ERROR;
299  }
300 
301  return RF_CORE_CMD_OK;
302 }
303 /*---------------------------------------------------------------------------*/
304 uint8_t
305 rf_core_stop_rat(void)
306 {
307  rfc_CMD_SYNC_STOP_RAT_t cmd_stop;
308  uint32_t cmd_status;
309 
310  rf_core_init_radio_op((rfc_radioOp_t *)&cmd_stop, sizeof(cmd_stop), CMD_SYNC_STOP_RAT);
311 
312  int ret = rf_core_send_cmd((uint32_t)&cmd_stop, &cmd_status);
313  if(ret != RF_CORE_CMD_OK) {
314  PRINTF("rf_core_get_rat_rtc_offset: SYNC_STOP_RAT fail, ret %d CMDSTA=0x%08lx\n",
315  ret, cmd_status);
316  return ret;
317  }
318 
319  /* Wait until done */
320  ret = rf_core_wait_cmd_done(&cmd_stop);
321  if(ret != RF_CORE_CMD_OK) {
322  PRINTF("rf_core_cmd_ok: SYNC_STOP_RAT wait, CMDSTA=0x%08lx, status=0x%04x\n",
323  cmd_status, cmd_stop.status);
324  return ret;
325  }
326 
327  if(!rat_offset_known) {
328  /* save the offset, but only if this is the first time */
329  rat_offset_known = true;
330  rat_offset = cmd_stop.rat0;
331  }
332 
333  return RF_CORE_CMD_OK;
334 }

336 void
337 rf_core_power_down()
338 {
339  bool interrupts_disabled = ti_lib_int_master_disable();
340  ti_lib_int_disable(INT_RFC_CPE_0);
341  ti_lib_int_disable(INT_RFC_CPE_1);
342 
343  if(rf_core_is_accessible()) {
344  HWREG(RFC_DBELL_NONBUF_BASE + RFC_DBELL_O_RFCPEIFG) = 0x0;
345  HWREG(RFC_DBELL_NONBUF_BASE + RFC_DBELL_O_RFCPEIEN) = 0x0;
346 
347  /* need to send FS_POWERDOWN or analog components will use power */
348  fs_powerdown();
349  }
350 
351  rf_core_stop_rat();
352 
353  /* Shut down the RFCORE clock domain in the MCU VD */
354  ti_lib_prcm_domain_disable(PRCM_DOMAIN_RFCORE);
355  ti_lib_prcm_load_set();
356  while(!ti_lib_prcm_load_get());
357 
358  /* Turn off RFCORE PD */
359  ti_lib_prcm_power_domain_off(PRCM_DOMAIN_RFCORE);
360  while(ti_lib_prcm_power_domain_status(PRCM_DOMAIN_RFCORE)
361  != PRCM_DOMAIN_POWER_OFF);
362 
363  ti_lib_int_pend_clear(INT_RFC_CPE_0);
364  ti_lib_int_pend_clear(INT_RFC_CPE_1);
365  ti_lib_int_enable(INT_RFC_CPE_0);
366  ti_lib_int_enable(INT_RFC_CPE_1);
367  if(!interrupts_disabled) {
368  ti_lib_int_master_enable();
369  }
370 }

436 void
437 rf_core_setup_interrupts(bool poll_mode)
438 {
439  bool interrupts_disabled;
440  const uint32_t enabled_irqs = poll_mode ? ENABLED_IRQS_POLL_MODE : ENABLED_IRQS;
441 
442  /* We are already turned on by the caller, so this should not happen */
443  if(!rf_core_is_accessible()) {
444  PRINTF("setup_interrupts: No access\n");
445  return;
446  }
447 
448  /* Disable interrupts */
449  interrupts_disabled = ti_lib_int_master_disable();
450 
451  /* Set all interrupt channels to CPE0 channel, error to CPE1 */
452  HWREG(RFC_DBELL_NONBUF_BASE + RFC_DBELL_O_RFCPEISL) = ERROR_IRQ;
453 
454  /* Acknowledge configured interrupts */
455  HWREG(RFC_DBELL_NONBUF_BASE + RFC_DBELL_O_RFCPEIEN) = enabled_irqs;
456 
457  /* Clear interrupt flags, active low clear(?) */
458  HWREG(RFC_DBELL_NONBUF_BASE + RFC_DBELL_O_RFCPEIFG) = 0x0;
459 
460  ti_lib_int_pend_clear(INT_RFC_CPE_0);
461  ti_lib_int_pend_clear(INT_RFC_CPE_1);
462  ti_lib_int_enable(INT_RFC_CPE_0);
463  ti_lib_int_enable(INT_RFC_CPE_1);
464 
465  if(!interrupts_disabled) {
466  ti_lib_int_master_enable();
467  }
468 }

578 /*---------------------------------------------------------------------------*/
579 void
580 cc26xx_rf_cpe0_isr(void)
581 {
582  ENERGEST_ON(ENERGEST_TYPE_IRQ);
583 
584  if(!rf_core_is_accessible()) {
585  printf("RF ISR called but RF not ready... PANIC!!\n");
586  if(rf_core_power_up() != RF_CORE_CMD_OK) {
587  PRINTF("rf_core_power_up() failed\n");
588  return;
589  }
590  }
591 
592  ti_lib_int_master_disable();
593 
594  if(HWREG(RFC_DBELL_NONBUF_BASE + RFC_DBELL_O_RFCPEIFG) & RX_FRAME_IRQ) {
595  /* Clear the RX_ENTRY_DONE interrupt flag */
596  HWREG(RFC_DBELL_NONBUF_BASE + RFC_DBELL_O_RFCPEIFG) = 0xFF7FFFFF;
597  process_poll(&rf_core_process);
598  }
