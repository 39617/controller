75 PROCESS_THREAD(remote_grove_gyro_process, ev, data)
76 {
77  PROCESS_BEGIN();
78 
79  uint8_t aux;
80 
81  /* Use Contiki's sensor macro to enable the sensor */
82  SENSORS_ACTIVATE(grove_gyro);
83 
84  /* The sensor itself is in low-power mode, to power on just the sensor and not
85  * the 3 gyroscope axis use GROVE_GYRO_SENSOR. Alternatively the value
86  * GROVE_GYRO_ALL could also be used to power everything at once
87  */
88  grove_gyro.configure(GROVE_GYRO_POWER_ON, GROVE_GYRO_SENSOR);
89 
90  /* Read back the configured sensor I2C address to check if the sensor is
91  * working OK, this is the only case in which the value() returns a value
92  */
93  aux = grove_gyro.value(GROVE_GYRO_ADDR);
94  if(aux == GROVE_GYRO_EXPECTED_ADDR) {
95  printf("Gyro sensor started with addr 0x%02X\n", GROVE_GYRO_EXPECTED_ADDR);
96  } else {
97  printf("Gyro sensor with unrecognized address 0x%02X\n", aux);
98  PROCESS_EXIT();
99  }
100 
101  /* Register the interrupt handler */
102  GROVE_GYRO_REGISTER_INT(gyro_interrupt_callback);
103 
104  /* The gyroscope sensor should be on now but the three gyroscope axis should
105  * be off, to enable a single axis or any combination of the 3 you can use as
106  * argument either GROVE_GYRO_X, GROVE_GYRO_Y, GROVE_GYRO_Z. To enable or
107  * disable the three axis alternatively use GROVE_GYRO_XYZ
108  */
109  grove_gyro.configure(GROVE_GYRO_POWER_ON, GROVE_GYRO_XYZ);
110 
111  /* Calibrate the sensor taking 200 samples every 5ms for the zero-point offset
112  * value, this has to be done manually and is up to the user
113  */
114  grove_gyro.configure(GROVE_GYRO_CALIBRATE_ZERO, 1);
115 
116  /* Enabling the data interrupt will feed data directly to the extern data
117  * structure and notify us about it, depending on the sampling rate and
118  * divisor this could generate many interrupts/second. A zero argument would
119  * disable the interrupts
120  */
121  grove_gyro.configure(GROVE_GYRO_DATA_INTERRUPT, 1);
122 
123  /* And periodically poll the sensor, values are in ยบ/s */
124 
125  while(1) {
126  etimer_set(&et, SENSOR_READ_INTERVAL);
127  PROCESS_WAIT_EVENT_UNTIL(etimer_expired(&et));
128 
129  /* This sensor has a different operation from others using Contiki's sensor
130  * API, to make data acquisition we write the readings directly to the
131  * extern data structure, allowing to write more than 1 value at the same
132  * operation, and also allowing upon a data interrupt event to immediatly
133  * access the data. The return value of the value() call is then the status
134  * result of the read operation
135  */
136  if(grove_gyro.value(GROVE_GYRO_XYZ) == GROVE_GYRO_SUCCESS) {
137 
138  /* Converted values with a 2-digit precision */
139  printf("Gyro: X: %u.%u, Y: %u.%u, Z: %u.%u\n", gyro_values.x / 100,
140  gyro_values.x % 100,
141  gyro_values.y / 100,
142  gyro_values.y % 100,
143  gyro_values.z / 100,
144  gyro_values.z % 100);
145  } else {
146  printf("Error, enable the DEBUG flag in the grove-gyro driver for info,");
147  printf(" or check if the sensor is properly connected\n");
148  PROCESS_EXIT();
149  }
150 
151  if(grove_gyro.value(GROVE_GYRO_TEMP) == GROVE_GYRO_SUCCESS) {
152  printf("Gyro: temperature %d.%02u C\n", gyro_values.temp / 100,
153  gyro_values.temp % 100);

155  printf("Error, enable the DEBUG flag in the grove-gyro driver for info,");
156  printf(" or check if the sensor is properly connected\n");
157  PROCESS_EXIT();
158  }
159  }
160  PROCESS_END();
161 }
