359 get_cca_info(void)
360 {
361  uint32_t cmd_status;
362  int8_t rssi;
363  rfc_CMD_IEEE_CCA_REQ_t cmd;

396 static radio_value_t
397 get_rssi(void)
398 {
399  uint32_t cmd_status;
400  uint8_t was_off = 0;
401  rfc_CMD_IEEE_CCA_REQ_t cmd;
402 
403  /* If we are off, turn on first */
404  if(!rf_is_on()) {
405  was_off = 1;
406  if(on() != RF_CORE_CMD_OK) {
407  PRINTF("get_rssi: on() failed\n");
408  return RF_CMD_CCA_REQ_RSSI_UNKNOWN;
409  }
410  }
411 
412  memset(&cmd, 0x00, sizeof(cmd));
413  cmd.ccaInfo.ccaEnergy = RF_CMD_CCA_REQ_CCA_STATE_INVALID;
414 
415  while(cmd.ccaInfo.ccaEnergy == RF_CMD_CCA_REQ_CCA_STATE_INVALID) {
416  memset(&cmd, 0x00, sizeof(cmd));
417  cmd.commandNo = CMD_IEEE_CCA_REQ;
418 
419  if(rf_core_send_cmd((uint32_t)&cmd, &cmd_status) == RF_CORE_CMD_ERROR) {
420  PRINTF("get_rssi: CMDSTA=0x%08lx\n", cmd_status);
421 
422  /* Make sure to return RSSI unknown */
423  cmd.currentRssi = RF_CMD_CCA_REQ_RSSI_UNKNOWN;
424  break;
425  }
426  }
427 
428  /* If we were off, turn back off */
429  if(was_off) {
430  off();
431  }
432 
433  return cmd.currentRssi;
434 }

443 /*
444  * Set TX power to 'at least' power dBm
445  * This works with a lookup table. If the value of 'power' does not exist in
446  * the lookup table, TXPOWER will be set to the immediately higher available
447  * value
448  */
449 static void
450 set_tx_power(radio_value_t power)
451 {
452  uint32_t cmd_status;
453  int i;
454  rfc_CMD_SET_TX_POWER_t cmd;
455 
456  /* First, find the correct setting and save it */
457  for(i = OUTPUT_CONFIG_COUNT - 1; i >= 0; --i) {
458  if(power <= output_power[i].dbm) {
459  tx_power_current = &output_power[i];
460  break;
461  }
462  }
463 
464  /*
465  * If the core is not accessible, the new setting will be applied next
466  * time we send CMD_RADIO_SETUP, so we don't need to do anything further.
467  * If the core is accessible, we can apply the new setting immediately with
468  * CMD_SET_TX_POWER
469  */
470  if(rf_core_is_accessible() == RF_CORE_NOT_ACCESSIBLE) {
471  return;
472  }
473 
474  memset(&cmd, 0x00, sizeof(cmd));
475  cmd.commandNo = CMD_SET_TX_POWER;
476  cmd.txPower.IB = output_power[i].register_ib;
477  cmd.txPower.GC = output_power[i].register_gc;
478  cmd.txPower.tempCoeff = output_power[i].temp_coeff;
479 
480  if(rf_core_send_cmd((uint32_t)&cmd, &cmd_status) == RF_CORE_CMD_ERROR) {
481  PRINTF("set_tx_power: CMDSTA=0x%08lx\n", cmd_status);
482  }
483 }

776 static uint8_t
777 check_rat_overflow(bool first_time)
778 {
779  static uint32_t last_value;
780  uint32_t current_value;
781  uint8_t interrupts_disabled;
782 
783  /* Bail out if the RF is not on */
784  if(!rf_is_on()) {
785  return 0;
786  }
787 
788  interrupts_disabled = ti_lib_int_master_disable();
789  if(first_time) {
790  last_value = HWREG(RFC_RAT_BASE + RATCNT);
791  } else {
792  current_value = HWREG(RFC_RAT_BASE + RATCNT);
793  if(current_value + RAT_RANGE / 4 < last_value) {
794  /* Overflow detected */
795  last_rat_overflow = RTIMER_NOW();
796  rat_overflow_counter++;
797  }
798  last_value = current_value;
799  }
800  if(!interrupts_disabled) {
801  ti_lib_int_master_enable();
802  }
803  return 1;
804 }

806 static void
807 handle_rat_overflow(void *unused)
808 {
809  uint8_t success;
810  uint8_t was_off = 0;
811 
812  if(!rf_is_on()) {
813  was_off = 1;
814  if(on() != RF_CORE_CMD_OK) {
815  PRINTF("overflow: on() failed\n");
816  ctimer_set(&rat_overflow_timer, CLOCK_SECOND,
817  handle_rat_overflow, NULL);
818  return;
819  }
820  }
821 
822  success = check_rat_overflow(false);
823 
824  if(was_off) {
825  off();
826  }
827 
828  if(success) {
829  /* Retry after half of the interval */
830  ctimer_set(&rat_overflow_timer, RAT_OVERFLOW_PERIOD_SECONDS * CLOCK_SECOND / 2,
831  handle_rat_overflow, NULL);
832  } else {
833  /* Retry sooner */
834  ctimer_set(&rat_overflow_timer, CLOCK_SECOND,
835  handle_rat_overflow, NULL);
836  }
837 }

888 /*---------------------------------------------------------------------------*/
889 static int
890 transmit(unsigned short transmit_len)
891 {
892  int ret;
893  uint8_t was_off = 0;
894  uint32_t cmd_status;
895  uint16_t stat;
896  uint8_t tx_active = 0;
897  rtimer_clock_t t0;
898  volatile rfc_CMD_IEEE_TX_t cmd;
899 
900  if(!rf_is_on()) {
901  was_off = 1;
902  if(on() != RF_CORE_CMD_OK) {
903  PRINTF("transmit: on() failed\n");
904  return RADIO_TX_ERR;
905  }
906  }
907 
908  /*
909  * We are certainly not TXing a frame as a result of CMD_IEEE_TX, but we may
910  * be in the process of TXing an ACK. In that case, wait for the TX to finish
911  * or return after approx TX_WAIT_TIMEOUT
912  */
913  t0 = RTIMER_NOW();
914 
915  do {
916  tx_active = transmitting();
917  } while(tx_active == 1 &&
918  (RTIMER_CLOCK_LT(RTIMER_NOW(), t0 + TX_WAIT_TIMEOUT)));
919 
920  if(tx_active) {
921  PRINTF("transmit: Already TXing and wait timed out\n");
922 
923  if(was_off) {
924  off();
925  }
926 
927  return RADIO_TX_COLLISION;
928  }
929 
930  /* Send the CMD_IEEE_TX command */
931  rf_core_init_radio_op((rfc_radioOp_t *)&cmd, sizeof(cmd), CMD_IEEE_TX);
932 
933  cmd.payloadLen = transmit_len;
934  cmd.pPayload = &tx_buf[TX_BUF_HDR_LEN];
935 
936  cmd.startTime = 0;
937  cmd.startTrigger.triggerType = TRIG_NOW;
938 
939  /* Enable the LAST_FG_COMMAND_DONE interrupt, which will wake us up */
940  rf_core_cmd_done_en(true, poll_mode);
941 
942  ret = rf_core_send_cmd((uint32_t)&cmd, &cmd_status);
943 
944  if(ret) {
945  /* If we enter here, TX actually started */
946  ENERGEST_OFF(ENERGEST_TYPE_LISTEN);
947  ENERGEST_ON(ENERGEST_TYPE_TRANSMIT);
948 
949  /* Idle away while the command is running */
950  while((cmd.status & RF_CORE_RADIO_OP_MASKED_STATUS)
951  == RF_CORE_RADIO_OP_MASKED_STATUS_RUNNING) {
952  /* Note: for now sleeping while Tx'ing in polling mode is disabled.
953  * To enable it:
954  * 1) make the `lpm_sleep()` call here unconditional;
955  * 2) change the radio ISR priority to allow radio ISR to interrupt rtimer ISR.
956  */
957  if(!poll_mode) {
958  lpm_sleep();
959  }
960  }
961 
962  stat = cmd.status;
963 
964  if(stat == RF_CORE_RADIO_OP_STATUS_IEEE_DONE_OK) {
965  /* Sent OK */
966  RIMESTATS_ADD(lltx);
967  ret = RADIO_TX_OK;
968  } else {
969  /* Operation completed, but frame was not sent */
970  PRINTF("transmit: ret=%d, CMDSTA=0x%08lx, status=0x%04x\n", ret,
971  cmd_status, stat);
972  ret = RADIO_TX_ERR;
973  }
974  } else {
975  /* Failure sending the CMD_IEEE_TX command */
976  PRINTF("transmit: ret=%d, CMDSTA=0x%08lx, status=0x%04x\n",
977  ret, cmd_status, cmd.status);
978 
979  ret = RADIO_TX_ERR;
980  }
981 
982  /*
983  * Update ENERGEST state here, before a potential call to off(), which
984  * will correctly update it if required.
985  */
986  ENERGEST_OFF(ENERGEST_TYPE_TRANSMIT);
987  ENERGEST_ON(ENERGEST_TYPE_LISTEN);
988 
989  /*
990  * Disable LAST_FG_COMMAND_DONE interrupt. We don't really care about it
991  * except when we are transmitting
992  */
993  rf_core_cmd_done_dis(poll_mode);
994 
995  if(was_off) {
996  off();
997  }
998 
999  return ret;
1000 }

1022 static uint32_t
1023 calc_last_packet_timestamp(uint32_t rat_timestamp)
1024 {
1025  uint64_t rat_timestamp64;
1026  uint32_t adjusted_overflow_counter;
1027  uint8_t was_off = 0;
1028 
1029  if(!rf_is_on()) {
1030  was_off = 1;
1031  on();
1032  }
1033 
1034  if(rf_is_on()) {
1035  check_rat_overflow(false);
1036  if(was_off) {
1037  off();
1038  }
1039  }
1040 
1041  adjusted_overflow_counter = rat_overflow_counter;
1042 
1043  /* if the timestamp is large and the last oveflow was recently,
1044  assume that the timestamp refers to the time before the overflow */
1045  if(rat_timestamp > (uint32_t)(RAT_RANGE * 3 / 4)) {
1046  if(RTIMER_CLOCK_LT(RTIMER_NOW(),
1047  last_rat_overflow + RAT_OVERFLOW_PERIOD_SECONDS * RTIMER_SECOND / 4)) {
1048  adjusted_overflow_counter--;
1049  }
1050  }
1051 
1052  /* add the overflowed time to the timestamp */
1053  rat_timestamp64 = rat_timestamp + RAT_RANGE * adjusted_overflow_counter;
1054  /* correct timestamp so that it refers to the end of the SFD */
1055  rat_timestamp64 += TIMESTAMP_OFFSET;
1056 
1057  last_rat_timestamp64 = rat_timestamp64 - rat_offset;
1058 
1059  return RADIO_TO_RTIMER(rat_timestamp64 - rat_offset);
1060 }
1061 /*---------------------------------------------------------------------------*/
1062 static int
1063 read_frame(void *buf, unsigned short buf_len)
1064 {
1065  int len = 0;
1066  rfc_dataEntryGeneral_t *entry = (rfc_dataEntryGeneral_t *)rx_read_entry;
1067  uint32_t rat_timestamp;
1068 
1069  /* wait for entry to become finished */
1070  rtimer_clock_t t0 = RTIMER_NOW();
1071  while(entry->status == DATA_ENTRY_STATUS_BUSY
1072  && RTIMER_CLOCK_LT(RTIMER_NOW(), t0 + (RTIMER_SECOND / 250)));
1073 
1074  if(entry->status != DATA_ENTRY_STATUS_FINISHED) {
1075  /* No available data */
1076  return 0;
1077  }
1078 
1079  if(rx_read_entry[8] < 4) {
1080  PRINTF("RF: too short\n");
1081  RIMESTATS_ADD(tooshort);
1082 
1083  release_data_entry();
1084  return 0;
1085  }
1086 
1087  len = rx_read_entry[8] - 8;
1088 
1089  if(len > buf_len) {
1090  PRINTF("RF: too long\n");
1091  RIMESTATS_ADD(toolong);
1092 
1093  release_data_entry();
1094  return 0;
1095  }
1096 
1097  memcpy(buf, (char *)&rx_read_entry[9], len);
1098 
1099  last_rssi = (int8_t)rx_read_entry[9 + len + 2];
1100  last_corr_lqi = (uint8_t)rx_read_entry[9 + len + 3] & STATUS_CORRELATION;
1101 
1102  /* get the timestamp */
1103  memcpy(&rat_timestamp, (char *)rx_read_entry + 9 + len + 4, 4);
1104 
1105  last_packet_timestamp = calc_last_packet_timestamp(rat_timestamp);
1106 
1107  if(!poll_mode) {
1108  /* Not in poll mode: packetbuf should not be accessed in interrupt context.
1109  * In poll mode, the last packet RSSI and link quality can be obtained through
1110  * RADIO_PARAM_LAST_RSSI and RADIO_PARAM_LAST_LINK_QUALITY */
1111  packetbuf_set_attr(PACKETBUF_ATTR_RSSI, last_rssi);
1112  packetbuf_set_attr(PACKETBUF_ATTR_LINK_QUALITY, last_corr_lqi);
1113  }
1114  RIMESTATS_ADD(llrx);
1115 
1116  release_data_entry();
1117 
1118  return len;
1119 }

1121 static int
1122 channel_clear(void)
1123 {
1124  uint8_t was_off = 0;
1125  uint8_t cca_info;
1126  int ret = RF_CCA_CLEAR;
1127 
1128  /*
1129  * If we are in the middle of a BLE operation, we got called by ContikiMAC
1130  * from within an interrupt context. Indicate a clear channel
1131  */
1132  if(rf_ble_is_active() == RF_BLE_ACTIVE) {
1133  PRINTF("channel_clear: Interrupt context but BLE in progress\n");
1134  return RF_CCA_CLEAR;
1135  }
1136 
1137  if(rf_is_on()) {
1138  /*
1139  * Wait for potential leftover ACK still being sent.
1140  * Strictly speaking, if we are TXing an ACK then the channel is not clear.
1141  * However, channel_clear is only ever called to determine whether there is
1142  * someone else's packet in the air, not ours.
1143  *
1144  * We could probably even simply return that the channel is clear
1145  */
1146  while(transmitting());
1147  } else {
1148  was_off = 1;
1149  if(on() != RF_CORE_CMD_OK) {
1150  PRINTF("channel_clear: on() failed\n");
1151  if(was_off) {
1152  off();
1153  }
1154  return RF_CCA_CLEAR;
1155  }
1156  }
1157 
1158  cca_info = get_cca_info();
1159 
1160  if(cca_info == RF_GET_CCA_INFO_ERROR) {
1161  PRINTF("channel_clear: CCA error\n");
1162  ret = RF_CCA_CLEAR;
1163  } else {
1164  /*
1165  * cca_info bits 1:0 - ccaStatus
1166  * Return 1 (clear) if idle or invalid.
1167  */
1168  ret = (cca_info & 0x03) != RF_CMD_CCA_REQ_CCA_STATE_BUSY;
1169  }
1170 
1171  if(was_off) {
1172  off();
1173  }
1174 
1175  return ret;
1176 }

1219 static int
1220 pending_packet(void)
1221 {
1222  volatile rfc_dataEntry_t *entry = (rfc_dataEntry_t *)rx_data_queue.pCurrEntry;
1223  int rv = 0;
1224 
1225  /* Go through all RX buffers and check their status */
1226  do {
1227  if(entry->status == DATA_ENTRY_STATUS_FINISHED
1228  || entry->status == DATA_ENTRY_STATUS_BUSY) {
1229  rv = 1;
1230  if(!poll_mode) {
1231  process_poll(&rf_core_process);
1232  }
1233  }
1234 
1235  entry = (rfc_dataEntry_t *)entry->pNextEntry;
1236  } while(entry != (rfc_dataEntry_t *)rx_data_queue.pCurrEntry);
1237

1239  return rv;
1240 }

1243 on(void)
1244 {

1246  * If we are in the middle of a BLE operation, we got called by ContikiMAC
1247  * from within an interrupt context. Abort, but pretend everything is OK.
1248  */
1249  if(rf_ble_is_active() == RF_BLE_ACTIVE) {
1250  PRINTF("on: Interrupt context but BLE in progress\n");
1251  return RF_CORE_CMD_OK;
1252  }

1255  * Request the HF XOSC as the source for the HF clock. Needed before we can
1256  * use the FS. This will only request, it will _not_ perform the switch.
1257  */

1260  if(rf_is_on()) {
1261  PRINTF("on: We were on. PD=%u, RX=0x%04x \n", rf_core_is_accessible(),
1262  RF_RADIO_OP_GET_STATUS(cmd_ieee_rx_buf));
1263  return RF_CORE_CMD_OK;
1264  }

1267

1269  * Trigger a switch to the XOSC, so that we can subsequently use the RF FS
1270  * This will block until the XOSC is actually ready, but give how we
1271  * requested it early on, this won't be too long a wait.
1272  * This should be done before starting the RAT.
1273  */

1282

1284  PRINTF("on: radio_setup() failed\n");

1286  }

1289 }
