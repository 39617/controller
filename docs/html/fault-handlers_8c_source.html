38 fault_handlers_hard_fault_isr(void)
39 {
40  /*
41  * Workaround for (Im)precise Bus Faults caused under unknown circumstances,
42  * likely by access to RFC registers while the RF PD is off (which is
43  * something that should never happen because we do in fact check before
44  * accessing)
45  */
46  if((HWREG(CPU_SCS_BASE + CPU_SCS_O_CFSR) == CPU_SCS_CFSR_IMPRECISERR) ||
47  (HWREG(CPU_SCS_BASE + CPU_SCS_O_CFSR) & CPU_SCS_CFSR_PRECISERR)){
48  /* ToDo: Check BFARVALID and then BFAR to filter down even further */
49  return;
50  }
51 
52  while(1);
53 }
