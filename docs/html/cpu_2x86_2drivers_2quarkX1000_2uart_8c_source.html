59 quarkX1000_uart_init_port(quarkX1000_uart_dev_t dev, unsigned baud)
60 {
61  uint16_t dl;
62  pci_config_addr_t pci_addr;
63  uart_16x50_driver_t ATTR_KERN_ADDR_SPACE *drv;
64 
65  assert((dev == QUARK_X1000_UART_0) || (dev == QUARK_X1000_UART_1));
66 
67  pci_addr.raw = 0;
68 
69  /* PCI addresses from section 18.4 of Intel Quark SoC X1000 Datasheet. */
70  pci_addr.dev = 20;
71  pci_addr.func = (dev == QUARK_X1000_UART_0) ? 1 : 5;
72  pci_addr.reg_off = PCI_CONFIG_REG_BAR0;
73 
74  if(dev == QUARK_X1000_UART_0) {
75  drv = &quarkX1000_uart0;
76  PROT_DOMAINS_INIT_ID(quarkX1000_uart0);
77  } else {
78  drv = &quarkX1000_uart1;
79  PROT_DOMAINS_INIT_ID(quarkX1000_uart1);
80  }
81  /* Divisor setting from section 18.2.2 of Intel Quark SoC X1000 Datasheet. */
82  dl = QUARK_X1000_UART_FBASE / (16 * baud);
83  uart_16x50_init_port(drv, pci_addr, dl);
84 }
