182 static radio_result_t
183 get_value(radio_param_t param, radio_value_t *value)
184 {
185  int i, v;
186 
187  if(!value) {
188  return RADIO_RESULT_INVALID_VALUE;
189  }
190  switch(param) {
191  case RADIO_PARAM_POWER_MODE:
192  *value = receive_on ? RADIO_POWER_MODE_ON : RADIO_POWER_MODE_OFF;
193  return RADIO_RESULT_OK;
194  case RADIO_PARAM_CHANNEL:
195  *value = cc2420_get_channel();
196  return RADIO_RESULT_OK;
197  case RADIO_PARAM_RX_MODE:
198  *value = 0;
199  if(getreg(CC2420_MDMCTRL0) & ADR_DECODE) {
200  *value |= RADIO_RX_MODE_ADDRESS_FILTER;
201  }
202  if(getreg(CC2420_MDMCTRL0) & AUTOACK) {
203  *value |= RADIO_RX_MODE_AUTOACK;
204  }
205  if(poll_mode) {
206  *value |= RADIO_RX_MODE_POLL_MODE;
207  }
208  return RADIO_RESULT_OK;
209  case RADIO_PARAM_TX_MODE:
210  *value = 0;
211  if(send_on_cca) {
212  *value |= RADIO_TX_MODE_SEND_ON_CCA;
213  }
214  return RADIO_RESULT_OK;
215  case RADIO_PARAM_TXPOWER:
216  v = cc2420_get_txpower();
217  *value = OUTPUT_POWER_MIN;
218  /* Find the actual estimated output power in conversion table */
219  for(i = 0; i < OUTPUT_NUM; i++) {
220  if(v >= output_power[i].config) {
221  *value = output_power[i].power;
222  break;
223  }
224  }
225  return RADIO_RESULT_OK;
226  case RADIO_PARAM_CCA_THRESHOLD:
227  *value = get_cca_threshold() + RSSI_OFFSET;
228  return RADIO_RESULT_OK;
229  case RADIO_PARAM_RSSI:
230  /* Return the RSSI value in dBm */
231  *value = cc2420_rssi();
232  return RADIO_RESULT_OK;
233  case RADIO_PARAM_LAST_RSSI:
234  /* RSSI of the last packet received */
235  *value = cc2420_last_rssi;
236  return RADIO_RESULT_OK;
237  case RADIO_PARAM_LAST_LINK_QUALITY:
238  /* LQI of the last packet received */
239  *value = cc2420_last_correlation;
240  return RADIO_RESULT_OK;
241  case RADIO_CONST_CHANNEL_MIN:
242  *value = 11;
243  return RADIO_RESULT_OK;
244  case RADIO_CONST_CHANNEL_MAX:
245  *value = 26;
246  return RADIO_RESULT_OK;
247  case RADIO_CONST_TXPOWER_MIN:
248  *value = OUTPUT_POWER_MIN;
249  return RADIO_RESULT_OK;
250  case RADIO_CONST_TXPOWER_MAX:
251  *value = OUTPUT_POWER_MAX;
252  return RADIO_RESULT_OK;
253  default:
254  return RADIO_RESULT_NOT_SUPPORTED;
255  }
256 }

258 static radio_result_t
259 set_value(radio_param_t param, radio_value_t value)
260 {
261  int i;
262 
263  switch(param) {
264  case RADIO_PARAM_POWER_MODE:
265  if(value == RADIO_POWER_MODE_ON) {
266  cc2420_on();
267  return RADIO_RESULT_OK;
268  }
269  if(value == RADIO_POWER_MODE_OFF) {
270  cc2420_off();
271  return RADIO_RESULT_OK;
272  }
273  return RADIO_RESULT_INVALID_VALUE;
274  case RADIO_PARAM_CHANNEL:
275  if(value < 11 || value > 26) {
276  return RADIO_RESULT_INVALID_VALUE;
277  }
278  cc2420_set_channel(value);
279  return RADIO_RESULT_OK;
280  case RADIO_PARAM_RX_MODE:
281  if(value & ~(RADIO_RX_MODE_ADDRESS_FILTER |
282  RADIO_RX_MODE_AUTOACK | RADIO_RX_MODE_POLL_MODE)) {
283  return RADIO_RESULT_INVALID_VALUE;
284  }
285  set_frame_filtering((value & RADIO_RX_MODE_ADDRESS_FILTER) != 0);
286  set_auto_ack((value & RADIO_RX_MODE_AUTOACK) != 0);
287  set_poll_mode((value & RADIO_RX_MODE_POLL_MODE) != 0);
288  return RADIO_RESULT_OK;
289  case RADIO_PARAM_TX_MODE:
290  if(value & ~(RADIO_TX_MODE_SEND_ON_CCA)) {
291  return RADIO_RESULT_INVALID_VALUE;
292  }
293  set_send_on_cca((value & RADIO_TX_MODE_SEND_ON_CCA) != 0);
294  return RADIO_RESULT_OK;
295  case RADIO_PARAM_TXPOWER:
296  if(value < OUTPUT_POWER_MIN || value > OUTPUT_POWER_MAX) {
297  return RADIO_RESULT_INVALID_VALUE;
298  }
299  /* Find the closest higher PA_LEVEL for the desired output power */
300  for(i = 1; i < OUTPUT_NUM; i++) {
301  if(value > output_power[i].power) {
302  break;
303  }
304  }
305  cc2420_set_txpower(output_power[i - 1].config);
306  return RADIO_RESULT_OK;
307  case RADIO_PARAM_CCA_THRESHOLD:
308  cc2420_set_cca_threshold(value - RSSI_OFFSET);
309  return RADIO_RESULT_OK;
310  default:
311  return RADIO_RESULT_NOT_SUPPORTED;
312  }
313 }

365 /*---------------------------------------------------------------------------*/
366 /* Reads a register */
367 static uint16_t
368 getreg(enum cc2420_register regname)
369 {
370  uint16_t value;
371 
372  CC2420_SPI_ENABLE();
373  SPI_WRITE(regname | 0x40);
374  value = (uint8_t)SPI_RXBUF;
375  SPI_TXBUF = 0;
376  SPI_WAITFOREORx();
377  value = SPI_RXBUF << 8;
378  SPI_TXBUF = 0;
379  SPI_WAITFOREORx();
380  value |= SPI_RXBUF;
381  CC2420_SPI_DISABLE();
382 
383  return value;
384 }
385 /*---------------------------------------------------------------------------*/
391 static void
392 setreg(enum cc2420_register regname, uint16_t value)
393 {
394  CC2420_SPI_ENABLE();
395  SPI_WRITE_FAST(regname);
396  SPI_WRITE_FAST((uint8_t) (value >> 8));
397  SPI_WRITE_FAST((uint8_t) (value & 0xff));
398  SPI_WAITFORTx_ENDED();
399  SPI_WRITE(0);
400  CC2420_SPI_DISABLE();
401 }
402 /*---------------------------------------------------------------------------*/
403 static void
404 read_ram(uint8_t *buffer, uint16_t adr, uint16_t count)
405 {
406  uint8_t i;
407 
408  CC2420_SPI_ENABLE();
409  SPI_WRITE(0x80 | ((adr) & 0x7f));
410  SPI_WRITE((((adr) >> 1) & 0xc0) | 0x20);
411  SPI_RXBUF;
412  for(i = 0; i < count; i++) {
413  SPI_READ(((uint8_t*) buffer)[i]);
414  }
415  CC2420_SPI_DISABLE();
416 }
417 /*---------------------------------------------------------------------------*/
418 /* Write to RAM in the CC2420 */
419 static void
420 write_ram(const uint8_t *buffer,
421  uint16_t adr,
422  uint16_t count,
423  enum write_ram_order order)
424 {
425  uint8_t i;
426 
427  CC2420_SPI_ENABLE();
428  SPI_WRITE_FAST(0x80 | (adr & 0x7f));
429  SPI_WRITE_FAST((adr >> 1) & 0xc0);
430  if(order == WRITE_RAM_IN_ORDER) {
431  for(i = 0; i < count; i++) {
432  SPI_WRITE_FAST((buffer)[i]);
433  }
434  } else {
435  for(i = count; i > 0; i--) {
436  SPI_WRITE_FAST((buffer)[i - 1]);
437  }
438  }
439  SPI_WAITFORTx_ENDED();
440  CC2420_SPI_DISABLE();
441 }

499 static void
500 wait_for_status(uint8_t status_bit)
501 {
502  rtimer_clock_t t0;
503  t0 = RTIMER_NOW();
504  while(!(get_status() & status_bit)
505  && RTIMER_CLOCK_LT(RTIMER_NOW(), t0 + (RTIMER_SECOND / 10)));
506 }
507 /*---------------------------------------------------------------------------*/
508 static void
509 wait_for_transmission(void)
510 {
511  rtimer_clock_t t0;
512  t0 = RTIMER_NOW();
513  while((get_status() & BV(CC2420_TX_ACTIVE))
514  && RTIMER_CLOCK_LT(RTIMER_NOW(), t0 + (RTIMER_SECOND / 10)));
515 }
516 /*---------------------------------------------------------------------------*/
517 static void
518 on(void)
519 {
520  if(!poll_mode) {
521  CC2420_ENABLE_FIFOP_INT();
522  }
523 
524  strobe(CC2420_SRXON);
525 
526  ENERGEST_ON(ENERGEST_TYPE_LISTEN);
527  receive_on = 1;
528 }
529 /*---------------------------------------------------------------------------*/
530 static void
531 off(void)
532 {
533  /* PRINTF("off\n");*/
534  receive_on = 0;
535 
536  /* Wait for transmission to end before turning radio off. */
537  wait_for_transmission();
538 
539  ENERGEST_OFF(ENERGEST_TYPE_LISTEN);
540  strobe(CC2420_SRFOFF);
541  if(!poll_mode) {
542  CC2420_DISABLE_FIFOP_INT();
543  }
544 
545  if(!CC2420_FIFOP_IS_1) {
546  flushrx();
547  }
548 }
549 /*---------------------------------------------------------------------------*/
550 static uint8_t locked, lock_on, lock_off;
551 #define GET_LOCK() locked++
552 static void RELEASE_LOCK(void) {
553  if(locked == 1) {
554  if(lock_on) {
555  on();
556  lock_on = 0;
557  }
558  if(lock_off) {
559  off();
560  lock_off = 0;
561  }
562  }
563  locked--;
564 }

585 encrypt(uint8_t *plaintext_and_result)
586 {

588 
589  write_ram(plaintext_and_result,
590  CC2420RAM_SABUF,

608 set_txpower(uint8_t power)

616 /*---------------------------------------------------------------------------*/
617 int
618 cc2420_init(void)
619 {
620  uint16_t reg;
621  {
622  int s = splhigh();
623  cc2420_arch_init(); /* Initalize ports and SPI. */
624  CC2420_DISABLE_FIFOP_INT();
625  CC2420_FIFOP_INT_INIT();
626  splx(s);
627  }
628 
629  /* Turn on voltage regulator and reset. */
630  SET_VREG_ACTIVE();
631  clock_delay(250);
632  SET_RESET_ACTIVE();
633  clock_delay(127);
634  SET_RESET_INACTIVE();
635  clock_delay(125);
636 
637 
638  /* Turn on the crystal oscillator. */
639  strobe(CC2420_SXOSCON);
640  /* And wait until it stabilizes */
641  wait_for_status(BV(CC2420_XOSC16M_STABLE));
642 
643  /* Set auto-ack and frame filtering */
644  set_auto_ack(CC2420_CONF_AUTOACK);
645  set_frame_filtering(CC2420_CONF_AUTOACK);
646 
647  /* Enabling CRC in hardware; this is required by AUTOACK anyway
648  and provides us with RSSI and link quality indication (LQI)
649  information. */
650  reg = getreg(CC2420_MDMCTRL0);
651  reg |= AUTOCRC;
652  setreg(CC2420_MDMCTRL0, reg);
653 
654  /* Set transmission turnaround time to the lower setting (8 symbols
655  = 0.128 ms) instead of the default (12 symbols = 0.192 ms). */
656  /* reg = getreg(CC2420_TXCTRL);
657  reg &= ~(1 << 13);
658  setreg(CC2420_TXCTRL, reg);*/
659 
660 
661  /* Change default values as recomended in the data sheet, */
662  /* correlation threshold = 20, RX bandpass filter = 1.3uA. */
663  setreg(CC2420_MDMCTRL1, CORR_THR(20));
664  reg = getreg(CC2420_RXCTRL1);
665  reg |= RXBPF_LOCUR;
666  setreg(CC2420_RXCTRL1, reg);
667 
668  /* Set the FIFOP threshold to maximum. */
669  setreg(CC2420_IOCFG0, FIFOP_THR(127));
670 
671  init_security();
672 
673  cc2420_set_pan_addr(0xffff, 0x0000, NULL);
674  cc2420_set_channel(CC2420_CONF_CHANNEL);
675  cc2420_set_cca_threshold(CC2420_CONF_CCA_THRESH);
676 
677  flushrx();
678 
679  set_poll_mode(0);
680 
681  process_start(&cc2420_process, NULL);
682  return 1;
683 }
684 /*---------------------------------------------------------------------------*/
685 static int
686 cc2420_transmit(unsigned short payload_len)
687 {
688  int i, txpower;
689 
690  GET_LOCK();
691 
692  txpower = 0;
693  if(packetbuf_attr(PACKETBUF_ATTR_RADIO_TXPOWER) > 0) {
694  /* Remember the current transmission power */
695  txpower = cc2420_get_txpower();
696  /* Set the specified transmission power */
697  set_txpower(packetbuf_attr(PACKETBUF_ATTR_RADIO_TXPOWER) - 1);
698  }
699 
700  /* The TX FIFO can only hold one packet. Make sure to not overrun
701  * FIFO by waiting for transmission to start here and synchronizing
702  * with the CC2420_TX_ACTIVE check in cc2420_send.
703  *
704  * Note that we may have to wait up to 320 us (20 symbols) before
705  * transmission starts.
706  */
707 #ifndef CC2420_CONF_SYMBOL_LOOP_COUNT
708 #error CC2420_CONF_SYMBOL_LOOP_COUNT needs to be set!!!
709 #else
710 #define LOOP_20_SYMBOLS CC2420_CONF_SYMBOL_LOOP_COUNT
711 #endif
712 
713  if(send_on_cca) {
714  strobe(CC2420_SRXON);
715  wait_for_status(BV(CC2420_RSSI_VALID));
716  strobe(CC2420_STXONCCA);
717  } else {
718  strobe(CC2420_STXON);
719  }
720  for(i = LOOP_20_SYMBOLS; i > 0; i--) {
721  if(CC2420_SFD_IS_1) {
722 #if PACKETBUF_WITH_PACKET_TYPE
723  {
724  rtimer_clock_t sfd_timestamp;
725  sfd_timestamp = cc2420_sfd_start_time;
726  if(packetbuf_attr(PACKETBUF_ATTR_PACKET_TYPE) ==
727  PACKETBUF_ATTR_PACKET_TYPE_TIMESTAMP) {
728  /* Write timestamp to last two bytes of packet in TXFIFO. */
729  write_ram((uint8_t *) &sfd_timestamp, CC2420RAM_TXFIFO + payload_len - 1, 2, WRITE_RAM_IN_ORDER);
730  }
731  }
732 #endif /* PACKETBUF_WITH_PACKET_TYPE */
733 
734  if(!(get_status() & BV(CC2420_TX_ACTIVE))) {
735  /* SFD went high but we are not transmitting. This means that
736  we just started receiving a packet, so we drop the
737  transmission. */
738  RELEASE_LOCK();
739  return RADIO_TX_COLLISION;
740  }
741  if(receive_on) {
742  ENERGEST_OFF(ENERGEST_TYPE_LISTEN);
743  }
744  ENERGEST_ON(ENERGEST_TYPE_TRANSMIT);
745  /* We wait until transmission has ended so that we get an
746  accurate measurement of the transmission time.*/
747  wait_for_transmission();
748 
749 #ifdef ENERGEST_CONF_LEVELDEVICE_LEVELS
750  ENERGEST_OFF_LEVEL(ENERGEST_TYPE_TRANSMIT,cc2420_get_txpower());
751 #endif
752  ENERGEST_OFF(ENERGEST_TYPE_TRANSMIT);
753  if(receive_on) {
754  ENERGEST_ON(ENERGEST_TYPE_LISTEN);
755  } else {
756  /* We need to explicitly turn off the radio,
757  * since STXON[CCA] -> TX_ACTIVE -> RX_ACTIVE */
758  off();
759  }
760 
761  if(packetbuf_attr(PACKETBUF_ATTR_RADIO_TXPOWER) > 0) {
762  /* Restore the transmission power */
763  set_txpower(txpower & 0xff);
764  }
765 
766  RELEASE_LOCK();
767  return RADIO_TX_OK;
768  }
769  }
770 
771  /* If we send with cca (cca_on_send), we get here if the packet wasn't
772  transmitted because of other channel activity. */
773  RIMESTATS_ADD(contentiondrop);
774  PRINTF("cc2420: do_send() transmission never started\n");
775 
776  if(packetbuf_attr(PACKETBUF_ATTR_RADIO_TXPOWER) > 0) {
777  /* Restore the transmission power */
778  set_txpower(txpower & 0xff);
779  }
780 
781  RELEASE_LOCK();
782  return RADIO_TX_COLLISION;
783 }

951 /*---------------------------------------------------------------------------*/
952 static int
953 cc2420_read(void *buf, unsigned short bufsize)
954 {
955  uint8_t footer[FOOTER_LEN];
956  uint8_t len;
957 
958  if(!CC2420_FIFOP_IS_1) {
959  return 0;
960  }
961 
962  GET_LOCK();
963 
964  getrxdata(&len, 1);
965 
966  if(len > CC2420_MAX_PACKET_LEN) {
967  /* Oops, we must be out of sync. */
968  RIMESTATS_ADD(badsynch);
969  } else if(len <= FOOTER_LEN) {
970  RIMESTATS_ADD(tooshort);
971  } else if(len - FOOTER_LEN > bufsize) {
972  RIMESTATS_ADD(toolong);
973  } else {
974  getrxdata((uint8_t *) buf, len - FOOTER_LEN);
975  getrxdata(footer, FOOTER_LEN);
976 
977  if(footer[1] & FOOTER1_CRC_OK) {
978  cc2420_last_rssi = footer[0] + RSSI_OFFSET;
979  cc2420_last_correlation = footer[1] & FOOTER1_CORRELATION;
980  if(!poll_mode) {
981  /* Not in poll mode: packetbuf should not be accessed in interrupt context.
982  * In poll mode, the last packet RSSI and link quality can be obtained through
983  * RADIO_PARAM_LAST_RSSI and RADIO_PARAM_LAST_LINK_QUALITY */
984  packetbuf_set_attr(PACKETBUF_ATTR_RSSI, cc2420_last_rssi);
985  packetbuf_set_attr(PACKETBUF_ATTR_LINK_QUALITY, cc2420_last_correlation);
986  }
987 
988  RIMESTATS_ADD(llrx);
989  } else {
990  RIMESTATS_ADD(badcrc);
991  len = FOOTER_LEN;
992  }
993 
994  if(!poll_mode) {
995  if(CC2420_FIFOP_IS_1) {
996  if(!CC2420_FIFO_IS_1) {
997  /* Clean up in case of FIFO overflow! This happens for every
998  * full length frame and is signaled by FIFOP = 1 and FIFO =
999  * 0. */
1000  flushrx();
1001  } else {
1002  /* Another packet has been received and needs attention. */
1003  process_poll(&cc2420_process);
1004  }
1005  }
1006  }
1007 
1008  RELEASE_LOCK();
1009  return len - FOOTER_LEN;
1010  }
1011 
1012  flushrx();
1013  RELEASE_LOCK();
1014  return 0;
1015 }

1065 {
1066  int cca;

1068

1095
