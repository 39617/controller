54 PROCESS(ac_dimmer_int_process, "AC Dimmer zero-cross interrupt process");
55 /*---------------------------------------------------------------------------*/
56 PROCESS_THREAD(ac_dimmer_int_process, ev, data)
57 {
58  PROCESS_EXITHANDLER();
59  PROCESS_BEGIN();
60 
61  int dimtime;
62 
63  while(1) {
64  PROCESS_YIELD_UNTIL(ev == PROCESS_EVENT_POLL);
65  dimtime = (uint8_t)(100 - dimming);
66  dimtime *= 100;
67 
68  /* Off cycle */
69  clock_delay_usec(dimtime);
70  GPIO_SET_PIN(DIMMER_GATE_PORT_BASE, DIMMER_GATE_PIN_MASK);
71  /* Triac on propagation delay */
72  clock_delay_usec(DIMMER_DEFAULT_GATE_PULSE_US);
73  GPIO_CLR_PIN(DIMMER_GATE_PORT_BASE, DIMMER_GATE_PIN_MASK);
74  }
75  PROCESS_END();
76 }

79 dimmer_zero_cross_int_handler(uint8_t port, uint8_t pin)
80 {
81  if((port == DIMMER_SYNC_PORT) && (pin == DIMMER_SYNC_PIN)) {

84 }

108 /*---------------------------------------------------------------------------*/
109 static int
110 configure(int type, int value)
111 {
112  if(type != SENSORS_ACTIVE) {
113  return DIMMER_ERROR;
114  }
115 
116  if(value) {
117  /* This is the Triac's gate pin */
118  GPIO_SOFTWARE_CONTROL(DIMMER_GATE_PORT_BASE, DIMMER_GATE_PIN_MASK);
119  GPIO_SET_OUTPUT(DIMMER_GATE_PORT_BASE, DIMMER_GATE_PIN_MASK);
120  ioc_set_over(DIMMER_GATE_PORT, DIMMER_GATE_PIN, IOC_OVERRIDE_OE);
121  GPIO_CLR_PIN(DIMMER_GATE_PORT_BASE, DIMMER_GATE_PIN_MASK);
122 
123  /* This is the zero-crossing pin and interrupt */
124  GPIO_SOFTWARE_CONTROL(DIMMER_SYNC_PORT_BASE, DIMMER_SYNC_PIN_MASK);
125  GPIO_SET_INPUT(DIMMER_SYNC_PORT_BASE, DIMMER_SYNC_PIN_MASK);
126 
127  /* Pull-up resistor, detect rising edge */
128  GPIO_DETECT_EDGE(DIMMER_SYNC_PORT_BASE, DIMMER_SYNC_PIN_MASK);
129  GPIO_TRIGGER_SINGLE_EDGE(DIMMER_SYNC_PORT_BASE, DIMMER_SYNC_PIN_MASK);
130  GPIO_DETECT_RISING(DIMMER_SYNC_PORT_BASE, DIMMER_SYNC_PIN_MASK);
131  gpio_register_callback(dimmer_zero_cross_int_handler, DIMMER_SYNC_PORT,
132  DIMMER_SYNC_PIN);
133 
134  /* Spin process until an interrupt is received */
135  process_start(&ac_dimmer_int_process, NULL);
136 
137  /* Enable interrupts */
138  GPIO_ENABLE_INTERRUPT(DIMMER_SYNC_PORT_BASE, DIMMER_SYNC_PIN_MASK);
139  // ioc_set_over(DIMMER_SYNC_PORT, DIMMER_SYNC_PIN, IOC_OVERRIDE_PUE);
140  NVIC_EnableIRQ(DIMMER_INT_VECTOR);
141 
142  enabled = 1;
143  dimming = DIMMER_DEFAULT_START_VALUE;
144  return DIMMER_SUCCESS;
145  }
146 
147  /* Disable interrupt and pins */
148 
149  GPIO_DISABLE_INTERRUPT(DIMMER_SYNC_PORT_BASE, DIMMER_SYNC_PIN_MASK);
150  GPIO_SET_INPUT(DIMMER_GATE_PORT_BASE, DIMMER_GATE_PIN_MASK);
151  GPIO_SET_OUTPUT(DIMMER_SYNC_PORT_BASE, DIMMER_SYNC_PIN_MASK);
152  process_exit(&ac_dimmer_int_process);
153 
154  enabled = 0;
155  dimming = 0;
156  return DIMMER_SUCCESS;
157 }
