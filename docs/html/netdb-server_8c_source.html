89 static int
90 buffer_db_data(const char *format, ...)
91 {
92  va_list ap;
93  size_t len;
94  char tmp[MAX_BUFFER_SIZE + 1];
95 
96  va_start(ap, format);
97  len = vsnprintf(tmp, sizeof(tmp), format, ap);
98  va_end(ap);
99 
100  if(len < 0) {
101  return -1;
102  }
103 
104  if(len + buffer_offset > sizeof(buffer)) {
105  send_buffered_data();
106  }
107 
108  memcpy(&buffer[buffer_offset], tmp, len);
109  buffer_offset += len;
110 
111  return len;
112 }

138 PROCESS_THREAD(query_process, ev, data)
139 {
140  static db_handle_t handle;
141  db_result_t result;
142  static tuple_id_t matching;
143  static tuple_id_t processed;
144 #if !PREPARE_DB
145  static struct etimer sampling_timer;
146 #endif
147  static unsigned i, errors;
148 
149  PROCESS_BEGIN();
150 
151  printf("NetDB host\n");
152 
153  db_init();
154  db_set_output_function(buffer_db_data);
155 
156  db_query(NULL, "REMOVE RELATION samples;");
157  db_query(NULL, "CREATE RELATION samples;");
158  db_query(NULL, "CREATE ATTRIBUTE time DOMAIN INT IN samples;");
159  db_query(NULL, "CREATE ATTRIBUTE hum DOMAIN INT IN samples;");
160  db_query(NULL, "CREATE INDEX samples.time TYPE INLINE;");
161 
162 #if PREPARE_DB
163  printf("Preparing the DB with %d tuples...\n", CARDINALITY);
164  errors = 0;
165  for(i = 1; i <= CARDINALITY; i++) {
166  PROCESS_PAUSE();
167 
168  result = db_query(NULL, "INSERT (%u, %u) INTO samples;",
169  i, (unsigned)random_rand());
170  if(DB_ERROR(result)) {
171  errors++;
172  }
173  }
174  printf("Done. Insertion errors: %d\n", errors);
175  printf("Ready to process queries\n");
176 #else
177  etimer_set(&sampling_timer, SAMPLING_INTERVAL * CLOCK_SECOND);
178 #endif
179 
180  for(;;) {
181  PROCESS_WAIT_EVENT();
182 
183  if(ev == serial_line_event_message && data != NULL) {
184  printf("START %s\n", (char *)data);
185  result = db_query(&handle, data);
186  if(DB_ERROR(result)) {
187  buffer_db_data("Query error: %s\n", db_get_result_message(result));
188  stop_handler(NULL);
189  db_free(&handle);
190  continue;
191  }
192 
193  if(!db_processing(&handle)) {
194  buffer_db_data("OK\n");
195  send_buffered_data();
196  stop_handler(NULL);
197  continue;
198  }
199 
200  packetbuf_set_attr(PACKETBUF_ATTR_PACKET_TYPE,
201  PACKETBUF_ATTR_PACKET_TYPE_STREAM);
202 
203  db_print_header(&handle);
204 
205  matching = 0;
206  processed = 0;
207 
208  while(db_processing(&handle)) {
209  PROCESS_PAUSE();
210 
211  if(matching == RESPONSE_LIMIT) {
212  buffer_db_data("Response suppressed at %u tuples: limit reached\n",
213  RESPONSE_LIMIT);
214  stop_handler(NULL);
215  db_free(&handle);
216  break;
217  }
218 
219  result = db_process(&handle);
220  if(result == DB_GOT_ROW) {
221  /* The processed tuple matched the condition in the query. */
222  matching++;
223  processed++;
224  db_print_tuple(&handle);
225  } else if(result == DB_OK) {
226  /* A tuple was processed, but did not match the condition. */
227  processed++;
228  continue;
229  } else {
230  if(result == DB_FINISHED) {
231  /* The processing has finished. Wait for a new command. */
232  buffer_db_data("[%ld tuples returned; %ld tuples processed]\n",
233  (long)matching, (long)processed);
234  buffer_db_data("OK\n");
235  } else if(DB_ERROR(result)) {
236  buffer_db_data("Processing error: %s\n",
237  db_get_result_message(result));
238  }
239  stop_handler(NULL);
240  db_free(&handle);
241  }
242  }
243  send_buffered_data();
244  }
245 
246 #if !PREPARE_DB
247  if(etimer_expired(&sampling_timer)) {
248  take_sample();
249  etimer_reset(&sampling_timer);
250  }
251 #endif
252  }
253 
254  PROCESS_END();
255 }

268 static void
269 received(struct mesh_conn *c, const linkaddr_t *from, uint8_t hops)
270 {
271  char *data;
272  unsigned len;
273  static char query[MAX_BUFFER_SIZE + 1];
274 
275  data = (char *)packetbuf_dataptr();
276  len = packetbuf_datalen();
277 
278  if(len > MAX_BUFFER_SIZE) {
279  buffer_db_data("Too long query: %d bytes\n", len);
280  return;
281  }
282 
283  memcpy(query, data, len);
284  query[len] = '\0';
285 
286  printf("Query received from %d.%d: %s (%d hops)\n",
287  from->u8[0], from->u8[1], query, (int)hops);
288  linkaddr_copy(&reply_addr, from);
289 
290  process_post(&query_process, serial_line_event_message, query);
291 }
