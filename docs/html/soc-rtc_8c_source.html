71 void
72 soc_rtc_init(void)
73 {
74  bool interrupts_disabled;
75  uint32_t next;
76 
77  /* Disable and clear interrupts */
78  interrupts_disabled = ti_lib_int_master_disable();
79 
80  ti_lib_aon_rtc_disable();
81 
82  ti_lib_aon_rtc_event_clear(AON_RTC_CH0);
83  ti_lib_aon_rtc_event_clear(AON_RTC_CH1);
84  ti_lib_aon_rtc_event_clear(AON_RTC_CH2);
85 
86  /* Setup the wakeup event */
87  ti_lib_aon_event_mcu_wake_up_set(AON_EVENT_MCU_WU0, AON_EVENT_RTC_CH0);
88  ti_lib_aon_event_mcu_wake_up_set(AON_EVENT_MCU_WU1, AON_EVENT_RTC_CH1);
89  ti_lib_aon_event_mcu_wake_up_set(AON_EVENT_MCU_WU2, AON_EVENT_RTC_CH2);
90  ti_lib_aon_rtc_combined_event_config(AON_RTC_CH0 | AON_RTC_CH1 | AON_RTC_CH2);
91 
92  HWREG(AON_RTC_BASE + AON_RTC_O_SEC) = SOC_RTC_START_TICK_COUNT;
93 
94  next = ti_lib_aon_rtc_current_compare_value_get() + COMPARE_INCREMENT;
95 
96  /* Configure channel 1 to start generating clock ticks. First tick at 512 */
97  ti_lib_aon_rtc_compare_value_set(AON_RTC_CH1, next);
98 
99  /* Enable channel 1 and the RTC */
100  ti_lib_aon_rtc_channel_enable(AON_RTC_CH1);
101  ti_lib_aon_rtc_enable();
102 
103  ti_lib_rom_int_enable(INT_AON_RTC_COMB);
104 
105  /* Re-enable interrupts */
106  if(!interrupts_disabled) {
107  ti_lib_int_master_enable();
108  }
109 }

111 rtimer_clock_t
112 soc_rtc_get_next_trigger()
113 {
114  rtimer_clock_t ch1 = ti_lib_aon_rtc_compare_value_get(AON_RTC_CH1);
115 
116  if(HWREG(AON_RTC_BASE + AON_RTC_O_CHCTL) & AON_RTC_CHCTL_CH0_EN) {
117  rtimer_clock_t ch0 = ti_lib_aon_rtc_compare_value_get(AON_RTC_CH0);
118 
119  return RTIMER_CLOCK_LT(ch0, ch1) ? ch0 : ch1;
120  }
121 
122  return ch1;
123 }

126 soc_rtc_schedule_one_shot(uint32_t channel, uint32_t ticks)
127 {
128  if((channel != AON_RTC_CH0) && (channel != AON_RTC_CH1) && (channel != AON_RTC_CH2)) {
129  return;
130  }
131 
132  /* Set the channel to fire a one-shot compare event at time==ticks */
133  ti_lib_aon_rtc_compare_value_set(channel, ticks);
134  ti_lib_aon_rtc_channel_enable(channel);
135 }

143 /* The AON RTC interrupt handler */
144 void
145 soc_rtc_isr(void)
146 {
147  uint32_t next;
148 
149  ENERGEST_ON(ENERGEST_TYPE_IRQ);
150 
151  last_isr_time = RTIMER_NOW();
152 
153  /* Adjust the s/w tick counter irrespective of which event trigger this */
154  clock_update();
155 
156  if(ti_lib_aon_rtc_event_get(AON_RTC_CH1)) {
157  HWREG(AON_RTC_BASE + AON_RTC_O_EVFLAGS) = AON_RTC_EVFLAGS_CH1;
158 
159  /*
160  * We need to keep ticking while we are awake, so we schedule the next
161  * event on the next 512 tick boundary. If we drop to deep sleep before it
162  * happens, lpm_drop() will reschedule us in the 'distant' future
163  */
164  next = ((ti_lib_aon_rtc_current_compare_value_get() + 5) +
165  COMPARE_INCREMENT) & MULTIPLE_512_MASK;
166  ti_lib_aon_rtc_compare_value_set(AON_RTC_CH1, next);
167  }
168 
169  if(ti_lib_aon_rtc_event_get(AON_RTC_CH0)) {
170  ti_lib_aon_rtc_channel_disable(AON_RTC_CH0);
171  HWREG(AON_RTC_BASE + AON_RTC_O_EVFLAGS) = AON_RTC_EVFLAGS_CH0;
172  rtimer_run_next();
173  }
174 
175  if(ti_lib_aon_rtc_event_get(AON_RTC_CH2)) {
176  /* after sleep; since a rtimer is already scheduled, do nothing */
177  ti_lib_aon_rtc_channel_disable(AON_RTC_CH2);
178  HWREG(AON_RTC_BASE + AON_RTC_O_EVFLAGS) = AON_RTC_EVFLAGS_CH2;
179  }
180 
181  ENERGEST_OFF(ENERGEST_TYPE_IRQ);
182 }
