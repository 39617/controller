117 PT_THREAD(send_values(struct httpd_state *s))
118 {
119  PSOCK_BEGIN(&s->sout);

121  SEND_STRING(&s->sout, TOP);
122 
123  if(strncmp(s->filename, "/index", 6) == 0 ||
124  s->filename[1] == '\0') {
125  /* Default page: show latest sensor values as text (does not
126  require Internet connection to Google for charts). */
127  blen = 0;
128  float mybatt = get_mybatt();
129  float mytemp = get_mytemp();
130  ADD("<h1>Current readings</h1>\n"
131  "Battery: %ld.%03d V<br>"
132  "Temperature: %ld.%03d &deg; C",
133  (long)mybatt, (unsigned)((mybatt - floor(mybatt)) * 1000),
134  (long)mytemp, (unsigned)((mytemp - floor(mytemp)) * 1000));
135  SEND_STRING(&s->sout, buf);
136  } else if(s->filename[1] == '0') {
137  /* Turn off leds */
138  leds_off(LEDS_ALL);
139  SEND_STRING(&s->sout, "Turned off leds!");
140  } else if(s->filename[1] == '1') {
141  /* Turn on leds */
142  leds_on(LEDS_ALL);
143  SEND_STRING(&s->sout, "Turned on leds!");
144  } else {
145  if(s->filename[1] != 't') {
146  generate_chart("Battery", "mV", 0, 4000, battery1);
147  SEND_STRING(&s->sout, buf);
148  }
149  if(s->filename[1] != 'b') {
150  generate_chart("Temperature", "Celsius", 0, 50, temperature);
151  SEND_STRING(&s->sout, buf);
152  }
153  }
154 
155  SEND_STRING(&s->sout, BOTTOM);
156 
157  PSOCK_END(&s->sout);
158 }
