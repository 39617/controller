67 static void
68 rain_callback(uint16_t value)
69 {
70  /* To calculate ticks from mm of rain, divide by 0.2794 mm */
71  printf("*** Rain gauge over threshold (%u ticks)\n", value);
72  weather_meter.configure(WEATHER_METER_RAIN_GAUGE_INT_OVER,
73  (value + RAIN_GAUGE_THRESHOLD_TICK));
74 }
75 /*---------------------------------------------------------------------------*/
76 static void
77 wind_speed_callback(uint16_t value)
78 {
79  /* This checks for instant wind speed values (over a second), the minimum
80  * value is 1.2 Km/h (one tick), as the reference is 2.4KM/h per rotation, and
81  * the anemometer makes 2 ticks per rotation. Instant speed is calculated as
82  * multiples of this, so if you want to check for 16Km/h, then it would be 13
83  * ticks
84  */
85  printf("*** Wind speed over threshold (%u ticks)\n", value);
86 }

88 PROCESS_THREAD(test_weather_meter_sensors, ev, data)
89 {
90  PROCESS_BEGIN();
91 
92  static uint32_t rain;
93  static uint16_t wind_speed;
94  static uint16_t wind_dir;
95  static uint16_t wind_dir_avg_2m;
96  static uint16_t wind_speed_avg;
97  static uint16_t wind_speed_avg_2m;
98  static uint16_t wind_speed_max;
99 
100  printf("Weather meter test example, integration period %u\n",
101  WEATHER_METER_AVG_PERIOD);
102 
103  /* Register the callback handler when thresholds are met */
104  WEATHER_METER_REGISTER_ANEMOMETER_INT(wind_speed_callback);
105  WEATHER_METER_REGISTER_RAIN_GAUGE_INT(rain_callback);
106 
107  /* Enable the sensors, this has to be called before any of the interrupt calls
108  * like the ones below
109  */
110  SENSORS_ACTIVATE(weather_meter);
111 
112  /* And the upper threshold value to compare and generate an interrupt */
113  weather_meter.configure(WEATHER_METER_ANEMOMETER_INT_OVER,
114  ANEMOMETER_THRESHOLD_TICK);
115  weather_meter.configure(WEATHER_METER_RAIN_GAUGE_INT_OVER,
116  RAIN_GAUGE_THRESHOLD_TICK);
117 
118  etimer_set(&et, READ_SENSOR_PERIOD);
119 
120  while(1) {
121  PROCESS_WAIT_EVENT_UNTIL(etimer_expired(&et));
122 
123  rain = weather_meter.value(WEATHER_METER_RAIN_GAUGE);
124  wind_speed = weather_meter.value(WEATHER_METER_ANEMOMETER);
125  wind_dir = weather_meter.value(WEATHER_METER_WIND_VANE);
126  wind_dir_avg_2m = weather_meter.value(WEATHER_METER_WIND_VANE_AVG_X);
127  wind_speed_avg = weather_meter.value(WEATHER_METER_ANEMOMETER_AVG);
128  wind_speed_avg_2m = weather_meter.value(WEATHER_METER_ANEMOMETER_AVG_X);
129  wind_speed_max = weather_meter.value(WEATHER_METER_ANEMOMETER_MAX);
130 
131 #if WEATHER_METER_RAIN_RETURN_TICKS
132  rain *= WEATHER_METER_AUX_RAIN_MM;
133  if(rain > (WEATHER_METER_AUX_RAIN_MM * 3)) {
134  printf("Rain: %lu.%lu mm, ", (rain / 10000), (rain % 10000));
135 #else
136  if(rain >= 10) {
137  printf("Rain: %u.%u mm, ", (rain / 10), (rain % 10));
138 #endif
139  } else {
140  printf("Rain: 0.%lu mm, ", rain);
141  }
142 
143  printf("Wind dir: %u.%01u deg, ", (wind_dir / 10), (wind_dir % 10));
144  printf("(%u.%01u deg avg)\n", (wind_dir_avg_2m / 10), (wind_dir_avg_2m % 10));
145  printf("Wind speed: %u m/h ", wind_speed);
146  printf("(%u m/h avg, %u m/h 2m avg, %u m/h max)\n\n", wind_speed_avg,
147  wind_speed_avg_2m,
148  wind_speed_max);
149  etimer_reset(&et);
150  }
151 
152  PROCESS_END();
153 }
