278 static void
279 handle_fd(fd_set *rset, fd_set *wset)
280 {
281  /* Optional delay between outgoing packets */
282  /* Base delay times number of 6lowpan fragments to be sent */
283  /* delaymsec = 10; */
284  if(delaymsec) {
285  struct timeval tv;
286  int dmsec;
287  gettimeofday(&tv, NULL);
288  dmsec=(tv.tv_sec-delaystartsec)*1000+tv.tv_usec/1000-delaystartmsec;
289  if(dmsec<0) delaymsec=0;
290  if(dmsec>delaymsec) delaymsec=0;
291  }
292 
293  if(delaymsec==0) {
294  int size;
295 
296  if(FD_ISSET(tunfd, rset)) {
297  size = tun_input(&uip_buf[UIP_LLH_LEN], sizeof(uip_buf));
298  /* printf("TUN data incoming read:%d\n", size); */
299  uip_len = size;
300  tcpip_input();
301 
302  if(slip_config_basedelay) {
303  struct timeval tv;
304  gettimeofday(&tv, NULL) ;
305  delaymsec=slip_config_basedelay;
306  delaystartsec =tv.tv_sec;
307  delaystartmsec=tv.tv_usec/1000;
308  }
309  }
310  }
311 }
