56 static int
57 file_test(const char *filename, char *msg)
58 {
59  static uint8_t sequence;
60  int fd;
61  int r;
62  struct record {
63  char message[16];
64  uint8_t sequence;
65  } record;
66 
67  /*
68  * Coffee determines the file length by finding the last non-zero byte
69  * of the file. This I/O semantic requires that each record should end
70  * with a non-zero, if we are writing multiple records and closing the
71  * file descriptor in between.
72  *
73  * In this example, in which the file_test function can be called
74  * multiple times, we ensure that the sequence counter starts at 1.
75  */
76 
77  if(sequence == 255) {
78  /* No more sequence numbers available. */
79  return -1;
80  }
81  sequence++;
82 
83  strncpy(record.message, msg, sizeof(record.message) - 1);
84  record.message[sizeof(record.message) - 1] = '\0';
85  record.sequence = sequence;
86 
87  /* Obtain a file descriptor for the file, capable of handling both
88  reads and writes. */
89  fd = cfs_open(FILENAME, CFS_WRITE | CFS_APPEND | CFS_READ);
90  if(fd < 0) {
91  printf("failed to open %s\n", FILENAME);
92  return 0;
93  }
94 
95  r = cfs_write(fd, &record, sizeof(record));
96  if(r != sizeof(record)) {
97  printf("failed to write %d bytes to %s\n",
98  (int)sizeof(record), FILENAME);
99  cfs_close(fd);
100  return 0;
101  }
102 
103  printf("Wrote message \"%s\", sequence %u\n",
104  record.message, record.sequence);
105 
106  /* To read back the message, we need to move the file pointer to the
107  beginning of the file. */
108  if(cfs_seek(fd, 0, CFS_SEEK_SET) != 0) {
109  printf("seek failed\n");
110  cfs_close(fd);
111  return 0;
112  }
113 
114  /* Read all written records and print their contents. */
115  for(;;) {
116  r = cfs_read(fd, &record, sizeof(record));
117  if(r == 0) {
118  break;
119  } else if(r < sizeof(record)) {
120  printf("failed to read %d bytes from %s, got %d\n",
121  (int)sizeof(record), FILENAME, r);
122  cfs_close(fd);
123  return 0;
124  }
125 
126  printf("Read message \"%s\", sequence %u\n",
127  record.message, record.sequence);
128  }
129 
130  /* Release the internal resources held by Coffee for
131  the file descriptor. */
132  cfs_close(fd);
133 
134  return 1;
135 }
