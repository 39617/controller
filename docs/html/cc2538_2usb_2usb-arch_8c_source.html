253 static void
254 read_hw_buffer(uint8_t *to, uint8_t hw_ep, unsigned int len)
255 {
256  uint32_t fifo_addr = USB_F0 + (hw_ep << 3);
257 
258  if(USB_ARCH_CONF_DMA && len > UDMA_SIZE_THRESHOLD) {
259  /* Set the transfer source and destination addresses */
260  udma_set_channel_src(USB_ARCH_CONF_RX_DMA_CHAN, fifo_addr);
261  udma_set_channel_dst(USB_ARCH_CONF_RX_DMA_CHAN,
262  (uint32_t)(to) + len - 1);
263 
264  /* Configure the control word */
265  udma_set_channel_control_word(USB_ARCH_CONF_RX_DMA_CHAN,
266  UDMA_RX_FLAGS | udma_xfer_size(len));
267 
268  /* Enabled the RF RX uDMA channel */
269  udma_channel_enable(USB_ARCH_CONF_RX_DMA_CHAN);
270 
271  /* Trigger the uDMA transfer */
272  udma_channel_sw_request(USB_ARCH_CONF_RX_DMA_CHAN);
273 
274  /* Wait for the transfer to complete. */
275  while(udma_channel_get_mode(USB_ARCH_CONF_RX_DMA_CHAN));
276  } else {
277  while(len--) {
278  *to++ = REG(fifo_addr);
279  }
280  }
281 }

313 static void
314 reset(void)
315 {
316  uint8_t e;
317 
318  for(e = 0; e < USB_MAX_ENDPOINTS; e++) {
319  if(usb_endpoints[e].flags & USB_EP_FLAGS_ENABLED) {
320  usb_buffer *buffer = usb_endpoints[e].buffer;
321 
322  usb_endpoints[e].flags = 0;
323  usb_disable_endpoint(e);
324  while(buffer) {
325  buffer->flags &= ~USB_BUFFER_SUBMITTED;
326  buffer = buffer->next;
327  }
328  }
329  }
330  usb_arch_setup_control_endpoint(0);
331 }

342 /*---------------------------------------------------------------------------*/
343 /* Init USB */
344 void
345 usb_arch_setup(void)
346 {
347  uint8_t i;
348 
349  lpm_register_peripheral(permit_pm1);
350 
351  /* Switch on USB PLL & USB module */
352  REG(USB_CTRL) = USB_CTRL_USB_EN | USB_CTRL_PLL_EN;
353 
354  /* Wait until USB PLL is stable */
355  while(!(REG(USB_CTRL) & USB_CTRL_PLL_LOCKED));
356 
357  /* Enable pull-up on usb port if driven by GPIO */
358 #if defined(USB_PULLUP_PORT_BASE) && defined(USB_PULLUP_PIN_MASK)
359  GPIO_SET_OUTPUT(USB_PULLUP_PORT_BASE, USB_PULLUP_PIN_MASK);
360  GPIO_SET_PIN(USB_PULLUP_PORT_BASE, USB_PULLUP_PIN_MASK);
361 #endif
362 
363  for(i = 0; i < USB_MAX_ENDPOINTS; i++) {
364  usb_endpoints[i].flags = 0;
365  usb_endpoints[i].event_process = 0;
366  }
367 
368  reset();
369 
370  /* Disable all EP interrupts, EP0 interrupt will be enabled later */
371  REG(USB_IIE) = 0;
372  REG(USB_OIE) = 0;
373 
374  /* Initialise the USB control structures */
375  if(USB_ARCH_CONF_DMA) {
376  /* Disable peripheral triggers for our channels */
377  udma_channel_mask_set(USB_ARCH_CONF_RX_DMA_CHAN);
378  udma_channel_mask_set(USB_ARCH_CONF_TX_DMA_CHAN);
379  }
380 
381  NVIC_EnableIRQ(USB_IRQn);
382 }
383 /*---------------------------------------------------------------------------*/
384 void
385 usb_submit_recv_buffer(uint8_t addr, usb_buffer *buffer)
386 {
387  usb_buffer **tailp;
388  uint8_t flag;
389  usb_endpoint_t *ep = EP_STRUCT(addr);
390 
391  if(!(ep->flags & USB_EP_FLAGS_ENABLED)) {
392  return;
393  }
394 
395  if(buffer->data == NULL && EP_HW_NUM(addr) == 0) {
396  if(buffer->flags & USB_BUFFER_NOTIFY) {
397  notify_ep_process(ep, USB_EP_EVENT_NOTIFICATION);
398  }
399  return;
400  }
401 
402  flag = disable_irq();
403 
404  tailp = &ep->buffer;
405  while(*tailp) {
406  tailp = &(*tailp)->next;
407  }
408  *tailp = buffer;
409  while(buffer) {
410  buffer->flags |= USB_BUFFER_SUBMITTED;
411  buffer = buffer->next;
412  }
413 
414  REG(USB_INDEX) = EP_HW_NUM(addr);
415  if(!EP_HW_NUM(ep->addr)) {
416  if(REG(USB_CS0) & USB_CS0_OUTPKT_RDY) {
417  ep0_interrupt_handler();
418  }
419  } else {
420  if(REG(USB_CSOL) & USB_CSOL_OUTPKT_RDY) {
421  out_ep_interrupt_handler(EP_HW_NUM(ep->addr));
422  }
423  }
424 
425  restore_irq(flag);
426 }
427 /*---------------------------------------------------------------------------*/
428 void
429 usb_submit_xmit_buffer(uint8_t addr, usb_buffer *buffer)
430 {
431  usb_buffer **tailp;
432  uint8_t flag;
433  uint8_t res;
434  usb_endpoint_t *ep = EP_STRUCT(addr);
435 
436  if(!(ep->flags & USB_EP_FLAGS_ENABLED)) {
437  return;
438  }
439 
440  flag = disable_irq();
441 
442  if(EP_HW_NUM(addr) == 0) {
443  if(buffer->data == NULL) {
444  /* We are asked to send a STATUS packet.
445  * But the USB hardware will do this automatically
446  * as soon as we release the HW FIFO. */
447  REG(USB_INDEX) = 0;
448  REG(USB_CS0) = USB_CS0_CLR_OUTPKT_RDY | USB_CS0_DATA_END;
449  notify_ep_process(ep, USB_EP_EVENT_NOTIFICATION);
450  restore_irq(flag);
451  return;
452  } else {
453  /* Release the HW FIFO */
454  REG(USB_INDEX) = 0;
455  REG(USB_CS0) = USB_CS0_CLR_OUTPKT_RDY;
456  }
457  }
458 
459  tailp = &ep->buffer;
460  while(*tailp) {
461  tailp = &(*tailp)->next;
462  }
463  *tailp = buffer;
464  while(buffer) {
465  buffer->flags |= USB_BUFFER_SUBMITTED | USB_BUFFER_IN;
466  buffer = buffer->next;
467  }
468 
469  REG(USB_INDEX) = EP_HW_NUM(ep->addr);
470  if(EP_HW_NUM(ep->addr)) {
471  res = ep_tx(EP_HW_NUM(ep->addr));
472  } else {
473  res = ep0_tx();
474  }
475 
476  restore_irq(flag);
477 
478  if(res & USB_WRITE_NOTIFY) {
479  notify_ep_process(ep, USB_EP_EVENT_NOTIFICATION);
480  }
481 }

489 static void
490 in_ep_setup(uint8_t addr)
491 {
492  uint8_t ei = EP_HW_NUM(addr);
493  usb_endpoint_t *ep = EP_STRUCT(addr);
494 
495  /* Enable IN EP interrupt */
496  REG(USB_IIE) |= USBIIE_INEPxIE(ei);
497 
498  /* Set internal FIFO size */
499  REG(USB_MAXI) = ep->xfer_size / 8;
500 
501  if(IS_ISO_EP(ep)) {
502  REG(USB_CSIH) |= USB_CSOH_ISO;
503  } else {
504  REG(USB_CSIH) &= ~USB_CSOH_ISO;
505  }
506 }

528 ep_setup(uint8_t addr)
529 {

531  uint8_t flag;
532  usb_endpoint_t *ep = EP_STRUCT(addr);
533

542

671 static void
672 set_stall(uint8_t addr, uint8_t stall)
673 {
674  uint8_t ei = EP_HW_NUM(addr);
675 
676  REG(USB_INDEX) = ei;
677  if(ei == 0) {
678  /* Stall is automatically deasserted on EP0 */
679  if(stall) {
680  ep0status = EP_STATUS_IDLE;
681  REG(USB_CS0) |= USB_CS0_SEND_STALL | USB_CS0_OUTPKT_RDY;
682  }
683  } else {
684  if(addr & 0x80) {
685  if(stall) {
686  REG(USB_CSIL) |= USB_CSIL_SEND_STALL;
687  } else {
688  REG(USB_CSIL) &= ~USB_CSIL_SEND_STALL;
689  }
690  } else {
691  if(stall) {
692  REG(USB_CSOL) |= USB_CSOL_SEND_STALL;
693  } else {
694  REG(USB_CSOL) &= ~USB_CSOL_SEND_STALL;
695  }
696  }
697  }

830 static uint8_t
831 fill_buffers(usb_buffer *buffer, uint8_t hw_ep, unsigned int len,
832  uint8_t short_packet)
833 {
834  unsigned int t;
835  uint8_t res = 0;

837  do {
838  if(buffer->left < len) {
839  t = buffer->left;
840  } else {
841  t = len;
842  }
843  len -= t;
844  buffer->left -= t;
845 
846  read_hw_buffer(buffer->data, hw_ep, t);
847 
848  buffer->data += t;
849 
850  if(len == 0) {
851  break;
852  }
853 
854  buffer->flags &= ~(USB_BUFFER_SUBMITTED | USB_BUFFER_SHORT_PACKET);
855  if(buffer->flags & USB_BUFFER_NOTIFY) {
856  res |= USB_READ_NOTIFY;
857  }
858  buffer = buffer->next;
859  } while(1);

861  if(short_packet) {

863  }

866  buffer->flags &= ~USB_BUFFER_SUBMITTED;
867  if(buffer->flags & USB_BUFFER_NOTIFY) {
868  res |= USB_READ_NOTIFY;
869  }
870  buffer = buffer->next;
871  } else {
872  if(short_packet) {
873  if(buffer->left && !(buffer->flags & USB_BUFFER_SHORT_END)) {
874  buffer->flags |= USB_BUFFER_FAILED;
875  res |= USB_READ_FAIL;
876  }
877  buffer->flags &= ~USB_BUFFER_SUBMITTED;
878  if(buffer->flags & USB_BUFFER_NOTIFY) {
879  res |= USB_READ_NOTIFY;
880  }
881  buffer = buffer->next;
882  }
883  }
884 
885  usb_endpoints[hw_ep].buffer = buffer;
886 
887  return res;
888 }

924 static uint8_t
925 ep0_get_data_pkt(void)
926 {
927  uint8_t res = 0;
928  uint8_t short_packet = 0;
929  usb_buffer *buffer = usb_endpoints[0].buffer;
930  uint8_t len = REG(USB_CNT0);
931 
932  if(!buffer) {
933  return USB_READ_BLOCK;
934  }
935 
936  if(buffer->flags & (USB_BUFFER_SETUP | USB_BUFFER_IN)) {
937  buffer->flags |= USB_BUFFER_FAILED;
938  buffer->flags &= ~USB_BUFFER_SUBMITTED;
939  if(buffer->flags & USB_BUFFER_NOTIFY) {
940  res |= USB_READ_NOTIFY;
941  }
942  /* Flush the fifo */
943  while(len--) {
944  REG(USB_F0);
945  }
946  usb_endpoints[0].buffer = buffer->next;
947  /* Force data stage end */
948  REG(USB_CS0) |= USB_CS0_CLR_OUTPKT_RDY | USB_CS0_DATA_END;
949 
950  ep0status = EP_STATUS_IDLE;
951  return res;
952  }
953 
954  if(get_receive_capacity(buffer) < len) {
955  /* Wait until we queue more buffers */
956  return USB_READ_BLOCK;
957  }
958 
959  if(len < usb_endpoints[0].xfer_size) {
960  short_packet = 1;
961  }
962 
963  res = fill_buffers(buffer, 0, len, short_packet);
964 
965  if(short_packet) {
966  /* The usb-core will send a status packet, we will release the fifo at this stage */
967  ep0status = EP_STATUS_IDLE;
968  } else {
969  REG(USB_CS0) |= USB_CS0_CLR_OUTPKT_RDY;
970  }
971  return res;
972 }
973 /*---------------------------------------------------------------------------*/
974 static uint8_t
975 ep0_tx(void)
976 {
977  usb_buffer *buffer = usb_endpoints[0].buffer;
978  unsigned int len = usb_endpoints[0].xfer_size;
979  uint8_t data_end = 0;
980  uint8_t res = 0;
981 
982  /* If TX Fifo still busy or ep0 not in TX data stage don't do anything */
983  if((REG(USB_CS0) & USB_CS0_INPKT_RDY) || (ep0status != EP_STATUS_TX)) {
984  return 0;
985  }
986 
987  if(!buffer) {
988  return 0;
989  }
990 
991  if(!(buffer->flags & USB_BUFFER_IN)) {
992  /* We should TX but queued buffer is in RX */
993  return 0;
994  }
995 
996  while(buffer) {
997  unsigned int copy;
998 
999  if(buffer->left < len) {
1000  copy = buffer->left;
1001  } else {
1002  copy = len;
1003  }
1004 
1005  len -= copy;
1006  buffer->left -= copy;
1007  write_hw_buffer(0, buffer->data, copy);
1008  buffer->data += copy;
1009  if(buffer->left == 0) {
1010  if(buffer->flags & USB_BUFFER_SHORT_END) {
1011  if(len == 0) {
1012  break; // We keep the buffer in queue so we will send a ZLP next time.
1013  } else {
1014  data_end = 1;
1015  len = 0; // Stop looking for more data to send
1016  }
1017  }
1018  buffer->flags &= ~USB_BUFFER_SUBMITTED;
1019  if(buffer->flags & USB_BUFFER_NOTIFY) {
1020  res |= USB_WRITE_NOTIFY;
1021  }
1022  buffer = buffer->next;
1023  }
1024  if(len == 0) {
1025  break; // FIFO is full, send packet.
1026  }
1027  }
1028  if(len) {
1029  data_end = 1;
1030  }
1031  usb_endpoints[0].buffer = buffer;
1032 
1033  /*
1034  * Workaround the fact that the usb controller do not like to have DATA_END
1035  * set after INPKT_RDY for the last packet. Thus if no more is in the queue
1036  * set DATA_END
1037  */
1038  if(data_end || !buffer) {
1039  ep0status = EP_STATUS_IDLE;
1040  REG(USB_CS0) |= USB_CS0_INPKT_RDY | USB_CS0_DATA_END;
1041  } else {
1042  REG(USB_CS0) |= USB_CS0_INPKT_RDY;
1043  }
1044 
1045  return res;
1046 }
1047 /*---------------------------------------------------------------------------*/
1048 static void
1049 ep0_interrupt_handler(void)
1050 {
1051  uint8_t cs0;
1052  uint8_t res;
1053 
1054  REG(USB_INDEX) = 0;
1055  cs0 = REG(USB_CS0);
1056  if(cs0 & USB_CS0_SENT_STALL) {
1057  /* Ack the stall */
1058  REG(USB_CS0) = 0;
1059  ep0status = EP_STATUS_IDLE;
1060  }
1061  if(cs0 & USB_CS0_SETUP_END) {
1062  /* Clear it */
1063  REG(USB_CS0) = USB_CS0_CLR_SETUP_END;
1064  ep0status = EP_STATUS_IDLE;
1065  }
1066 
1067  if(cs0 & USB_CS0_OUTPKT_RDY) {
1068  if(ep0status == EP_STATUS_IDLE) {
1069  res = ep0_get_setup_pkt();
1070  } else {
1071  res = ep0_get_data_pkt();
1072  }
1073 
1074  if(res & USB_READ_NOTIFY) {
1075  notify_ep_process(&usb_endpoints[0], USB_EP_EVENT_NOTIFICATION);
1076  }
1077  if(res & USB_READ_BLOCK) {
1078  return;
1079  }
1080  }
1081 
1082  res = ep0_tx();
1083 
1084  if(res & USB_WRITE_NOTIFY) {
1085  notify_ep_process(&usb_endpoints[0], USB_EP_EVENT_NOTIFICATION);
1086  }
1087 }

1089 static uint8_t
1090 ep_tx(uint8_t ep_hw)
1091 {
1092  unsigned int len;
1093  uint8_t res = 0;
1094  usb_endpoint_t *ep = EP_STRUCT(ep_hw);
1095 
1096  len = ep->xfer_size;
1097 
1098  if(ep->halted) {
1099  return 0;
1100  }
1101 
1102  if(!ep->buffer || !(ep->buffer->flags & USB_BUFFER_IN)) {
1103  return 0;
1104  }
1105 
1106  while(ep->buffer) {
1107  unsigned int copy;
1108 
1109  if(ep->buffer->left < len) {
1110  copy = ep->buffer->left;
1111  } else {
1112  copy = len;
1113  }
1114 
1115  len -= copy;
1116  ep->buffer->left -= copy;
1117 
1118  /*
1119  * Delay somewhat if the previous packet has not yet left the IN FIFO,
1120  * making sure the dog doesn't bark while we're waiting
1121  */
1122  while(REG(USB_CSIL) & USB_CSIL_INPKT_RDY) {
1123  watchdog_periodic();
1124  }
1125 
1126  write_hw_buffer(EP_INDEX(ep_hw), ep->buffer->data, copy);
1127  ep->buffer->data += copy;
1128 
1129  if(ep->buffer->left == 0) {
1130  if(ep->buffer->flags & USB_BUFFER_SHORT_END) {
1131  if(len == 0) {
1132  /* We keep the buffer in queue so we will send a ZLP next */
1133  break;
1134  } else {
1135  /* Stop looking for more data to send */
1136  len = 0;
1137  }
1138  }
1139  ep->buffer->flags &= ~USB_BUFFER_SUBMITTED;
1140  if(ep->buffer->flags & USB_BUFFER_NOTIFY) {
1141  res |= USB_WRITE_NOTIFY;
1142  }
1143  ep->buffer = ep->buffer->next;
1144  }
1145  if(len == 0) {
1146  /* FIFO full, send */
1147  break;
1148  }
1149  }
1150 
1151  REG(USB_CSIL) |= USB_CSIL_INPKT_RDY;
1152 
1153  return res;
1154 }

1156 static uint8_t
1157 ep_get_data_pkt(uint8_t ep_hw)
1158 {
1159  uint16_t pkt_len;
1160  uint8_t res;
1161  uint8_t short_packet = 0;
1162  usb_endpoint_t *ep = EP_STRUCT(ep_hw);
1163 
1164  if(!ep->buffer) {
1165  return USB_READ_BLOCK;
1166  }
1167
