43 static int
44 slip_putchar(char c, FILE *stream)
45 {
46 #define SLIP_END 0300
47  static char debug_frame = 0;
48 
49  if (!debug_frame) { /* Start of debug output */
50  slip_arch_writeb(SLIP_END);
51  slip_arch_writeb('\r'); /* Type debug line == '\r' */
52  debug_frame = 1;
53  }
54 
55  slip_arch_writeb((unsigned char)c);
56 
57  /*
58  * Line buffered output, a newline marks the end of debug output and
59  * implicitly flushes debug output.
60  */
61  if (c == '\n') {
62  slip_arch_writeb(SLIP_END);
63  debug_frame = 0;
64  }
65 
66  return c;
67 }

72 void
73 slip_arch_init(unsigned long ubr)
74 {
75  uint8_t dummy;
76  spl_t s = splhigh();
77 
78  UBRR1L = ubr;
79  UBRR1H = ubr >> 8; /* Always zero */
80 
81  UCSR1C = BV(UCSZ1) | BV(UCSZ0); /* 1 start bit, no parity, 1 stop bit */
82  UCSR1B = BV(RXEN) | BV(TXEN) | BV(RXCIE);
83  dummy = UDR1; /* Flush RX buffer. */
84 
85  splx(s);
86 
87  stdout = &slip_stdout;
88 }
