206 int
207 ipaddr_sprintf(char *buf, uint8_t buf_len, const uip_ipaddr_t *addr)
208 {
209  uint16_t a;
210  uint8_t len = 0;
211  int i, f;
212  for(i = 0, f = 0; i < sizeof(uip_ipaddr_t); i += 2) {
213  a = (addr->u8[i] << 8) + addr->u8[i + 1];
214  if(a == 0 && f >= 0) {
215  if(f++ == 0) {
216  len += snprintf(&buf[len], buf_len - len, "::");
217  }
218  } else {
219  if(f > 0) {
220  f = -1;
221  } else if(i > 0) {
222  len += snprintf(&buf[len], buf_len - len, ":");
223  }
224  len += snprintf(&buf[len], buf_len - len, "%x", a);
225  }
226  }
227 
228  return len;
229 }

246 static void
247 pub_handler(const char *topic, uint16_t topic_len, const uint8_t *chunk,
248  uint16_t chunk_len)
249 {
250  DBG("Pub Handler: topic='%s' (len=%u), chunk_len=%u\n", topic, topic_len,
251  chunk_len);
252 
253  /* If we don't like the length, ignore */
254  if(topic_len != 23 || chunk_len != 1) {
255  printf("Incorrect topic or chunk len. Ignored\n");
256  return;
257  }
258 
259  /* If the format != json, ignore */
260  if(strncmp(&topic[topic_len - 4], "json", 4) != 0) {
261  printf("Incorrect format\n");
262  }
263 
264  if(strncmp(&topic[10], "leds", 4) == 0) {
265  if(chunk[0] == '1') {
266  leds_on(LEDS_RED);
267  } else if(chunk[0] == '0') {
268  leds_off(LEDS_RED);
269  }
270  return;
271  }

274 static void
275 mqtt_event(struct mqtt_connection *m, mqtt_event_t event, void *data)
276 {
277  switch(event) {
278  case MQTT_EVENT_CONNECTED: {
279  DBG("APP - Application has a MQTT connection\n");
280  timer_set(&connection_life, CONNECTION_STABLE_TIME);
281  state = STATE_CONNECTED;
282  break;
283  }
284  case MQTT_EVENT_DISCONNECTED: {
285  DBG("APP - MQTT Disconnect. Reason %u\n", *((mqtt_event_t *)data));
286 
287  state = STATE_DISCONNECTED;
288  process_poll(&mqtt_demo_process);
289  break;
290  }
291  case MQTT_EVENT_PUBLISH: {
292  msg_ptr = data;
293 
294  /* Implement first_flag in publish message? */
295  if(msg_ptr->first_chunk) {
296  msg_ptr->first_chunk = 0;
297  DBG("APP - Application received a publish on topic '%s'. Payload "
298  "size is %i bytes. Content:\n\n",
299  msg_ptr->topic, msg_ptr->payload_length);
300  }
301 
302  pub_handler(msg_ptr->topic, strlen(msg_ptr->topic), msg_ptr->payload_chunk,
303  msg_ptr->payload_length);
304  break;
305  }
306  case MQTT_EVENT_SUBACK: {
307  DBG("APP - Application is subscribed to topic successfully\n");
308  break;
309  }
310  case MQTT_EVENT_UNSUBACK: {
311  DBG("APP - Application is unsubscribed to topic successfully\n");
312  break;
313  }
314  case MQTT_EVENT_PUBACK: {
315  DBG("APP - Publishing complete\n");
316  break;
317  }
318  default:
319  DBG("APP - Application got a unhandled MQTT event: %i\n", event);
320  break;
321  }
322 }

446 static void
447 publish(void)
448 {
449  /* Publish MQTT topic in IBM quickstart format */
450  int len;
451  int remaining = APP_BUFFER_SIZE;
452 
453  seq_nr_value++;
454 
455  buf_ptr = app_buffer;
456 
457  len = snprintf(buf_ptr, remaining,
458  "{"
459  "\"d\":{"
460  "\"Seq #\":%d,"
461  "\"Uptime (sec)\":%lu", seq_nr_value, clock_seconds());
462 
463  if(len < 0 || len >= remaining) {
464  printf("Buffer too short. Have %d, need %d + \\0\n", remaining, len);
465  return;
466  }
467 
468  remaining -= len;
469  buf_ptr += len;
470 
471  /* Put our Default route's string representation in a buffer */
472  char def_rt_str[64];
473  memset(def_rt_str, 0, sizeof(def_rt_str));
474  ipaddr_sprintf(def_rt_str, sizeof(def_rt_str), uip_ds6_defrt_choose());
475 
476  len = snprintf(buf_ptr, remaining, ",\"Def Route\":\"%s\",\"RSSI (dBm)\":%d",
477  def_rt_str, def_rt_rssi);
478 
479  if(len < 0 || len >= remaining) {
480  printf("Buffer too short. Have %d, need %d + \\0\n", remaining, len);
481  return;
482  }
483  remaining -= len;
484  buf_ptr += len;
485 
486  len = snprintf(buf_ptr, remaining, ",\"On-Chip Temp (deg.C)\":%d",
487  temperature_sensor.value(0));
488 
489  if(len < 0 || len >= remaining) {
490  printf("Buffer too short. Have %d, need %d + \\0\n", remaining, len);
491  return;
492  }
493  remaining -= len;
494  buf_ptr += len;
495 
496  len = snprintf(buf_ptr, remaining, "}}");
497 
498  if(len < 0 || len >= remaining) {
499  printf("Buffer too short. Have %d, need %d + \\0\n", remaining, len);
500  return;
501  }
502 
503  mqtt_publish(&conn, NULL, pub_topic, (uint8_t *)app_buffer,
504  strlen(app_buffer), MQTT_QOS_LEVEL_0, MQTT_RETAIN_OFF);
505 
506  DBG("APP - Publish!\n");
507 }

529 /*---------------------------------------------------------------------------*/
530 static void
531 state_machine(void)
532 {
533  switch(state) {
534  case STATE_INIT:
535  /* If we have just been configured register MQTT connection */
536  mqtt_register(&conn, &mqtt_demo_process, client_id, mqtt_event,
537  MAX_TCP_SEGMENT_SIZE);
538 
539  /*
540  * If we are not using the quickstart service (thus we are an IBM
541  * registered device), we need to provide user name and password
542  */
543  if(strncasecmp(conf.org_id, QUICKSTART, strlen(conf.org_id)) != 0) {
544  if(strlen(conf.auth_token) == 0) {
545  printf("User name set, but empty auth token\n");
546  state = STATE_ERROR;
547  break;
548  } else {
549  mqtt_set_username_password(&conn, "use-token-auth",
550  conf.auth_token);
551  }
552  }
553 
554  /* _register() will set auto_reconnect. We don't want that. */
555  conn.auto_reconnect = 0;
556  connect_attempt = 1;
557 
558  state = STATE_REGISTERED;
559  DBG("Init\n");
560  /* Continue */
561  case STATE_REGISTERED:
562  if(uip_ds6_get_global(ADDR_PREFERRED) != NULL) {
563  /* Registered and with a public IP. Connect */
564  DBG("Registered. Connect attempt %u\n", connect_attempt);
565  ping_parent();
566  connect_to_broker();
567  } else {
568  leds_on(STATUS_LED);
569  ctimer_set(&ct, NO_NET_LED_DURATION, publish_led_off, NULL);
570  }
571  etimer_set(&publish_periodic_timer, NET_CONNECT_PERIODIC);
572  return;
573  break;
574  case STATE_CONNECTING:
575  leds_on(STATUS_LED);
576  ctimer_set(&ct, CONNECTING_LED_DURATION, publish_led_off, NULL);
577  /* Not connected yet. Wait */
578  DBG("Connecting (%u)\n", connect_attempt);
579  break;
580  case STATE_CONNECTED:
581  /* Don't subscribe unless we are a registered device */
582  if(strncasecmp(conf.org_id, QUICKSTART, strlen(conf.org_id)) == 0) {
583  DBG("Using 'quickstart': Skipping subscribe\n");
584  state = STATE_PUBLISHING;
585  }
586  /* Continue */
587  case STATE_PUBLISHING:
588  /* If the timer expired, the connection is stable. */
589  if(timer_expired(&connection_life)) {
590  /*
591  * Intentionally using 0 here instead of 1: We want RECONNECT_ATTEMPTS
592  * attempts if we disconnect after a successful connect
593  */
594  connect_attempt = 0;
595  }
596 
597  if(mqtt_ready(&conn) && conn.out_buffer_sent) {
598  /* Connected. Publish */
599  if(state == STATE_CONNECTED) {
600  subscribe();
601  state = STATE_PUBLISHING;
602  } else {
603  leds_on(STATUS_LED);
604  ctimer_set(&ct, PUBLISH_LED_ON_DURATION, publish_led_off, NULL);
605  publish();
606  }
607  etimer_set(&publish_periodic_timer, conf.pub_interval);
608 
609  DBG("Publishing\n");
610  /* Return here so we don't end up rescheduling the timer */
611  return;
612  } else {
613  /*
614  * Our publish timer fired, but some MQTT packet is already in flight
615  * (either not sent at all, or sent but not fully ACKd).
616  *
617  * This can mean that we have lost connectivity to our broker or that
618  * simply there is some network delay. In both cases, we refuse to
619  * trigger a new message and we wait for TCP to either ACK the entire
620  * packet after retries, or to timeout and notify us.
621  */
622  DBG("Publishing... (MQTT state=%d, q=%u)\n", conn.state,
623  conn.out_queue_full);
624  }
625  break;
626  case STATE_DISCONNECTED:
627  DBG("Disconnected\n");
628  if(connect_attempt < RECONNECT_ATTEMPTS ||
629  RECONNECT_ATTEMPTS == RETRY_FOREVER) {
630  /* Disconnect and backoff */
631  clock_time_t interval;
632  mqtt_disconnect(&conn);
633  connect_attempt++;
634 
635  interval = connect_attempt < 3 ? RECONNECT_INTERVAL << connect_attempt :
636  RECONNECT_INTERVAL << 3;
637 
638  DBG("Disconnected. Attempt %u in %lu ticks\n", connect_attempt, interval);
639 
640  etimer_set(&publish_periodic_timer, interval);
641 
642  state = STATE_REGISTERED;
643  return;
644  } else {
645  /* Max reconnect attempts reached. Enter error state */
646  state = STATE_ERROR;
647  DBG("Aborting connection after %u attempts\n", connect_attempt - 1);
648  }
649  break;
650  case STATE_CONFIG_ERROR:
651  /* Idle away. The only way out is a new config */
652  printf("Bad configuration\n");
653  return;
654  case STATE_ERROR:
655  default:
656  leds_on(STATUS_LED);
