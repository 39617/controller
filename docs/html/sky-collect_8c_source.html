93 PROCESS_THREAD(depth_blink_process, ev, data)
94 {
95  static struct etimer et;
96  static int count;
97 
98  PROCESS_BEGIN();
99 
100  while(1) {
101  etimer_set(&et, CLOCK_SECOND * 10);
102  PROCESS_WAIT_UNTIL(etimer_expired(&et));
103  count = collect_depth(&tc);
104  if(count == COLLECT_MAX_DEPTH) {
105  leds_on(LEDS_BLUE);
106  } else {
107  leds_off(LEDS_BLUE);
108  count /= COLLECT_LINK_ESTIMATE_UNIT;
109  while(count > 0) {
110  leds_on(LEDS_RED);
111  etimer_set(&et, CLOCK_SECOND / 32);
112  PROCESS_WAIT_UNTIL(etimer_expired(&et));
113  leds_off(LEDS_RED);
114  etimer_set(&et, CLOCK_SECOND / 8);
115  PROCESS_WAIT_UNTIL(etimer_expired(&et));
116  --count;
117  }
118  }
119  }
120 
121  PROCESS_END();
122 }

125  int channel[16];
126 };
127 #define NUM_SAMPLES 4
128 static struct spectrum rssi_samples[NUM_SAMPLES];
129 static int
130 do_rssi(void)
131 {
132  static int sample;
133  int channel;
134 
135  NETSTACK_MAC.off(0);
136 
137  cc2420_on();
138  for(channel = 11; channel <= 26; ++channel) {
139  cc2420_set_channel(channel);
140  rssi_samples[sample].channel[channel - 11] = cc2420_rssi() + 53;
141  }
142 
143  NETSTACK_MAC.on();
144 
145  sample = (sample + 1) % NUM_SAMPLES;
146 
147  {
148  int channel, tot;
149  tot = 0;
150  for(channel = 0; channel < 16; ++channel) {
151  int max = -256;
152  int i;
153  for(i = 0; i < NUM_SAMPLES; ++i) {
154  max = MAX(max, rssi_samples[i].channel[channel]);
155  }
156  tot += max / 20;
157  }
158  return tot;
159  }
160 }

162 static void
163 recv(const linkaddr_t *originator, uint8_t seqno, uint8_t hops)
164 {
165  struct sky_collect_msg *msg;
166 
167  msg = packetbuf_dataptr();
168  printf("%u %u %u %u %u %u %u %u %u %u %u %lu %lu %lu %lu %lu ",
169  (originator->u8[1] << 8) + originator->u8[0],
170  seqno, hops,
171  msg->light1, msg->light2, msg->temperature, msg->humidity,
172  msg->rssi,
173 
174  (msg->best_neighbor.u8[0] << 8) + msg->best_neighbor.u8[1],
175  msg->best_neighbor_etx, msg->best_neighbor_rtmetric,
176  msg->energy_lpm, msg->energy_cpu, msg->energy_rx, msg->energy_tx, msg->energy_rled
177  );
178  printf("%u %u %u %u %u %u %u %u %u %u %u %u %u %u %u %u %u %u ",
179  msg->tx, msg->rx, msg->reliabletx, msg->reliablerx, msg->rexmit,
180  msg->acktx, msg->noacktx, msg->ackrx, msg->timedout, msg->badackrx,
181  msg->toolong, msg->tooshort, msg->badsynch, msg->badcrc,
182  msg->contentiondrop, msg->sendingdrop, msg->lltx, msg->llrx);
183 #if TIMESYNCH_CONF_ENABLED
184  printf("%u", timesynch_time() - msg->timestamp);
185 #else
186  printf("%u", RTIMER_NOW() - msg->timestamp);
187 #endif /* TIMESYNCH_CONF_ENABLED */
188  printf("\n");
189 
190 }
