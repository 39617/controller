87 /*
88  * Handle the response to the observe request and the following notifications
89  */
90 static void
91 notification_callback(coap_observee_t *obs, void *notification,
92  coap_notification_flag_t flag)
93 {
94  int len = 0;
95  const uint8_t *payload = NULL;
96 
97  printf("Notification handler\n");
98  printf("Observee URI: %s\n", obs->url);
99  if(notification) {
100  len = coap_get_payload(notification, &payload);
101  }
102  switch(flag) {
103  case NOTIFICATION_OK:
104  printf("NOTIFICATION OK: %*s\n", len, (char *)payload);
105  break;
106  case OBSERVE_OK: /* server accepeted observation request */
107  printf("OBSERVE_OK: %*s\n", len, (char *)payload);
108  break;
109  case OBSERVE_NOT_SUPPORTED:
110  printf("OBSERVE_NOT_SUPPORTED: %*s\n", len, (char *)payload);
111  obs = NULL;
112  break;
113  case ERROR_RESPONSE_CODE:
114  printf("ERROR_RESPONSE_CODE: %*s\n", len, (char *)payload);
115  obs = NULL;
116  break;
117  case NO_REPLY_FROM_SERVER:
118  printf("NO_REPLY_FROM_SERVER: "
119  "removing observe registration with token %x%x\n",
120  obs->token[0], obs->token[1]);
121  obs = NULL;
122  break;
123  }
124 }

129 void

164  /* toggle observation every time the timer elapses or the button is pressed */
165  while(1) {

179  }
