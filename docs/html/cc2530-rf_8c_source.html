216 static radio_value_t
217 get_rssi(void)
218 {
219  int8_t rssi;
220 
221  /* If we are off, turn on first */
222  if(RXENABLE == 0) {
223  rf_flags |= WAS_OFF;
224  on();
225  }
226 
227  /* Wait on RSSI_VALID */
228  while((RSSISTAT & RSSISTAT_RSSI_VALID) == 0);
229 
230  rssi = (radio_value_t)RSSI - RSSI_OFFSET;
231 
232  /* If we were off, turn back off */
233  if((rf_flags & WAS_OFF) == WAS_OFF) {
234  rf_flags &= ~WAS_OFF;
235  off();
236  }
237 
238  return rssi;
239 }

255 /* Returns the current TX power in dBm */
256 static radio_value_t
257 get_tx_power(void)
258 {
259  int i;
260  uint8_t reg_val = TXPOWER;
261 
262  if(TXCTRL == CC2530_RF_TX_POWER_TXCTRL_MIN_VAL) {
263  return OUTPUT_POWER_MIN;
264  }
265 
266  /*
267  * Find the TXPOWER value in the lookup table
268  * If the value has been written with set_tx_power, we should be able to
269  * find the exact value. However, in case the register has been written in
270  * a different fashion, we return the immediately lower value of the lookup
271  */
272  for(i = 0; i < OUTPUT_CONFIG_COUNT; i++) {
273  if(reg_val >= output_power[i].txpower_val) {
274  return output_power[i].power;
275  }
276  }
277  return OUTPUT_POWER_MIN;
278 }
279 /*---------------------------------------------------------------------------*/
280 /*
281  * Set TX power to 'at least' power dBm
282  * This works with a lookup table. If the value of 'power' does not exist in
283  * the lookup table, TXPOWER will be set to the immediately higher available
284  * value
285  */
286 static void
287 set_tx_power(radio_value_t power)
288 {
289  int i;
290 
291  if(power <= output_power[OUTPUT_CONFIG_COUNT - 1].power) {
292  TXCTRL = CC2530_RF_TX_POWER_TXCTRL_MIN_VAL;
293  TXPOWER = output_power[OUTPUT_CONFIG_COUNT - 1].txpower_val;
294  return;
295  }
296 
297  for(i = OUTPUT_CONFIG_COUNT - 2; i >= 0; --i) {
298  if(power <= output_power[i].power) {
299  /* Perhaps an earlier call set TXCTRL to 0x09. Restore */
300  TXCTRL = CC2530_RF_TX_POWER_TXCTRL_DEF_VAL;
301  TXPOWER = output_power[i].txpower_val;
302  return;
303  }
304  }
305 }

329 static int
330 init(void)
331 {
332  PUTSTRING("RF: Init\n");
333 
334  if(rf_flags & RF_ON) {
335  return 0;
336  }
337 
338 #if CC2530_RF_LOW_POWER_RX
339  /* Reduce RX power consumption current to 20mA at the cost of sensitivity */
340  RXCTRL = 0x00;
341  FSCTRL = 0x50;
342 #else
343  RXCTRL = 0x3F;
344  FSCTRL = 0x55;
345 #endif /* CC2530_RF_LOW_POWER_RX */
346 
347  CCACTRL0 = CC2530_RF_CCA_THRES;
348 
349  /*
350  * According to the user guide, these registers must be updated from their
351  * defaults for optimal performance
352  *
353  * Table 23-6, Sec. 23.15.1, p. 259
354  */
355  TXFILTCFG = 0x09; /* TX anti-aliasing filter */
356  AGCCTRL1 = 0x15; /* AGC target value */
357  FSCAL1 = 0x00; /* Reduce the VCO leakage */
358 
359  /* Auto ACKs and CRC calculation, default RX and TX modes with FIFOs */
360  FRMCTRL0 = FRMCTRL0_AUTOCRC;
361 #if CC2530_RF_AUTOACK
362  FRMCTRL0 |= FRMCTRL0_AUTOACK;
363 #endif
364 
365  /* Disable source address matching and autopend */
366  SRCMATCH = 0; /* investigate */
367 
368  /* MAX FIFOP threshold */
369  FIFOPCTRL = CC2530_RF_MAX_PACKET_LEN;
370 
371  TXPOWER = CC2530_RF_TX_POWER;
372 
373  RF_TX_LED_OFF();
374  RF_RX_LED_OFF();
375 
376  rf_flags |= RF_ON;
377 
378  return 1;
379 }
380 /*---------------------------------------------------------------------------*/
381 static int
382 prepare(const void *payload, unsigned short payload_len)
383 {
384  uint8_t i;
385 
386  PUTSTRING("RF: Prepare 0x");
387  PUTHEX(payload_len + CHECKSUM_LEN);
388  PUTSTRING(" bytes\n");
389 
390  /*
391  * When we transmit in very quick bursts, make sure previous transmission
392  * is not still in progress before re-writing to the TX FIFO
393  */
394  while(FSMSTAT1 & FSMSTAT1_TX_ACTIVE);
395 
396  if((rf_flags & RX_ACTIVE) == 0) {
397  on();
398  }
399 
400  CC2530_CSP_ISFLUSHTX();
401 
402  PUTSTRING("RF: data = ");
403  /* Send the phy length byte first */
404  RFD = payload_len + CHECKSUM_LEN; /* Payload plus FCS */
405  for(i = 0; i < payload_len; i++) {
406  RFD = ((unsigned char *)(payload))[i];
407  PUTHEX(((unsigned char *)(payload))[i]);
408  }
409  PUTSTRING("\n");
410 
411  /* Leave space for the FCS */
412  RFD = 0;
413  RFD = 0;
414 
415  return 0;
416 }
417 /*---------------------------------------------------------------------------*/
418 static int
419 transmit(unsigned short transmit_len)
420 {
421  uint8_t counter;
422  int ret = RADIO_TX_ERR;
423  rtimer_clock_t t0;
424  transmit_len; /* hush the warning */
425 
426  if(!(rf_flags & RX_ACTIVE)) {
427  t0 = RTIMER_NOW();
428  on();
429  rf_flags |= WAS_OFF;
430  while(RTIMER_CLOCK_LT(RTIMER_NOW(), t0 + ONOFF_TIME));
431  }
432 
433  if(channel_clear() == CC2530_RF_CCA_BUSY) {
434  RIMESTATS_ADD(contentiondrop);
435  return RADIO_TX_COLLISION;
436  }
437 
438  /*
439  * prepare() double checked that TX_ACTIVE is low. If SFD is high we are
440  * receiving. Abort transmission and bail out with RADIO_TX_COLLISION
441  */
442  if(FSMSTAT1 & FSMSTAT1_SFD) {
443  RIMESTATS_ADD(contentiondrop);
444  return RADIO_TX_COLLISION;
445  }
446 
447  /* Start the transmission */
448  RF_TX_LED_ON();
449  ENERGEST_OFF(ENERGEST_TYPE_LISTEN);
450  ENERGEST_ON(ENERGEST_TYPE_TRANSMIT);
451 
452  CC2530_CSP_ISTXON();
453 
454  counter = 0;
455  while(!(FSMSTAT1 & FSMSTAT1_TX_ACTIVE) && (counter++ < 3)) {
456  clock_delay_usec(6);
457  }
458 
459  if(!(FSMSTAT1 & FSMSTAT1_TX_ACTIVE)) {
460  PUTSTRING("RF: TX never active.\n");
461  CC2530_CSP_ISFLUSHTX();
462  ret = RADIO_TX_ERR;
463  } else {
464  /* Wait for the transmission to finish */
465  while(FSMSTAT1 & FSMSTAT1_TX_ACTIVE);
466  ret = RADIO_TX_OK;
467  }
468  ENERGEST_OFF(ENERGEST_TYPE_TRANSMIT);
469  ENERGEST_ON(ENERGEST_TYPE_LISTEN);
470 
471  if(rf_flags & WAS_OFF) {
472  off();
473  }
474 
475  RIMESTATS_ADD(lltx);
476 
477  RF_TX_LED_OFF();
478 
479  /* OK, sent. We are now ready to send more */
480  return ret;
481 }

489 /*---------------------------------------------------------------------------*/
490 static int
491 read(void *buf, unsigned short bufsize)
492 {
493  uint8_t i;
494  uint8_t len;
495  uint8_t crc_corr;
496  int8_t rssi;
497 
498  PUTSTRING("RF: Read\n");
499 
500  /* Check the length */
501  len = RFD;
502 
503  /* Check for validity */
504  if(len > CC2530_RF_MAX_PACKET_LEN) {
505  /* Oops, we must be out of sync. */
506  PUTSTRING("RF: bad sync\n");
507 
508  RIMESTATS_ADD(badsynch);
509  CC2530_CSP_ISFLUSHRX();
510  return 0;
511  }
512 
513  if(len <= CC2530_RF_MIN_PACKET_LEN) {
514  PUTSTRING("RF: too short\n");
515 
516  RIMESTATS_ADD(tooshort);
517  CC2530_CSP_ISFLUSHRX();
518  return 0;
519  }
520 
521  if(len - CHECKSUM_LEN > bufsize) {
522  PUTSTRING("RF: too long\n");
523 
524  RIMESTATS_ADD(toolong);
525  CC2530_CSP_ISFLUSHRX();
526  return 0;
527  }
528 
529  RF_RX_LED_ON();
530 
531  PUTSTRING("RF: read (0x");
532  PUTHEX(len);
533  PUTSTRING(" bytes) = ");
534  len -= CHECKSUM_LEN;
535  for(i = 0; i < len; ++i) {
536  ((unsigned char *)(buf))[i] = RFD;
537  PUTHEX(((unsigned char *)(buf))[i]);
538  }
539  PUTSTRING("\n");
540 
541  /* Read the RSSI and CRC/Corr bytes */
542  rssi = ((int8_t) RFD) - RSSI_OFFSET;
543  crc_corr = RFD;
544 
545  /* MS bit CRC OK/Not OK, 7 LS Bits, Correlation value */
546  if(crc_corr & CRC_BIT_MASK) {
547  packetbuf_set_attr(PACKETBUF_ATTR_RSSI, rssi);
548  packetbuf_set_attr(PACKETBUF_ATTR_LINK_QUALITY, crc_corr & LQI_BIT_MASK);
549  RIMESTATS_ADD(llrx);
550  } else {
551  RIMESTATS_ADD(badcrc);
552  CC2530_CSP_ISFLUSHRX();
553  RF_RX_LED_OFF();
554  return 0;
555  }
556 
557  /* If FIFOP==1 and FIFO==0 then we had a FIFO overflow at some point. */
558  if((FSMSTAT1 & (FSMSTAT1_FIFO | FSMSTAT1_FIFOP)) == FSMSTAT1_FIFOP) {
559  /*
560  * If we reach here means that there might be more intact packets in the
561  * FIFO despite the overflow. This can happen with bursts of small packets.
562  *
563  * Only flush if the FIFO is actually empty. If not, then next pass we will
564  * pick up one more packet or flush due to an error.
565  */
566  if(!RXFIFOCNT) {
567  CC2530_CSP_ISFLUSHRX();
568  }
569  }
570 
571  RF_RX_LED_OFF();
572 
573  return (len);
574 }

606 on(void)
607 {
608  if(!(rf_flags & RX_ACTIVE)) {
609  CC2530_CSP_ISFLUSHRX();
610  CC2530_CSP_ISRXON();
611 
612  rf_flags |= RX_ACTIVE;
613  }
614

631 static radio_result_t
632 get_value(radio_param_t param, radio_value_t *value)
633 {
634  if(!value) {
635  return RADIO_RESULT_INVALID_VALUE;
636  }
637 
638  switch(param) {
639  case RADIO_PARAM_POWER_MODE:
640  *value = RXENABLE == 0 ? RADIO_POWER_MODE_OFF : RADIO_POWER_MODE_ON;
641  return RADIO_RESULT_OK;
642  case RADIO_PARAM_CHANNEL:
643  *value = (radio_value_t)get_channel();
644  return RADIO_RESULT_OK;
645  case RADIO_PARAM_PAN_ID:
646  *value = get_pan_id();
647  return RADIO_RESULT_OK;
648  case RADIO_PARAM_16BIT_ADDR:
649  *value = get_short_addr();
650  return RADIO_RESULT_OK;
651  case RADIO_PARAM_RX_MODE:
652  *value = 0;
653  if(FRMFILT0 & FRMFILT0_FRAME_FILTER_EN) {
654  *value |= RADIO_RX_MODE_ADDRESS_FILTER;
655  }
656  if(FRMCTRL0 & FRMCTRL0_AUTOACK) {
657  *value |= RADIO_RX_MODE_AUTOACK;
658  }
659  return RADIO_RESULT_OK;
660  case RADIO_PARAM_TXPOWER:
661  *value = get_tx_power();
662  return RADIO_RESULT_OK;
663  case RADIO_PARAM_CCA_THRESHOLD:
664  *value = get_cca_threshold();
665  return RADIO_RESULT_OK;
666  case RADIO_PARAM_RSSI:
667  *value = get_rssi();
668  return RADIO_RESULT_OK;
669  case RADIO_CONST_CHANNEL_MIN:
670  *value = CC2530_RF_CHANNEL_MIN;
671  return RADIO_RESULT_OK;
672  case RADIO_CONST_CHANNEL_MAX:
673  *value = CC2530_RF_CHANNEL_MAX;
674  return RADIO_RESULT_OK;
675  case RADIO_CONST_TXPOWER_MIN:
676  *value = OUTPUT_POWER_MIN;
677  return RADIO_RESULT_OK;
678  case RADIO_CONST_TXPOWER_MAX:
679  *value = OUTPUT_POWER_MAX;
680  return RADIO_RESULT_OK;
681  default:
682  return RADIO_RESULT_NOT_SUPPORTED;
683  }
684 }
685 /*---------------------------------------------------------------------------*/
686 static radio_result_t
687 set_value(radio_param_t param, radio_value_t value)
688 {
689  switch(param) {
690  case RADIO_PARAM_POWER_MODE:
691  if(value == RADIO_POWER_MODE_ON) {
692  on();
693  return RADIO_RESULT_OK;
694  }
695  if(value == RADIO_POWER_MODE_OFF) {
696  off();
697  return RADIO_RESULT_OK;
698  }
699  return RADIO_RESULT_INVALID_VALUE;
700  case RADIO_PARAM_CHANNEL:
701  if(value < CC2530_RF_CHANNEL_MIN || value > CC2530_RF_CHANNEL_MAX) {
702  return RADIO_RESULT_INVALID_VALUE;
703  }
704  if(set_channel(value) == CC2530_RF_CHANNEL_SET_ERROR) {
705  return RADIO_RESULT_ERROR;
706  }
707  return RADIO_RESULT_OK;
708  case RADIO_PARAM_PAN_ID:
709  set_pan_id(value & 0xffff);
710  return RADIO_RESULT_OK;
711  case RADIO_PARAM_16BIT_ADDR:
712  set_short_addr(value & 0xffff);
713  return RADIO_RESULT_OK;
714  case RADIO_PARAM_RX_MODE:
715  if(value & ~(RADIO_RX_MODE_ADDRESS_FILTER |
716  RADIO_RX_MODE_AUTOACK)) {
