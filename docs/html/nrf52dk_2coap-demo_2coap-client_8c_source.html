88 static void
89 notification_callback(coap_observee_t *obs, void *notification,
90  coap_notification_flag_t flag)
91 {
92  int len = 0;
93  const uint8_t *payload = NULL;
94 
95 
96  PRINTF("Notification handler\n");
97  PRINTF("Observee URI: %s\n", obs->url);
98 
99  if (notification) {
100  len = coap_get_payload(notification, &payload);
101  }
102 
103  (void)len;
104 
105  switch (flag) {
106  case NOTIFICATION_OK:
107  PRINTF("NOTIFICATION OK: %*s\n", len, (char *)payload);
108  if (*payload == '1') {
109  leds_on(OBS_LED);
110  } else {
111  leds_off(OBS_LED);
112  }
113  break;
114  case OBSERVE_OK: /* server accepeted observation request */
115  PRINTF("OBSERVE_OK: %*s\n", len, (char *)payload);
116  if (*payload == '1') {
117  leds_on(OBS_LED);
118  } else {
119  leds_off(OBS_LED);
120  }
121  leds_on(SUBS_LED);
122  ctimer_set(&ct, CLOCK_SECOND, observe_led_off, NULL);
123  break;
124  case OBSERVE_NOT_SUPPORTED:
125  PRINTF("OBSERVE_NOT_SUPPORTED: %*s\n", len, (char *)payload);
126  obs = NULL;
127  break;
128  case ERROR_RESPONSE_CODE:
129  PRINTF("ERROR_RESPONSE_CODE: %*s\n", len, (char *)payload);
130  obs = NULL;
131  break;
132  case NO_REPLY_FROM_SERVER:
133  PRINTF("NO_REPLY_FROM_SERVER: "
134  "removing observe registration with token %x%x\n",
135  obs->token[0], obs->token[1]);
136  obs = NULL;
137  break;
138  }
139 }
