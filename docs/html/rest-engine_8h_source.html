137 /*
138  * Macro to define a periodic resource.
139  * The corresponding [name]_periodic_handler() function will be called every period.
140  * For instance polling a sensor and publishing a changed value to subscribed clients would be done there.
141  * The subscriber list will be maintained by the final_handler rest_subscription_handler() (see rest-mapping header file).
142  */
143 #define PERIODIC_RESOURCE(name, attributes, get_handler, post_handler, put_handler, delete_handler, period, periodic_handler) \
144  periodic_resource_t periodic_##name; \
145  resource_t name = { NULL, NULL, IS_OBSERVABLE | IS_PERIODIC, attributes, get_handler, post_handler, put_handler, delete_handler, { .periodic = &periodic_##name } }; \
146  periodic_resource_t periodic_##name = { NULL, &name, period, { { 0 } }, periodic_handler };
