127  PRINTF("Server IPv6 addresses: ");
128  for(i = 0; i < UIP_DS6_ADDR_NB; i++) {
129  state = uip_ds6_if.addr_list[i].state;
130  if(state == ADDR_TENTATIVE || state == ADDR_PREFERRED) {
131  PRINT6ADDR(&uip_ds6_if.addr_list[i].ipaddr);
132  PRINTF("\n");
133  /* hack to make address "final" */
