282 #define HEX_TO_INT(x) (isdigit(x) ? x - '0' : x - 'W')
283 static size_t
284 url_unescape(const char *src, size_t srclen, char *dst, size_t dstlen)
285 {
286  size_t i, j;
287  int a, b;
288 
289  for(i = j = 0; i < srclen && j < dstlen - 1; i++, j++) {
290  if(src[i] == '%' && isxdigit(*(unsigned char *)(src + i + 1))
291  && isxdigit(*(unsigned char *)(src + i + 2))) {
292  a = tolower(*(unsigned char *)(src + i + 1));
293  b = tolower(*(unsigned char *)(src + i + 2));
294  dst[j] = ((HEX_TO_INT(a) << 4) | HEX_TO_INT(b)) & 0xff;
295  i += 2;
296  } else if(src[i] == '+') {
297  dst[j] = ' ';
298  } else {
299  dst[j] = src[i];
300  }
301  }
302 
303  dst[j] = '\0';
304 
305  return i == srclen;
306 }

336 static
337 PT_THREAD(enqueue_chunk(struct httpd_state *s, uint8_t immediate,
338  const char *format, ...))
339 {
340  va_list ap;
341 
342  PSOCK_BEGIN(&s->sout);
343 
344  va_start(ap, format);
345 
346  s->tmp_buf_len = vsnprintf(s->tmp_buf, TMP_BUF_SIZE, format, ap);
347 
348  va_end(ap);
349 
350  if(s->blen + s->tmp_buf_len < HTTPD_SIMPLE_MAIN_BUF_SIZE) {
351  /* Enough space for the entire chunk. Copy over */
352  memcpy(&s->buf[s->blen], s->tmp_buf, s->tmp_buf_len);
353  s->blen += s->tmp_buf_len;
354  } else {
355  memcpy(&s->buf[s->blen], s->tmp_buf, HTTPD_SIMPLE_MAIN_BUF_SIZE - s->blen);
356  s->tmp_buf_copied = HTTPD_SIMPLE_MAIN_BUF_SIZE - s->blen;
357  s->blen = HTTPD_SIMPLE_MAIN_BUF_SIZE;
358  PSOCK_SEND(&s->sout, (uint8_t *)s->buf, s->blen);
359  s->blen = 0;
360  if(s->tmp_buf_copied < s->tmp_buf_len) {
361  memcpy(s->buf, &s->tmp_buf[s->tmp_buf_copied],
362  s->tmp_buf_len - s->tmp_buf_copied);
363  s->blen += s->tmp_buf_len - s->tmp_buf_copied;
364  }
365  }
366 
367  if(immediate != 0 && s->blen > 0) {
368  PSOCK_SEND(&s->sout, (uint8_t *)s->buf, s->blen);
369  s->blen = 0;
370  }
371 
372  PSOCK_END(&s->sout);
373 }

907 static void
908 parse_post_request_chunk(char *buf, int buf_len, int last_chunk)
909 {
910  int i;
911  int finish;
912 
913  for(i = 0; i < buf_len; i++) {
914  switch(state) {
915  case PARSE_POST_STATE_INIT:
916  state = PARSE_POST_STATE_MORE;
917  /* continue */
918  case PARSE_POST_STATE_MORE:
919  memset(key, 0, PARSE_POST_BUF_SIZES);
920  memset(val, 0, PARSE_POST_BUF_SIZES);
921  memset(val_escaped, 0, PARSE_POST_BUF_SIZES);
922  key_len = 0;
923  val_len = 0;
924  state = PARSE_POST_STATE_READING_KEY;
925  /* continue */
926  case PARSE_POST_STATE_READING_KEY:
927  if(buf[i] == ISO_equal) {
928  state = PARSE_POST_STATE_READING_VAL;
929  } else if(buf[i] == ISO_amp) {
930  /* Don't accept an amp while reading a key */
931  state = PARSE_POST_STATE_ERROR;
932  } else {
933  /* Make sure we don't overshoot key's boundary */
934  if(key_len <= PARSE_POST_MAX_POS) {
935  key[key_len] = buf[i];
936  key_len++;
937  } else {
938  /* Not enough space for the key. Abort */
939  state = PARSE_POST_STATE_ERROR;
940  }
941  }
942  break;
943  case PARSE_POST_STATE_READING_VAL:
944  finish = 0;
945  if(buf[i] == ISO_amp) {
946  finish = 1;
947  } else if(buf[i] == ISO_equal) {
948  /* Don't accept an '=' while reading a val */
949  state = PARSE_POST_STATE_ERROR;
950  } else {
951  /* Make sure we don't overshoot key's boundary */
952  if(val_len <= PARSE_POST_MAX_POS) {
953  val[val_len] = buf[i];
954  val_len++;
955  /* Last character of the last chunk */
956  if((i == buf_len - 1) && (last_chunk == 1)) {
957  finish = 1;
958  }
959  } else {
960  /* Not enough space for the value. Abort */
961  state = PARSE_POST_STATE_ERROR;
962  }
963  }
964 
965  if(finish == 1) {
966  /*
967  * Done reading a key=value pair, either because we encountered an amp
968  * or because we reached the end of the message body.
969  *
970  * First, unescape the value.
971  *
972  * Then invoke handlers. We will bail out with PARSE_POST_STATE_ERROR,
973  * unless the key-val gets correctly processed
974  */
975  url_unescape(val, val_len, val_escaped, PARSE_POST_BUF_SIZES);
976  val_len = strlen(val_escaped);
977 
978  for(handler = list_head(post_handlers); handler != NULL;
979  handler = list_item_next((void *)handler)) {
980  if(handler->handler != NULL) {
981  finish = handler->handler(key, key_len, val_escaped, val_len);
982  }
983  if(finish == HTTPD_SIMPLE_POST_HANDLER_ERROR) {
984  state = PARSE_POST_STATE_ERROR;
985  break;
986  } else if(finish == HTTPD_SIMPLE_POST_HANDLER_OK) {
987  /* Restart the state machine to expect the next pair */
988  state = PARSE_POST_STATE_MORE;
989 
990  /*
991  * At least one handler returned OK, therefore we must generate a
992  * new config event when we're done.
993  */
994  config_ok = 1;
995  break;
996  }
997  /* Else, continue */
998  }
999  }
1000  break;
1001  case PARSE_POST_STATE_ERROR:
1002  /* If we entered the error state earlier, do nothing */
1003  return;
1004  default:
1005  break;
1006  }
1007  }
1008 }

1097  http_content_type_plain,
1098  NULL,
1099  http_header_con_close));
1100  PT_WAIT_THREAD(&s->outputpt, send_string(s, "Service Unavailable\n"));

1122  http_content_type_html,
1123  NULL,
1124  http_header_con_close));
1125  PT_WAIT_THREAD(&s->outputpt, s->script(s));
1126  }
1127  }

1130  PT_END(&s->outputpt);
