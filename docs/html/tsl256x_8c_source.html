66 static uint16_t
67 calculate_lux(uint8_t *buf)
68 {
69  uint32_t ch0, ch1, chscale = 0;
70  uint32_t ratio = 0;
71  uint32_t lratio, tmp = 0;
72  uint16_t buffer[2];
73 
74  /* The calculations below assume the integration time is 402ms and the gain
75  * is 16x (nominal), if not then it is required to normalize the reading
76  * before converting to lux
77  */
78 
79  buffer[0] = (buf[1] << 8 | (buf[0]));
80  buffer[1] = (buf[3] << 8 | (buf[2]));
81 
82  switch(timming) {
83  case TSL256X_TIMMING_INTEG_402MS:
84  chscale = (1 << CH_SCALE);
85  break;
86  case TSL256X_TIMMING_INTEG_101MS:
87  chscale = CHSCALE_TINT1;
88  break;
89  case TSL256X_TIMMING_INTEG_13_7MS:
90  chscale = CHSCALE_TINT0;
91  break;
92  }
93 
94  if(!gain) {
95  chscale = chscale << 4;
96  }
97 
98  ch0 = (buffer[0] * chscale) >> CH_SCALE;
99  ch1 = (buffer[1] * chscale) >> CH_SCALE;
100 
101  if(ch0 > 0) {
102  ratio = (ch1 << CH_SCALE);
103  ratio = ratio / ch0;
104  }
105 
106  lratio = (ratio + 1) >> 1;
107 
108  if((lratio >= 0) && (lratio <= K1T)) {
109  tmp = (ch0 * B1T) - (ch1 * M1T);
110  } else if(lratio <= K2T) {
111  tmp = (ch0 * B2T) - (ch1 * M2T);
112  } else if(lratio <= K3T) {
113  tmp = (ch0 * B3T) - (ch1 * M3T);
114  } else if(lratio <= K4T) {
115  tmp = (ch0 * B4T) - (ch1 * M4T);
116  } else if(lratio <= K5T) {
117  tmp = (ch0 * B5T) - (ch1 * M5T);
118  } else if(lratio <= K6T) {
119  tmp = (ch0 * B6T) - (ch1 * M6T);
120  } else if(lratio <= K7T) {
121  tmp = (ch0 * B7T) - (ch1 * M7T);
122  } else if(lratio > K8T) {
123  tmp = (ch0 * B8T) - (ch1 * M8T);
124  }
125 
126  if(tmp < 0) {
127  tmp = 0;
128  }
129 
130  tmp += (1 << (LUX_SCALE - 1));
131  return tmp >> LUX_SCALE;
132 }

134 static int
135 tsl256x_read_reg(uint8_t reg, uint8_t *buf, uint8_t regNum)
136 {
137  i2c_master_enable();
138  if(i2c_single_send(TSL256X_ADDR, reg) == I2C_MASTER_ERR_NONE) {
139  while(i2c_master_busy());
140  if(i2c_burst_receive(TSL256X_ADDR, buf, regNum) == I2C_MASTER_ERR_NONE) {
141  return TSL256X_SUCCESS;
142  }
143  }
144  return TSL256X_ERROR;
145 }

162 static int
163 tsl256x_on(void)
164 {
165  uint8_t buf[2];
166  buf[0] = (TSL256X_COMMAND + TSL256X_CONTROL);
167  buf[1] = TSL256X_CONTROL_POWER_ON;
168 
169  if(tsl256x_write_reg(buf, 2) == I2C_MASTER_ERR_NONE) {
170  if(i2c_single_receive(TSL256X_ADDR, &buf[0]) == I2C_MASTER_ERR_NONE) {
171  if((buf[0] & 0x0F) == TSL256X_CONTROL_POWER_ON) {
172  PRINTF("TSL256X: powered on\n");
173  return TSL256X_SUCCESS;
174  }
175  }
176  }
177 
178  PRINTF("TSL256X: failed to power on\n");
179  return TSL256X_ERROR;
180 }

221 static int
222 tsl256x_read_sensor(uint16_t *lux)
223 {
224  uint8_t buf[4];
225 
226  /* This is hardcoded to use word write/read operations */
227  if(tsl256x_read_reg((TSL256X_COMMAND + TSL256X_D0LOW),
228  &buf[0], 2) == TSL256X_SUCCESS) {
229  if(tsl256x_read_reg((TSL256X_COMMAND + TSL256X_D1LOW),
230  &buf[2], 2) == TSL256X_SUCCESS) {
231 
232  PRINTF("TSL256X: CH0 0x%02X%02X CH1 0x%02X%02X\n", buf[1], buf[0],
233  buf[3], buf[2]);
234  *lux = calculate_lux(buf);
235  return TSL256X_SUCCESS;
236  }
237  }
238  PRINTF("TSL256X: failed to read\n");
239  return TSL256X_ERROR;
240 }

244 PROCESS_THREAD(tsl256x_int_process, ev, data)
245 {
246  PROCESS_EXITHANDLER();
247  PROCESS_BEGIN();
248 
249  while(1) {
250  PROCESS_YIELD_UNTIL(ev == PROCESS_EVENT_POLL);
251  tsl256x_clear_interrupt();
252  tsl256x_int_callback(0);
253  }
254  PROCESS_END();
255 }

266 static int
267 configure(int type, int value)
268 {
269  uint8_t buf[3];
270 
271  if((type != TSL256X_ACTIVE) && (type != TSL256X_INT_OVER) &&
272  (type != TSL256X_INT_BELOW) && (type != TSL256X_INT_DISABLE) &&
273  (type != TSL256X_TIMMING_CFG)) {
274  PRINTF("TSL256X: invalid start value\n");
275  return TSL256X_ERROR;
276  }
277 
278  /* As default the power-on values of the sensor are gain 1X, 402ms integration
279  * time (not nominal), with manual control disabled
280  */
281 
282  if(type == TSL256X_ACTIVE) {
283  if(value) {
284  i2c_init(I2C_SDA_PORT, I2C_SDA_PIN, I2C_SCL_PORT, I2C_SCL_PIN,
285  I2C_SCL_NORMAL_BUS_SPEED);
286 
287  /* Initialize interrupts handlers */
288  tsl256x_int_callback = NULL;
289 
290  /* Power on the sensor and check for the part number */
291  if(tsl256x_on() == TSL256X_SUCCESS) {
292  if(tsl256x_id_register(&buf[0]) == TSL256X_SUCCESS) {
293  if((buf[0] & TSL256X_ID_PARTNO_MASK) == TSL256X_EXPECTED_PARTNO) {
294 
295  /* Read the timming/gain configuration */
296  if(tsl256x_read_reg((TSL256X_COMMAND + TSL256X_TIMMING),
297  &buf[0], 1) == TSL256X_SUCCESS) {
298  gain = buf[0] & TSL256X_TIMMING_GAIN;
299  timming = buf[0] & TSL256X_TIMMING_INTEG_MASK;
300  PRINTF("TSL256X: enabled, timming %u gain %u\n", timming, gain);
301 
302  /* Restart the over interrupt threshold */
303  buf[0] = (TSL256X_COMMAND + TSL256X_THRHIGHLOW);
304  buf[1] = 0xFF;
305  buf[2] = 0xFF;
306 
307  if(tsl256x_write_reg(buf, 3) != TSL256X_SUCCESS) {
308  PRINTF("TSL256X: failed to clear over interrupt\n");
309  return TSL256X_ERROR;
310  }
311 
312  /* Restart the below interrupt threshold */
313  buf[0] = (TSL256X_COMMAND + TSL256X_THRLOWLOW);
314  buf[1] = 0x00;
315  buf[2] = 0x00;
316 
317  if(tsl256x_write_reg(buf, 3) != TSL256X_SUCCESS) {
318  PRINTF("TSL256X: failed to clear below interrupt\n");
319  return TSL256X_ERROR;
320  }
321 
322  /* Clear any pending interrupt */
323  if(tsl256x_clear_interrupt() == TSL256X_SUCCESS) {
324  enabled = 1;
325  return TSL256X_SUCCESS;
326  }
327  }
328  }
329  }
330  }
331  return TSL256X_ERROR;
332  } else {
333  if(tsl256x_off() == TSL256X_SUCCESS) {
334  PRINTF("TSL256X: stopped\n");
335  enabled = 0;
336  return TSL256X_SUCCESS;
337  }
338  return TSL256X_ERROR;
339  }
340  }
341 
342  if(!enabled) {
343  PRINTF("TSL256X: sensor not started\n");
344  return TSL256X_ERROR;
345  }
346 
347  if(type == TSL256X_INT_DISABLE) {
348 
349  /* Ensure the GPIO doesn't generate more interrupts, this may affect others
350  * I2C digital sensors using the bus and sharing this pin, so an user may
351  * comment the line below
352  */
353  GPIO_DISABLE_INTERRUPT(TSL256X_INT_PORT_BASE, TSL256X_INT_PIN_MASK);
354 
355  /* This also wipes out the persistance value, to be reconfigured when
356  * enabling back the interruption
357  */
358  buf[0] = (TSL256X_COMMAND + TSL256X_INTERRUPT);
359  buf[1] = TSL256X_INTR_DISABLED;
360 
361  if(tsl256x_write_reg(buf, 2) != TSL256X_SUCCESS) {
362  PRINTF("TSL256X: failed to disable the interrupt\n");
363  return TSL256X_ERROR;
364  }
365  return TSL256X_SUCCESS;
366  }
367 
368  /* Configure the timming and gain */
369  if(type == TSL256X_TIMMING_CFG) {
370  if((value != TSL256X_G16X_402MS) && (value != TSL256X_G1X_402MS) &&
371  (value != TSL256X_G1X_101MS) && (value != TSL256X_G1X_13_7MS)) {
372  PRINTF("TSL256X: invalid timming configuration values\n");
373  return TSL256X_ERROR;
374  }
375 
376  buf[0] = (TSL256X_COMMAND + TSL256X_TIMMING);
377  buf[1] = value;
378 
379  if(tsl256x_write_reg(buf, 2) == TSL256X_SUCCESS) {
380  if(value == TSL256X_G16X_402MS) {
381  gain = 1;
382  }
383 
384  switch(value) {
385  case TSL256X_G16X_402MS:
386  case TSL256X_G1X_402MS:
387  timming = TSL256X_TIMMING_INTEG_402MS;
388  break;
389  case TSL256X_G1X_101MS:
390  timming = TSL256X_TIMMING_INTEG_101MS;
391  break;
392  case TSL256X_G1X_13_7MS:
393  timming = TSL256X_TIMMING_INTEG_13_7MS;
394  break;
395  }
396 
397  PRINTF("TSL256X: new timming %u gain %u\n", timming, gain);
398  return TSL256X_SUCCESS;
399  }
400  PRINTF("TSL256X: failed to configure timming\n");
401  return TSL256X_ERROR;
402  }
403 
404  /* From here we handle the interrupt configuration, it requires the interrupt
405  * callback handler to have been previously set using the TSL256X_REGISTER_INT
406  * macro
407  */
408 
409  buf[1] = ((uint8_t *)&value)[0];
410  buf[2] = ((uint8_t *)&value)[1];
411 
412  if(type == TSL256X_INT_OVER) {
413  buf[0] = (TSL256X_COMMAND + TSL256X_THRHIGHLOW);
414  } else if(type == TSL256X_INT_BELOW) {
415  buf[0] = (TSL256X_COMMAND + TSL256X_THRLOWLOW);
416  }
417 
418  if(tsl256x_write_reg(buf, 3) != TSL256X_SUCCESS) {
419  PRINTF("TSL256X: failed to set interrupt level\n");
420  return TSL256X_ERROR;
421  }
422 
423  /* Now configure the interruption register (level interrupt, 2 integration
424  * cycles after threshold has been reached (roughly 804ms if timming is 402ms)
425  */
426  buf[0] = (TSL256X_COMMAND + TSL256X_INTERRUPT);
427  buf[1] = (TSL256X_INTR_LEVEL << TSL256X_INTR_SHIFT);
428  buf[1] += TSL256X_INT_PERSIST_2_CYCLES;
429 
430  if(tsl256x_write_reg(buf, 2) != TSL256X_SUCCESS) {
431  PRINTF("TSL256X: failed to enable interrupt\n");
432  return TSL256X_ERROR;
433  }
434 
435  /* Configure the interrupts pins */
436  GPIO_SOFTWARE_CONTROL(TSL256X_INT_PORT_BASE, TSL256X_INT_PIN_MASK);
437  GPIO_SET_INPUT(TSL256X_INT_PORT_BASE, TSL256X_INT_PIN_MASK);
438 
439  /* Pull-up resistor, detect falling edge */
440  GPIO_DETECT_EDGE(TSL256X_INT_PORT_BASE, TSL256X_INT_PIN_MASK);
441  GPIO_TRIGGER_SINGLE_EDGE(TSL256X_INT_PORT_BASE, TSL256X_INT_PIN_MASK);
442  GPIO_DETECT_FALLING(TSL256X_INT_PORT_BASE, TSL256X_INT_PIN_MASK);
443  gpio_register_callback(tsl256x_interrupt_handler, I2C_INT_PORT, I2C_INT_PIN);
444 
445  /* Spin process until an interrupt is received */
446  process_start(&tsl256x_int_process, NULL);
447 
448  /* Enable interrupts */
449  GPIO_ENABLE_INTERRUPT(TSL256X_INT_PORT_BASE, TSL256X_INT_PIN_MASK);
450 
451  /* The RE-Mote revision A has this pin shared and with a pull-down resistor,
452  * for other platforms (like the firefly), change to enable pull-up internal
453  * resistor instead if no external pull-up is present.
454  */
455  ioc_set_over(I2C_INT_PORT, I2C_INT_PIN, IOC_OVERRIDE_PUE);
456  NVIC_EnableIRQ(I2C_INT_VECTOR);
457 
458  PRINTF("TSL256X: Interrupt configured\n");
459  return TSL256X_SUCCESS;
460 }
