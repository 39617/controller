76 rest_init_engine(void)
77 {
78  /* avoid initializing twice */
79  static uint8_t initialized = 0;
80 
81  if(initialized) {
82  PRINTF("REST engine process already running - double initialization?\n");

117 void
118 rest_activate_resource(resource_t *resource, char *path)
119 {
120  resource->url = path;
121  list_add(restful_services, resource);
122 
123  PRINTF("Activating: %s\n", resource->url);
124 
125  /* Only add periodic resources with a periodic_handler and a period > 0. */
126  if(resource->flags & IS_PERIODIC && resource->periodic->periodic_handler
127  && resource->periodic->period) {
128  PRINTF("Periodic resource: %p (%s)\n", resource->periodic,
129  resource->periodic->resource->url);
130  list_add(restful_periodic_services, resource->periodic);
131  if(process_is_running(&rest_engine_process)) {
132  PRINTF("Periodic: Set timer for /%s to %lu\n",
133  resource->url, resource->periodic->period);
134  PROCESS_CONTEXT_BEGIN(&rest_engine_process);
135  etimer_set(&resource->periodic->periodic_timer,
136  resource->periodic->period);
137  PROCESS_CONTEXT_END(&rest_engine_process);
138  }
139  }

150 int
151 rest_invoke_restful_service(void *request, void *response, uint8_t *buffer,
152  uint16_t buffer_size, int32_t *offset)
153 {
154  uint8_t found = 0;
155  uint8_t allowed = 1;
156 
157  resource_t *resource = NULL;
158  const char *url = NULL;
159  int url_len, res_url_len;
160 
161  url_len = REST.get_url(request, &url);
162 
163  for(resource = (resource_t *)list_head(restful_services);
164  resource; resource = resource->next) {
165 
166  /* if the web service handles that kind of requests and urls matches */
167  res_url_len = strlen(resource->url);
168 
169  if((url_len == res_url_len
170  || (url_len > res_url_len
171  && (resource->flags & HAS_SUB_RESOURCES)
172  && url[res_url_len] == '/'))
173  && strncmp(resource->url, url, res_url_len) == 0) {
174  found = 1;
175 
176  rest_resource_flags_t method = REST.get_method_type(request);
177 
178  PRINTF("/%s, method %u, resource->flags %u\n", resource->url,
179  (uint16_t)method, resource->flags);
180 
181  if((method & METHOD_GET) && resource->get_handler != NULL) {
182  /* call handler function */
183  resource->get_handler(request, response, buffer, buffer_size, offset);
184  } else if((method & METHOD_POST) && resource->post_handler != NULL) {
185  /* call handler function */
186  resource->post_handler(request, response, buffer, buffer_size,
187  offset);
188  } else if((method & METHOD_PUT) && resource->put_handler != NULL) {
189  /* call handler function */
190  resource->put_handler(request, response, buffer, buffer_size, offset);
191  } else if((method & METHOD_DELETE) && resource->delete_handler != NULL) {
192  /* call handler function */
193  resource->delete_handler(request, response, buffer, buffer_size,
194  offset);
195  } else {
196  allowed = 0;
197  REST.set_response_status(response, REST.status.METHOD_NOT_ALLOWED);
198  }
199  break;
200  }
201  }
202  if(!found) {
203  REST.set_response_status(response, REST.status.NOT_FOUND);
204  } else if(allowed) {
205  /* final handler for special flags */
206  if(resource->flags & IS_OBSERVABLE) {
207  REST.subscription_handler(resource, request, response);
208  }
209  }
210  return found & allowed;
211 }

213 PROCESS_THREAD(rest_engine_process, ev, data)
214 {
215  PROCESS_BEGIN();
216 
217  /* pause to let REST server finish adding resources. */
218  PROCESS_PAUSE();
219 
220  /* initialize the PERIODIC_RESOURCE timers, which will be handled by this process. */
221  periodic_resource_t *periodic_resource = NULL;
222 
223  for(periodic_resource =
224  (periodic_resource_t *)list_head(restful_periodic_services);
225  periodic_resource; periodic_resource = periodic_resource->next) {
226  if(periodic_resource->periodic_handler && periodic_resource->period) {
227  PRINTF("Periodic: Set timer for /%s to %lu\n",
228  periodic_resource->resource->url, periodic_resource->period);
229  etimer_set(&periodic_resource->periodic_timer,
230  periodic_resource->period);
231  }
232  }
233  while(1) {
234  PROCESS_WAIT_EVENT();
235 
236  if(ev == PROCESS_EVENT_TIMER) {
237  for(periodic_resource =
238  (periodic_resource_t *)list_head(restful_periodic_services);
239  periodic_resource; periodic_resource = periodic_resource->next) {
240  if(periodic_resource->period
241  && etimer_expired(&periodic_resource->periodic_timer)) {
242 
243  PRINTF("Periodic: etimer expired for /%s (period: %lu)\n",
244  periodic_resource->resource->url, periodic_resource->period);
245 
246  /* Call the periodic_handler function, which was checked during adding to list. */
247  (periodic_resource->periodic_handler)();
248 
249  etimer_reset(&periodic_resource->periodic_timer);
250  }
251  }
252  }
253  }
254 
255  PROCESS_END();
256 }
