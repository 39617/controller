255  //Dest address
256  uip_lladdr_t destAddr;
257  uip_lladdr_t *destAddrPtr = NULL;
258 
259  PRINTF("Packet type: %x\n", ((struct uip_eth_hdr *) ethHeader)->type);
260 
261  //RUM doesn't support sending data
262  #if UIP_CONF_USE_RUM
263  return;
264  #endif
265 
266  //If not IPv6 we don't do anything
267  if (((struct uip_eth_hdr *) ethHeader)->type != UIP_HTONS(UIP_ETHTYPE_IPV6)) {
268  PRINTF("eth2low: Packet is not IPv6, dropping\n");
269 /* rndis_stat.txbad++; */
270  uip_clear_buf();

272  }
273 
274  // In sniffer mode we don't ever send anything
275  if (usbstick_mode.sendToRf == 0) {
276  uip_clear_buf();
277  return;
278  }
279 
280  /* IPv6 uses 33-33-xx-xx-xx-xx prefix for multicast ND stuff */
281  if ( (((struct uip_eth_hdr *) ethHeader)->dest.addr[0] == 0x33) &&
282  (((struct uip_eth_hdr *) ethHeader)->dest.addr[1] == 0x33) )
283  {
284  PRINTF("eth2low: Ethernet multicast packet received\n");
285  ;//Do Nothing
286  } else if ( (((struct uip_eth_hdr *) ethHeader)->dest.addr[0] == 0xFF) &&
287  (((struct uip_eth_hdr *) ethHeader)->dest.addr[1] == 0xFF) &&
288  (((struct uip_eth_hdr *) ethHeader)->dest.addr[2] == 0xFF) &&
289  (((struct uip_eth_hdr *) ethHeader)->dest.addr[3] == 0xFF) &&
290  (((struct uip_eth_hdr *) ethHeader)->dest.addr[4] == 0xFF) &&
291  (((struct uip_eth_hdr *) ethHeader)->dest.addr[5] == 0xFF) ) {
292  /* IPv6 does not use broadcast addresses, hence this should not happen */
293  PRINTF("eth2low: Ethernet broadcast address received, should not happen?\n");
294 /* rndis_stat.txbad++; */
295  uip_clear_buf();
296  return;
297  } else {
298  PRINTF("eth2low: Addressed packet received... ");
299  //Check this returns OK
300  if (mac_createSicslowpanLongAddr( &(((struct uip_eth_hdr *) ethHeader)->dest.addr[0]), &destAddr) == 0) {
301  PRINTF(" translation failed\n");
302 /* rndis_stat.txbad++; */
303  uip_clear_buf();
304  return;
305  }
306  PRINTF(" translated OK\n");
307  destAddrPtr = &destAddr;
308  }
309 
310  //Remove header from length before passing onward
311  uip_len -= UIP_LLH_LEN;
312 
313  //Some IP packets have link layer in them, need to change them around!
314  if (usbstick_mode.translate) {
315  uint8_t transReturn =
316  mac_translateIPLinkLayer(ll_802154_type);
317  PRINTF("IPTranslation: returns %d\n", transReturn);
318  }
319 
320  if (usbstick_mode.sendToRf){
321  tcpip_output(destAddrPtr);
322 /* rndis_stat.txok++; */
323  }
324 
325  uip_clear_buf();
326

427 int8_t mac_translateIcmpLinkLayer(lltype_t target)
428 {
429  uint16_t icmp_opt_offset = 0;
430  int16_t len = UIP_IP_BUF->len[1] | (UIP_IP_BUF->len[0] << 8);
431 
432  uint16_t iplen;
433 
434  uint8_t i;
435 
436  int16_t sizechange;
437 
438  uint8_t llbuf[16];
439 
440  //Figure out offset to start of options
441  switch(UIP_ICMP_BUF->type) {
442  case ICMP6_NS:
443  case ICMP6_NA:
444  icmp_opt_offset = 24;
445  break;
446 
447  case ICMP6_RS:
448  icmp_opt_offset = 8;
449  break;
450 
451  case ICMP6_RA:
452  icmp_opt_offset = 16;
453  break;
454 
455  case ICMP6_REDIRECT:
456  icmp_opt_offset = 40;
457  break;
458 
460  case ICMP6_DST_UNREACH:
461  case ICMP6_PACKET_TOO_BIG:
462  case ICMP6_TIME_EXCEEDED:
463  case ICMP6_PARAM_PROB:
464  case ICMP6_ECHO_REQUEST:
465  case ICMP6_ECHO_REPLY:
466  return 0;
467  break;
468 
469  default:
470  return -1;
471  }
472 
473  //Figure out length of options
474  len -= icmp_opt_offset;
475 
476  //Sanity check
477  if (len < 8) return -2;
478 
479  //While we have options to do...
480  while (len >= 8){
481 
482  //If we have one of these, we have something useful!
483  if (((UIP_ICMP_OPTS(icmp_opt_offset)->type) == UIP_ND6_OPT_SLLAO) ||
484  ((UIP_ICMP_OPTS(icmp_opt_offset)->type) == UIP_ND6_OPT_TLLAO) ) {
485 
486  /* Shrinking the buffer may thrash things, so we store the old
487  link-layer address */
488  for(i = 0; i < (UIP_ICMP_OPTS(icmp_opt_offset)->length*8 - 2); i++) {
489  llbuf[i] = UIP_ICMP_OPTS(icmp_opt_offset)->data[i];
490  }
491 
492  //Shrink/grow buffer as needed
493  if (target == ll_802154_type) {
494  //Current is 802.3, Hence current link-layer option is 6 extra bytes
495  sizechange = 8;
496  slide(UIP_ICMP_OPTS(icmp_opt_offset)->data + 6, len - 6, sizechange);
497  } else if (target == ll_8023_type) {
498  /* Current is 802.15.4, Hence current link-layer option is 14 extra
499  * bytes.
500  * (Actual LL is 8 bytes, but total option length is in multiples of
501  * 8 Bytes, hence 8 + 2 = 10. Closest is 16 bytes, then 16 bytes for
502  * total optional length - 2 bytes for type + length leaves 14 )
503  */
504  sizechange = -8;
505  slide(UIP_ICMP_OPTS(icmp_opt_offset)->data + 14, len - 14, sizechange);
506  } else {
507  return -3; //Uh-oh!
508  }
509 
510  //Translate addresses
511  if (target == ll_802154_type) {
512  mac_createSicslowpanLongAddr(llbuf, (uip_lladdr_t *)UIP_ICMP_OPTS(icmp_opt_offset)->data);
513  } else {
514  mac_createEthernetAddr(UIP_ICMP_OPTS(icmp_opt_offset)->data, (uip_lladdr_t *)llbuf);
515  }
516 
517  //Adjust the length
518  if (target == ll_802154_type) {
519  UIP_ICMP_OPTS(icmp_opt_offset)->length = 2;
520  } else {
521  UIP_ICMP_OPTS(icmp_opt_offset)->length = 1;
522  }
523 
524  //Adjust the IP header length, as well as uIP length
525  iplen = UIP_IP_BUF->len[1] | (UIP_IP_BUF->len[0]<<8);
526  iplen += sizechange;
527  len += sizechange;
528 
529  UIP_IP_BUF->len[1] = (uint8_t)iplen;
530  UIP_IP_BUF->len[0] = (uint8_t)(iplen >> 8);
531 
532  uip_len += sizechange;
533 
534  //We broke ICMP checksum, be sure to fix that
535  UIP_ICMP_BUF->icmpchksum = 0;
536  UIP_ICMP_BUF->icmpchksum = ~uip_icmp6chksum();
537 
538  //Finally set up next run in while loop
539  len -= 8 * UIP_ICMP_OPTS(icmp_opt_offset)->length;
540  icmp_opt_offset += 8 * UIP_ICMP_OPTS(icmp_opt_offset)->length;
541  } else {
542 
543  //Not an option we care about, ignore it
544  len -= 8 * UIP_ICMP_OPTS(icmp_opt_offset)->length;
545 
546  //This shouldn't happen!
547  if (UIP_ICMP_OPTS(icmp_opt_offset)->length == 0) {
548  PRINTF("Option in ND packet has length zero, error?\n");
549  len = 0;
550  }
551 
552  icmp_opt_offset += 8 * UIP_ICMP_OPTS(icmp_opt_offset)->length;
553 
554  } //If ICMP_OPT is one we care about
555 
556  } //while(len >= 8)
557 
558  return 0;
559 
560 }

736 void slide(uint8_t * data, uint8_t length, int16_t slide)
737 {
738  //Sanity checks
739  if (!length) return;
740  if (!slide) return;
741 
742  uint8_t i = 0;
743 
744  while(length) {
745  length--;
746 
747  //If we are sliding up, we do from the top of the buffer down
748  if (slide > 0) {
749  *(data + length + slide) = *(data + length);
750 
751  //If we are sliding down, we do from the bottom of the buffer up
752  } else {
753  *(data + slide + i) = *(data + i);
754  }
755 
756  i++;
757  }
758 }
