66 res_any_handler(void *request, void *response, uint8_t *buffer, uint16_t preferred_size, int32_t *offset)
67 {
68  /* The ETag and Token is copied to the header. */
69  uint8_t opaque[] = { 0x0A, 0xBC, 0xDE };
70 
71  /* Strings are not copied, so use static string buffers or strings in .text memory (char *str = "string in .text";). */
72  static char location[] = { '/', 'f', '/', 'a', '?', 'k', '&', 'e', 0 };
73 
74  /* No default my be assumed for the Content-Format. (Unsigned -1 means all bits set.) */
75  unsigned int content_format = -1;
76 
77  /* The other getters copy the value (or string/array pointer) to the given pointers and return 1 for success or the length of strings/arrays. */
78  uint32_t longint = 0;
79  const char *str = NULL;
80  const uint8_t *bytes = NULL;
81  uint32_t block_num = 0;
82  uint8_t block_more = 0;
83  uint16_t block_size = 0;
84  int len = 0;
85 
86  /* Mirror the received header options in the response payload. Unsupported getters (e.g., rest_get_header_observe() with HTTP) will return 0. */
87 
88  int strpos = 0;
89  /* snprintf() counts the terminating '\0' to the size parameter.
90  * The additional byte is taken care of by allocating REST_MAX_CHUNK_SIZE+1 bytes in the REST framework.
91  * Add +1 to fill the complete buffer, as the payload does not need a terminating '\0'. */
92  if(REST.get_header_content_type(request, &content_format)) {
93  strpos += snprintf((char *)buffer, REST_MAX_CHUNK_SIZE + 1, "CF %u\n", content_format);
94  }
95  if(strpos <= REST_MAX_CHUNK_SIZE && (len = REST.get_header_accept(request, &content_format))) {
96  strpos += snprintf((char *)buffer + strpos, REST_MAX_CHUNK_SIZE - strpos + 1, "Ac %u\n", content_format);
97  /* Some getters such as for ETag or Location are omitted, as these options should not appear in a request.
98  * Max-Age might appear in HTTP requests or used for special purposes in CoAP. */
99  }
100  if(strpos <= REST_MAX_CHUNK_SIZE && REST.get_header_max_age(request, &longint)) {
101  strpos += snprintf((char *)buffer + strpos, REST_MAX_CHUNK_SIZE - strpos + 1, "MA %lu\n", longint);
102  /* For HTTP this is the Length option, for CoAP it is the Size option. */
103  }
104  if(strpos <= REST_MAX_CHUNK_SIZE && REST.get_header_length(request, &longint)) {
105  strpos += snprintf((char *)buffer + strpos, REST_MAX_CHUNK_SIZE - strpos + 1, "SZ %lu\n", longint);
106  }
107  if(strpos <= REST_MAX_CHUNK_SIZE && (len = REST.get_header_host(request, &str))) {
108  strpos += snprintf((char *)buffer + strpos, REST_MAX_CHUNK_SIZE - strpos + 1, "UH %.*s\n", len, str);
109  }
110  if(strpos <= REST_MAX_CHUNK_SIZE && (len = REST.get_url(request, &str))) {
111  strpos += snprintf((char *)buffer + strpos, REST_MAX_CHUNK_SIZE - strpos + 1, "UP %.*s\n", len, str);
112  }
113  if(strpos <= REST_MAX_CHUNK_SIZE && (len = REST.get_query(request, &str))) {
114  strpos += snprintf((char *)buffer + strpos, REST_MAX_CHUNK_SIZE - strpos + 1, "UQ %.*s\n", len, str);
115  /* Undefined request options for debugging: actions not required for normal RESTful Web service. */
116  }
117  if(strpos <= REST_MAX_CHUNK_SIZE && (len = coap_get_header_location_path(request, &str))) {
118  strpos += snprintf((char *)buffer + strpos, REST_MAX_CHUNK_SIZE - strpos + 1, "LP %.*s\n", len, str);
119  }
120  if(strpos <= REST_MAX_CHUNK_SIZE && (len = coap_get_header_location_query(request, &str))) {
121  strpos += snprintf((char *)buffer + strpos, REST_MAX_CHUNK_SIZE - strpos + 1, "LQ %.*s\n", len, str);
122  /* CoAP-specific example: actions not required for normal RESTful Web service. */
123  }
124  coap_packet_t *const coap_pkt = (coap_packet_t *)request;
125 
126  if(strpos <= REST_MAX_CHUNK_SIZE && coap_pkt->token_len > 0) {
127  strpos += snprintf((char *)buffer + strpos, REST_MAX_CHUNK_SIZE - strpos + 1, "To 0x");
128  int index = 0;
129  for(index = 0; index < coap_pkt->token_len; ++index) {
130  strpos += snprintf((char *)buffer + strpos, REST_MAX_CHUNK_SIZE - strpos + 1, "%02X", coap_pkt->token[index]);
131  }
132  strpos += snprintf((char *)buffer + strpos, REST_MAX_CHUNK_SIZE - strpos + 1, "\n");
133  }
134 
135  if(strpos <= REST_MAX_CHUNK_SIZE && IS_OPTION(coap_pkt, COAP_OPTION_OBSERVE)) {
136  strpos += snprintf((char *)buffer + strpos, REST_MAX_CHUNK_SIZE - strpos + 1, "Ob %lu\n", coap_pkt->observe);
137  }
138  if(strpos <= REST_MAX_CHUNK_SIZE && IS_OPTION(coap_pkt, COAP_OPTION_ETAG)) {
139  strpos += snprintf((char *)buffer + strpos, REST_MAX_CHUNK_SIZE - strpos + 1, "ET 0x");
140  int index = 0;
141  for(index = 0; index < coap_pkt->etag_len; ++index) {
142  strpos += snprintf((char *)buffer + strpos, REST_MAX_CHUNK_SIZE - strpos + 1, "%02X", coap_pkt->etag[index]);
143  }
144  strpos += snprintf((char *)buffer + strpos, REST_MAX_CHUNK_SIZE - strpos + 1, "\n");
145  }
146  if(strpos <= REST_MAX_CHUNK_SIZE && coap_get_header_block2(request, &block_num, &block_more, &block_size, NULL)) { /* This getter allows NULL pointers to get only a subset of the block parameters. */
147  strpos += snprintf((char *)buffer + strpos, REST_MAX_CHUNK_SIZE - strpos + 1, "B2 %lu%s (%u)\n", block_num, block_more ? "+" : "", block_size);
148  }
149  if(strpos <= REST_MAX_CHUNK_SIZE && coap_get_header_block1(request, &block_num, &block_more, &block_size, NULL)) {
150  strpos += snprintf((char *)buffer + strpos, REST_MAX_CHUNK_SIZE - strpos + 1, "B1 %lu%s (%u)\n", block_num, block_more ? "+" : "", block_size);
151  }
152  if(strpos <= REST_MAX_CHUNK_SIZE && (len = REST.get_request_payload(request, &bytes))) {
153  strpos += snprintf((char *)buffer + strpos, REST_MAX_CHUNK_SIZE - strpos + 1, "%.*s", len, bytes);
154  }
155  if(strpos >= REST_MAX_CHUNK_SIZE) {
156  buffer[REST_MAX_CHUNK_SIZE - 1] = 0xBB; /* 'Â»' to indicate truncation */
157  }
158  REST.set_response_payload(response, buffer, strpos);
159 
160  PRINTF("/mirror options received: %s\n", buffer);
161 
162  /* Set dummy header options for response. Like getters, some setters are not implemented for HTTP and have no effect. */
163  REST.set_header_content_type(response, REST.type.TEXT_PLAIN);
164  REST.set_header_max_age(response, 17); /* For HTTP, browsers will not re-request the page for 17 seconds. */
165  REST.set_header_etag(response, opaque, 2);
166  REST.set_header_location(response, location); /* Initial slash is omitted by framework */
167  REST.set_header_length(response, strpos); /* For HTTP, browsers will not re-request the page for 10 seconds. CoAP action depends on the client. */
168 
169 /* CoAP-specific example: actions not required for normal RESTful Web service. */
170  coap_set_header_uri_host(response, "tiki");
171  coap_set_header_observe(response, 10);
172  coap_set_header_proxy_uri(response, "ftp://x");
173  coap_set_header_block2(response, 42, 0, 64); /* The block option might be overwritten by the framework when blockwise transfer is requested. */
174  coap_set_header_block1(response, 23, 0, 16);
175  coap_set_header_accept(response, TEXT_PLAIN);
176  coap_set_header_if_none_match(response);
177 }
