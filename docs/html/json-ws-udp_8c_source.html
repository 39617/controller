115 json_ws_udp_send(struct jsontree_value *tree, const char *path)
116 {
117  struct jsontree_context json;
118  /* maxsize = 70 bytes */
119  char buf[70];
120 
121  udp_buf = buf;
122 
123  /* reset state and set max-size */
124  /* NOTE: packet will be truncated at 70 bytes */
125  pos = 0;
126  size = sizeof(buf);
127 
128  json.values[0] = (struct json_value *)tree;
129  jsontree_reset(&json);
130  find_json_path(&json, path);
131  json.path = json.depth;
132  json.putchar = putchar_udp;
133  while(jsontree_print_next(&json) && json.path <= json.depth);
134 
135  printf("Real UDP size: %d\n", pos);
136  buf[pos] = 0;
137 
138  uip_udp_packet_sendto(client_conn, &buf, pos,
139  &server_ipaddr, UIP_HTONS(server_port));
140 }
