78 /*
79 * Variables begin and end manage the buffer space in a cyclic
80 * fashion. The first used byte is at begin and end is one byte past
81 * the last. I.e. [begin, end) is the actively used space.
82 *
83 * If begin != pkt_end we have a packet at [begin, pkt_end),
84 * furthermore, if state == STATE_TWOPACKETS we have one more packet at
85 * [pkt_end, end). If more bytes arrive in state STATE_TWOPACKETS
86 * they are discarded.
87 */

178 int from_slipbuffer(uint8_t *outbuf, uint16_t blen)
179 {
180 
181  /*
182  * Interrupt can not change begin but may change pkt_end.
183  * If pkt_end != begin it will not change again.
184  */
185  if(begin != pkt_end) {
186  uint16_t len;
187 
188  if(begin < pkt_end) {
189  len = pkt_end - begin;
190  if(len > blen) {
191  len = 0;
192  } else {
193  memcpy(outbuf, &rxbuf[begin], len);
194  }
195  } else {
196  len = (RX_BUFSIZE - begin) + (pkt_end - 0);
197  if(len > blen) {
198  len = 0;
199  } else {
200  unsigned i;
201  for(i = begin; i < RX_BUFSIZE; i++) {
202  *outbuf++ = rxbuf[i];
203  }
204  for(i = 0; i < pkt_end; i++) {
205  *outbuf++ = rxbuf[i];
206  }
207  }
208  }
209 
210  /* Remove data from buffer together with the copied packet. */
211  begin = pkt_end;
212  if(state == STATE_TWOPACKETS) {
213  pkt_end = end;
214  state = STATE_OK; /* Assume no bytes where lost! */
215 
216  /* One more packet is buffered, need to be polled again! */
217  process_poll(&slip_process);
218  }
219  return len;
220  }
221 
222  return 0;
223 }

323 PROCESS_THREAD(slip_process, ev, data)
324 {
325  PROCESS_BEGIN();
326 
327  rxbuf_init();
328 
329  ctimer_set(&t, 2*CLOCK_SECOND, request_mac, NULL);
330 
331  while(1) {
332  PROCESS_YIELD_UNTIL(ev == PROCESS_EVENT_POLL);
333 
334  slip_active = 1;
335 
336  /* Move packet from rxbuf to buffer provided by uIP. */
337  uip_len = slip_poll_handler(&uip_buf[UIP_LLH_LEN],
338  UIP_BUFSIZE - UIP_LLH_LEN);
339 #if !NETSTACK_CONF_WITH_IPV6
340  if(uip_len == 4 && strncmp((char*)&uip_buf[UIP_LLH_LEN], "?IPA", 4) == 0) {
341  char buf[8];
342  memcpy(&buf[0], "=IPA", 4);
343  memcpy(&buf[4], &uip_hostaddr, 4);
344  if(input_callback) {
345  input_callback();
346  }
347  slip_write(buf, 8);
348  } else if(uip_len > 0
349  && uip_len == (((uint16_t)(BUF->len[0]) << 8) + BUF->len[1])
350  && uip_ipchksum() == 0xffff) {
351 #define IP_DF 0x40
352  if(BUF->ipid[0] == 0 && BUF->ipid[1] == 0 && BUF->ipoffset[0] & IP_DF) {
353  static uint16_t ip_id;
354  uint16_t nid = ip_id++;
355  BUF->ipid[0] = nid >> 8;
356  BUF->ipid[1] = nid;
357  nid = htons(nid);
358  nid = ~nid; /* negate */
359  BUF->ipchksum += nid; /* add */
360  if(BUF->ipchksum < nid) { /* 1-complement overflow? */
361  BUF->ipchksum++;
362  }
363  }
364  if(tcpip_input_callback) {
365  tcpip_input_callback();
366  } else {
367  tcpip_input();
368  }
369  } else {
370  uip_clear_buf();
371  SLIP_STATISTICS(slip_ip_drop++);
372  }
373 #else /* NETSTACK_CONF_WITH_IPV6 */
374  if(uip_len > 0) {
375  if(tcpip_input_callback) {

387 int
388 slip_input_byte(unsigned char c)
389 {
390  switch(state) {
391  case STATE_RUBBISH:
392  if(c == SLIP_END) {
393  state = STATE_OK;
394  }
395  return 0;
396 
397  case STATE_TWOPACKETS: /* Two packets are already buffered! */
398  return 0;
399 
400  case STATE_ESC:
401  if(c == SLIP_ESC_END) {
402  c = SLIP_END;
403  } else if(c == SLIP_ESC_ESC) {
404  c = SLIP_ESC;
405  } else {
406  state = STATE_RUBBISH;
407  SLIP_STATISTICS(slip_rubbish++);
408  end = pkt_end; /* remove rubbish */
409  return 0;
410  }
411  state = STATE_OK;
412  break;
413 
414  case STATE_OK:
415  if(c == SLIP_ESC) {
416  state = STATE_ESC;
417  return 0;
418  } else if(c == SLIP_END) {
419  /*
420  * We have a new packet, possibly of zero length.
421  *
422  * There may already be one packet buffered.
423  */
424  if(end != pkt_end) { /* Non zero length. */
425  if(begin == pkt_end) { /* None buffered. */
426  pkt_end = end;
427  } else {
428  state = STATE_TWOPACKETS;
429  SLIP_STATISTICS(slip_twopackets++);
430  }
431  process_poll(&slip_process);
432  return 1;
433  }
434  return 0;
435  }
436  break;
437  }
438 
439  /* add_char: */
440  {
441  unsigned next;
442  next = end + 1;
443  if(next == RX_BUFSIZE) {
444  next = 0;
445  }
446  if(next == begin) { /* rxbuf is full */
447  state = STATE_RUBBISH;
448  SLIP_STATISTICS(slip_overflow++);
449  end = pkt_end; /* remove rubbish */
450  return 0;
451  }
452  rxbuf[end] = c;
453  end = next;
454  }
455 
456  /* There could be a separate poll routine for this. */
457  if(c == 'T' && rxbuf[begin] == 'C') {
458  process_poll(&slip_process);
459  return 1;
460  }
461 
462  return 0;
463 }

483  i++;
484  }
485  }
