55 res_post_handler(void *request, void *response, uint8_t *buffer, uint16_t preferred_size, int32_t *offset)
56 {
57  /* Example allows only one request on time. There are no checks for multiply access !!! */
58  if(*offset == 0) {
59  /* Incoming Data */
60  if(coap_block1_handler(request, response, big_msg, &big_msg_len, MAX_DATA_LEN)) {
61  /* More Blocks will follow. Example waits for
62  * the last block and stores data into big_msg.
63  */
64  return;
65  }
66  /* Last block was received. */
67  coap_separate_accept(request, &request_metadata);
68 
69  /* Need Time for calculation now */
70  unsigned i;
71  for(i = 0; i <= 4096; i++) {
72  printf("\r%4u\r", i);
73  }
74  printf("\n");
75 
76  /* Send first block */
77  coap_transaction_t *transaction = NULL;
78  if((transaction = coap_new_transaction(request_metadata.mid, &request_metadata.addr, request_metadata.port))) {
79  coap_packet_t resp[1]; /* This way the packet can be treated as pointer as usual. */
80 
81  /* Restore the request information for the response. */
82  coap_separate_resume(resp, &request_metadata, CONTENT_2_05);
83 
84  /* Set payload and block info */
85  coap_set_payload(resp, big_msg, big_msg_len > request_metadata.block2_size ? request_metadata.block2_size : big_msg_len);
86  if(big_msg_len > request_metadata.block2_size) {
87  coap_set_header_block2(resp, 0, 1, request_metadata.block2_size);
88  }
89 
90  /* Warning: No check for serialization error. */
91  transaction->packet_len = coap_serialize_message(resp, transaction->packet);
92  coap_send_transaction(transaction);
93  }
94  } else {
95  /* request for more blocks */
96  if(*offset >= big_msg_len) {
97  coap_set_status_code(response, BAD_OPTION_4_02);
98  coap_set_payload(response, "BlockOutOfScope", 15);
99  return;
100  }
101 
102  memcpy(buffer, big_msg + *offset, 32);
103  if(big_msg_len - *offset < preferred_size) {
104  preferred_size = big_msg_len - *offset;
105  *offset = -1;
106  } else {
107  *offset += preferred_size;
108  }
109  coap_set_payload(response, buffer, preferred_size);
110  }
111 }
