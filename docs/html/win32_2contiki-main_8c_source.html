104 {
105  _set_fmode(O_BINARY);
106 
107  contiki_argc = argc;
108  contiki_argv = argv;
109 
110  /* The first one or two args are used for wpcap configuration */
111  /* so this needs to be "removed" from contiki_args. */
112  contiki_argc--;

117 #endif
118 
119  process_init();

123 #ifdef PLATFORM_BUILD
124  program_handler_add(&directory_dsc, "Directory", 1);
125  program_handler_add(&www_dsc, "Web browser", 1);
126 #endif /* PLATFORM_BUILD */
127 
128  autostart_start(autostart_processes);
129 
130 #if !NETSTACK_CONF_WITH_IPV6
131  {
132  uip_ipaddr_t addr;
133  uip_ipaddr(&addr, 192,168,0,111);
134  uip_sethostaddr(&addr);
135  log_message("IP Address: ", inet_ntoa(*(struct in_addr*)&addr));
136 
137  uip_ipaddr(&addr, 255,255,255,0);
138  uip_setnetmask(&addr);
139  log_message("Subnet Mask: ", inet_ntoa(*(struct in_addr*)&addr));
140 
141  uip_ipaddr(&addr, 192,168,0,1);
142  uip_setdraddr(&addr);
143  log_message("Def. Router: ", inet_ntoa(*(struct in_addr*)&addr));
144 
145  uip_ipaddr(&addr, 192,168,0,1);
146  uip_nameserver_update(&addr, UIP_NAMESERVER_INFINITE_LIFETIME);
147  log_message("DNS Server: ", inet_ntoa(*(struct in_addr*)&addr));
148  }
149 
150 #else /* NETSTACK_CONF_WITH_IPV6 */
151 
152 #if !UIP_CONF_IPV6_RPL
153 #ifdef HARD_CODED_ADDRESS
154  uip_ipaddr_t ipaddr;
155  uiplib_ipaddrconv(HARD_CODED_ADDRESS, &ipaddr);
156  if ((ipaddr.u16[0]!=0) || (ipaddr.u16[1]!=0) || (ipaddr.u16[2]!=0) || (ipaddr.u16[3]!=0)) {
157 #if UIP_CONF_ROUTER
158  uip_ds6_prefix_add(&ipaddr, UIP_DEFAULT_PREFIX_LEN, 0, 0, 0, 0);
159 #else /* UIP_CONF_ROUTER */
160  uip_ds6_prefix_add(&ipaddr, UIP_DEFAULT_PREFIX_LEN, 0);
161 #endif /* UIP_CONF_ROUTER */
162 #if !UIP_CONF_IPV6_RPL
163  uip_ds6_set_addr_iid(&ipaddr, &uip_lladdr);
164  uip_ds6_addr_add(&ipaddr, 0, ADDR_AUTOCONF);
165 #endif
166  }
167 #endif /* HARD_CODED_ADDRESS */
168 #endif
169 #endif
170 
171  while(1) {
172 
173  process_run();
174 
175  etimer_request_poll();
176 
177  /* Allow user-mode APC to execute. */
178  SleepEx(10, TRUE);
179 
180 #ifdef PLATFORM_BUILD
181  if(console_resize()) {
182  ctk_restore();
183  }
184 #endif /* PLATFORM_BUILD */
185  }
186 }
