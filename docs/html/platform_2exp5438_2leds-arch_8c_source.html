50 void
51 leds_arch_init(void)
52 {
53  /* only red and green for now */
54  P1DIR |= LEDS_CONF_RED;
55  P1DIR |= LEDS_CONF_GREEN;
56 }
57 /*---------------------------------------------------------------------------*/
58 unsigned char
59 leds_arch_get(void)
60 {
61  return ((P1OUT & LEDS_CONF_RED) ? 0 : LEDS_RED)
62  | ((P1OUT & LEDS_CONF_GREEN) ? 0 : LEDS_GREEN);
63 }
64 /*---------------------------------------------------------------------------*/
65 void
66 leds_arch_set(unsigned char leds)
67 {
68  P1OUT = (P1OUT & ~LEDS_CONF_RED) | ((leds & LEDS_RED) ? LEDS_CONF_RED : 0);
69  P1OUT = (P1OUT & ~LEDS_CONF_GREEN) |
70  ((leds & LEDS_GREEN) ? LEDS_CONF_GREEN : 0);
71 }
