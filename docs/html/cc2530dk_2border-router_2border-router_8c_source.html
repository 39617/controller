97 void
98 set_prefix_64(uip_ipaddr_t *prefix_64)
99 {
100  rpl_dag_t *dag;
101  uip_ipaddr_t ipaddr;
102  memcpy(&ipaddr, prefix_64, 16);
103  prefix_set = 1;
104  uip_ds6_set_addr_iid(&ipaddr, &uip_lladdr);
105  uip_ds6_addr_add(&ipaddr, 0, ADDR_AUTOCONF);
106 
107  /* Become root of a new DODAG with ID our global v6 address */
108  dag = rpl_set_root(RPL_DEFAULT_INSTANCE, &ipaddr);
109  if(dag != NULL) {
110  rpl_set_prefix(dag, &ipaddr, 64);
111  PUTSTRING("Created a new RPL dag with ID: ");
112  PRINT6ADDR(&dag->dag_id);
113  PUTCHAR('\n');
114  }
115 }
