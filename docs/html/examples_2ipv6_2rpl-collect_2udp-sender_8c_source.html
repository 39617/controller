96 void
97 collect_common_send(void)
98 {
99  static uint8_t seqno;
100  struct {
101  uint8_t seqno;
102  uint8_t for_alignment;
103  struct collect_view_data_msg msg;
104  } msg;
105  /* struct collect_neighbor *n; */
106  uint16_t parent_etx;
107  uint16_t rtmetric;
108  uint16_t num_neighbors;
109  uint16_t beacon_interval;
110  rpl_parent_t *preferred_parent;
111  linkaddr_t parent;
112  rpl_dag_t *dag;
113 
114  if(client_conn == NULL) {
115  /* Not setup yet */
116  return;
117  }
118  memset(&msg, 0, sizeof(msg));
119  seqno++;
120  if(seqno == 0) {
121  /* Wrap to 128 to identify restarts */
122  seqno = 128;
123  }
124  msg.seqno = seqno;
125 
126  linkaddr_copy(&parent, &linkaddr_null);
127  parent_etx = 0;
128 
129  /* Let's suppose we have only one instance */
130  dag = rpl_get_any_dag();
131  if(dag != NULL) {
132  preferred_parent = dag->preferred_parent;
133  if(preferred_parent != NULL) {
134  uip_ds6_nbr_t *nbr;
135  nbr = uip_ds6_nbr_lookup(rpl_get_parent_ipaddr(preferred_parent));
136  if(nbr != NULL) {
137  /* Use parts of the IPv6 address as the parent address, in reversed byte order. */
138  parent.u8[LINKADDR_SIZE - 1] = nbr->ipaddr.u8[sizeof(uip_ipaddr_t) - 2];
139  parent.u8[LINKADDR_SIZE - 2] = nbr->ipaddr.u8[sizeof(uip_ipaddr_t) - 1];
140  parent_etx = rpl_get_parent_rank((uip_lladdr_t *) uip_ds6_nbr_get_ll(nbr)) / 2;
141  }
142  }
143  rtmetric = dag->rank;
144  beacon_interval = (uint16_t) ((2L << dag->instance->dio_intcurrent) / 1000);
145  num_neighbors = uip_ds6_nbr_num();
146  } else {
147  rtmetric = 0;
148  beacon_interval = 0;
149  num_neighbors = 0;
150  }
151 
152  /* num_neighbors = collect_neighbor_list_num(&tc.neighbor_list); */
153  collect_view_construct_message(&msg.msg, &parent,
154  parent_etx, rtmetric,
155  num_neighbors, beacon_interval);
156 
157  uip_udp_packet_sendto(client_conn, &msg, sizeof(msg),
158  &server_ipaddr, UIP_HTONS(UDP_SERVER_PORT));
159 }

175  int i;

177 
178  PRINTF("Client IPv6 addresses: ");
179  for(i = 0; i < UIP_DS6_ADDR_NB; i++) {
180  state = uip_ds6_if.addr_list[i].state;
181  if(uip_ds6_if.addr_list[i].isused &&
182  (state == ADDR_TENTATIVE || state == ADDR_PREFERRED)) {
183  PRINT6ADDR(&uip_ds6_if.addr_list[i].ipaddr);
184  PRINTF("\n");
