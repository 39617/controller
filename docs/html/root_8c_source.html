74 static void
75 multicast_send(void)
76 {
77  uint32_t id;
78 
79  id = uip_htonl(seq_id);
80  memset(buf, 0, MAX_PAYLOAD_LEN);
81  memcpy(buf, &id, sizeof(seq_id));
82 
83  PRINTF("Send to: ");
84  PRINT6ADDR(&mcast_conn->ripaddr);
85  PRINTF(" Remote Port %u,", uip_ntohs(mcast_conn->rport));
86  PRINTF(" (msg=0x%08lx)", (unsigned long)uip_ntohl(*((uint32_t *)buf)));
87  PRINTF(" %lu bytes\n", (unsigned long)sizeof(id));
88 
89  seq_id++;
90  uip_udp_packet_send(mcast_conn, buf, sizeof(id));
91 }

93 static void
94 prepare_mcast(void)
95 {
96  uip_ipaddr_t ipaddr;
97 
98  /*
99  * IPHC will use stateless multicast compression for this destination
100  * (M=1, DAC=0), with 32 inline bits (1E 89 AB CD)
101  */
102  uip_ip6addr(&ipaddr, 0xFF1E,0,0,0,0,0,0x89,0xABCD);
103  mcast_conn = udp_new(&ipaddr, UIP_HTONS(MCAST_SINK_UDP_PORT), NULL);
104 }

117 
118  PRINTF("Our IPv6 addresses:\n");
119  for(i = 0; i < UIP_DS6_ADDR_NB; i++) {
120  state = uip_ds6_if.addr_list[i].state;
121  if(uip_ds6_if.addr_list[i].isused && (state == ADDR_TENTATIVE || state
122  == ADDR_PREFERRED)) {
123  PRINTF(" ");
124  PRINT6ADDR(&uip_ds6_if.addr_list[i].ipaddr);
125  PRINTF("\n");
126  if(state == ADDR_TENTATIVE) {
127  uip_ds6_if.addr_list[i].state = ADDR_PREFERRED;
128  }
129  }
130  }
131 
132  /* Become root of a new DODAG with ID our global v6 address */
133  dag = rpl_set_root(RPL_DEFAULT_INSTANCE, &ipaddr);
134  if(dag != NULL) {
135  rpl_set_prefix(dag, &ipaddr, 64);
136  PRINTF("Created a new RPL dag with ID: ");
137  PRINT6ADDR(&dag->dag_id);
138  PRINTF("\n");
139  }
140 }

143 {

146  PROCESS_BEGIN();

156  etimer_set(&et, START_DELAY * CLOCK_SECOND);
157  while(1) {
158  PROCESS_YIELD();

160  if(seq_id == ITERATIONS) {
161  etimer_stop(&et);
162  } else {
163  multicast_send();
164  etimer_set(&et, SEND_INTERVAL);
165  }
166  }
167  }
168 
169  PROCESS_END();
170 }
