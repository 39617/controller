84 PROCESS_THREAD(border_router_process, ev, data)
85 {
86  static struct etimer et;
87 
88  PROCESS_BEGIN();
89 
90 /* While waiting for the prefix to be sent through the SLIP connection, the future
91  * border router can join an existing DAG as a parent or child, or acquire a default
92  * router that will later take precedence over the SLIP fallback interface.
93  * Prevent that by turning the radio off until we are initialized as a DAG root.
94  */
95  prefix_set = 0;
96 
97  PROCESS_PAUSE();
98 
99  PRINTF("RPL-Border router started\n");
100 
101 
102  /* Request prefix until it has been received */
103  while(!prefix_set) {
104  etimer_set(&et, CLOCK_SECOND);
105  request_prefix();
106  PROCESS_WAIT_EVENT_UNTIL(etimer_expired(&et));
107  PRINTF("Waiting for prefix\n");
108  }
109 
110  PRINTF("Obtained prefix: ");
111  uip_debug_ipaddr_print(&prefix);
112  PRINTF("\n");
113 
114  rpl_tools_init(&prefix);
115 
116  etimer_set(&et, CLOCK_SECOND * 60);
117  while(1) {
118  print_network_status();
119  PROCESS_YIELD_UNTIL(etimer_expired(&et));
120  etimer_reset(&et);
121  }
