517 static int
518 galileo_pinmux_set_pin(uint8_t pin, GALILEO_PINMUX_FUNC func)
519 {
520  struct mux_path *mux_path;
521  uint8_t index, i;
522 
523  if(pin >= GALILEO_NUM_PINS) {
524  return -1;
525  }
526 
527  index = PINMUX_NUM_FUNCS * pin;
528  index += func;
529 
530  mux_path = &galileo_pinmux_paths[index];
531 
532  for(i = 0; i < PINMUX_NUM_PATHS; i++) {
533  struct gpio_pcal9535a_data *exp = NULL;
534  switch(mux_path->path[i].chip) {
535  case EXP0:
536  exp = &data.exp0;
537  break;
538  case EXP1:
539  exp = &data.exp1;
540  break;
541  case EXP2:
542  exp = &data.exp2;
543  break;
544  case PWM0:
545  if(pwm_pca9685_set_duty_cycle(&data.pwm0, mux_path->path[i].pin, mux_path->path[i].level ? 100 : 0) < 0) {
546  return -1;
547  }
548  continue;
549  case NONE:
550  continue;
551  }
552 
553  assert(exp != NULL);
554 
555  if(gpio_pcal9535a_write(exp, mux_path->path[i].pin, mux_path->path[i].level) < 0) {
556  return -1;
557  }
558  if(gpio_pcal9535a_config(exp, mux_path->path[i].pin, mux_path->path[i].cfg) < 0) {
559  return -1;
560  }
561  }
562 
563  return 0;
564 }

575 int
576 galileo_brd_to_cpu_gpio_pin(unsigned pin, bool *sus)
577 {
578  static const int SUS = 0x100;
579  unsigned pins[GALILEO_NUM_DIGITAL_PINS] = {
580  3, 4, 5, 6,
581  SUS | 4, 8, 9, SUS | 0,
582  SUS | 1, SUS | 2, 2, SUS | 3,
583  7, SUS | 5
584  };
585  int cpu_pin;
586 
587  /* GPIOs in the analog pin space are implemented by EXP2, not the CPU. */
588  assert(pin < GALILEO_NUM_DIGITAL_PINS);
589  cpu_pin = pins[pin];
590 
591  *sus = (cpu_pin & SUS) == SUS;
592 
593  return cpu_pin & ~SUS;
594 }
