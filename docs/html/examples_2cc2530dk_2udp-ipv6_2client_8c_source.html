98  }
99 
100  PRINTF("Client to: ");
101  PRINT6ADDR(&this_conn->ripaddr);
102 
103  memcpy(buf, &seq_id, sizeof(seq_id));
104 
105  PRINTF(" Remote Port %u,", UIP_HTONS(this_conn->rport));
106  PRINTF(" (msg=0x%04x), %u bytes\n", *(uint16_t *) buf, sizeof(seq_id));
107 
108  uip_udp_packet_send(this_conn, buf, sizeof(seq_id));
109  leds_off(LEDS_RED);

132 
133  uip_ip6addr(&ipaddr, UIP_DS6_DEFAULT_PREFIX, 0, 0, 0, 0x0215, 0x2000, 0x0002, 0x2145);
134  g_conn = udp_new(&ipaddr, UIP_HTONS(3000), NULL);
135  if(!g_conn) {
136  PRINTF("udp_new g_conn error.\n");
137  }
138  udp_bind(g_conn, UIP_HTONS(GLOBAL_CONN_PORT));
139 
140  PRINTF("Global connection with ");

142  PRINTF(" local/remote port %u/%u\n",
143  UIP_HTONS(g_conn->lport), UIP_HTONS(g_conn->rport));
144 
145  etimer_set(&et, SEND_INTERVAL);
146 
147  while(1) {

153  tcpip_handler();

157  PROCESS_END();
