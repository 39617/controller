70 static void
71 res_get_handler(void *request, void *response, uint8_t *buffer, uint16_t preferred_size, int32_t *offset)
72 {
73  /*
74  * Example allows only one open separate response.
75  * For multiple, the application must manage the list of stores.
76  */
77  if(separate_active >= COAP_MAX_OPEN_SEPARATE) {
78  coap_separate_reject();
79  } else {
80  ++separate_active;
81 
82  /* Take over and skip response by engine. */
83  coap_separate_accept(request, &separate_store->request_metadata);
84  /* Be aware to respect the Block2 option, which is also stored in the coap_separate_t. */
85 
86  /*
87  * At the moment, only the minimal information is stored in the store (client address, port, token, MID, type, and Block2).
88  * Extend the store, if the application requires additional information from this handler.
89  * buffer is an example field for custom information.
90  */
91  snprintf(separate_store->buffer, sizeof(separate_store->buffer), "StoredInfo");
92  }
93 }
94 static void
95 res_resume_handler()
96 {
97  if(separate_active) {
98  coap_transaction_t *transaction = NULL;
99  if((transaction = coap_new_transaction(separate_store->request_metadata.mid, &separate_store->request_metadata.addr, separate_store->request_metadata.port))) {
100  coap_packet_t response[1]; /* This way the packet can be treated as pointer as usual. */
101 
102  /* Restore the request information for the response. */
103  coap_separate_resume(response, &separate_store->request_metadata, REST.status.OK);
104 
105  coap_set_payload(response, separate_store->buffer, strlen(separate_store->buffer));
106 
107  /*
108  * Be aware to respect the Block2 option, which is also stored in the coap_separate_t.
109  * As it is a critical option, this example resource pretends to handle it for compliance.
110  */
111  coap_set_header_block2(response, separate_store->request_metadata.block2_num, 0, separate_store->request_metadata.block2_size);
112 
113  /* Warning: No check for serialization error. */
114  transaction->packet_len = coap_serialize_message(response, transaction->packet);
115  coap_send_transaction(transaction);
116  /* The engine will clear the transaction (right after send for NON, after acked for CON). */
117 
118  /* FIXME there could me more! */
119  separate_active = 0;
120  } else {
121  /*
122  * Set timer for retry, send error message, ...
123  * The example simply waits for another button press.
124  */
125  }
126  } /* if (separate_active) */
127 }
