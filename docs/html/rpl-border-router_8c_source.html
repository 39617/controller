131 PROCESS_THREAD(border_router_process, ev, data)
132 {
133  static struct etimer et;
134 
135  PROCESS_BEGIN();
136 
137 /* While waiting for the prefix to be sent through the SLIP connection, the future
138  * border router can join an existing DAG as a parent or child, or acquire a default
139  * router that will later take precedence over the SLIP fallback interface.
140  * Prevent that by turning the radio off until we are initialized as a DAG root.
141  */
142  prefix_set = 0;
143 
144  PROCESS_PAUSE();
145 
146  PRINTF("RPL-Border router started\n");
147 
148  /* Request prefix until it has been received */
149  while(!prefix_set) {
150  etimer_set(&et, CLOCK_SECOND);
151  request_prefix();
152  PROCESS_WAIT_EVENT_UNTIL(etimer_expired(&et));
153  PRINTF("Waiting for prefix\n");
154  }
155 
156  PRINTF("Obtained prefix: ");
157  uip_debug_ipaddr_print(&prefix);
158  PRINTF("\n");
159 
160  rpl_tools_init(&prefix);
161 
162  rest_init_engine();
163  rest_activate_resource(&resource_get_rssi, "Get-RSSI");
164  rest_activate_resource(&resource_get_last_rssi, "Get-Last-RSSI");
165 
166 
167  PROCESS_END();
168 }
