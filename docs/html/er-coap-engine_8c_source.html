67 static int
68 coap_receive(void)
69 {
70  erbium_status_code = NO_ERROR;
71 
72  PRINTF("handle_incoming_data(): received uip_datalen=%u \n",
73  (uint16_t)uip_datalen());
74 
75  /* static declaration reduces stack peaks and program code size */
76  static coap_packet_t message[1]; /* this way the packet can be treated as pointer as usual */
77  static coap_packet_t response[1];
78  static coap_transaction_t *transaction = NULL;
79 
80  if(uip_newdata()) {
81 
82  PRINTF("receiving UDP datagram from: ");
83  PRINT6ADDR(&UIP_IP_BUF->srcipaddr);
84  PRINTF(":%u\n Length: %u\n", uip_ntohs(UIP_UDP_BUF->srcport),
85  uip_datalen());
86 
87  erbium_status_code =
88  coap_parse_message(message, uip_appdata, uip_datalen());
89 
90  if(erbium_status_code == NO_ERROR) {
91 
92  /*TODO duplicates suppression, if required by application */
93 
94  PRINTF(" Parsed: v %u, t %u, tkl %u, c %u, mid %u\n", message->version,
95  message->type, message->token_len, message->code, message->mid);
96  PRINTF(" URL: %.*s\n", message->uri_path_len, message->uri_path);
97  PRINTF(" Payload: %.*s\n", message->payload_len, message->payload);
98 
99  /* handle requests */
100  if(message->code >= COAP_GET && message->code <= COAP_DELETE) {
101 
102  /* use transaction buffer for response to confirmable request */
103  if((transaction =
104  coap_new_transaction(message->mid, &UIP_IP_BUF->srcipaddr,
105  UIP_UDP_BUF->srcport))) {
106  uint32_t block_num = 0;
107  uint16_t block_size = COAP_MAX_BLOCK_SIZE;
108  uint32_t block_offset = 0;
109  int32_t new_offset = 0;
110 
111  /* prepare response */
112  if(message->type == COAP_TYPE_CON) {
113  /* reliable CON requests are answered with an ACK */
114  coap_init_message(response, COAP_TYPE_ACK, CONTENT_2_05,
115  message->mid);
116  } else {
117  /* unreliable NON requests are answered with a NON as well */
118  coap_init_message(response, COAP_TYPE_NON, CONTENT_2_05,
119  coap_get_mid());
120  /* mirror token */
121  } if(message->token_len) {
122  coap_set_token(response, message->token, message->token_len);
123  /* get offset for blockwise transfers */
124  }
125  if(coap_get_header_block2
126  (message, &block_num, NULL, &block_size, &block_offset)) {
127  PRINTF("Blockwise: block request %lu (%u/%u) @ %lu bytes\n",
128  block_num, block_size, COAP_MAX_BLOCK_SIZE, block_offset);
129  block_size = MIN(block_size, COAP_MAX_BLOCK_SIZE);
130  new_offset = block_offset;
131  }
132 
133  /* invoke resource handler */
134  if(service_cbk) {
135 
136  /* call REST framework and check if found and allowed */
137  // Select our interface
138  rest_select_if(COAP_IF);
139  if(service_cbk
140  (message, response, transaction->packet + COAP_MAX_HEADER_SIZE,
141  block_size, &new_offset)) {
142 
143  if(erbium_status_code == NO_ERROR) {
144 
145  /* TODO coap_handle_blockwise(request, response, start_offset, end_offset); */
146 
147  /* resource is unaware of Block1 */
148  if(IS_OPTION(message, COAP_OPTION_BLOCK1)
149  && response->code < BAD_REQUEST_4_00
150  && !IS_OPTION(response, COAP_OPTION_BLOCK1)) {
151  PRINTF("Block1 NOT IMPLEMENTED\n");
152 
153  erbium_status_code = NOT_IMPLEMENTED_5_01;
154  coap_error_message = "NoBlock1Support";
155 
156  /* client requested Block2 transfer */
157  } else if(IS_OPTION(message, COAP_OPTION_BLOCK2)) {
158 
159  /* unchanged new_offset indicates that resource is unaware of blockwise transfer */
160  if(new_offset == block_offset) {
161  PRINTF
162  ("Blockwise: unaware resource with payload length %u/%u\n",
163  response->payload_len, block_size);
164  if(block_offset >= response->payload_len) {
165  PRINTF
166  ("handle_incoming_data(): block_offset >= response->payload_len\n");
167 
168  response->code = BAD_OPTION_4_02;
169  coap_set_payload(response, "BlockOutOfScope", 15); /* a const char str[] and sizeof(str) produces larger code size */
170  } else {
171  coap_set_header_block2(response, block_num,
172  response->payload_len -
173  block_offset > block_size,
174  block_size);
175  coap_set_payload(response,
176  response->payload + block_offset,
177  MIN(response->payload_len -
178  block_offset, block_size));
179  } /* if(valid offset) */
180 
181  /* resource provides chunk-wise data */
182  } else {
183  PRINTF("Blockwise: blockwise resource, new offset %ld\n",
184  new_offset);
185  coap_set_header_block2(response, block_num,
186  new_offset != -1
187  || response->payload_len >
188  block_size, block_size);
189 
190  if(response->payload_len > block_size) {
191  coap_set_payload(response, response->payload,
192  block_size);
193  }
194  } /* if(resource aware of blockwise) */
195 
196  /* Resource requested Block2 transfer */
197  } else if(new_offset != 0) {
198  PRINTF
199  ("Blockwise: no block option for blockwise resource, using block size %u\n",
200  COAP_MAX_BLOCK_SIZE);
201 
202  coap_set_header_block2(response, 0, new_offset != -1,
203  COAP_MAX_BLOCK_SIZE);
204  coap_set_payload(response, response->payload,
205  MIN(response->payload_len,
206  COAP_MAX_BLOCK_SIZE));
207  } /* blockwise transfer handling */
208  } /* no errors/hooks */
209  /* successful service callback */
210  /* serialize response */
211  }
212  if(erbium_status_code == NO_ERROR) {
213  if((transaction->packet_len = coap_serialize_message(response,
214  transaction->
215  packet)) ==
216  0) {
217  erbium_status_code = PACKET_SERIALIZATION_ERROR;
218  }
219  }
220  } else {
221  erbium_status_code = NOT_IMPLEMENTED_5_01;
222  coap_error_message = "NoServiceCallbck"; /* no 'a' to fit into 16 bytes */
223  } /* if(service callback) */
224  } else {
225  erbium_status_code = SERVICE_UNAVAILABLE_5_03;
226  coap_error_message = "NoFreeTraBuffer";
227  } /* if(transaction buffer) */
228 
229  /* handle responses */
230  } else {
231 
232  if(message->type == COAP_TYPE_CON && message->code == 0) {
233  PRINTF("Received Ping\n");
234  erbium_status_code = PING_RESPONSE;
235  } else if(message->type == COAP_TYPE_ACK) {
236  /* transactions are closed through lookup below */
237  PRINTF("Received ACK\n");
238  } else if(message->type == COAP_TYPE_RST) {
239  PRINTF("Received RST\n");
240  /* cancel possible subscriptions */
241  coap_remove_observer_by_mid(&UIP_IP_BUF->srcipaddr,
242  UIP_UDP_BUF->srcport, message->mid);
243  }
244 
245  if((transaction = coap_get_transaction_by_mid(message->mid))) {
246  /* free transaction memory before callback, as it may create a new transaction */
247  restful_response_handler callback = transaction->callback;
248  void *callback_data = transaction->callback_data;
249 
250  coap_clear_transaction(transaction);
251 
252  /* check if someone registered for the response */
253  if(callback) {
254  callback(callback_data, message);
255  }
256  }
257  /* if(ACKed transaction) */
258  transaction = NULL;
259 
260 #if COAP_OBSERVE_CLIENT
261  /* if observe notification */
262  if((message->type == COAP_TYPE_CON || message->type == COAP_TYPE_NON)
263  && IS_OPTION(message, COAP_OPTION_OBSERVE)) {
264  PRINTF("Observe [%ld]\n", message->observe);
265  coap_handle_notification(&UIP_IP_BUF->srcipaddr, UIP_UDP_BUF->srcport,
266  message);
267  }
268 #endif /* COAP_OBSERVE_CLIENT */
269  } /* request or response */
270  } /* parsed correctly */
271 
272  /* if(parsed correctly) */
273  if(erbium_status_code == NO_ERROR) {
274  if(transaction) {
275  coap_send_transaction(transaction);
276  }
277  } else if(erbium_status_code == MANUAL_RESPONSE) {
278  PRINTF("Clearing transaction for manual response");
279  coap_clear_transaction(transaction);
280  } else {
281  coap_message_type_t reply_type = COAP_TYPE_ACK;
282 
283  PRINTF("ERROR %u: %s\n", erbium_status_code, coap_error_message);
284  coap_clear_transaction(transaction);
285 
286  if(erbium_status_code == PING_RESPONSE) {
287  erbium_status_code = 0;
288  reply_type = COAP_TYPE_RST;
289  } else if(erbium_status_code >= 192) {
290  /* set to sendable error code */
291  erbium_status_code = INTERNAL_SERVER_ERROR_5_00;
292  /* reuse input buffer for error message */
293  }
294  coap_init_message(message, reply_type, erbium_status_code,
295  message->mid);
296  coap_set_payload(message, coap_error_message,
297  strlen(coap_error_message));
298  coap_send_message(&UIP_IP_BUF->srcipaddr, UIP_UDP_BUF->srcport,
299  uip_appdata, coap_serialize_message(message,
300  uip_appdata));
301  }
302  }
303 
304  /* if(new data) */
305  return erbium_status_code;
306 }

337 PROCESS_THREAD(coap_engine, ev, data)
338 {
339  PROCESS_BEGIN();
340  PRINTF("Starting %s receiver...\n", coap_rest_implementation.name);
341 
342  rest_activate_resource(&res_well_known_core, ".well-known/core");
343 
344  coap_register_as_transaction_handler();
345  coap_init_connection(SERVER_LISTEN_PORT);
346 
347  while(1) {
348  PROCESS_YIELD();
349 
350  if(ev == tcpip_event) {
351  coap_receive();
352  } else if(ev == PROCESS_EVENT_TIMER) {
353  /* retransmissions are handled here */
354  coap_check_transactions();
355  }
356  } /* while (1) */
357 
358  PROCESS_END();
359 }

372 PT_THREAD(coap_blocking_request
373  (struct request_state_t *state, process_event_t ev,
374  uip_ipaddr_t *remote_ipaddr, uint16_t remote_port,
375  coap_packet_t *request,
376  blocking_response_handler request_callback))
377 {
378  PT_BEGIN(&state->pt);
379 
380  static uint8_t more;
381  static uint32_t res_block;
382  static uint8_t block_error;
383 
384  state->block_num = 0;
385  state->response = NULL;
386  state->process = PROCESS_CURRENT();
387 
388  more = 0;
389  res_block = 0;
390  block_error = 0;
391 
392  do {
393  request->mid = coap_get_mid();
394  if((state->transaction = coap_new_transaction(request->mid, remote_ipaddr,
395  remote_port))) {
396  state->transaction->callback = coap_blocking_request_callback;
397  state->transaction->callback_data = state;
398 
399  if(state->block_num > 0) {
400  coap_set_header_block2(request, state->block_num, 0,
401  REST_MAX_CHUNK_SIZE);
402  }
403  state->transaction->packet_len = coap_serialize_message(request,
404  state->
405  transaction->
406  packet);
407 
408  coap_send_transaction(state->transaction);
409  PRINTF("Requested #%lu (MID %u)\n", state->block_num, request->mid);
410 
411  PT_YIELD_UNTIL(&state->pt, ev == PROCESS_EVENT_POLL);
412 
413  if(!state->response) {
414  PRINTF("Server not responding\n");
415  PT_EXIT(&state->pt);
416  }
417 
418  coap_get_header_block2(state->response, &res_block, &more, NULL, NULL);
419 
420  PRINTF("Received #%lu%s (%u bytes)\n", res_block, more ? "+" : "",
421  state->response->payload_len);
422 
423  if(res_block == state->block_num) {
424  request_callback(state->response);
425  ++(state->block_num);
426  } else {
427  PRINTF("WRONG BLOCK %lu/%lu\n", res_block, state->block_num);
428  ++block_error;
429  }
430  } else {
431  PRINTF("Could not allocate transaction buffer");
432  PT_EXIT(&state->pt);
433  }
434  } while(more && block_error < COAP_MAX_ATTEMPTS);
435 
436  PT_END(&state->pt);
437 }
