174 static void
175 save_config()
176 {
177  /* Dump current running config to flash */
178 #if BOARD_SENSORTAG || BOARD_LAUNCHPAD
179  int rv;
180  cc26xx_web_demo_sensor_reading_t *reading = NULL;
181 
182  rv = ext_flash_open();
183 
184  if(!rv) {
185  printf("Could not open flash to save config\n");
186  ext_flash_close();
187  return;
188  }
189 
190  rv = ext_flash_erase(CONFIG_FLASH_OFFSET, sizeof(cc26xx_web_demo_config_t));
191 
192  if(!rv) {
193  printf("Error erasing flash\n");
194  } else {
195  cc26xx_web_demo_config.magic = CONFIG_MAGIC;
196  cc26xx_web_demo_config.len = sizeof(cc26xx_web_demo_config_t);
197  cc26xx_web_demo_config.sensors_bitmap = 0;
198 
199  for(reading = list_head(sensor_list);
200  reading != NULL;
201  reading = list_item_next(reading)) {
202  if(reading->publish) {
203  cc26xx_web_demo_config.sensors_bitmap |= (1 << reading->type);
204  }
205  }
206 
207  rv = ext_flash_write(CONFIG_FLASH_OFFSET, sizeof(cc26xx_web_demo_config_t),
208  (uint8_t *)&cc26xx_web_demo_config);
209  if(!rv) {
210  printf("Error saving config\n");
211  }
212  }
213 
214  ext_flash_close();
215 #endif
216 }

218 static void
219 load_config()
220 {
221 #if BOARD_SENSORTAG || BOARD_LAUNCHPAD
222  /* Read from flash into a temp buffer */
223  cc26xx_web_demo_config_t tmp_cfg;
224  cc26xx_web_demo_sensor_reading_t *reading = NULL;
225 
226  int rv = ext_flash_open();
227 
228  if(!rv) {
229  printf("Could not open flash to load config\n");
230  ext_flash_close();
231  return;
232  }
233 
234  rv = ext_flash_read(CONFIG_FLASH_OFFSET, sizeof(tmp_cfg),
235  (uint8_t *)&tmp_cfg);
236 
237  ext_flash_close();
238 
239  if(!rv) {
240  printf("Error loading config\n");
241  return;
242  }
243 
244  if(tmp_cfg.magic == CONFIG_MAGIC && tmp_cfg.len == sizeof(tmp_cfg)) {
245  memcpy(&cc26xx_web_demo_config, &tmp_cfg, sizeof(cc26xx_web_demo_config));
246  }
247 
248  for(reading = list_head(sensor_list);
249  reading != NULL;
250  reading = list_item_next(reading)) {
251  if(cc26xx_web_demo_config.sensors_bitmap & (1 << reading->type)) {
252  reading->publish = 1;
253  } else {
254  reading->publish = 0;
255  snprintf(reading->converted, CC26XX_WEB_DEMO_CONVERTED_LEN, "\"N/A\"");
256  }
257  }
258 #endif
259 }

264 int
265 cc26xx_web_demo_ipaddr_sprintf(char *buf, uint8_t buf_len,
266  const uip_ipaddr_t *addr)
267 {
268  uint16_t a;
269  uint8_t len = 0;
270  int i, f;
271  for(i = 0, f = 0; i < sizeof(uip_ipaddr_t); i += 2) {
272  a = (addr->u8[i] << 8) + addr->u8[i + 1];
273  if(a == 0 && f >= 0) {
274  if(f++ == 0) {
275  len += snprintf(&buf[len], buf_len - len, "::");
276  }
277  } else {
278  if(f > 0) {
279  f = -1;
280  } else if(i > 0) {
281  len += snprintf(&buf[len], buf_len - len, ":");
282  }
283  len += snprintf(&buf[len], buf_len - len, "%x", a);
284  }
285  }
286 
287  return len;
288 }

354 static int
355 sensor_readings_handler(char *key, int key_len, char *val, int val_len)
356 {
357  cc26xx_web_demo_sensor_reading_t *reading = NULL;
358  int rv;
359 
360  for(reading = list_head(sensor_list);
361  reading != NULL;
362  reading = list_item_next(reading)) {
363  if(key_len == strlen(reading->form_field) &&
364  strncmp(reading->form_field, key, strlen(key)) == 0) {
365 
366  rv = atoi(val);
367 
368  /* Be pedantic: only accept 0 and 1, not just any non-zero value */
369  if(rv == 0) {
370  reading->publish = 0;
371  snprintf(reading->converted, CC26XX_WEB_DEMO_CONVERTED_LEN, "\"N/A\"");
372  } else if(rv == 1) {
373  reading->publish = 1;
374  } else {
375  return HTTPD_SIMPLE_POST_HANDLER_ERROR;
376  }
377 
378  return HTTPD_SIMPLE_POST_HANDLER_OK;
379  }
380  }
381 
382  return HTTPD_SIMPLE_POST_HANDLER_UNKNOWN;
383 }

437 static void
438 get_batmon_reading(void *data)
439 {
440  int value;
441  char *buf;
442  clock_time_t next = SENSOR_READING_PERIOD +
443  (random_rand() % SENSOR_READING_RANDOM);
444 
445  if(batmon_temp_reading.publish) {
446  value = batmon_sensor.value(BATMON_SENSOR_TYPE_TEMP);
447  if(value != CC26XX_SENSOR_READING_ERROR) {
448  batmon_temp_reading.raw = value;
449 
450  buf = batmon_temp_reading.converted;
451  memset(buf, 0, CC26XX_WEB_DEMO_CONVERTED_LEN);
452  snprintf(buf, CC26XX_WEB_DEMO_CONVERTED_LEN, "%d", value);
453  }
454  }
455 
456  if(batmon_volt_reading.publish) {
457  value = batmon_sensor.value(BATMON_SENSOR_TYPE_VOLT);
458  if(value != CC26XX_SENSOR_READING_ERROR) {
459  batmon_volt_reading.raw = value;
460 
461  buf = batmon_volt_reading.converted;
462  memset(buf, 0, CC26XX_WEB_DEMO_CONVERTED_LEN);
463  snprintf(buf, CC26XX_WEB_DEMO_CONVERTED_LEN, "%d", (value * 125) >> 5);
464  }
465  }
466 
467  ctimer_set(&batmon_timer, next, get_batmon_reading, NULL);
468 }
