80 static uip_ds6_maddr_t *
81 join_mcast_group(void)
82 {
83  uip_ipaddr_t addr;
84  uip_ds6_maddr_t *rv;
85 
86  /* First, set our v6 global */
87  uip_ip6addr(&addr, UIP_DS6_DEFAULT_PREFIX, 0, 0, 0, 0, 0, 0, 0);
88  uip_ds6_set_addr_iid(&addr, &uip_lladdr);
89  uip_ds6_addr_add(&addr, 0, ADDR_AUTOCONF);
90 
91  /*
92  * IPHC will use stateless multicast compression for this destination
93  * (M=1, DAC=0), with 32 inline bits (1E 89 AB CD)
94  */
95  uip_ip6addr(&addr, 0xFF1E,0,0,0,0,0,0x89,0xABCD);
96  rv = uip_ds6_maddr_add(&addr);
97 
98  if(rv) {
99  PRINTF("Joined multicast group ");
100  PRINT6ADDR(&uip_ds6_maddr_lookup(&addr)->ipaddr);
101  PRINTF("\n");
102  }
103  return rv;
104 }
