55 ping6handler()
56 {
57  if(count < PING6_NB) {
58  UIP_IP_BUF->vtc = 0x60;
59  UIP_IP_BUF->tcflow = 1;
60  UIP_IP_BUF->flow = 0;
61  UIP_IP_BUF->proto = UIP_PROTO_ICMP6;
62  UIP_IP_BUF->ttl = uip_ds6_if.cur_hop_limit;
63  uip_ipaddr_copy(&UIP_IP_BUF->destipaddr, &dest_addr);
64  uip_ds6_select_src(&UIP_IP_BUF->srcipaddr, &UIP_IP_BUF->destipaddr);
65 
66  UIP_ICMP_BUF->type = ICMP6_ECHO_REQUEST;
67  UIP_ICMP_BUF->icode = 0;
68  /* set identifier and sequence number to 0 */
69  memset((uint8_t *)UIP_ICMP_BUF + UIP_ICMPH_LEN, 0, 4);
70  /* put one byte of data */
71  memset((uint8_t *)UIP_ICMP_BUF + UIP_ICMPH_LEN + UIP_ICMP6_ECHO_REQUEST_LEN,
72  count, PING6_DATALEN);
73 
74 
75  uip_len = UIP_ICMPH_LEN + UIP_ICMP6_ECHO_REQUEST_LEN + UIP_IPH_LEN
76  + PING6_DATALEN;
77  UIP_IP_BUF->len[0] = (uint8_t)((uip_len - 40) >> 8);
78  UIP_IP_BUF->len[1] = (uint8_t)((uip_len - 40) & 0x00FF);
79 
80  UIP_ICMP_BUF->icmpchksum = 0;
81  UIP_ICMP_BUF->icmpchksum = ~uip_icmp6chksum();
82 
83 
84  PRINTF("Echo Request to ");
85  PRINT6ADDR(&UIP_IP_BUF->destipaddr);
86  PRINTF(" from ");
87  PRINT6ADDR(&UIP_IP_BUF->srcipaddr);
88  PRINTF("\n");
89  UIP_STAT(++uip_stat.icmp.sent);
90 
91  tcpip_ipv6_output();
92 
93  count++;
94  etimer_set(&ping6_periodic_timer, 3 * CLOCK_SECOND);
95  } else {
96  count = 0;
97  }
98 }
