79 void
80 config_read(char *filename)
81 {
82  int file;
83 
84  file = cfs_open(filename, CFS_READ);
85  if(file < 0) {
86  log_message(filename, ": File not found");
87  error_exit();
88  }
89 
90  if(cfs_read(file, &config, sizeof(config)) < sizeof(uip_ipaddr_t) * 4
91  + sizeof(uint16_t)) {
92  log_message(filename, ": No config file");
93  error_exit();
94  }
95 
96  cfs_close(file);
97 
98  log_message("IP Address: ", ipaddrtoa(&config.hostaddr, uip_buf));
99  log_message("Subnet Mask: ", ipaddrtoa(&config.netmask, uip_buf));
100  log_message("Def. Router: ", ipaddrtoa(&config.draddr, uip_buf));
101  log_message("DNS Server: ", ipaddrtoa(&config.resolvaddr, uip_buf));
102 
103 #ifdef STATIC_DRIVER
104  #define _stringize(arg) #arg
105  #define stringize(arg) _stringize(arg)
106 #if WITH_SLIP
107  log_message("SLIP Driver: ", stringize(STATIC_DRIVER));
108 #else /* WITH_SLIP */
109  log_message("Eth. Driver: ", stringize(STATIC_DRIVER));
110 #endif /* WITH_SLIP */
111  #undef _stringize
112  #undef stringize
113 #else /* STATIC_DRIVER */
114  log_message("Eth. Driver: ", config.ethernet.name);
115 #endif /* STATIC_DRIVER */
116 #if !WITH_SLIP
117  log_message("Driver Port: $", utoa(config.ethernet.addr, uip_buf, 16));
118 #endif /* !WITH_SLIP */
119 
120  uip_sethostaddr(&config.hostaddr);
121  uip_setnetmask(&config.netmask);
122  uip_setdraddr(&config.draddr);
123 #if WITH_DNS
124  uip_nameserver_update(&config.resolvaddr, UIP_NAMESERVER_INFINITE_LIFETIME);
125 #endif /* WITH_DNS */
126 }
