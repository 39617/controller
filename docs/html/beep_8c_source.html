159  /* What's next on the playlist? */
160  switch (tuneindex++) {

163  case 3 :noteptr=furelise;break;
164  case 4 :noteptr=sandman2;break;
165  case 5 :noteptr=axel;break;

167  }

169  /* Turn on the power to the speaker. */
170  ENTER_DDR |= (1 << PE7);
171  ENTER_PUR |= (1 << PE7);
172 
173  /* Setup port pin */
174  BEEP_DDR |= (1 << BEEP_BIT);
175 
176  /* Start with no tone */
177  icnt = 0;
178  tone = NONE;
179 
180  /* Clock 8 bit timer at maximum frequency (CS0=1), interrupt overflow */
181  /* 8MHz / 256 = 31250 interrupts per second */
182  TCCR0A |= _BV(CS00);
183  TCNT0 = 0;
184  TIMSK0 |= _BV(TOIE0);
185 
186  /* Play all the notes */
187  for (;;) {
188  tone=pgm_read_byte(noteptr++);
189  if (tone==0xff) break;
190  for (i = pgm_read_byte(noteptr++);i > 0; i--) _delay_us(1000UL*TONE_LENGTH);
191  tone = NONE;_delay_us(1000UL*TONE_GAP);
192  }
193 
194  /* Turn off interrupts and speaker */
195  TIMSK0 &= ~_BV(TOIE0);
196  BEEP_PORT &= ~(1 << BEEP_BIT);
197  ENTER_PUR &= ~(1 << PE7);
