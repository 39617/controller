87  linkaddr_copy(&msg.src, packetbuf_addr(PACKETBUF_ADDR_SENDER));
88  linkaddr_copy(&msg.dest, packetbuf_addr(PACKETBUF_ADDR_RECEIVER));

93 static void

125 PROCESS_THREAD(shell_sniff_process, ev, data)
126 {
