183 static void
184 init_pcap(struct in_addr addr)
185 {
186  struct pcap_if *interfaces;
187  struct pcap_addr *paddr;
188  char error[256];
189 
190  if(pcap_findalldevs(&interfaces, error) == -1) {
191  error_exit(error);
192  }
193 
194  while(interfaces != NULL) {
195  log_message("init_pcap: found interface: ", interfaces->description);
196 
197  if(interfaces->addresses != NULL) {
198  for(paddr = interfaces->addresses;
199  paddr != NULL;
200  paddr = paddr->next) {
201  if(paddr->addr != NULL && paddr->addr->sa_family == AF_INET) {
202 
203  struct in_addr interface_addr;
204  interface_addr = ((struct sockaddr_in *)paddr->addr)->sin_addr;
205  log_message("init_pcap: with address: ", inet_ntoa(interface_addr));
206 
207  if(interface_addr.s_addr == addr.s_addr) {
208  pcap = pcap_open_live(interfaces->name, BUFSIZE, 0, -1, error);
209  if(pcap == NULL) {
210  error_exit(error);
211  }
212  return;
213  }
214  }
215  }
216  }
217  interfaces = interfaces->next;
218  }
219 
220  if(interfaces == NULL) {
221  error_exit("no interface found with ip addr specified on cmdline\n");
222  }
223 }

304 static void
305 uip_arp_update(uip_ipaddr_t *ipaddr, struct uip_eth_addr *ethaddr)
306 {
307  struct arp_entry *tabptr;
308  int i, tmpage, c;
309 
310  /* Walk through the ARP mapping table and try to find an entry to
311  update. If none is found, the IP -> MAC address mapping is
312  inserted in the ARP table. */
313  for(i = 0; i < UIP_ARPTAB_SIZE; ++i) {
314 
315  tabptr = &arp_table[i];
316  /* Only check those entries that are actually in use. */
317  if(!uip_ipaddr_cmp(&tabptr->ipaddr, &all_zeroes_addr)) {
318 
319  /* Check if the source IP address of the incoming packet matches
320  the IP address in this ARP table entry. */
321  if(uip_ipaddr_cmp(ipaddr, &tabptr->ipaddr)) {
322 
323  /* An old entry found, update this and return. */
324  memcpy(tabptr->ethaddr.addr, ethaddr->addr, 6);
325  tabptr->time = arptime;
326 
327  return;
328  }
329  }
330  }
331 
332  /* If we get here, no existing ARP table entry was found, so we
333  create one. */
334 
335  /* First, we try to find an unused entry in the ARP table. */
336  for(i = 0; i < UIP_ARPTAB_SIZE; ++i) {
337  tabptr = &arp_table[i];
338  if(uip_ipaddr_cmp(&tabptr->ipaddr, &all_zeroes_addr)) {
339  break;
340  }
341  }
342 
343  /* If no unused entry is found, we try to find the oldest entry and
344  throw it away. */
345  if(i == UIP_ARPTAB_SIZE) {
346  tmpage = 0;
347  c = 0;
348  for(i = 0; i < UIP_ARPTAB_SIZE; ++i) {
349  tabptr = &arp_table[i];
350  if(arptime - tabptr->time > tmpage) {
351  tmpage = arptime - tabptr->time;
352  c = i;
353  }
354  }
355  i = c;
356  tabptr = &arp_table[i];
357  }
358 
359  /* Now, i is the ARP table entry which we will fill with the new
360  information. */
361  uip_ipaddr_copy(&tabptr->ipaddr, ipaddr);
362  memcpy(tabptr->ethaddr.addr, ethaddr->addr, 6);
363  tabptr->time = arptime;
364 }

471 void *
472 do_arp(void *buf, int len)
473 {
474  struct arp_hdr *hdr;
475 
476  hdr = (struct arp_hdr *)buf;
477  if(hdr->ethhdr.type == UIP_HTONS(UIP_ETHTYPE_ARP)) {
478  if(hdr->opcode == UIP_HTONS(ARP_REQUEST)) {
479  /* Check if the ARP is for our network */
480  /* fprintf(stderr, "ARP for %d.%d.%d.%d we are %d.%d.%d.%d/%d.%d.%d.%d\n",
481  uip_ipaddr_to_quad(&hdr->dipaddr),
482  uip_ipaddr_to_quad(&netaddr),
483  uip_ipaddr_to_quad(&netmask));*/
484  if(uip_ipaddr_maskcmp(&hdr->dipaddr, &netaddr, &netmask)) {
485  uip_ipaddr_t tmpaddr;
486 
487  /* fprintf(stderr, "ARP for us.\n");*/
488  uip_arp_update(&hdr->sipaddr, &hdr->shwaddr);
489 
490  hdr->opcode = UIP_HTONS(ARP_REPLY);
491 
492  memcpy(&hdr->dhwaddr.addr, &hdr->shwaddr.addr, 6);
493  memcpy(&hdr->shwaddr.addr, &uip_lladdr.addr, 6);
494  memcpy(&hdr->ethhdr.src.addr, &uip_lladdr.addr, 6);
495  memcpy(&hdr->ethhdr.dest.addr, &hdr->dhwaddr.addr, 6);
496 
497  uip_ipaddr_copy(&tmpaddr, &hdr->dipaddr);
498  uip_ipaddr_copy(&hdr->dipaddr, &hdr->sipaddr);
499  uip_ipaddr_copy(&hdr->sipaddr, &tmpaddr);
500 
501  hdr->ethhdr.type = UIP_HTONS(UIP_ETHTYPE_ARP);
502  raw_send(hdr, sizeof(struct arp_hdr));
503  return NULL;
504  }
505  } else if(hdr->opcode == UIP_HTONS(ARP_REPLY)) {
506  /* ARP reply. We insert or update the ARP table if it was meant
507  for us. */
508  if(uip_ipaddr_maskcmp(&hdr->dipaddr, &netaddr, &netmask)) {
509  uip_arp_update(&hdr->sipaddr, &hdr->shwaddr);
510  }
511  }
512  }
513  if(hdr->ethhdr.type == UIP_HTONS(UIP_ETHTYPE_IP)) {
514  return &((struct ethip_hdr *)hdr)->vhl;
515  }
516  return NULL;
517 }
