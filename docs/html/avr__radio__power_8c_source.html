76 PROCESS_THREAD(power_use_process, ev, data)
77 {
78  static int i, j;
79  PROCESS_EXITHANDLER(broadcast_close(&broadcast);)
80  PROCESS_BEGIN();
81 
82  /* powertrace_start(CLOCK_SECOND * 2); */
83  broadcast_open(&broadcast, 129, &broadcast_call);
84  leds_init();
85  rf230_set_channel(25);
86 
87  for(i=0; i < 2; i++) { /* Loop over min and max rpc settings */
88 
89  NETSTACK_RADIO.off(); /* Radio off for rpc change */
90  NETSTACK_RADIO.off();
91  etimer_set(&et, CLOCK_SECOND * 8);
92  PROCESS_WAIT_EVENT_UNTIL(etimer_expired(&et));
93  if(i == 0)
94  rf230_set_rpc(0x0); /* Disbable all RPC features */
95  else
96  rf230_set_rpc(0xFF); /* Enable all RPC features. Only XRFR2 radios */
97  NETSTACK_RADIO.on();
98 
99  /* Loop over the different TX power settings 0-15 */
100 
101  for(j=15; j >= 0; j--) {
102  NETSTACK_RADIO.on();
103  rf230_set_txpower(j);
104  ps = 0;
105  printf("TX with PWR=%d RPC=0x%02x\n", rf230_get_txpower(), rf230_get_rpc());
106  ps = 1;
107 
108  etimer_set(&et, CLOCK_SECOND * TEST_PERIOD);
109 
110  while(1) {
111  if(etimer_expired(&et))
112  break;
113  packetbuf_copyfrom(buf, sizeof(buf)); /* Dummy data */
114  broadcast_send(&broadcast);
115  PROCESS_PAUSE();
116  //leds_on(LEDS_RED);
117  }
118  /* Barrier so we can see next run */
119  NETSTACK_RADIO.off();
120  etimer_set(&et, CLOCK_SECOND * TEST_PERIOD);
121  PROCESS_WAIT_EVENT_UNTIL(etimer_expired(&et));
122  }
123 
124  /* RX */
125  NETSTACK_RADIO.on();
126  ps = 0;
127  printf("RX radio on RPC=0x%02x\n", rf230_get_rpc());
128  ps = 1;
129  etimer_set(&et, CLOCK_SECOND * TEST_PERIOD);
130  PROCESS_WAIT_EVENT_UNTIL(etimer_expired(&et));
131  NETSTACK_RADIO.off();
132  }
133 
134  /* Last just RX OFF*/
135  ps = 0;
136  printf("RX radio off\n");
137  ps = 1;
138  etimer_set(&et, CLOCK_SECOND * TEST_PERIOD);
139  PROCESS_WAIT_EVENT_UNTIL(etimer_expired(&et));
140 
141  PROCESS_END();
142 }
