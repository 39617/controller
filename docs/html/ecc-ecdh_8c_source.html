57 static void
58 ecc_set_random(uint32_t *secret)
59 {
60  int i;
61  for(i = 0; i < 8; ++i) {
62  secret[i] = (uint32_t)random_rand() | (uint32_t)random_rand() << 16;
63  }
64 }
