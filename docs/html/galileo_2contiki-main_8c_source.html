88 int
89 main(void)
90 {
91  uintptr_t *func_ptr;
92 
93 #ifdef X86_CONF_RESTRICT_DMA
94  quarkX1000_imr_conf();
95 #endif
96  irq_init();
97  quarkX1000_uart_init();
98  /* Initialize UART connected to Galileo Gen1 3.5mm audio-style jack or
99  * Galileo Gen2 FTDI header
100  */
101  quarkX1000_uart_init_port(QUARK_X1000_UART_1, 115200);
102  SET_EXCEPTION_HANDLER(13, 1, gp_fault_handler);
103  clock_init();
104  rtimer_init();
105 
106  pci_root_complex_init();
107  quarkX1000_eth_init();
108  quarkX1000_i2c_init();
109  quarkX1000_i2c_configure(QUARKX1000_I2C_SPEED_STANDARD,
110  QUARKX1000_I2C_ADDR_MODE_7BIT);
111  /* The GPIO subsystem must be initialized prior to configuring pinmux, because
112  * the pinmux configuration automatically performs GPIO configuration for the
113  * relevant pins.
114  */
115  quarkX1000_gpio_init();
116  /* use default pinmux configuration */
117  if(galileo_pinmux_initialize() < 0) {
118  fprintf(stderr, "Failed to initialize pinmux\n");
119  }
120  shared_isr_init();
121 
122  /* The ability to remap interrupts is not needed after this point and should
123  * thus be disabled according to the principle of least privilege.
124  */
125  pci_root_complex_lock();
126 
127  func_ptr = (uintptr_t *)&_sdata_kern_startup_func;
128  while(func_ptr != (uintptr_t *)&_edata_kern_startup_func) {
129  ((void (*)(void))*func_ptr)();
130 
131  func_ptr++;
132  }
133 
134  prot_domains_leave_main();
135 
136  return 0;
137 }
