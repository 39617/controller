127 static radio_result_t
128 get_value(radio_param_t param, radio_value_t *value)
129 {
130  if(!value) {
131  return RADIO_RESULT_INVALID_VALUE;
132  }
133  switch(param) {
134  case RADIO_PARAM_POWER_MODE:
135  *value = receive_on ? RADIO_POWER_MODE_ON : RADIO_POWER_MODE_OFF;
136  return RADIO_RESULT_OK;
137  case RADIO_PARAM_CHANNEL:
138  *value = cc2520_get_channel();
139  return RADIO_RESULT_OK;
140  case RADIO_CONST_CHANNEL_MIN:
141  *value = 11;
142  return RADIO_RESULT_OK;
143  case RADIO_CONST_CHANNEL_MAX:
144  *value = 26;
145  return RADIO_RESULT_OK;
146  default:
147  return RADIO_RESULT_NOT_SUPPORTED;
148  }
149 }
150 static radio_result_t
151 set_value(radio_param_t param, radio_value_t value)
152 {
153  switch(param) {
154  case RADIO_PARAM_POWER_MODE:
155  if(value == RADIO_POWER_MODE_ON) {
156  cc2520_on();
157  return RADIO_RESULT_OK;
158  }
159  if(value == RADIO_POWER_MODE_OFF) {
160  cc2520_off();
161  return RADIO_RESULT_OK;
162  }
163  return RADIO_RESULT_INVALID_VALUE;
164  case RADIO_PARAM_CHANNEL:
165  if(value < 11 || value > 26) {
166  return RADIO_RESULT_INVALID_VALUE;
167  }
168  cc2520_set_channel(value);
169  return RADIO_RESULT_OK;
170  default:
171  return RADIO_RESULT_NOT_SUPPORTED;
172  }

244 static void
245 on(void)
246 {
247  CC2520_ENABLE_FIFOP_INT();
248  strobe(CC2520_INS_SRXON);
249 
250  BUSYWAIT_UNTIL(status() & (BV(CC2520_XOSC16M_STABLE)), RTIMER_SECOND / 100);
251 
252  ENERGEST_ON(ENERGEST_TYPE_LISTEN);
253  receive_on = 1;

255 static void
256 off(void)
257 {
258  /* PRINTF("off\n");*/
259  receive_on = 0;
260 
261  /* Wait for transmission to end before turning radio off. */
262  BUSYWAIT_UNTIL(!(status() & BV(CC2520_TX_ACTIVE)), RTIMER_SECOND / 10);
263 
264  ENERGEST_OFF(ENERGEST_TYPE_LISTEN);
265  strobe(CC2520_INS_SRFOFF);
266  CC2520_DISABLE_FIFOP_INT();
267 
268  if(!CC2520_FIFOP_IS_1) {
269  flushrx();
270  }
271 }

317 int
318 cc2520_init(void)
319 {
320  {
321  int s = splhigh();
322  cc2520_arch_init(); /* Initalize ports and SPI. */
323  CC2520_DISABLE_FIFOP_INT();
324  CC2520_FIFOP_INT_INIT();
325  splx(s);
326  }
327 
328  SET_VREG_INACTIVE();
329  clock_delay(250);
330  /* Turn on voltage regulator and reset. */
331  SET_VREG_ACTIVE();
332  clock_delay(250);
333  SET_RESET_ACTIVE();
334  clock_delay(127);
335  SET_RESET_INACTIVE();
336  clock_delay(125);
337  /* Turn on the crystal oscillator. */
338  strobe(CC2520_INS_SXOSCON);
339  clock_delay(125);
340 
341  BUSYWAIT_UNTIL(status() & (BV(CC2520_XOSC16M_STABLE)), RTIMER_SECOND / 100);
342 
343  /* Change default values as recommended in the data sheet, */
344  /* correlation threshold = 20, RX bandpass filter = 1.3uA.*/
345 
346  setreg(CC2520_TXCTRL, 0x94);
347  setreg(CC2520_TXPOWER, 0x13); /* Output power 1 dBm */
348 
349  /*
350 
351  valeurs de TXPOWER
352  0x03 -> -18 dBm
353  0x2C -> -7 dBm
354  0x88 -> -4 dBm
355  0x81 -> -2 dBm
356  0x32 -> 0 dBm
357  0x13 -> 1 dBm
358  0x32 -> 0 dBm
359  0x13 -> 1 dBm
360  0xAB -> 2 dBm
361  0xF2 -> 3 dBm
362  0xF7 -> 5 dBm
363  */
364  setreg(CC2520_CCACTRL0, 0xF8); /* CCA treshold -80dBm */
365 
366  /* Recommended RX settings */
367  setreg(CC2520_MDMCTRL0, 0x84); /* Controls modem */
368  setreg(CC2520_MDMCTRL1, 0x14); /* Controls modem */
369  setreg(CC2520_RXCTRL, 0x3F); /* Adjust currents in RX related analog modules */
370  setreg(CC2520_FSCTRL, 0x5A); /* Adjust currents in synthesizer. */
371  setreg(CC2520_FSCAL1, 0x2B); /* Adjust currents in VCO */
372  setreg(CC2520_AGCCTRL1, 0x11); /* Adjust target value for AGC control loop */
373  setreg(CC2520_AGCCTRL2, 0xEB);
374 
375  /* Disable external clock */
376  setreg(CC2520_EXTCLOCK, 0x00);
377 
378  /* Tune ADC performance */
379  setreg(CC2520_ADCTEST0, 0x10);
380  setreg(CC2520_ADCTEST1, 0x0E);
381  setreg(CC2520_ADCTEST2, 0x03);
382 
383  /* Set auto CRC on frame. */
384 #if CC2520_CONF_AUTOACK
385  setreg(CC2520_FRMCTRL0, AUTOCRC | AUTOACK);
386  setreg(CC2520_FRMFILT0, FRAME_MAX_VERSION | FRAME_FILTER_ENABLE);
387 #else
388  /* setreg(CC2520_FRMCTRL0, 0x60); */
389  setreg(CC2520_FRMCTRL0, AUTOCRC);
390  /* Disable filter on @ (remove if you want to address specific wismote) */
391  setreg(CC2520_FRMFILT0, 0x00);
392 #endif /* CC2520_CONF_AUTOACK */
393  /* SET_RXENMASK_ON_TX */
394  setreg(CC2520_FRMCTRL1, 1);
395  /* Set FIFOP threshold to maximum .*/
396  setreg(CC2520_FIFOPCTRL, FIFOP_THR(0x7F));
397 
398  cc2520_set_pan_addr(0xffff, 0x0000, NULL);
399  cc2520_set_channel(26);
400 
401  flushrx();
402 
403  process_start(&cc2520_process, NULL);
404  return 1;
405 }

407 static int
408 cc2520_transmit(unsigned short payload_len)
409 {
410  int i, txpower;
411 
412  GET_LOCK();
413 
414  txpower = 0;
415  if(packetbuf_attr(PACKETBUF_ATTR_RADIO_TXPOWER) > 0) {
416  /* Remember the current transmission power */
417  txpower = cc2520_get_txpower();
418  /* Set the specified transmission power */
419  set_txpower(packetbuf_attr(PACKETBUF_ATTR_RADIO_TXPOWER) - 1);
420  }
421 
422  /* The TX FIFO can only hold one packet. Make sure to not overrun
423  * FIFO by waiting for transmission to start here and synchronizing
424  * with the CC2520_TX_ACTIVE check in cc2520_send.
425  *
426  * Note that we may have to wait up to 320 us (20 symbols) before
427  * transmission starts.
428  */
429 #ifndef CC2520_CONF_SYMBOL_LOOP_COUNT
430 #error CC2520_CONF_SYMBOL_LOOP_COUNT needs to be set!!!
431 #else
432 #define LOOP_20_SYMBOLS CC2520_CONF_SYMBOL_LOOP_COUNT
433 #endif
434 
435 #if WITH_SEND_CCA
436  strobe(CC2520_INS_SRXON);
437  BUSYWAIT_UNTIL(status() & BV(CC2520_RSSI_VALID), RTIMER_SECOND / 10);
438  strobe(CC2520_INS_STXONCCA);
439 #else /* WITH_SEND_CCA */
440  strobe(CC2520_INS_STXON);
441 #endif /* WITH_SEND_CCA */
442  for(i = LOOP_20_SYMBOLS; i > 0; i--) {
443  if(CC2520_SFD_IS_1) {
444 #if PACKETBUF_WITH_PACKET_TYPE
445  {
446  rtimer_clock_t sfd_timestamp;
447  sfd_timestamp = cc2520_sfd_start_time;
448  if(packetbuf_attr(PACKETBUF_ATTR_PACKET_TYPE) ==
449  PACKETBUF_ATTR_PACKET_TYPE_TIMESTAMP) {
450  /* Write timestamp to last two bytes of packet in TXFIFO. */
451  CC2520_WRITE_RAM(&sfd_timestamp, CC2520RAM_TXFIFO + payload_len - 1, 2);
452  }
453  }
454 #endif /* PACKETBUF_WITH_PACKET_TYPE */
455 
456  if(!(status() & BV(CC2520_TX_ACTIVE))) {
457  /* SFD went high but we are not transmitting. This means that
458  we just started receiving a packet, so we drop the
459  transmission. */
460  RELEASE_LOCK();
461  return RADIO_TX_COLLISION;
462  }
463  if(receive_on) {
464  ENERGEST_OFF(ENERGEST_TYPE_LISTEN);
465  }
466  ENERGEST_ON(ENERGEST_TYPE_TRANSMIT);
467  /* We wait until transmission has ended so that we get an
468  accurate measurement of the transmission time.*/
469  /* BUSYWAIT_UNTIL(getreg(CC2520_EXCFLAG0) & TX_FRM_DONE , RTIMER_SECOND / 100); */
470  BUSYWAIT_UNTIL(!(status() & BV(CC2520_TX_ACTIVE)), RTIMER_SECOND / 10);
471 
472 #ifdef ENERGEST_CONF_LEVELDEVICE_LEVELS
473  ENERGEST_OFF_LEVEL(ENERGEST_TYPE_TRANSMIT, cc2520_get_txpower());
474 #endif
475  ENERGEST_OFF(ENERGEST_TYPE_TRANSMIT);
476  if(receive_on) {
477  ENERGEST_ON(ENERGEST_TYPE_LISTEN);
478  } else {
479  /* We need to explicitly turn off the radio,
480  * since STXON[CCA] -> TX_ACTIVE -> RX_ACTIVE */
481  off();
482  }
483 
484  if(packetbuf_attr(PACKETBUF_ATTR_RADIO_TXPOWER) > 0) {
485  /* Restore the transmission power */
486  set_txpower(txpower & 0xff);
487  }
488 
489  RELEASE_LOCK();
490 
491  return RADIO_TX_OK;
492  }
493  }
494 
495  /* If we are using WITH_SEND_CCA, we get here if the packet wasn't
496  transmitted because of other channel activity. */
497  RIMESTATS_ADD(contentiondrop);
498  PRINTF("cc2520: do_send() transmission never started\n");
499 
500  if(packetbuf_attr(PACKETBUF_ATTR_RADIO_TXPOWER) > 0) {
501  /* Restore the transmission power */
502  set_txpower(txpower & 0xff);
503  }
504 
505  RELEASE_LOCK();
506  return RADIO_TX_COLLISION;
507 }

542 /*---------------------------------------------------------------------------*/
543 int
544 cc2520_off(void)
545 {
546  /* Don't do anything if we are already turned off. */
547  if(receive_on == 0) {
548  return 1;
549  }
550 
551  /* If we are called when the driver is locked, we indicate that the
552  radio should be turned off when the lock is unlocked. */
553  if(locked) {
554  /* printf("Off when locked (%d)\n", locked);*/
555  lock_off = 1;
556  return 1;
557  }
558 
559  GET_LOCK();
560  /* If we are currently receiving a packet (indicated by SFD == 1),
561  we don't actually switch the radio off now, but signal that the
562  driver should switch off the radio once the packet has been
563  received and processed, by setting the 'lock_off' variable. */
564  if(status() & BV(CC2520_TX_ACTIVE)) {
565  lock_off = 1;
566  } else {
567  off();
568  }
569  RELEASE_LOCK();
570  return 1;
571 }

573 int
574 cc2520_on(void)
575 {
576  if(receive_on) {
577  return 1;
578  }
579  if(locked) {
580  lock_on = 1;
581  return 1;
582  }
583 
584  GET_LOCK();
585  on();
586  RELEASE_LOCK();
587  return 1;
588 }

596 int
597 cc2520_set_channel(int c)
598 {
599  uint16_t f;
600 
601  GET_LOCK();
602  /*
603  * Subtract the base channel (11), multiply by 5, which is the
604  * channel spacing. 357 is 2405-2048 and 0x4000 is LOCK_THR = 1.
605  */
606  channel = c;
607 
608  f = MIN_CHANNEL + ((channel - MIN_CHANNEL) * CHANNEL_SPACING);
609  /*
610  * Writing RAM requires crystal oscillator to be stable.
611  */
612  BUSYWAIT_UNTIL((status() & (BV(CC2520_XOSC16M_STABLE))), RTIMER_SECOND / 10);
613 
614  /* Wait for any transmission to end. */
615  BUSYWAIT_UNTIL(!(status() & BV(CC2520_TX_ACTIVE)), RTIMER_SECOND / 10);
616 
617  /* Define radio channel (between 11 and 25) */
618  setreg(CC2520_FREQCTRL, f);
619 
620  /* If we are in receive mode, we issue an SRXON command to ensure
621  that the VCO is calibrated. */
622  if(receive_on) {
623  strobe(CC2520_INS_SRXON);
624  }
625 
626  RELEASE_LOCK();
627  return 1;
628 }

630 void
631 cc2520_set_pan_addr(unsigned pan,
632  unsigned addr,
633  const uint8_t *ieee_addr)
634 {
635  uint8_t tmp[2];
636 
637  GET_LOCK();
638 
639  /*
640  * Writing RAM requires crystal oscillator to be stable.
641  */
642  BUSYWAIT_UNTIL(status() & (BV(CC2520_XOSC16M_STABLE)), RTIMER_SECOND / 10);
643 
644  tmp[0] = pan & 0xff;
645  tmp[1] = pan >> 8;
646  CC2520_WRITE_RAM(&tmp, CC2520RAM_PANID, 2);
647 
648  tmp[0] = addr & 0xff;
649  tmp[1] = addr >> 8;
650  CC2520_WRITE_RAM(&tmp, CC2520RAM_SHORTADDR, 2);
651  if(ieee_addr != NULL) {
652  int f;
653  uint8_t tmp_addr[8];
654  /* LSB first, MSB last for 802.15.4 addresses in CC2520 */
655  for(f = 0; f < 8; f++) {
656  tmp_addr[7 - f] = ieee_addr[f];
657  }
658  CC2520_WRITE_RAM(tmp_addr, CC2520RAM_IEEEADDR, 8);
659  }
660  RELEASE_LOCK();
661 }

676 /*---------------------------------------------------------------------------*/
677 PROCESS_THREAD(cc2520_process, ev, data)
678 {
679  int len;
680  PROCESS_BEGIN();
681 
682  PRINTF("cc2520_process: started\n");
683 
684  while(1) {
685  PROCESS_YIELD_UNTIL(ev == PROCESS_EVENT_POLL);
686

700 /*---------------------------------------------------------------------------*/
701 static int
702 cc2520_read(void *buf, unsigned short bufsize)
703 {
704  uint8_t footer[2];
705  uint8_t len;
706 
707  if(!CC2520_FIFOP_IS_1) {
708  return 0;
709  }
710 
711  GET_LOCK();
712 
713  cc2520_packets_read++;
714 
715  getrxbyte(&len);
716 
717  if(len > CC2520_MAX_PACKET_LEN) {
718  /* Oops, we must be out of sync. */
719  flushrx();
720  RIMESTATS_ADD(badsynch);
721  RELEASE_LOCK();
722  return 0;
723  }
724 
725  if(len <= FOOTER_LEN) {
726  flushrx();
727  RIMESTATS_ADD(tooshort);
728  RELEASE_LOCK();
729  return 0;
730  }
731 
732  if(len - FOOTER_LEN > bufsize) {
733  flushrx();
734  RIMESTATS_ADD(toolong);
735  RELEASE_LOCK();
736  return 0;
737  }
738 
739  getrxdata(buf, len - FOOTER_LEN);
740  getrxdata(footer, FOOTER_LEN);
741 
742  if(footer[1] & FOOTER1_CRC_OK) {
743  cc2520_last_rssi = footer[0] - RSSI_OFFSET;
744  cc2520_last_correlation = footer[1] & FOOTER1_CORRELATION;
745 
746  packetbuf_set_attr(PACKETBUF_ATTR_RSSI, cc2520_last_rssi);
747  packetbuf_set_attr(PACKETBUF_ATTR_LINK_QUALITY, cc2520_last_correlation);
748 
749  RIMESTATS_ADD(llrx);
750  } else {
751  RIMESTATS_ADD(badcrc);
752  len = FOOTER_LEN;
753  }
754 
755  if(CC2520_FIFOP_IS_1) {
756  if(!CC2520_FIFO_IS_1) {
757  /* Clean up in case of FIFO overflow! This happens for every
758  * full length frame and is signaled by FIFOP = 1 and FIFO =
759  * 0. */
760  flushrx();
761  } else {
762  /* Another packet has been received and needs attention. */
763  process_poll(&cc2520_process);
764  }
765  }
766 
767  RELEASE_LOCK();
768 
769  if(len < FOOTER_LEN) {
770  return 0;
771  }
772 
773  return len - FOOTER_LEN;
774 }

873

875 
876  if(radio_was_off) {

878  }
