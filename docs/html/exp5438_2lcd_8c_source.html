87 lcd_write_char(char c)
88 {
89  char string[2];
90 
91  if(WITH_LCD) {
92  if(c == '\n') {
93  inc_y();
94  } else {
95  string[0] = c;
96  string[1] = 0;
97 
98  if(ypos == Y_MAX) {
99  lcd_clear();
100  ypos = xpos = 0;
101  }
102 
103  halLcdPrintXY(string, xpos * X_CHAR_SIZE, ypos * Y_CHAR_SIZE, 0);
104 
105  if(xpos == X_MAX) {
106  inc_y();
107  } else {
108  xpos++;
109  }
110  }
111  }
112 }
