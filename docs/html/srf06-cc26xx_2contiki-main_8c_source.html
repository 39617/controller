107 static void
108 fade(unsigned char l)
109 {
110  volatile int i;
111  int k, j;
112  for(k = 0; k < 800; ++k) {
113  j = k > 400 ? 800 - k : k;
114 
115  leds_on(l);
116  for(i = 0; i < j; ++i) {
117  __asm("nop");
118  }
119  leds_off(l);
120  for(i = 0; i < 400 - j; ++i) {
121  __asm("nop");
122  }
123  }
124 }

126 static void
127 set_rf_params(void)
128 {
129  uint16_t short_addr;
130  uint8_t ext_addr[8];
131  radio_value_t val = 0;
132 
133  ieee_addr_cpy_to(ext_addr, 8);
134 
135  short_addr = ext_addr[7];
136  short_addr |= ext_addr[6] << 8;
137 
138  /* In case the OUI contain a broadcast bit, we mask that out here. */
139  ext_addr[0] = (ext_addr[0] & 0xfe);
140  /* Set the U/L bit, in order to create a locally administered MAC address
141  as stated by the RFC 2464 */
142  ext_addr[0] = (ext_addr[0] | 0x02);
143 
144  // Just to make it compatible with our ethernet translation.
145  ext_addr[3] = 0xFF;
146  ext_addr[4] = 0xFE;
147 
148 #if UIP_CONF_DS6_INTERFACES_NUMBER > 1
149  translate_lowpan_to_eth(&ethernet_if_addr, (uip_lladdr_t *)&ext_addr);
150 #endif /* UIP_CONF_DS6_INTERFACES_NUMBER > 1 */
151 
152  /* Populate linkaddr_node_addr. Maintain endianness */
153  memcpy(&linkaddr_node_addr, &ext_addr[8 - LINKADDR_SIZE], LINKADDR_SIZE);
154 
155  NETSTACK_RADIO.set_value(RADIO_PARAM_PAN_ID, IEEE802154_PANID);
156  NETSTACK_RADIO.set_value(RADIO_PARAM_16BIT_ADDR, short_addr);
157  NETSTACK_RADIO.set_value(RADIO_PARAM_CHANNEL, RF_CORE_CHANNEL);
158  NETSTACK_RADIO.set_object(RADIO_PARAM_64BIT_ADDR, ext_addr, 8);
159 
160  NETSTACK_RADIO.get_value(RADIO_PARAM_CHANNEL, &val);
161  printf(" RF: Channel %d\n", val);
162 
163  int i;
164  printf("* Link layer addr: ");
165  for(i = 0; i < LINKADDR_SIZE - 1; i++) {
166  printf("%02x:", linkaddr_node_addr.u8[i]);
167  }
168  printf("%02x\n", linkaddr_node_addr.u8[i]);
169 
170  /* also set the global node id */
171  node_id = short_addr;
172  printf(" Node ID: %d\n", node_id);
173 }
