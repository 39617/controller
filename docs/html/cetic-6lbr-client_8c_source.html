74 {
75  uint16_t a;

79 
80  for(i = 0, f = 0; i < sizeof(uip_ipaddr_t); i += 2) {
81  a = (addr->u8[i] << 8) + addr->u8[i + 1];
82  if(a == 0 && f >= 0) {
83  if(f++ == 0) {
84  p += sprintf(p, "::");
85  }
86  } else {

88  f = -1;

91  }
