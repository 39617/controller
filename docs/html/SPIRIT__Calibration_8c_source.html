182 void SpiritCalibrationSetRcoCalWords(uint8_t cRwt, uint8_t cRfb)
183 {
184  uint8_t tempRegValue[2];
185 
186  /* Build the value of RWT and the MSbits of the RFB word */
187  tempRegValue[0] = (cRwt << 4) | (cRfb >> 1);
188 
189  /* Reads the register value to update the LSbit of RFB */
190  g_xStatus = SpiritSpiReadRegisters(RCO_VCO_CALIBR_IN1_BASE, 1, &tempRegValue[1]);

194
