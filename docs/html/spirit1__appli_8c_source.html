606 static void SystemClockConfig_STOP(void)
607 {
608  RCC_ClkInitTypeDef RCC_ClkInitStruct;
609  RCC_OscInitTypeDef RCC_OscInitStruct;
610 
611  /* Enable Power Control clock */
612  __PWR_CLK_ENABLE();
613 
614  /* The voltage scaling allows optimizing the power consumption when the device is
615  clocked below the maximum system frequency, to update the voltage scaling value
616  regarding system frequency refer to product datasheet. */
617  __HAL_PWR_VOLTAGESCALING_CONFIG(PWR_REGULATOR_VOLTAGE_SCALE1);
618 
619  /* Get the Oscillators configuration according to the internal RCC registers */
620  HAL_RCC_GetOscConfig(&RCC_OscInitStruct);
621 
622  /* After wake-up from STOP reconfigure the system clock: Enable HSI and PLL */
623  RCC_OscInitStruct.OscillatorType = RCC_OSCILLATORTYPE_HSI;
624  RCC_OscInitStruct.HSEState = RCC_HSE_OFF;
625  RCC_OscInitStruct.HSIState = RCC_HSI_ON;
626  RCC_OscInitStruct.PLL.PLLState = RCC_PLL_ON;
627  RCC_OscInitStruct.PLL.PLLSource = RCC_PLLSOURCE_HSI;
628  RCC_OscInitStruct.PLL.PLLMUL = RCC_PLLMUL_4;
629  RCC_OscInitStruct.PLL.PLLDIV = RCC_PLLDIV_2;
630  RCC_OscInitStruct.HSICalibrationValue = 0x10;
631  HAL_RCC_OscConfig(&RCC_OscInitStruct);
632 
633  /* Select PLL as system clock source and configure the HCLK, PCLK1 and PCLK2
634  clocks dividers */
