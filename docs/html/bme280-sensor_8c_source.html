41 static int
42 value(int type)
43 {
44  /* Read all measurements with one burst read */
45  bme280_read(BME280_MODE_WEATHER);
46 
47  /* Return a la Contiki API */
48  switch(type) {
49 
50  case BME280_SENSOR_TEMP:
51  return bme280_mea.t_overscale100 / 100;
52 
53  case BME280_SENSOR_HUMIDITY:
54  return bme280_mea.h_overscale1024 >> 10;
55 
56  case BME280_SENSOR_PRESSURE:
57  /* Scale down w. 10 not to overslow the signed int */
58 #ifdef BME280_64BIT
59  return bme280_mea.p_overscale256 / (256 * 10);
60 #else
61  return bme280_mea.p / 10;
62 #endif
63  }
64  return 0;
65 }
