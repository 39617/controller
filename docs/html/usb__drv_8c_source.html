187 U8 usb_select_enpoint_interrupt(void)
188 {
189 U8 interrupt_flags;
190 U8 ep_num;
191 
192  ep_num = 0;
193  interrupt_flags = Usb_interrupt_flags();
194 
195  while(ep_num < MAX_EP_NB)
196  {
197  if (interrupt_flags & 1)
198  {
199  return (ep_num);
200  }
201  else
202  {
203  ep_num++;
204  interrupt_flags = interrupt_flags >> 1;
205  }
206  }
207  return 0;
208 }
