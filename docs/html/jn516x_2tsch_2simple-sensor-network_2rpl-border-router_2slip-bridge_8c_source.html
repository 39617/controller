65 slip_input_callback(void)
66 {
67  PRINTF("SIN: %u\n", uip_len);
68  if(uip_buf[0] == '!') {
69  PRINTF("Got configuration message of type %c\n", uip_buf[1]);
70  uip_len = 0;
71  if(uip_buf[1] == 'P') {
72  uip_ipaddr_t prefix;
73  /* Here we set a prefix !!! */
74  memset(&prefix, 0, 16);
75  memcpy(&prefix, &uip_buf[2], 8);
76  PRINTF("Setting prefix ");
77  PRINT6ADDR(&prefix);
78  PRINTF("\n");
79  set_prefix_64(&prefix);
80  }
81  } else if (uip_buf[0] == '?') {
82  PRINTF("Got request message of type %c\n", uip_buf[1]);
83  if(uip_buf[1] == 'M') {
84  char* hexchar = "0123456789abcdef";
85  int j;
86  /* this is just a test so far... just to see if it works */
87  uip_buf[0] = '!';
88  for(j = 0; j < 8; j++) {
89  uip_buf[2 + j * 2] = hexchar[uip_lladdr.addr[j] >> 4];
90  uip_buf[3 + j * 2] = hexchar[uip_lladdr.addr[j] & 15];
91  }
92  uip_len = 18;
93  slip_send();
94 
95  }
96  uip_len = 0;
97  }
98  /* Save the last sender received over SLIP to avoid bouncing the
99  packet back if no route is found */
100  uip_ipaddr_copy(&last_sender, &UIP_IP_BUF->srcipaddr);
101 }
