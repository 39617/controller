146 {

149 
150  P4DIR |= BV(FLASH_CS) | BV(FLASH_HOLD) | BV(FLASH_PWR);
151  P4OUT |= BV(FLASH_PWR); /* P4.3 Output, turn on power! */
152 
153  /* Release from Deep Power-down */
154  s = splhigh();
155  SPI_FLASH_ENABLE();
156  SPI_WRITE_FAST(SPI_FLASH_INS_RES);
157  SPI_WAITFORTx_ENDED();
158  SPI_FLASH_DISABLE(); /* Unselect flash. */
159  splx(s);
160 
161  SPI_FLASH_UNHOLD();
162 }

229 {

231  const unsigned long end = addr + size;
232  unsigned long i, next_page;
233 
234  ENERGEST_ON(ENERGEST_TYPE_FLASH_WRITE);
235 
236  for(i = addr; i < end;) {
237  next_page = (i | 0xff) + 1;
238  if(next_page > end) {
239  next_page = end;
240  }
241  p = program_page(i, p, next_page - i);
242  i = next_page;
243  }
244 
245  ENERGEST_OFF(ENERGEST_TYPE_FLASH_WRITE);
246 
247  return size;
248 }
