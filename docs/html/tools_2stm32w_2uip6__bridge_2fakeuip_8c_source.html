72 static uint16_t
73 chksum(uint16_t sum, const uint8_t *data, uint16_t len)
74 {
75  uint16_t t;
76  const uint8_t *dataptr;
77  const uint8_t *last_byte;
78 
79  dataptr = data;
80  last_byte = data + len - 1;
81 
82  while(dataptr < last_byte) { /* At least two more bytes */
83  t = (dataptr[0] << 8) + dataptr[1];
84  sum += t;
85  if(sum < t) {
86  sum++; /* carry */
87  }
88  dataptr += 2;
89  }
90 
91  if(dataptr == last_byte) {
92  t = (dataptr[0] << 8) + 0;
93  sum += t;
94  if(sum < t) {
95  sum++; /* carry */
96  }
97  }
98 
99  /* Return sum in host byte order. */
100  return sum;
101 }
