149 {
150  /* Escape SLIP control characters */

218 /* Upper half does the polling. */
219 static uint16_t
220 slip_poll_handler(uint8_t *outbuf, uint16_t blen)
221 {
222  uint16_t len;
223  uint16_t pos;
224  uint8_t c;
225  uint8_t state;
226 
227  if(end_counter == 0 && is_full == 0) {
228  return 0;
229  }
230  for(len = 0, pos = begin, state = c = SLIP_NEUTRAL;
231  len < blen + 1; /* +1 for SLIP_END! */
232  ) {
233 
234  c = rxbuf[pos++];
235 
236  if(pos == RX_BUFSIZE) {
237  /* Circular buffer: warp around */
238  pos = 0;
239  }
240  if(c == SLIP_END) {
241  /* End of packet */
242  break;
243  }
244  if(len >= blen) {
245  /* End of buffer with no SLIP_END
246  * ==> something wrong happened */
247  break;
248  }
249  switch(c) {
250  case SLIP_ESC:
251  state = SLIP_ESC;
252  break;
253  case SLIP_ESC_END:
254  if(state == SLIP_ESC) {
255  outbuf[len++] = SLIP_END;
256  state = SLIP_NEUTRAL;
257  } else {
258  outbuf[len++] = c;
259  } break;
260  case SLIP_ESC_ESC:
261  if(state == SLIP_ESC) {
262  outbuf[len++] = SLIP_ESC;
263  state = SLIP_NEUTRAL;
264  } else {
265  outbuf[len++] = c;
266  } break;
267 #if UART_XONXOFF_FLOW_CTRL
268  case SLIP_ESC_XON:
269  if(state == SLIP_ESC) {
270  outbuf[len++] = XON;
271  state = SLIP_NEUTRAL;
272  } else {
273  outbuf[len++] = c;
274  } break;
275  case SLIP_ESC_XOFF:
276  if(state == SLIP_ESC) {
277  outbuf[len++] = XOFF;
278  state = SLIP_NEUTRAL;
279  } else {
280  outbuf[len++] = c;
281  } break;
282 #endif /* UART_XONXOFF_FLOW_CTRL */
283  default:
284  outbuf[len++] = c;
285  state = SLIP_NEUTRAL;
286  break;
287  }
288  }
289 
290  /* Update counters */
291  if(c == SLIP_END) {
292  ATOMIC(begin = pos;
293  if(end_counter) {
294  end_counter--;
295  }
296  )
297  PUTCHAR('P');
298  } else {
299  /* Something went wrong, no SLIP_END found, drop everything */
300  ATOMIC(rxbuf_init();
301  is_dropping = 1;
302  )
303  SLIP_STATISTICS(slip_error_drop++);
304  len = 0;
305  PRINTF("SLIP: *** out of sync!\n");
306  }
307 
308  if(end_counter > 0) {
309  /* One more packet is buffered, need to be polled again! */
310  process_poll(&slip_process);
311  }
312  return len;
313 }

315 PROCESS_THREAD(slip_process, ev, data)
316 {
317  PROCESS_BEGIN();
318 
319  rxbuf_init();
320 
321  while(1) {
322  PROCESS_YIELD_UNTIL(ev == PROCESS_EVENT_POLL);
323 
324  /* Move packet from rxbuf to buffer provided by uIP. */
325  uip_len = slip_poll_handler(&uip_buf[UIP_LLH_LEN],
326  UIP_BUFSIZE - UIP_LLH_LEN);
327 
328  PRINTF("SLIP: recv bytes %u frames RECV: %u. is_full %u, is_dropping %u.\n",
329  end_counter, uip_len, is_full, is_dropping);
330 
331  /* We have free space now, resume slip RX */
332  if(is_full) {
333  is_full = 0;
334  ENABLE_UART_INTERRUPTS();
335  }
336 
337  if(uip_len > 0) {
338  if(input_callback) {
339  input_callback();
340  }
341 #ifdef SLIP_CONF_TCPIP_INPUT
342  SLIP_CONF_TCPIP_INPUT();
343 #else
344  tcpip_input();
345 #endif
346  }
347  }
348 
349  PROCESS_END();
350 }

352 /* Return status from slip_input_byte:
353  * -1 means RX buffer overflow ==> stop reading
354  * 0 means do not exit power saving mode
355  * 1 means exit power saving mode
356  **/
357 int
358 slip_input_byte(unsigned char c)
359 {
360  static int in_frame = 0;
361  uint16_t next, next_next;
362  int error_return_code = is_full ? -1 : 0;
363  int success_return_code = is_full ? -1 : 1;
364 
365  SLIP_STATISTICS(slip_received++);
366 
367 #if UART_XONXOFF_FLOW_CTRL
368  if(c == XOFF || c == XON) {
369  xonxoff_state = c;
370  return 1;
371  } else {
372  /* ANY char would be XON */
373  xonxoff_state = XON;
374  }
375 #endif /* UART_XONXOFF_FLOW_CTRL */
376 
377  if(is_dropping) {
378  /* Make sure to drop full frames when overflow or
379  * out of sync happens */
380  if(c != SLIP_END) {
381  SLIP_STATISTICS(slip_drop_bytes++);
382  } else {
383  is_dropping = 0;
384  in_frame = 0;
385  }
386  return error_return_code;
387  }
388 
389  if(!in_frame && c == SLIP_END) {
390  /* Ignore slip end when not receiving frame */
391  return error_return_code;
392  /* increment and wrap */
393  }
394  next = end + 1;
395  if(next >= RX_BUFSIZE) {
396  next = 0;
397  }
398  next_next = next + 1;
399  if(next_next >= RX_BUFSIZE) {
400  next_next = 0;
401  /* Next byte will overflow. Stop accepting. */
402  }
403  if(next_next == begin) {
404  is_full = 1;
405  /* disable UART interrupts */
406  DISABLE_UART_INTERRUPTS();
407  process_poll(&slip_process);
408  }
409 
410  /* Buffer is full. We can't store anymore.
411  * Shall not happen normally,
412  * because of overflow protection above. */
413  if(next == begin) {
414  is_dropping = 1;
415  SLIP_STATISTICS(slip_overflow++);
416  is_full = 1;
417  /* disable UART interrupts */
418  DISABLE_UART_INTERRUPTS();
419  process_poll(&slip_process);
420  return -1;
421  }
422 
423  rxbuf[end] = c;
424  end = next;
425  in_frame = 1;
426 
427  if(c == SLIP_END) {
428  in_frame = 0;
429  end_counter++;
430  SLIP_STATISTICS(slip_frames++);
431  process_poll(&slip_process);
432  return success_return_code;
433  }
434  return error_return_code;
435 }
