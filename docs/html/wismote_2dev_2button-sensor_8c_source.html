46 {

49  if(BUTTON_CHECK_IRQ() && timer_expired(&debouncetimer)) {

64 static int
65 configure(int type, int value)
66 {
67  if(type == SENSORS_ACTIVE) {
68  if(value == 0) {
69  /* Deactivate button sensor */
70  BUTTON_DISABLE_IRQ();
71  } else {
72  /* Activate button sensor */
73  if(!status(SENSORS_ACTIVE)) {
74  timer_set(&debouncetimer, 0);
75  BUTTON_IRQ_EDGE_SELECTD();
76 
77  BUTTON_SELECT();
78  BUTTON_MAKE_INPUT();
79 
80  BUTTON_ENABLE_IRQ();
81  }
82  }
83  return 1;
84  }
85  return 0;
86 }
