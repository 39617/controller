120 static void
121 ipaddr_add(const uip_ipaddr_t *addr)
122 {
123  uint16_t a;
124  int i, f;
125  for(i = 0, f = 0; i < sizeof(uip_ipaddr_t); i += 2) {
126  a = (addr->u8[i] << 8) + addr->u8[i + 1];
127  if(a == 0 && f >= 0) {
128  if(f++ == 0 && sizeof(buf) - blen >= 2) {
129  buf[blen++] = ':';
130  buf[blen++] = ':';
131  }
132  } else {
133  if(f > 0) {
134  f = -1;
135  } else if(i > 0 && blen < sizeof(buf)) {
136  buf[blen++] = ':';
137  }
138  ADD("%x", a);
139  }
140  }
141 }

264 /* Format: <name=value>;<name=value>;...;<name=value>*/
265 /* this function just cut at ; and store in the sensor array */
266 void
267 border_router_set_sensors(const char *data, int len)
268 {
269  int i;
270  int last_pos = 0;
271  int sc = 0;
272  for(i = 0;i < len; i++) {
273  if(data[i] == ';') {
274  sensors[sc][i - last_pos] = 0;
275  memcpy(sensors[sc++], &data[last_pos], i - last_pos);
276  last_pos = i + 1; /* skip the ';' */
277  }
278  if(sc == MAX_SENSORS) {
279  sensor_count = sc;
280  return;
281  }
282  }
283  sensors[sc][len - last_pos] = 0;
284  memcpy(sensors[sc++], &data[last_pos], len - last_pos);
285  sensor_count = sc;
286 }
