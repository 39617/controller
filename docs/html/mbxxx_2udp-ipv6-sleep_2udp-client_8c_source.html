132 PROCESS_THREAD(udp_client_process, ev, data)
133 {
134  static struct etimer et, wake_timer, periodic_timer;
135  uip_ipaddr_t ipaddr;
136 
137  PROCESS_BEGIN();
138  PRINTF("UDP client process started\n");
139 
140 #if UIP_CONF_ROUTER
141  set_global_address();
142 #endif
143 
144  print_local_addresses();
145 
146  set_connection_address(&ipaddr);
147 
148  /* new connection with remote host */
149  client_conn = udp_new(&ipaddr, UIP_HTONS(3000), NULL);
150  udp_bind(client_conn, UIP_HTONS(3001));
151 
152  PRINTF("Created a connection with the server ");
153  PRINT6ADDR(&client_conn->ripaddr);
154  PRINTF(" local/remote port %u/%u\n",
155  UIP_HTONS(client_conn->lport), UIP_HTONS(client_conn->rport));
156 
157  etimer_set(&et, CLOCK_SECOND*10);
158  PROCESS_WAIT_UNTIL(etimer_expired(&et)); // Wait for DAD and Router Discovery procedure to end.
159 
160  etimer_set(&et, SEND_INTERVAL);
161  etimer_set(&wake_timer, AWAKE_INTERVAL);
162  etimer_set(&periodic_timer, 1);
163 
164  while(1) {
165  PROCESS_YIELD();
166  if(etimer_expired(&wake_timer)){ // if timer hasn't expired do not go in deep sleep, in order to receive a response.
167  printf("Sleeping...\r\n");
168  halBoardPowerDown();//sensorsPowerDown();
169  sleep_seconds(SLEEP_INTERVAL_SECONDS); // Put system in deep sleep mode for a while.
170  halBoardPowerUp();//sensorsPowerUp();
171  printf("Awake\r\n");
172  }
173  if(etimer_expired(&et)) {
174  timeout_handler();
175  etimer_restart(&et);
176  etimer_restart(&wake_timer);
177  } else if(ev == tcpip_event) {
178  tcpip_handler();
179  }
180 
181  /* Make the process be called almost immediately,
182  * so that it can force the system to go into deep sleep. */
183  etimer_restart(&periodic_timer);
184  }
185 
186  PROCESS_END();
187 }
