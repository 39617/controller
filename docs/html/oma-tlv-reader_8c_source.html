63 static size_t
64 read_string(const lwm2m_context_t *ctx, const uint8_t *inbuf, size_t len,
65  uint8_t *value, size_t stringlen)
66 {
67  oma_tlv_t tlv;
68  size_t size;
69  size = oma_tlv_read(&tlv, inbuf, len);
70  if(size > 0) {
71  if(stringlen <= tlv.length) {
72  /* The outbuffer can not contain the full string including ending zero */
73  return 0;
74  }
75  memcpy(value, tlv.value, tlv.length);
76  value[tlv.length] = '\0';
77  }
78  return size;
79 }
