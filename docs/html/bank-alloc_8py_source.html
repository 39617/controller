44 # Open a module object file (.rel) and read it's code size
45 def retrieve_module_size(file_name):
46  size_pat = re.compile('^A\s+(?:HOME|BANK[0-9])\s+size\s+([1-9A-F][0-9A-F]*)')
47  for code_line in open(file_name):
48  matches = size_pat.search(code_line)
49  if matches is not None:
50  return int(matches.group(1), 16)
51  return 0

53 # Searches for a code segment rule for file_name in the segment_rules file
54 # If there is a rule, we respect it. Otherwise, we can move the file around

119 # Allocate bankable modules to banks according to a simple
120 # 'first fit, decreasing' bin packing heuristic.
121 def bin_pack(modules, bins, offset, log):
122  if offset==1:
123  bins['HOME'][1] -= 4096
124 
125  # Sort by size, descending, in=place
126  modules.sort(key=operator.itemgetter(1), reverse=True)
127 
128  for module in modules:
129  # We want to iterate in a specific order and dict.keys() won't do that
130  for bin_id in ['HOME', 'BANK1', 'BANK2', 'BANK3', 'BANK4', 'BANK5', 'BANK6', 'BANK7']:
131  if bins[bin_id][0] + module[1] < bins[bin_id][1]:
132  bins[bin_id][0] += module[1]
133  module[2] = bin_id
134  log.writelines(' '.join([module[2].ljust(8), \
135  str(module[1]).rjust(5), module[0], '\n']))
136  break
137  else:
138  if bin_id == 'BANK7':

140  "to a code bank. This is fatal"
