37 /* Change this to 0x21 if J2 is set to 1-2

46 /* Cache the current state of each port to obviate the need for reading before

53 write_reg(uint8_t reg, uint8_t data)
54 {
55  uint8_t pkt[] = { reg, data };
56  assert(quarkX1000_i2c_polling_write(pkt, sizeof(pkt), I2C_ADDR) == 0);

60 read_reg(uint8_t reg)
61 {
62  uint8_t data;
63  assert(quarkX1000_i2c_polling_write(&reg, 1, I2C_ADDR) == 0);
64  assert(quarkX1000_i2c_polling_read(&data, 1, I2C_ADDR) == 0);
65  return data;
66 }

76  status = read_reg(0x2E);

87 void
88 cy8c9540a_set_port_dir(cy8c9540a_bit_addr_t addr, cy8c9540a_port_dir_t dir)
89 {
90  uint8_t mask;
91 
92  assert(addr.port < PORT_CNT);
93 
94  write_reg(REG_PORT_SEL, addr.port);
95  mask = read_reg(REG_PORT_DIR);
96  mask &= ~(1 << addr.pin);
97  mask |= ((uint8_t)dir) << addr.pin;
98  write_reg(REG_PORT_DIR, mask);
99 }

115 cy8c9540a_read(cy8c9540a_bit_addr_t addr)
116 {
117  assert(addr.port < PORT_CNT);

121 /*---------------------------------------------------------------------------*/
122 void
123 cy8c9540a_write(cy8c9540a_bit_addr_t addr, bool val)
124 {
125  assert(addr.port < PORT_CNT);
126 
127  out_cache[addr.port] &= ~(1 << addr.pin);
128  out_cache[addr.port] |= ((uint8_t)val) << addr.pin;
129  write_reg(8 + addr.port, out_cache[addr.port]);
