50 ctk_mouse_init(void)
51 {
52 #ifdef STATIC_MOUSE
53 
54  okay = mouse_install(&mouse_def_callbacks, &STATIC_MOUSE) == MOUSE_ERR_OK;
55  if(okay) {
56  atexit((void (*)(void))mouse_uninstall);
57  }
58 
59 #else /* STATIC_MOUSE */
60 
61  struct mod_ctrl module_control = {cfs_read};
62 
63  module_control.callerdata = cfs_open("contiki.mou", CFS_READ);
64  okay = module_control.callerdata >= 0;
65  if(okay) {
66  okay = mod_load(&module_control) == MLOAD_OK;
67  if(okay) {
68  okay = mouse_install(&mouse_def_callbacks, module_control.module) == MOUSE_ERR_OK;
69  if(okay) {
70  atexit((void (*)(void))mouse_uninstall);
71  } else {
72  mod_free(module_control.module);
73  }
74  }
75  cfs_close(module_control.callerdata);
76  }
77 
78 #endif /* STATIC_MOUSE */
79 }
