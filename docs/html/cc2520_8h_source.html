101 /* Write to a register in the CC2520 */
102 /* Note: the SPI_WRITE(0) seems to be needed for getting the */
103 /* write reg working on the Z1 / MSP430X platform */
104 #define CC2520_WRITE_REG(adr,data) \
105  do { \
106  CC2520_SPI_ENABLE(); \
107  SPI_WRITE_FAST(CC2520_INS_MEMWR | ((adr>>8)&0xFF)); \
108  SPI_WRITE_FAST(adr & 0xff); \
109  SPI_WRITE_FAST((uint8_t) data); \
110  SPI_WAITFORTx_ENDED(); \
111  CC2520_SPI_DISABLE(); \
112  } while(0)

115 /* Read a register in the CC2520 */
116 #define CC2520_READ_REG(adr,data) \
117  do { \
118  CC2520_SPI_ENABLE(); \
119  SPI_WRITE((CC2520_INS_MEMRD | ((adr>>8)&0xFF))); \
