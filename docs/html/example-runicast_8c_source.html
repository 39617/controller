58 /* OPTIONAL: Sender history.
59  * Detects duplicate callbacks at receiving nodes.
60  * Duplicates appear when ack messages are lost. */
61 struct history_entry {
62  struct history_entry *next;
63  linkaddr_t addr;
64  uint8_t seq;
65 };
66 LIST(history_table);
67 MEMB(history_mem, struct history_entry, NUM_HISTORY_ENTRIES);
68 /*---------------------------------------------------------------------------*/
69 static void
70 recv_runicast(struct runicast_conn *c, const linkaddr_t *from, uint8_t seqno)
71 {
72  /* OPTIONAL: Sender history */
73  struct history_entry *e = NULL;
74  for(e = list_head(history_table); e != NULL; e = e->next) {
75  if(linkaddr_cmp(&e->addr, from)) {
76  break;
77  }
78  }
79  if(e == NULL) {
80  /* Create new history entry */
81  e = memb_alloc(&history_mem);
82  if(e == NULL) {
83  e = list_chop(history_table); /* Remove oldest at full history */
84  }
85  linkaddr_copy(&e->addr, from);
86  e->seq = seqno;
87  list_push(history_table, e);
88  } else {
89  /* Detect duplicate callback */
90  if(e->seq == seqno) {
91  printf("runicast message received from %d.%d, seqno %d (DUPLICATE)\n",
92  from->u8[0], from->u8[1], seqno);
93  return;
94  }
95  /* Update existing history entry */
96  e->seq = seqno;
97  }
98 
99  printf("runicast message received from %d.%d, seqno %d\n",
100  from->u8[0], from->u8[1], seqno);
101 }
